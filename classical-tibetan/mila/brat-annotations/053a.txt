བླ་མའི་ཞལ་ནས། བུ་དེ་རྣམས་ཡིན། ཕ་རྒན་ངས་ཁྱོད་རང་ལ་འཆད་པའི་སྙིང་གཏམ་གྱི་ཐ་མ་འདི་རྣམས་མ་བརྗེད་པར་སེམས་ལ་ཆོངས་ཞིག་གསུང་། ཕྱག་ངའི་མགོ་ཐོག་ཏུ་བཞག་ནས་མགུར་འདི་གསུངས་སོ། ། ། ། རྗེ་བླ་མ་རྣམས་ལ་ཕྱག་འཚལ་ལོ། ། ལས་ཅན་ཆོས་དང་འཐུན་པའི་བུ། ། ཆོས་སྐུ་སངས་རྒྱས་གྲུབ་པར་ཤོག ངག་རྡོ་རྗེའི་བཟླས་པ་བདུད་རྩིའི་རོས། ། རྟེན་འབྲེལ་ལོངས་སྐུར་གྲུབ་པར་ཤོག སེམས་རྩ་བ་བྱང་ཆུབ་སྡོང་བོ་ལ། ། སྤྲུལ་སྐུའི་ལོ་འདབ་རྒྱས་པར་ཤོག བླ་མའི་བཀའ་ཚིག་རྡོ་རྗེའི་གསུང་། ། སྙིང་ལ་བརྗེད་མེད་གནས་པར་ཤོག ཡི་དམ་མཁའ་འགྲོའི་བྱིན་རླབས་དེ། ། སྲོག་གི་རྩ་བར་གནས་པར་ཤོག ཆོས་སྐྱོང་བསྲུངས་མའི་རྒྱབ་རྟེན་དེས། ། འབྲལ་བ་མེད་པར་སྐྱོང་བར་ཤོག རྟེན་འབྲེལ་ཟབ་མོའི་སྨོན་ལམ་དེ། ། མྱུར་བ་ཉིད་དུ་གྲུབ་པར་ཤོག ཆོས་མཛད་ཀུན་གྱི་ཐུགས་རྗེ་དེས། ། དུས་གསུམ་ཀུན་ཏུ་འཛིན་པར་ཤོག གཙང་སིལ་མ་ལ་ཡི་ལ་ཐོག་ཏུ། ། བརྟན་མ་བཅུ་གཉིས་ཀྱི་སུན་མ་ཡོད། ། སང་ཉིན་འགྲོ་ཡི་ལམ་བར་ལ། ། དཔའ་བོ་མཁའ་འགྲོའི་སྐྱེལ་མ་ཡོད། ། ཕ་ཡུལ་ཞེན་པའི་མཁར་ཞིང་ན། ། མི་རྟག་སྒྱུ་མའི་སློབ་དཔོན་ཡོད། ། ཨ་ནེ་སྲིང་མོའི་ཉེ་འཁོར་ལ། ། སྒྱུ་མ་ཞིག་པའི་སློབ་དཔོན་ཡོད། ། མི་མེད་ལུང་པའི་བྲག་ཕུག་ན། ། འཁོར་འདས་བརྗེད་བའི་ཚོང་འདུས་ཡོད། ། སྙིང་རུས་ལུས་ཀྱི་དགོན་པ་ན། ། ལྷ་བདེ་གཤེགས་འདུས་པའི་མཆོད་ཁང་ཡོད། ། ནད་མེད་ཟ་ཡི་ཚོགས་འཁོར་ལ། ། མཁའ་འགྲོ་མཉེས་པའི་བདུད་རྩི་ཡོད། ། འཕྲུལ་འཁོར་གནད་ཀྱི་ཁ་བསྒྱུར་ལ། ། འབྲས་རིན་ཆེན་སྐྱེ་བའི་སོ་ནམ་ཡོད། ། ཡུལ་མི་རྩིགས་ཆུང་རང་ཡུལ་དེར། ། འཕྲལ་གཡེང་བ་མེད་པའི་དགེ་སྦྱོར་ཡོད། ། འཚམས་དམ་པོ་མི་མེད་ཁྱི་མེད་ན། ། རྟགས་མྱུར་དུ་མཐོང་བའི་སྒྲོན་མེ་ཡོད། ། ཟས་དཀོར་ཟས་མེད་པའི་རང་ཟས་ལ། ། བདེ་བ་ལྷ་ཡི་ཕམ་ཕབས་ཡོད། །
