ང་རི་ཁྲོད་ལ་སྒྲུབ་པ་བྱེད་པ་འདི་བླ་མའི་ཞལ་ཏའི་མཐིལ་ཡང་ཡིན། རྣམ་ཐར་ཡང་དེ་ཀས་འཁྱོངས། ཐུགས་དགོངས་ཡང་སྒྲུབ་པས་རྫོགས་པ་ཡོད། བསྟན་པ་དང་སེམས་ཅན་ལ་ཡང་སྒྲུབ་པས་ཕན། ཕ་མ་ཡང་སྒྲུབ་པས་འདྲོངས། རང་དོན་ཡང་སྒྲུབ་པས་གྲུབ། ངས་སྒྲུབ་པ་མིན་པ་གཞན་ཤེས་ཡང་མི་ཤེས། ཡོང་ཡང་མི་ཡོང་། སེམས་ཡང་མི་སེམས་པར་གདའ། ཁྱད་པར་ཡུལ་དུ་སླེབ་ནས་ཕ་མས་ཁང་གཞིས་བཟུང་། ནོར་བསགས་པའི་རྗེས་རྣམས་ཀྱི་དཔེས་གསལ་བཏབ་པས་སྒྲུབ་པ་བྱེད་པའི་བསམ་པ་དྲག་ཏུ་རང་སོང་བས་ཁོང་ན་མེ་འབར་བ་ཙམ་གདའ། གཞན་སྡུག་བསྔལ་འདི་འདྲ་ཐོག་ཏུ་མ་བབ་པ་དང་། འཆི་བ་དང་ངན་སོང་གི་སྡུག་བསྔལ་མ་དྲན་པ་རྣམས་ལ་སྐྱིད་ཆོས་ཀྱིས་ཆོག་པར་གདའ་སྟེ། བདག་ལྟོ་གོས་གཏམ་གསུམ་ལ་མ་ལྟོས་པའི་སྒྲུབ་པ་བྱེད་སྙམ་པ་རྐྱེན་འདི་དག་གིས་བསྐུལ་ཞིང་། དོན་འདི་རྣམས་ལགས་བྱས་མཆི་མ་འཁྲུག་བཞིན་པའི་ངང་ནས་གླུ་འདི་བླངས་སོ། ། སྐྱེས་མཆོག་མར་པའི་ཞབས་ལ་འདུད། ། སྤྲང་མངོན་ཞེན་ལྡོག་པར་བྱིན་གྱིས་རློབས། ། ཀྱེ་མ་ཀྱེ་མ་ཀྱི་ཧུད་ཀྱི་ཧུད་ཨང་། ། འཁོར་བའི་ཆོས་ལ་བློ་གཏད་བྱེད་པ་རྣམས། ། བསམས་ཤིང་བསམས་ཤིང་ཡི་མུག་ཡང་ཡང་ལྡང་། ། སྤྱད་ཅིང་སྤྱད་ཅིང་སྡུག་བསྔལ་གཏིང་ནས་འཁྲུག སྐོར་ཞིང་སྐོར་ཞིང་འཁོར་བའི་གཏིང་ལ་འཕེན། ། འདི་འདྲའི་སྙིང་རླུང་ལས་ཀྱིས་ཟིན་པ་རྣམས། ། ཙུག་བྱེད་ཙུག་བྱེད་ཆོས་ལས་དྲག་པ་མེད། ། རྗེ་མི་བསྐྱོད་ཀྱི་ངོ་བོ་རྡོ་རྗེ་འཆང་། ། སྤྲང་རི་ཁྲོད་ཐུབ་པར་བྱིན་གྱིས་རློབས། ། མི་རྟག་སྒྱུ་མའི་གྲོང་ཁྱེར་ན། ། ཤུལ་རིངས་ཀྱི་འགྲོན་པོ་སྙིང་རླུང་ལངས། ། ཡུལ་ཡ་མཚན་ཅན་གུང་ཐང་གི་སྐོར་ལུང་ལ། ། ནོར་ལུག་རྣམས་བ་རའི་ཁ་ཟས་རྩ། ། དུས་དིང་སང་འབྱུང་པོས་བདག་ཏུ་བཟུང་། ། འདི་ཡང་མི་རྟག་སྒྱུ་མའི་དཔེ། ། དཔེ་འདིས་ཀྱང་རྣལ་འབྱོར་ང་སྒྲུབ་པ་བྱེད། །
