བསོད་ནམས་འཛད་ཐབས་བསམ་ངན་དང་། ། སྦྱོར་རྩུབ་སུ་ཡང་མི་བྱེད་ཤོག ། བསམ་པ་བཟང་པོའི་བྱང་ཆུབ་སེམས། ། སེམས་ཅན་ཀུན་གྱིས་ཐོབ་པར་ཤོག ། ཅེས་གསུངས་པས། ཁོ་དགའ་སྤྲོ་དཔག་ཏུ་མེད་པ་ཐོབ་ནས། རྗེ་བཙུན་གྱི་བཀའ་བཞིན་ད་ཕྱིན་ཆད་ཆོས་མིན་གྱི་ལས་མི་བགྱིད་པར་ཚེ་དང་སྒྲུབ་པ་སྙོམ་ལགས། སྔར་གྱི་སྡིག་པ་ཡང་ཁང་གཞིས་དང་ནོར་གྱིས་ལན་པ་ལགས་པས། ད་བདག་རང་ལའང་ཁང་ཞིང་དང་། ནོར་གྱིས་དགོས་པ་མེད་པས། རྗེ་བཙུན་མི་བཞེས་རུང་བུ་སློབ་རྣམས་ཀྱི་སྒྲུབ་པའི་འཐུན་རྐྱེན་དུ་ཅིས་ཀྱང་བཞེས་དགོས་ཞེས། ཞུ་བ་ནན་གྱིས་ཕུལ་རུང་རྗེ་བཙུན་གྱིས་མ་བཞེས་པ། ཕྱིས་བུ་སློབ་རྣམས་ཀྱིས་བཞེས་ནས་རྗེ་བཙུན་གྱི་དུས་མཆོད་ཀྱི་བཞི་རྐྱེན་དུ་བཙུགས་པའི་མཆོད་པ་ཆུ་བར་ན་དུས་ད་ལྟ་ཡང་ཡོད་དོ། ། དགེ་བཤེས་རྩག་ཕུ་པ་དེ་ཡང་ཕྱིས་ཚེ་འདི་བློས་བཏང་བའི་ཆོས་པ་དགའ་མོ་ཞིག་བྱུང་བ་ཡིན་ནོ། ། དེ་ནས་རྗེ་བཙུན་གྱི་ཞལ་ནས། གནས་འདིར་བསྡད་པའི་དགོས་པ་སྡིག་པོ་ཆེས་འགྱོད་ཚངས་ཕུལ་ནས་སྡིག་བ་ལས་གྲོལ་བ་དེ་ཡིན། ང་ཡང་སྒོམ་ཆེན་གྲོང་ཡུལ་དུ་འཆི་བ་རྒྱལ་པོ་དམངས་ཁྱིམ་ཏུ་འཆི་བ་དང་འདྲ་བས། ད་ཆུ་བར་དུ་དུར་ས་སྙོག་ཏུ་འགྲོ་གསུང་བ་ལ། སེ་བན་རས་པས། འོ་ན་སྙུང་གཞི་འདིས་འབྱོན་པ་འོ་རྒྱལ་བ་གདའ་བས་འདོ་ལི་ལ་གདན་འདྲེན་དགོས་ཞུས་པས། ང་ནའི་ཡོང་བའི་ངེས་པ་ཡང་མེད། འཆི་བའི་ངེས་པ་ཡང་མེད་དེ། ན་ཚུལ་འདིར་བསྟན། འཆི་ཚུལ་སྟོན་དུ་ཆུ་བར་དུ་འགྲོ། དེ་ལ་འདོ་ལི་མི་དགོས། རས་པ་གཞོན་པ་འགའ་ཆུ་བར་དུ་སྔོན་ལ་སོང་གསུང་། དེར་གཞོན་པ་རྣམས་ཀྱིས་ཆུ་བར་དུ་ཕྱིན་པས། རྗེ་བཙུན་འབྲི་ལྕེ་ཕུག་ན་སྔོན་ལ་ཕེབས་འདུག གྲྭ་པ་བགྲེས་པ་རྣམས་ལ་ཕྱག་ཕྱི་བྱས་ཡོང་རྒྱུ་ཅིག་བྱུང་། ཕ་བོང་རེག་པ་དུག་ཅན་གྱི་སྟེང་ན་སྙུང་ཚུལ་སྟོན་མཁན་གཅིག ཆུ་བར་དུ་ཕྱིས་མཇལ་དུ་འགྲོ་བའི་ཉ་མ་དད་པ་ཅན་རྣམས་ལ་ཕྱགས་ཕྱི་བྱེད་རྒྱུ་གཅིག སྡིང་མའི་ཡོན་བདག་ཕྱིན་ཡོད་པ་རྣམས་ལ་བྲག་མཁར་སྟེང་དུ་ཆོས་གསུང་མཁན་གཅིག ནང་ན་གནས་པ་རྣམས་རེ་རེའི་མདུན་ན་བསྙེན་བཀུར་གསོལ་མཁན་རེ་རེ་བྱུང་།
