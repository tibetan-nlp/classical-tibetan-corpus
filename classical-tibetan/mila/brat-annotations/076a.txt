མཚན་མ་ཡོད་པ་སུས་ཀྱང་ཤེས། ། ཁྲེལ་དང་ངོ་ཚ་བསྲུང་བ་རྣམས། ། འཇིག་རྟེན་ཕལ་ལ་མི་འདུག་སྟེ། ། ངོ་ཚའི་བུ་མོ་རིན་གྱིས་ཉོས། ། ངོ་ཚའི་བུ་ཕོ་པང་དུ་བླངས། ། བརྣབ་སེམས་གནོད་སེམས་ལོག་ལྟའི་འབྲས། ། ངན་སྤྱོད་གཡོ་སྒྱུ་རྐུ་འཕྲོག་དང་། ། བློ་གཏད་ཉེ་བ་སླུ་བ་རྣམས། ། ཁྲེལ་ངོ་ཚ་ཡིན་ཏེ་འཛེམ་མཁན་ཉུང་། ། ཚེ་འདི་བློས་བཏང་གི་སྒོམ་ཆེན་ཀུན། ། ཉམས་ལེན་གནད་ཀྱི་མན་ངག་མཆོག ། ཟབ་མོ་རྡོ་རྗེ་གསང་སྤྱོད་ཀྱིས། ། མི་ཚེ་ཆོས་ལ་སྐྱེལ་བ་ལ། ། བཅོས་མའི་ངོ་ཚ་བྱེད་དོན་མེད། ། དེ་ཕྱིར་རང་སྡུག་མ་ཤོམ་པར། ། པེ་ཏ་ཤེས་པ་རང་སོར་ཞོག ། བྱས་པས། ངོ་ནག་ཐུག་གེ་བའི་ངང་ནས་སློངས་མོ་ལས་བྱུང་བའི་བརྒྱགས་ཆུ་སྡོར་རྣམས་ཕུལ་ནས། གང་བྱས་རུང་ང་མནོས་པ་བཞིན་ཨ་ཇོས་མི་བྱེད་པར་འདུག་སྟེ། ངས་ཨ་ཇོ་བློས་མི་ཐོངས་པར་འདུག་པས་འདི་ཀུན་གསོལ་མཛོད། ངས་ད་དུང་ཅི་རྙེད་བྱས་ཡོང་གིས་ཟེར་འགྲོ་གྲབས་བྱེད་ཅིང་འདུག་པ་ལ། ངས་པེ་ཏ་བློ་ཆོས་ལ་འགྱུར་དུ་ཨེ་རུང་བསམ། ཁྱོད་རང་ཡང་ཆས་འདི་རྣམས་མ་ཟད་བར་ཆོས་མི་བྱེད་རུང་སྡིག་མ་སོག་པར་སྡོད་བྱས་བསྡད་པའི་བར་ལ། རྣ་བར་ལས་རྒྱུ་འབྲས་ཀྱི་ཆོས་ཅི་འགྲོ་བྱས་པས། ཆོས་ལ་ངེས་ཤེས་འགྲོངས་ནས་བློ་སྣ་ཅུང་ཟད་ཐུང་དུ་སོང་བའི་ཚེ། ཨ་ནེ་ཡང་ཨ་ཁུ་ཤི་རྟིང་ཁ་ཞེ་མེད་པའི་འགྱོད་པ་སྐྱེས་ཏེ། མཇོ་ཡུ་གཅིག་ལ་ཁལ་བཀལ་ནས་བྲིན་གྱི་བར་དུ་རྩད་གཅོད་ཅིང་ཡོངས། བྲིན་དུ་སླེབ་པ་དང་། ཆས་སྤྱི་དང་མཛོ་བཅོལ་བའི་ཐོག་ནས། ཨ་ནེ་རང་གིས་ཅི་ཐེག་ཅིག་ཁྱེར་ནས་ཡོངས་པ་པེ་ཏ་སྤོ་མཐོ་ན་ཡོད་པས་མཐོང་སྟེ། ཨ་ནེ་ཡིན་པར་ཤེས་ནས། ཨ་ནེས་འོ་ཅག་མ་བུ་ལ་སྡུག་སྣ་དགུ་བཏང་བས་མ་ཕྲད་པ་གྲག་ཟེར་ཕུག་པའི་འགག་ན་བཅྭའ་ཟམ་ཞིག་ཡོད་པ་དེ་བསྡོགས་པས།
