འཁོས་མེད་ཀྱི་རྣལ་འབྱོར་ངོ་ཤེས་ནས། ། ཞེ་སྡང་གནོད་སེམས་བསམ་སྦྱོར་ནི། ། ཁྱི་ཕྱོ་ཕྱོ་ཟེར་ཞིང་སྦོད་དུ་བྱུང་། ། སྦྲ་ཡི་ཚུག་ཤིང་བེར་ཀ་ཡིས། ། ལུས་སྙེ་ཕུང་བཞིན་དུ་བརྡུངས་པས་ན། ། ཆབ་རྐོ་མོའི་ནང་དུ་ཁ་བུབ་འགྱེལ། ། རིན་ཆེན་སྲོག་དང་ངོ་བྲལ་བྱུང་། ། རླ་འཚོང་བྲེང་ཞིང་འཐབ་པ་དང་། ། རྐང་འདྲེན་ཟེར་ཞིང་འབར་བ་ཡི། ། ཚིག་རྩུབ་མཚོན་གྱིས་སྙིང་སྣད་པས། ། སྡུག་བསྔལ་ཡི་ཆད་སྙིང་རླུང་སླང་། ། དབུགས་སྟོད་དུ་འཚངས་ནས་སྐད་མ་ཐོན། ། དེ་རྗེས་བདག་ལ་དགོས་མེད་ཀྱང་། ། གཡོ་ཐབས་དུ་མས་ཁང་ཞིང་ཕྲོགས། ། ཨ་ནེའི་ལུས་ལ་བདུད་མོའི་སེམས། ། ཨ་ནེའི་ཁོ་ཐག་དེ་ནས་ཆོད། ། ཨ་ཁུའི་སྒོ་རུ་སླེབ་པའི་ཚེ། ། བསམ་ངན་སྦྱོར་རྩུབ་དབེན་ཚིག་ནི། ། ཡུལ་གྱི་ཕུང་འདྲེས་སླེབས་ཡོད་ཟེར། ། ཁྱིམ་མཛེས་གསོད་པའི་ར་མདར་སྦྲན། ། མི་སྙན་ཁ་ནས་ཐོན་གསུམ་སྨྲས། ། གོར་མོ་རྡོ་ཡི་ཆར་བ་ཕབ། ། ཉག་ཕྲན་མདའ་ཡི་ཟམ་པ་བཙུགས། ། སྙིང་ལ་བཟོད་བྲལ་ནད་ཀྱིས་བཏབ། ། དེར་ཡང་སྲོག་དང་དོ་བྲལ་བྱུང་། ། ཨ་ཁུའི་ལུས་ལ་གཤེད་མའི་སེམས། ། ཨ་ཁུའི་ཁོ་ཐག་དེ་ནས་ཆོད། ། རང་འཁོས་མེད་ཀྱི་ཉེ་བ་དགྲ་བས་སྡང་། ། དེ་རྗེས་རི་ལ་སྒོམ་པའི་ཚེ། ། ཕྱི་ཐག་ཅན་གྱི་མཛེས་སེ་དེས། ། མ་ཕོད་བརྩེ་བས་ཕྲད་དུ་བྱུང་། ། བློ་དང་འཚམ་པའི་སྙིང་གཏམ་གྱིས། ། སྡུག་བསྔལ་ཅན་ལ་སེམས་གསོ་བཏང་། ། རོ་བཅུད་ལྡན་པའི་ཁ་ཟས་ཀྱིས། ། དབུལ་ཕོངས་བདག་གི་བཀྲེས་སྐོམ་བསལ། ། བཀའ་དྲིན་ཆེ་བར་ཁུམས་ལགས་ཀྱང་། ། སྐྱེས་བུ་ཆོས་མཛད་མ་ལགས་པ། ། མཛེས་སེ་བྱུང་རུང་ཕྲད་དོན་མེད། ། ཨ་ནེ་ཕྲད་དོན་དེ་བས་ཆུང་། ། ད་ལྟ་ཉིད་དུ་སླར་ལོག་ལ། ། སྔ་བའི་དུས་སུ་འགྲོ་འདུག་རན། ། བྱས་པས་ཨ་ནེ་མཆི་མ་མང་པོ་གཏོང་ཞིང་ཕྱག་མང་དུ་བཙལ། སྔར་ཕན་ཆད་ཚ་བོ་བདེན་བཟོད་པ་བཞེས་པར་ཞུ་ཞེས་ཁ་ཞེ་མེད་པའི་བཤགས་པ་ཕུལ།
