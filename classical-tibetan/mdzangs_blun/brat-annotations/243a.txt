རོལ་མོ་སྙན་པ་ཐོས་ནས་དེ་དག་དགའ་སྟེ། ཁྱོད་ངའི་སྐྱེད་མོས་ཚལ་དུ་རོལ་མོ་བྱེད་ཅིང་སྐྱེད་མོས་ཚལ་ཡང་སྲུང་ན། མི་ནུས་སམ་ཞེས་དྲིས་ན། བདག་གིས་མིག་ལོང་ནས་མི་ནུས་སོ་ཞེས་སྨྲས་སོ། ། དེས་སྨྲས་པ། ཁྱོད་མིག་མ་མཐོང་ཡང་། ཤིང་གི་ཁ་ན་ཐེ་གུ་ལ་དྲིལ་བུ་བཏགས་ཏེ། བྱའི་སྐད་གྲགས་ན་ཐེ་གུ་དྲངས་པས་ཆོག་གི། ངའི་སྐྱེད་མོས་ཚལ་དུ་འདོང་ངོ་ཞེས་སྨྲས་སོ། ། རྒྱལ་བུ་དེའི་ཚལ་དུ་འདུག་ཅིང་རོལ་མོ་བྱེད་དོ། ། དེའི་ཚེ་ན་སྡིག་དོན་ཕྱིར་འོངས་བ་དང་། རྒྱལ་པོས་ཁྱོད་ཀྱི་ཕུ་བོ་དང་ཚོང་པ་གཞན་དག་ག་རེ་ཞེས་དྲིས་ན། བདག་གི་ཕུ་བོ་དགེ་དོན་དང་ཚོང་པ་གཞན་ནི་ནོར་བུ་རིན་པོ་ཆེ་མང་དུ་རྙེད་ནས་གྲུ་དང་བཅས་ཏེ། ཆུར་ནུབ་བོ། ། བདག་ནི་རྐྱལ་འཚལ་བས་བདག་འབའ་ཞིག་ཐར་རོ། ། ཞེས་སྨྲས་པ་དང་། རྒྱལ་པོ་ཡབ་དང་བཙུན་མོ་གཉི་ག་འཁམས་ནས་ས་ལ་འགྱེལ་ལོ། ། ཡུལ་ན་མི་འཁོད་པ་གཞན་དག་ཀྱང་དེ་སྐད་ཐོས་ནས་ངོ་མཚར་དུ་གྱུར་ནས་མྱ་ངན་གྱིས་བརྣངས་སོ། ། རྒྱལ་པོ་ཡབ་དང་བཙུན་མོ་གཉིས་མྱ་ངན་གྱིས་ཡིད་གདུངས་ནས། སྡིག་དོན་ཁྱོད་ཀྱང་ཕུ་བོ་དང་ལྷན་ཅིག་འགྲོ་ན་ཅི་ཉེས་ཅེས་བསྒོའོ། ། རྒྱལ་བུ་དགེ་དོན་འདུག་པའི་ཚེ་བྱ་ངང་པ་གཅིག་གསོས་པ་དེ་ལ་རྒྱལ་པོས་འདི་སྐད་ཅེས་བསྒོའོ། ། རྒྱལ་བུ་དགེ་དོན་རྒྱ་མཚོར་རིན་པོ་ཆེ་ལེན་དུ་སོང་བ་ལས། རྒྱ་མཚོར་ནུབ་སྟེ་དགེ་དོན་དག་ཅེས་ཟེར་ཏེ། ཁྱོད་རྟོག་ཅིང་ཚོལ་དུ་སོང་ཤིག་ཅེས་བསྒོ་བ་དང་། ཡི་གེ་བྲིས་ཏེ་མགུལ་དུ་བཏགས་ནས་སོང་ངོ་། ། དེ་ནས་བྱ་ངང་པ་ཀུན་ཏུ་འཕུར་ཞིང་ཚོལ་ཚོལ་བ་ལས། སྐྱེད་མོས་ཚལ་དེ་ན་རྒྱལ་བུའི་སྐད་གྲགས་ནས། དེར་བབས་ཏེ་བལྟས་ན་རྒྱལ་བུ་དེར་འདུག་པ་དང་ཕྲད་དོ། ། རྒྱལ་བུས་ཀྱང་ངང་པའི་སྐད་ཤེས་པས་མགུལ་ནས་ཡི་གེ་བཀྲོལ་ཏེ་ཀློག་ཏུ་བཅུག་གོ། ། དེ་ནས་ཡི་གེ་ལན་བྲིས་ནས་སྡིག་དོན་གྱིས་ཅི་ལྟར་མིག་ཏུ་ཚེར་མ་བཙུགས་ཏེ་བྲོས་པ་ཡང་ཞིབ་ཏུ་སྤྲིང་ངོ་། ། དེའི་ཚེ་ན་རྒྱལ་པོ་ལི་ཤི་བར་དེ་ལ་བུ་མོ་འཇིག་རྟེན་ན་དཀོན་པ་ཞིག་ཡོད་དེ། བུ་མོ་སྐྱེད་མོས་ཚལ་དུ་འཆག་ཅིང་སོང་སོང་བ་ལས་རྒྱལ་བུ་དགེ་དོན་འདུག་པ་དང་ཕྲད་ནས་རྒྱལ་བུ་དགེ་དོན་སྐྲ་ནི་འཛིངས།
