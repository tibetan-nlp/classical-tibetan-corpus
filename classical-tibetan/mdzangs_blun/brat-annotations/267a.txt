ད་སངས་རྒྱས་དང་ཕྲད་དེ། རབ་ཏུ་བྱུང་ནས་ལུང་བསྟན་པ་ཡིན་ནོ། ། དེའི་ཚེ་འཁོར་མང་པོ་རྣམས་སངས་རྒྱས་ཀྱིས་གསུངས་པ་ལ་རབ་ཏུ་དགའ་བར་གྱུར་ཏོ། ། དེའི་ཚེ་རྒྱལ་པོ་དང་། བློན་པོ་དང་། མི་མང་པོ་དག་གིས་བུད་མེད་དབུལ་མོས་སངས་རྒྱས་ལ་མར་མེ་ཕུལ་བས་སངས་རྒྱས་སུ་ལུང་བསྟན་ཏོ་ཞེས་ཐོས་ནས་ཐམས་ཅད་ཀྱིས་བཀུར་སྟི་བྱས་ཏེ། འཚོ་བའི་ཡོ་བྱད་མཆོག་དམ་པ་རྣམ་པ་བཞི་ཆོག་པར་སྦྱར་ཏེ་ཕུལ་ལོ། ། དེའི་ཚེ་ཡུལ་དེའི་ཁྱེའུ་དང་། བུ་མོ་ཡ་རབས་དང་། མ་རབས་ཐམས་ཅད་ཀྱིས་མར་མེ་དྲི་བསུང་ཅན་སྦྱར་ཏེ། རྒྱལ་བུ་རྒྱལ་བྱེད་ཀྱི་ཚལ་དུ་བཅོམ་ལྡན་འདས་ལ་ཕུལ་ལོ། ། མར་མེ་ཕུལ་བའི་མི་མང་པོས་མེ་མར་གྱིས་རྒྱལ་བུ་རྒྱལ་བྱེད་ཀྱི་ཚལ་ཀུན་ཡོངས་སུ་གང་སྟེ། ཕྱོགས་བཞི་ནས་སྐར་མ་ནམ་མཁའ་ལ་སྣང་བ་དང་འདྲ་བར་ཉིན་གཅིག་བཞིན་དུ་ཕུལ་ཏེ། དེ་ལྟ་བུར་རྒྱུན་མ་ཆད་པར་ཉིན་ཞག་བདུན་གྱི་བར་དུ་ཕུལ་ལོ། ། དེའི་ཚེ་ཀུན་དགའ་བོ་ཤིན་ཏུ་དགའ་ནས་ངོ་མཚར་དུ་གྱུར་ཏེ། བཅོམ་ལྡན་འདས་ཀྱི་ཡོན་ཏན་ནི་འདི་སྙེད་ཅིག་མངའོ་ཞེས་བསྔགས་ནས། བཅོམ་ལྡན་འདས་ལ་འདི་སྐད་ཅེས་གསོལ་ཏོ། ། བཅོམ་ལྡན་འདས་ཀྱིས་སྔོན་ཕྲིན་ལས་ཅི་ཞིག་བགྱིས་ན། འདི་ལྟ་བུའི་མར་མེ་ཚད་མེད་པ་དང་ལྡན་པར་གྱུར། བཅོམ་ལྡན་འདས་ཀྱིས་ཀུན་དགའ་བོ་ལ་བཀའ་སྩལ་པ། སྔོན་འདས་པའི་དུས་ན་འཛམ་བུའི་གླིང་འདིར་རྒྱལ་པོ་པ་སི་ལི་ཞེས་བྱ་བ་ཞིག་རྒྱལ་ཕྲན་བརྒྱད་ཁྲི་བཞི་སྟོང་སྙེད་ལ་དབང་བྱེད་དོ། ། རྒྱལ་པོའི་བཙུན་མོ་ལ་སྲས་གཅིག་བྱུང་སྟེ། ལུས་ཀྱི་མདོག་གསེར་ལྟ་བུར་འདུག་ལ་མཚན་སུམ་ཅུ་རྩ་གཉིས་དང་། དཔེ་བྱད་བཟང་པོ་བརྒྱད་ཅུར་ལྡན་ཏེ། སྤྱི་བོ་ལ་རིན་པོ་ཆེ་ཁ་དོག་གསལ་བ། བལྟ་ན་མིག་སྦྲིད་པ་ཞིག་ཡོད་དོ། ། མཚན་མཁན་བོས་ནས་བཟང་ངན་བརྟགས་ཏེ། མིང་བཏགས་པ་ལས། མཚན་མཁན་གྱིས་བལྟས་ན། མཚན་ངོ་མཚར་དུ་བཟུང་བས། ལག་པ་བཏེགས་ནས་ཤིན་ཏུ་ལེགས་སོ། ། ཞེས་སྨྲས་ཏེ། རྒྱལ་བུ་འདི་ནི་འཇིག་རྟེན་ན་ལྷ་དང་མིའི་ནང་དུ་བསྒྲུན་དུ་མེད་དོ། །
