གལ་ཏེ་ཆུང་མར་མ་བླངས་ན་བདག་ཀྱང་འགུམ་མོ་ཞེས་སྨྲས་པ་དང་། ཕ་མས་ཀྱང་བུའི་ངོ་མ་བཟློག་པས་བུ་མོ་སློང་དུ་བཏང་ངོ་། ། བུ་མོའི་ཕ་མས་སྨྲས་པ། ཁྱོད་ནི་ཆོ་རིགས་བཙུན་པ་ཡིན་ཏེ། བདག་ནི་རིགས་ཆུང་ཞིང་ངན་པ་ཡིན་པས་གཉེན་ཟླ་ཡང་མ་ཡིན་ན། ཅིའི་ཕྱིར་བུ་མོ་སློང་ཞེས་བྱས་སོ། ། འོན་ཀྱང་ཁྱེའུ་དེ་བུ་མོ་དེ་ལ་ཆགས་སོ། ། ཡང་བསྐྱར་ཏེ་བུ་མོ་སློང་དུ་བཏང་བ་དང་། བུ་མོའི་ཕ་མས་འདི་སྐད་སྨྲས་སོ། ། གལ་ཏེ་བདག་གི་དམག་པར་བྱ་བ་རྩལ་སྣ་ཚོགས་ཀྱང་ཡོད་ལ། གླུ་དང་གར་དང་རྩེད་མོའི་བག་ཅི་ཡང་ཤེས་ཤིང་རྒྱལ་པོ་ཅི་ལ་དགྱེས་པ་ཀུན་ཤེས་ན་གཉེན་བྱའོ། ། ཞེས་སྨྲས་པ་དང་། ཁྱེའུ་དེ་རོལ་མོ་ལ་སོགས་པའི་ལས་བསླབ་པ་ལ་ངོ་ཚ་བ་དང་བསྡོས་ནས། རིང་པོར་མ་ལོན་པར་ཀུན་བྱང་བར་བསླབས་སོ། ། དུས་གཞན་ཞིག་ན་རྒྱལ་པོས་རོལ་མོ་བྱས་ཏེ། རྩལ་ཆེ་བ་ཀུན་བསགས་ནས་ཤིང་འཛེགས་པ་དང་། མཆོངས་པ་དང་། ཐག་འཛེགས་ལ་སོགས་པའི་རྩེད་མོ་སྣ་ཚོགས་བྱེད་པ་ལས། ཁྱིམ་བདག་གི་བུ་ཡང་དེར་འོངས་ནས་ཐག་པ་ལ་འཛེགས་ཏེ། ཐག་པ་ལ་རྒྱུག་ཅིང་ཐག་འཛེགས་བྱས་པ་དང་། རྒྱལ་པོ་ཡེངས་ཏེ་མ་མཐོང་བས་གདོད་བྱོས་ཤིག་ཅེས་བསྒོའོ། ། ཁྱེའུ་ངལ་བས་ཐག་འཛེགས་བྱས་པ་དང་། ཐག་པའི་དབུས་ཙམ་དུ་ཕྱིན་ནས་ཐག་པ་ལས་ལྷུང་སྟེ། ཤིན་ཏུ་སྐྲག་པས་སྐྱབས་མེད་དོ་ཞེས་སྨྲས་སོ། ། དེའི་ཚེ་མཽད་གལ་གྱི་བུ་དེར་འོངས་ནས་ས་ལ་མ་ལྷུང་བར་བཟུང་སྟེ་འདི་སྐད་ཅེས་སྨྲས་སོ། ། ཁྱོད་ད་ལུས་དང་སྲོག་མ་ཡལ་བར་རབ་ཏུ་བྱུང་ངམ། འོན་ཏེ་ས་ལ་ལྷུང་སྟེ་བུ་མོ་ཆུང་མར་བླང་བར་བྱ། སྨྲས་པ་སྲོག་མ་ཡལ་ན་ཆུང་མ་མི་འདོད་དོ་ཞེས་སྨྲས་པ་དང་། མཽད་གལ་གྱིས་ནམ་མཁའ་ལ་ས་གཞི་མཉམ་པར་སྤྲུལ་ནས་བདེ་བར་བཞག་པ་དང་། ཁྱེའུ་དེ་འཇིགས་པ་མེད་པར་གྱུར་ཏེ། བདེ་བར་ས་ལ་བཞག་པས་ཁྱེའུ་རབ་ཏུ་དགའ་ནས། མཽད་གལ་གྱི་ཕྱི་བཞིན་བཅོམ་ལྡན་འདས་གང་ན་བ་དེར་སོང་ངོ་། ། དེར་ཕྱིན་པ་དང་། བཅོམ་ལྡན་འདས་ཀྱི་ཞབས་ལ་མགོ་བོས་ཕྱག་འཚལ་ཏེ། སངས་རྒྱས་ལ་མཆོད་པ་བྱས་སོ། །
