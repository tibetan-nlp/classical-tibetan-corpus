རྒྱལ་པོ་བྱི་མ་སེ་ནས་ཐོས་ནས་མི་དགའ་སྟེ་འཁམས་ནས་སྙིང་གས་ཏེ་ཚེ་འཕོས་སོ། ། དེའི་ཚེ་ན་བྲམ་ཟེས་རྒྱལ་པོའི་མགོ་ཁྱེར་ཏེ་སོང་བའི་ཚེ་རྒྱལ་ཕྲན་དང་བློན་པོ་དང་བཙུན་མོ་དང་། རྒྱལ་བུ་རྣམས་ཀྱིས་ལུས་ས་ལ་བརྡབས་ཏེ་ཆོ་ངེས་བཏབ་ནས། ལ་ལ་ཁྲག་ཏུ་སྐྱུགས་ནས་ཚེ་འཕོས་སོ། ། ཁ་ཅིག་ནི་ཅི་ཡང་མི་དྲན་པར་མྱོས་སོ། ། ཁ་ཅིག་ནི་རང་གི་སྐྲ་ཀུན་བལ་ལོ། ། ལ་ལ་ནི་རང་གི་གོས་ཀུན་དྲལ་ལོ། ། ཁ་ཅིག་ནི་རང་གི་ལག་པ་གཉིས་ཀྱིས་གདོང་ཀུན་བྲད་བྲད་ནས་ཆོ་ངེས་བཏབ་ཅིང་ས་ལ་འགྲེའོ། ། དེའི་ཚེ་ན་བྲམ་ཟེ་དེས་རྒྱལ་པོའི་དབུ་མི་སྡུག་ཅིང་མནམ་མོ་ཞེས་མི་འདོད་ནས་ས་ལ་བོར་ཏེ་སོང་བ་དང་། དེ་ལ་བྲམ་ཟེ་ཁ་ཅིག་གིས་ཁྱོད་ཤིན་ཏུ་གདུག་པ་ཅན་ཡིན་པ་ལྟ་ཐོག་མ་ནས་དགོས་པ་མེད་ན། བསླངས་ཏེ་ཅི་བྱ་ཞེས་སྤྱོས་སོ། ། བྲམ་ཟེ་དེ་ལམ་དུ་ཞུགས་ནས་ཕྲད་པ་ཐམས་ཅད་ཀྱིས་སྤྱོས་ནས། ཁ་ཟས་མི་སྟེར་བས་ཤིན་ཏུ་ཉམ་ཐག་པར་གྱུར་ལ། ལམ་ནས་བྱི་མ་སེ་ནཡང་ཚེ་འཕོས་པར་གཏམ་ཐོས་པས་ཐོག་མར་རེ་བའི་སེམས་ལྟར་མ་གྱུར་ནས་མྱ་ངན་གྱིས་ཡི་མུག་སྟེ་སྙིང་ཚལ་པ་བདུན་དུ་གས་ནས། ཁྲག་སྐྱུགས་ནས་ཚེ་འཕོས་སོ། ། རྒྱལ་པོ་བྱི་མ་སེ་ནདང་། བྲམ་ཟེ་གཉིས་ཚེ་འཕོས་ནས་མཚམས་མེད་པའི་སེམས་ཅན་དམྱལ་བར་སྐྱེས་སོ། ། རྒྱལ་པོ་ཟླ་འོད་ཀྱི་མྱ་ངན་གྱིས་གདུངས་ཏེ་ཚེ་འཕོས་པ་རྣམས་ནི་མཐོ་རིས་ཀྱི་ལྷར་སྐྱེས་སོ། ། ཀུན་དགའ་བོ་དེའི་ཚེ་དེའི་དུས་ན་རྒྱལ་པོ་ཟླ་འོད་དེ་ནི་ད་ལྟར་ང་ཡིན་ནོ། ། རྒྱལ་པོ་བྱི་མ་སེ་ན་དེ་ནི་བདུད་སྡིག་ཏོ་ཅན་ཡིན་ནོ། ། དེའི་དུས་ན་བྲམ་ཟེ་དེ་ནི་ད་ལྟར་ལྷ་སྦྱིན་ཡིན་ནོ། ། དེའི་ཚེ་ན་ཤིང་གི་ལྷ་མོ་དེ་ནི་ད་ལྟར་མཽད་གལ་གྱི་བུ་ཡིན་ནོ། ། དེའི་ཚེ་ན་བློན་པོ་ཟླ་བ་ཆེན་པོ་དེ་ནི་ཤཱ་རིའི་བུ་ཡིན་ནོ། ། དེའི་ཚེ་ཡང་ང་ཚེ་འཕོས་པར་མཐོང་བར་མི་བཟོད་པས་སྔར་ཚེ་འཕོས་སོ། ། དེང་གི་དུས་འདིར་ཡང་ང་མྱ་ངན་ལས་འདའ་བར་མཐོང་བར་མི་བཟོད་ནས། སྔར་མྱ་ངན་ལས་འདས་པ་ཡིན་ནོ། ། བཅོམ་ལྡན་འདས་ཀྱིས་དེ་སྐད་ཅེས་བཀའ་སྩལ་ནས། ཚེ་དང་ལྡན་པ་ཀུན་དགའ་བོ་དང་། འཁོར་གཞན་དག་སྐད་འཐུན་པར་དེ་བཞིན་གཤེགས་པའི་ཡོན་ཏན་ནི་ངོ་མཚར་རྨད་དོ་ཞེས་བསྔགས་ནས།
