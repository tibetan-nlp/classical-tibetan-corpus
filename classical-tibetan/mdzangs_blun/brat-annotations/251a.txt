རྒྱལ་པོས་ཀྱང་དེ་བཞིན་དུ་བྱའོ་ཞེས་སྨྲས་ནས། བྲམ་ཟེ་ལོང་བ་འབོད་དུ་བཏང་ངོ་། ། བྲམ་ཟེ་ལོང་བ་འོངས་ནས། རྒྱལ་པོས་ལོང་བ་ལ་འདི་སྐད་ཅེས་སྨྲས་སོ། ། ཁོ་བོས་ཁྱོད་ལ་བསྒྲུབ་ཏུ་གསོལ་བ་ཞིག་ཡོད་ཀྱིས་ཁྱོད་ཀྱིས་བྱ་དགོས་སོ་ཞེས་བསྒོ་བ་དང་། བྲམ་ཟེས་སྨྲས་པ། བདག་མིག་གཉི་ག་ལོང་ན་ཅི་ཞིག་བསྒྲུབ་ཏུ་ཡོད་ཅེས་གསོལ་ཏོ། ། རྒྱལ་པོས་བསྒོ་བ་རྒྱལ་པོ་མིག་འབྱེད་དཔུང་གི་ཚོགས་བཤམས་ཏེ། འདིར་འོངས་པས་ཁོ་བོ་ནི་གཞོན་པས་བྲོས་ཀྱང་ཐར་ན། ཁྱོད་མིག་མེད་པ་གར་བྲོས་པའི་ས་ཡང་མེད་པས། རྒྱལ་པོ་དེས་སྦྱིན་པ་གཏོང་བ་ལ་ཕ་རོལ་གྱི་ཡིད་འཐུན་པར་གཏོང་ངོ་ཞེས་གྲག་གི། ཁྱོད་སོང་ལ་དེ་ལ་དེའི་མིག་སློང་ཤིག་དང་། གལ་ཏེ་སྟེར་ན་དམག་ཀྱང་མི་འོང་བར་འགྱུར་གྱིས། ཁྱོད་དེ་ལ་མིག་སློང་དུ་སོང་ཤིག་ཅེས་བསྒོ་བ་དང་། བྲམ་ཟེས་བདག་མིག་མི་མཐོང་ན་ཅི་ལྟར་བྱ་ཞེས་གསོལ་ཏོ། ། རྒྱལ་པོས་བསྒོ་བ་ཁྱོད་སོང་ཤིག་དང་། ལམ་རྒྱགས་དང་། ལམ་འདྲེན་པ་དང་། ཡོ་བྱད་གཞན་ཡང་ཅི་དགོས་པ་སྦྱར་རོ། ། གལ་ཏེ་མིག་ཐོབ་ན་བྱ་དགའ་ཡང་ཆེར་སྦྱིན་ནོ་ཞེས་བསྒོ་ནས་སོང་ངོ་། ། དེའི་ཚེ་ན་རྒྱལ་པོ་མིག་འབྱེད་ཀྱི་ཡུལ་ན་ལྟས་ངན་པ་རྣམ་པ་སྣ་ཚོགས་བྱུང་ངོ་། ། འདི་ལྟ་སྟེ། གནམ་གྲུམ་པ་དང་གནམ་ལྕགས་འབབ་པ་དང་། སྐར་མ་ལྷུང་བ་དང་། ནམ་རྨུགས་པ་དང་། གནམ་ལྕགས་བབ་ནས་ས་ཀུན་གས་པ་དང་། བྱ་རྣམས་ནམ་མཁའ་ལ་ངུ་བའི་སྐད་འབྱིན་ཅིང་ས་ལ་ལྟུང་བ་དང་། སེང་གེ་དང་སྟག་དང་། འཕར་སྤྱང་དག་སྙིང་རྗེའི་སྐད་འབྱིན་ཅིང་མིའི་ནང་དུ་འོངས་ཏེ། ས་ལ་འགྲེ་བ་ལ་སོགས་པའི་ལྟས་ངན་པ་སྣ་ཚོགས་བྱུང་ངོ་། ། དེ་ནས་བྲམ་ཟེ་ལོང་བ་ཡང་མཐར་གྱིས་ཡུལ་དེར་ཕྱིན་ཏེ། ཕོ་བྲང་གི་སྒོར་འོངས་ནས། སྐད་ཆེན་པོས་འདི་སྐད་ཅེས་སྨྲས་སོ། ། བདག་ནི་ཡུལ་གཞན་ནས་འདིར་མཆིས་ཀྱི། རྒྱལ་པོ་ཁྱོད་ཀྱི་ཡོན་ཏན་ཀུན་ཏུ་གྲགས་པ་ནི་སློང་བ་འོངས་ན་ཅི་ཡང་གཏོང་སྟེ། སློང་བའི་ཡིད་དང་མི་འགལ་བར་མཛད་ཅེས་ཐོས་ནས། ཉོན་མོངས་པ་དང་བསྡོས་ཏེ་འདིར་སློང་དུ་འོངས་ཀྱིས། ཅི་འདོད་པ་བྱིན་ཅིག་ཅེས་སྨྲས་སོ། །
