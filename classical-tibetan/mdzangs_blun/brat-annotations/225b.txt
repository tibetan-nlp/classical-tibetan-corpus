བདག་ཅག་ནི་ཕ་མེས་རྒྱུད་ཀྱི་འདི་འདྲ་བའི་གསོད་པའི་ལས་བྱེད་དེ། འདི་མ་བྱས་ན་མི་འཚོའོ་ཞེས་ཟེར་རོ། ། དེ་ལ་སྦྱིན་པ་ཆེན་པོ་ངོ་མཚར་དུ་གྱུར་ནས། ལྟ་མི་ཕོད་པར་སོང་ངོ་། ། དེ་ནས་ཕ་ཟད་དུ་ཕྱིན་པ་དང་། ཞིང་པ་དག་ཅིག་ཞིང་རྨོས་ཏེ། ཞིང་རྨོས་པའི་ཤུལ་དུ་སྲིན་བུ་མང་དུ་བྱུང་བ་ལས་བྱ་མང་པོས་འབུ་དག་ཐུ་ཞིང་ཟ་བ་མཐོང་བ་དང་དེ་དག་ལ་ཅི་བྱེད་ཅེས་དྲིས་ན་བདག་ཅག་ནི་ཞིང་རྨོས་ཏེ་ས་རློགས་པའི་ཤུལ་དུ་ས་བོན་བཏབ་ནས་ཕྱིས་མང་དུ་སྐྱེས་ཏེ། དེས་མང་དུ་འཚོ་ཞིང་རྒྱལ་པོའི་ཕྱིར་ཡང་འབུལ་ལོ་ཞེས་ཟེར་རོ། ། སྦྱིན་པ་ཆེན་པོས་དེ་ལ་ཡང་ངོ་མཚར་དུ་གྱུར་ནས། ཕ་ཟད་སོང་བ་དང་། རྔོན་པ་དག་བྱ་རྒྱ་བྲེས་པ་དང་། རྙི་བཙུགས་ནས། བྱ་རྣམས་བཤོར་བ་དང་སེམས་ཅན་དག་བྱ་རྒྱ་དང་རྙིས་ཟིན་ནས་ལུས་བརྡབས་ཤིང་འཐེན་ཀྱང་མ་ཤོར་ཏེ་སྙིང་རྗེའི་སྐད་ཕྱུང་སྟེ་ཤིན་ཏུ་འཇིགས་སྐྲག་པ་དག་མཐོང་ནས། ཁྱོད་ཅི་བྱེད་ཅེས་དྲིས་ན་བདག་ཅག་ཀྱང་རྒྱུད་ཀྱི་འདི་འདྲ་བ་གསོད་པའི་ལས་བྱེད་པས་འཚོའོ་ཞེས་ཟེར་རོ། ། སྦྱིན་པ་ཆེན་པོས་དེ་སྐད་སྨྲས་པ་ཐོས་ནས། ཤིན་ཏུ་ཡི་མུག་སྟེ། ཡང་ཕ་ཟད་སོང་བ་དང་། ཉ་པ་དག་ཉ་རྒྱ་བཙུགས་པ་དང་མང་དུ་ཟིན་ཏེ། ཐང་ལ་ཕུང་པོར་སྤུངས་ནས་ད་དུང་ཡང་འགུལ་བ་དག་མཐོང་ནས། འདི་དག་ཅི་བྱེད་ཅེས་དྲིས་ན། བདག་ཅག་རྒྱུད་ཀྱིས་ལས་གཞན་མེད་དོ། ། འདི་ལྟར་ཉ་བཤོར་ཞིང་འཚོའོ་ཞེས་ཟེར་རོ། ། སྦྱིན་པ་ཆེན་པོས་འདི་ལྟ་བུ་དག་མཐོང་ནས་འདི་སྙམ་དུ་བསམས་སོ། ། སེམས་ཅན་འདི་དག་ནི་དབུལ་ཞིང་ཕོངས་པས་བགོ་བ་དང་། བཟའ་བའི་ཕྱིར་འདི་ལྟ་བུའི་ཁ་ན་མ་ཐོ་བའི་ལས་གསོད་པ་ལ་སོགས་པ་བྱེད་དེ། འདི་དག་ནི་ཚེ་འཕོས་མ་ཐག་ཏུ་ངན་སོང་གསུམ་དུ་ལྷུང་སྟེ་མུན་པ་ནས་མུན་པར་འགྲོ་ཞིང་ཐར་པའི་དུས་མེད་དོ་སྙམ་བསམས་ནས། ཕྱིར་ཁྱིམ་དུ་འོངས་ཏེ་ཉིན་མཚན་དུ་དེ་སེམས་ཤིང་མི་དགའ་ནས་ཕའི་དྲུང་དུ་སོང་སྟེ། ཕ་ལ་བདག་གསོལ་པ་ཞིག་མཆིས་ཀྱིས་གནོང་ཤིག་ཅེས་སྨྲས་སོ། ། ཕས་ཀྱང་བུ་ཁྱོད་ཅི་ཟེར་སྨྲས་པ་བཞིན་དུ་བྱའོ་ཞེས་སྨྲས་སོ། ། ཡབ་བདག་སྔོན་ཕྱི་རོལ་ཏུ་འཆག་ཏུ་མཆི་མཆི་བ་ལས། མི་མང་པོས་གོས་དང་ཟས་ཀྱི་ཕྱིར་ཐམས་ཅད་ཉོན་མོངས་ཏེ།
