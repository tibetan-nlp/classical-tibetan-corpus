སུ་བྷ་ཥི་ཏ་ལེགས་པར་གསུངས་པ་དང་། པྲཛྙ་ཝརྨྨ་དང་། དཱི་པཾ་ཀ་ར་ཤྲཱི་ཛྙཱ་ན་མར་མེ་མཛད་ཡེ་ཤེས་ཏེ་ཇོ་བོ་རྗེ་དང་། དེའི་དབོན་པོ་དཱ་ན་ཤྲཱི་དང་། སྨྲྀ་ཏི་ཛྙཱ་ན་ཀིརྟི་དྲན་པའི་ཡེ་ཤེས་གྲགས་པ་དང་། ཕྲ་ལ་རིང་བ་དང་། སུ་ནྱ་ཤྲཱི་ལེགས་པར་རིག་པའི་དཔལ་དང་། ཁ་ཆེ་ཛྙཱ་ན་ཤྲཱི་དང་། ཙནྟྲ་རཱ་ཧུ་ལ་དང་། དྷི་ར་པཱ་ལ་བརྟན་སྐྱོང་སྟེ་འབུཾ་ཕྲག་གསུམ་པ་དང་། ཨ་ཏུ་ལྱེ་དཱ་ཤ་མི་མཉམ་ཁོལ་པ་དང་། སུ་མ་ཏི་ཀཱི་རྟི་བློ་བཟང་གྲགས་པ་དང་། ཨ་མ་ར་ཙནྟྲ་དང་། ཐིག་ལེ་བུམ་པ་དང་། གཞོན་ནུ་བུམ་པ་དང་། ཀ་ན་ཀ་ཝརྨ་དང་། དམ་པ་རྒྱ་གར་དང་། ཛ་ཡ་ཨ་ནནྟ་དང་། ག་ཡ་དྷ་ར་དང་། དོན་ཡོད་རྡོ་རྗེ་དང་། ཟླ་བ་མགོན་པོ་དང་། སྟོང་ཉིད་ཏིང་ངེ་འཛིན་རྡོ་རྗེ་དང་། ཡེ་ཤེས་རྡོ་རྗེ་དང་། ཤེས་རབ་གསང་བ་དང་། མ་ཧཱ་ཛ་ན་དང་། ཝ་ལ་ཙནྟྲ་དང་། མནྟྲ་ཀ་ལཀྵུ་དང་། སུ་ག་ཏ་ཤྲཱི་དང་། ཀརྨྨ་བཛྲ་དང་། ཛ་མཱ་རི་དང་། བཻ་རོ་ཙ་ན་བཛྲ་དང་། མཉྫུ་གྷོ་ཥ་དང་། སཱུརྱ་ཀིརྟི་དང་། པྲ་ཛྙཱ་ཤྲཱི་ན་དང་། གང་གཱ་དྷ་ར་དང་། ནོར་སྦས་དང་། ས་མནྟ་ཤྲཱི་དང་། ནིསྐ་ལང་ཀ་དེ་ཝ་དང་། ཛ་ག་ཏ་མི་ཏྲ་ཨ་ནནྟ་དང་། བུད་དྷཱ་ཤྲཱི་ཛྙཱ་ན་དང་། ཤཱཀྱ་ཤྲཱི་བྷ་དྲ་དང་། བི་བྷུ་ཏི་ཙནྡྲ་དང་། དཱ་ན་ཤཱི་ལ་དང་། སཾ་གྷ་ཤྲཱི་དང་། ལོངས་སྤྱོད་རྡོ་རྗེ་དང་། རཏྣ་ཤྲཱི་དང་། བལ་པོ་མཧཱ་པ་ན་དང་། བལ་པོ་བཛྲ་ཀཱིརྟི་དང་། བལ་པོ་ཀ་ཡ་ཤྲཱི་དང་། བལ་པོ་གྲགས་པ་དཔལ་དང་། ཀུ་མཱ་ར་དང་། ཐམས་ཅད་མཁྱེན་པའི་དཔལ་དང་། སཱ་དྷུ་ཀཱིརྟི་དང་། བི་ན་ཡ་ཤྲཱི་དང་། ཤཱི་ལ་ཤྲཱི་དང་། མཾ་ག་ལ་ཤྲཱི་དང་། བི་མ་ལ་ཤྲཱི་དང་། དར་པན་ཨཱ་ཙརྱ་དང་། ཛ་ཡ་དེ་བ་དང་། ལཀྵྨཱ་ཀ་ར་དང་། རཏྣ་ཤྲཱི་དང་། ཨ་ནནྟ་ཤྲཱི་
