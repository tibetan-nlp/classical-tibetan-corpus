ཆོས་ཀྱི་རྫོགས་ལོངས་སྤྱོད་འདི་ནི། ། རྒྱལ་སྲས་རྣམས་ནི་སྤྱོད་པ་ཡིན། ཞེས་སམ། བྱང་ཆུབ་སེམས་དཔའ་ས་བཅུ་ལ་བཞུགས་པ་རྣམས་དང་ཐབས་ཅིག་ཅེས་པས་སོ། ། སྤྲུལ་སྐུ་ལ་མཆོག་གི་སྤྲུལ་སྐུ་ནི་དྲོད་ཡན་ཆད་ནས་ནམ་འདོད་དུ་མཐོང་ཞིང་ཚོགས་ལམ་ཆེན་པོ་ནས་ཕལ་ཆེར་མཐོང་སྟེ། རྒྱན་ནས། དེ་ལྟར་མཐོང་དཀའི་ཐུབ་པ་རྟག་མཐོང་ཞིང་། ཞེས་པས་སོ། ། དེ་མན་ཆད་དུའང་མཐོང་བ་སྲིད། ཅེས་འབྱུང་ངོ་། ། སྐྱེ་བའི་སྤྲུལ་པ་ལ་སོགས་པ་ནི། ། གང་ལ་གང་འདུལ་ལ་སྣང་སྟེ། ། གང་དང་གང་ལ་གང་དང་ནི། ། གང་ཚེ་བྱ་བ་ཕན་གྱུར་པ། ། དེ་དང་དེ་ལ་དེ་སྣང་ངོ་། ། ཞེས་སོ། ། གདུལ་བྱ་མིན་པའམ། མི་སྣང་བས། གདུལ་བྱ་རྣམས་ལ་མི་སྣང་སྟེ། ། རྒྱན་ལས། ཇི་ལྟར་ཆུ་ཡི་སྣོད་ཆག་ན། ། ཟླ་བའི་གཟུགས་བརྙན་མི་སྣང་བ། ། དེ་བཞིན་སེམས་ཅད་ངན་པ་ལ། ། སངས་རྒྱས་གཟུགས་ཀྱང་མི་སྣང་ངོ་། ། ཞེས་སོ། ། ལྔ་པ་དབྱེ་བ་ལ། ཆོས་སྐུ་ལ། གསེར་འོད་ལས། དེ་ཁོ་ན་ཉིད་དང་ཡང་དག་པའི་ཡེ་ཤེས་ཉིད་ཙམ་དུ་གནས་པ་དེ་ནི་ཆོས་ཀྱི་སྐུ་ཞེས་བྱའོ་ཞེས་པས། ཆོས་ཅན་ཆོས་ཉིད་དམ་དབྱིངས་ཡེ་ཤེས་སམ། འདུས་བྱས་འདུས་མ་བྱས་གཉིས་སོ། ། ལོངས་སྐུ་ལ་དབུ་མ་པ་ཆོས་ཀྱི་བཤེས་གཉེན་གྱིས། ལོངས་སྤྱོད་རྫོགས་པ་ཆེན་པོ་ས་བཅུ་པ་ཐོབ་པ་ཡན་ཆད་ཀྱི་ཡུལ་དང་། ས་དང་པོ་ཐོབ་པའི་བྱང་སེམས་ཡན་ཆད་ཀྱི་ཡུལ་སྤྲུལ་སྐུ་ཁྱད་པར་ཅན་ལ་ལོངས་སྐུར་བཏགས་པ་གཉིས་སོ། ། ཞེས་གསུངས་སོ། ། དང་པོ་ནི་ངེས་པ་ལྔ་ལྡན་ཏེ། གནས་ངེས་པ་འོག་མིན་ནི་བཀས་བཅད་བར་པར་འོག་མིན་གཙང་མའི་གནས་ཀྱི་ལོགས་ཤིག་ན་ཡོད་པར་བཤད། ལང་ཀར་གཤེགས་པར། རིན་ཆེན་སྣ་ཚོགས་མཛེས་པ་ཡིས། ། འོག་མིན་གནས་ནི་ཉམས་དགའ་བར། །
