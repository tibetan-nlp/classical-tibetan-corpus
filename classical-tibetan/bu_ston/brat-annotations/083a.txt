མྱ་ངན་ལས་འདས་པའི་ཕྱོགས་བཞི་བསྟན་པར་བྱའོ། ། དགེ་སློང་དག། དཀོན་མཆོག་གསུམ་དང་བདེན་པ་བཞི་ལ་ཐེ་ཚོམ་ཡོད་ན་ཞུས་ཤིག། དེ་ནས་སྐུའི་སྟོད་གཡོགས་བསལ་ཏེ། དགེ་སློང་དག། དེ་བཞིན་གཤེགས་པ་ལྟ་བ་ནི་རྙེད་པར་དཀའ་ཡིས། དེ་བཞིན་གཤེགས་པའི་སྐུ་ལ་ལྟོས་ཤིག། དགེ་སློང་དག། རེ་ཞིག་ཅང་མ་སྨྲ་ཞིག་དང་། འདི་ལྟར་འདུས་བྱས་ཐམས་ཅད་ནི་འཇིག་པའི་ཆོས་ཅན་ཡིན་ཏེ། དེ་ལ་འདི་ནི་དེ་བཞིན་གཤེགས་པའི་གསུང་གི་ཐ་མ་ཡིན་ནོ་ཞེས་གསུངས་ནས། བསམ་གཏན་བཞི། གཟུགས་མེད་བཞི། འགོག་སྙོམས་ལ། ལུགས་འབྱུང་ལུགས་ཟློག་ལ་སྙོམས་པར་ཞུགས་ནས། སླར་བསམ་གཏན་བཞི་ལ་ཞུགས་ཏེ། རབ་མཐའ་ལས་མྱ་ངན་ལས་འདས་སོ། ། སློབ་དཔོན་ཚུལ་ཁྲིམས་བསྐྱངས་ཀྱིས། ཕྲན་ཚེགས་འགྲེལ་པ་ལས། ཁ་ཆེ་བ་དང་བསྟུན་ནས། རབ་མཐའི་རྗེས་ལ་ལུང་མ་བསྟན་གྱི། སེམས་ལས་མྱ་ངན་ལས་འདས་པ་ཡིན་ཡང་། རབ་མཐའ་དང་ཉེ་བས་རབ་མཐའ་ཞེས་བརྗོད་པར་བཤད་དོ། ། དེ་མ་ཐག། ས་གཡོས། སྐར་མདའ་ལྷུང་། ཕྱོགས་ཚིག། ལྷའི་རོལ་མོའི་སྒྲ་གྲག་གོ། དེའི་ཚེ་འོད་སྲུངས་ཆེན་པོ་རྒྱལ་པོའི་ཁབ་ན་གནས་པས། སྟོན་པ་འདས་པར་ཤེས་ཏེ། འདུས་བྱས་ཀྱི་ཆོས་ཉིད་འདི་འདྲའོ་སྙམ་སྟེ། དེ་མ་སྐྱེས་དགྲས་ཐོས་ན་འཆི་བར་འགྱུར་བས། དེའི་ཐབས་ཤིག་བྱའོ་སྙམ་སྟེ། བྲམ་ཟེ་དབྱར་བྱེད་ལ། ཁྱོད་མྱུར་དུ་སྐྱེད་ཚལ་དུ་སོང་ལ། སྟོན་པ་ལྷུམས་སུ་ཞུགས་པ་དང་། སངས་རྒྱས་པ་དང་། འཁོར་ལོ་བསྐོར་བ་དང་། མཉན་ཡོད་དུ་ཆོ་འཕྲུལ་བསྟན་པ་དང་། གྲོང་ཁྱེར་སྐྲ་ཅན་དུ་ལྷ་ལས་བབས་པ་དང་།
