རྐྱེན་ལ་བྱིངས་ལས་མིང་བསྒྲུབ་པ་བྱེད་ལྡན་གྱི་རྐྱེན། མིང་ལས་མིང་གཞན་སྒྲུབ་པ་དེ་ཕན་གྱི་རྐྱེན། དེ་ལའང་སྤྱི་དང་གྲངས་མེད་དང་དངོས་པོའི་རྐྱེན་གསུམ་མོ། ། བྱིངས་ལས་ཚིག་བསྒྲུབ་ཅིང་དུས་རྣམ་པར་འབྱེད་པ་ཏིང་དང་། མིང་ལས་ཚིག་བསྒྲུབ་ཅིང་དོན་རྣམ་པར་འབྱེད་པ་སུ་པ་དང་། རང་བཞིན་དང་རྐྱེན་གྱི་བར་དུ་གཞུག་པ་ཨ་ག་མ་དང་། བྱིངས་གཞན་དུ་སྒྱུར་བའི་ཉེར་སྒྱུར་རྣམས་སོ། ། ཉེར་བསྒྱུར་ནི་རང་བཞིན་དུ་རིགས་སོ། ། རྣམ་འགྱུར་ནི་མཚམས་སྦྱོར་ལ་སོགས་པའི་དབང་གིས་ཡི་གེ་དབྱི་བ་དང་བསྒྱུར་བ་ལ་སོགས་དང་ཚིག་གི་བསྡུ་བ་རྣམས་སོ། ། དེ་དག་གི་སྒྲའི་བརྗོད་བྱ་བསྡུས་པའམ། ཡང་ན་མཚམས་སྦྱོར་དང་མིང་དང་ཀུན་བཤད་པ་དང་། ཀྲྀཏ་དང་བཞིས་མདོའི་དོན་རྩ་བ་བསྟན་ལ། བྱིངས་ཉེ་བསྒྱུར་ཨུ་ཎ་ལ་སོགས་ཡན་ལག་སྟེ། དེ་དག་གིས་འཆད་པ་ཀ་ལཱ་པ་ལྟ་བུའམ། ཡི་གེའི་ཚོགས། མིང་གི་ཚོགས། ཚིག་གི་ཚོགས་གསུམ་དུ་བསྡུས་ནས་འཆད་པ་སྨྲ་སྒོ་ལྟ་བུ་རྣམས་སོ། ། དེ་དག་ཀྱང་སོ་སོ་ཡང་དག་པར་རིག་པ་བཞི་ཐོབ་པའི་རྒྱུར་བསླབ་པར་བྱའི་སྡེ་སྣོད་གསུམ་གང་ཡང་མིན་ནོ། ། དེ་དག་དང་འབྲེལ་བ་ཡི་གེ་ལྕེ་ཡང་གི་བྱེ་བྲག་གི་སྦྱོར་ཚུལ་དང་། པྲཏཱ་ར་དགོད་པ་སྟོན་པ་སྡེབ་སྦྱོར་གྱི་བསྟན་བཅོས་རིན་ཆེན་འབྱུང་གནས་ལྟ་བུ་དང་། མིང་གི་རྣམ་གྲངས་དང་རྟགས་གསུམ་གྱི་དབྱེ་བ་དང་། མིང་གཅིག་དོན་མང་པོ་ལ་འཟུག་པའི་ཚུལ་ལ་སོགས་པ་སྟོན་པ་མངོན་བརྗོད་ཀྱི་བསྟན་བཅོས་འཆི་མེད་མཛོད་ལྟ་བུ་དང་། སྙན་དངགས་ཀྱི་བསྟན་བཅོས་ལ་སྙན་དངགས་ཀྱི་མཚན་ཉིད་དང་སྙན་དངགས་ཀྱི་གྲུབ་མཐའ་འདོད་ཚུལ་མི་འདྲ་བའི་ཚུལ་དང་། རྒྱན་སུམ་ཅུ་རྩ་ལྔ་དང་།
