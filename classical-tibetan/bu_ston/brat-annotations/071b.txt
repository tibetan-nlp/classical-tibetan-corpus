དེ་ནས་རངས་བྱེད་ཀྱི་བུ་ལྷག་སྤྱོད་སློབ་མ་བདུན་བརྒྱ་ལ་སྲིད་རྩེའི་སེམས་སྟོན་པ་མཐོང་ནས། བདག་འདི་ལྟ་བུའི་བརྟུལ་ཞུགས་དང་དཀའ་ཐུབ་བརྩམས་ནས། བདག་ཁྱད་པར་དུ་འཕགས་པའི་འདུ་ཤེས་བསྐྱེད་ཅིང་། ཟག་བཅས་ཀྱི་བསམ་གཏན་སུན་དབྱུང་ངོ་སྙམ་སྟེ། དེའི་སློབ་མར་ཁས་བླངས་ནས། མྱུར་དུ་སྲིད་རྩེའི་སེམས་ཁོང་དུ་ཆུད་ནས་སྔར་བཞིན་བརྗོད་ནས། འདིས་ནི་མྱང་འདས་སུ་འགྱུར་བ་མ་ཡིན་ནོ་ཞེས་སུན་ཕྱུང་སྟེ་སོང་ངོ་། ། དེའི་ཚེ་ལྔ་སྡེ་ཡང་བྱང་ཆུབ་སེམས་དཔའི་རྗེས་སུ་འབྲངས་ཏེ་རིམ་གྱིས་རི་ག་ཡའི་རྩེ་མོར། སྔར་མ་ཤེས་པའི་དཔེ་གསུམ་མངོན་དུ་གྱུར་ཏེ། ཆུ་བོ་ནེ་རཉྫ་ནའི་འགྲམ་དུ་ཕྱིན་ནས། བདག་སྙིགས་མ་ལྔའི་དུས་དམན་པ་ལ་མོས་པ། བརྟུལ་ཞུགས་ངན་པས་དག་པར་འཛིན་པ་སུན་དབྱུང་བའི་ཕྱིར། བརྟུལ་ཞུགས་དང་དཀའ་ཐུབ་བརྩམ་མོ་སྙམ་སྟེ། ལོ་དྲུག་ཏུ་མཁའ་ཁྱབ་ཀྱི་ཏིང་ངེ་འཛིན་ལ་བཞག་སྟེ། ཏིལ་འབྲུ་གཅིག་དང་། རྒྱ་ཤུག་འབྲུ་གཅིག་དང་། འབྲས་འབྲུ་གཅིག་གིས་ལོ་དྲུག་ཏུ་དཀའ་བ་སྤྱད་དེ། དབུགས་རྒྱུ་བ་བཅད་དེ་བཞུགས་སོ། ། དེའི་ཚེ་ལྷའི་བུ་ཅིག་གིས་ལྷ་མོ་སྒྱུ་འཕྲུལ་ལ་གཞོན་ནུ་ཤིའོ་ཞེས་བརྗོད་པས། ལྷ་མོས་འོངས་པ་དང་། བྱང་ཆུབ་སེམས་དཔའི་ལུས་བསྐམས་པ་དང་། ཤི་བ་དང་འདྲ་བ་མཐོང་ནས། བལྟམས་མ་ཐག་ཏུ་ངའི་སྐྱེ་བའི་མཐའ་མ་འདིའོ་ཞེས་པའི་དམ་བཅའ་ཡང་མ་རྫོགས་དྲང་སྲོང་ནག་པོའི་ལུང་བསྟན་ཡང་མི་བདེན་ནོ། ། ཞེས་སྨྲེ་སྔགས་འདོན་པ་ལ། བྱང་ཆུབ་སེམས་དཔས་སྨྲས་པ། ཉི་ཟླ་སྐར་ཚོགས་ས་ལ་ལྟུང་འགྱུར་སྲིད། ། སོ་སོ་སྐྱེ་བོ་ཡིན་ཡང་ང་མི་འཆི། ། དེ་ཕྱིར་འདི་ལ་མྱ་ངན་ཁྱོད་མ་བྱེད། ། རིང་པོར་མི་ཐོགས་བྱང་ཆུབ་སངས་རྒྱས་མཐོང་། ། ཞེས་སྨྲས་པས་དགའ་སྟེ་གནས་སུ་སོང་ངོ་། །
