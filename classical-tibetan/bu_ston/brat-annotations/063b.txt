སྐྱེ་བ་དང་རྒ་བ་དང་འཆི་བ་མཐར་དབྱུང་ངོ་། ། བྱང་དུ་སེམས་ཅན་ཐམས་ཅད་ཀྱི་ནང་ན་བླ་ན་མེད་པར་བྱའོ། ། འོག་ཏུ་བདུད་དང་བདུད་ཀྱི་སྡེ་གཞོམ་པར་བྱའོ། ། སེམས་ཅན་དམྱལ་བ་པ་ཐམས་ཅད་ཀྱི་སེམས་ཅན་དམྱལ་བའི་མེ་བསད་པའི་ཆོས་ཀྱི་ཆར་སྤྲིན་ཆེན་པོའི་ཆར་ཕབ་སྟེ། དེ་དག་བདེ་བས་ཚིམ་པར་བྱའོ། ། སྟེང་དུ་སེམས་ཅན་ཐམས་ཅད་ཀྱིས་གྱེན་དུ་ལྟ་བར་འགྱུར་བར་བྱའོ་ཞེས་གསུངས་ཏེ། གོམ་པ་བདུན་བདུན་བོར་བ་དང་པདྨ་བྱུང་ངོ་། ། དེའི་ཚེ་བཀྲ་ཤིས་པའི་ལྟས་ཕྱོགས་ཐམས་ཅད་སྣང་བས་གང་བ་དང་། ཡུལ་ཆེན་པོ་བཞིར་རྒྱལ་ཕྲན་བཞི་བཙས། ཡ་རབས་ཀྱི་རིགས་ལྔ་བརྒྱ། གྲགས་ལྡན་མ་ལ་སོགས་པ་བརྒྱད་བརྒྱ། འདུན་པ་ལ་སོགས་པ་བྲན་ལྔ་བརྒྱ། སྔགས་ལྡན་ལ་སོགས་རྟེའུ་ཁྲི། རྟེའུ་མོ་ཁྲི་དང་། བ་ལང་ཁྲི་ཡང་བཙས། གླིང་གི་དབུས་སུ་བྱང་ཆུབ་ཀྱི་ཤིང་སྐྱེས། སྐྱེད་ཚལ་ལྔ་བརྒྱ་དང་། གཏེར་ཁ་ལྔ་བརྒྱ་ཡང་བྱུང་སྟེ། རྒྱལ་བུ་དོན་ཐམས་ཅད་གྲུབ་པར་བཏགས་སོ། ། དེ་ནས་རིགས་ཀྱི་ལྷ་མཆོད་པའི་སར་ཕྱིན་ཏེ། ཤཱཀྱ་མུ་ཅོར་སྨྲ་བ་རྣམས་ཐུབ་པ་བཞིན་འདུག་པས། ཤཱཀྱ་ཐུབ་པར་བཏགས། ཤཀྱ་འཕེལ་གྱིས་ཕྱག་བཙལ་བའམ། ལྷ་རྣམས་ཀྱིས་ཕྱག་བཙལ་བས་ལྷའི་ལྷར་བཏགས་སོ། ། ཞག་བདུན་ནས་ཡུམ་ཚེའི་དུས་བྱས་ཏེ་སུམ་ཅུ་རྩ་གསུམ་དུ་སྐྱེས་སོ། ། དེ་བྱང་ཆུབ་སེམས་དཔས་ནོངས་པ་མིན་གྱི། སྲས་མངོན་པར་འབྱུང་བ་གཟིགས་ན། ཡུམ་སྙིང་འགས་པར་འགྱུར་བས་ཚེ་ཚད་མཐར་ཐུག་པའི་ཕྱིར་རོ། ། མདོ་གཞན་ནས་རི་བོ་ཀུན་འཛིན་ལ་གནས་པའི་དྲང་སྲོང་ཉོན་མོངས་མེད་ཀྱི་ཚ་བོ་མེས་བྱིན་གྱི་བུས་ཤེས་ཏེ། མཁན་པོ་འདིར་ནི་ཉི་མ་ཐམས་ཅད་དག། ཅིག་ཅར་བྱུང་བར་གྱུར་ཏམ་ཅི། ། འདི་ལྟར་རི་ནི་ཕུག་བཅས་པ། །
