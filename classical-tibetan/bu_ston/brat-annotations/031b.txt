རབ་རྒྱས་མ་བསླབས་བདག་གིས་ཞེས། ། ཐོག་མར་ཚིག་དེ་བརྗོད་པར་བྱ། ། ཁྱོད་ནི་ཤེས་ཤིང་མཁས་ཡིན་ན། ། བདག་ཉིད་ཆེན་པོའི་མདུན་དུ་ནི། ། བདག་གིས་ཇི་ལྟར་བཤད་ནུས་ཞེས། ། ཁྱོད་ཀྱིས་དེ་སྐད་བརྗོད་པར་བྱ། ། འཕྲལ་དུ་བརྗོད་པར་མི་བྱ་ཡི། ། སྣོད་ལ་བརྟག་པ་བྱས་ནས་བྱ། ། གལ་ཏེ་སྣོད་ནི་ཤེས་གྱུར་ན། ། གསོལ་མ་བཏབ་ཀྱང་བཤད་པར་བྱ། ། འཁོར་ནི་མང་པོ་འཁོད་ནང་དུ། ། གལ་ཏེ་ངན་ཚུལ་ངན་མཐོང་ན། ། ཁྱོད་ཀྱི་ཡོ་བྱད་བསྙུངས་མི་བཤད། ། སྦྱིན་པའི་བསྔགས་པ་བརྗོད་པར་བྱ། ། གལ་ཏེ་འདོད་པ་ཆུང་ཞིག་སྟེ། ། ཚུལ་ཁྲིམས་དག་པར་གནས་ཡིན་ན། ། ཁྱོད་ཀྱིས་བྱམས་པའི་སེམས་བསྐྱེད་ལ། ། ཡོ་བྱད་བསྙུངས་པའི་གཏམ་གྱིས་ཤིག། གལ་ཏེ་སྡིག་འདོད་ཆུང་གྱུར་ཅིང་། ། ཚུལ་ཁྲིམས་ལྡན་པ་མང་གྱུར་ལ། ། གཉེན་པོ་རྙེད་པ་དེ་ཡི་ཚེ། ། ཚུལ་ཁྲིམས་བསྔགས་པ་བརྗོད་པར་བྱ། ། ཞེས་བཤད་པ་བཞིན་ནོ། ། གཉིས་པ་ནི། དགོས་པ་དོན་དུ་གཉེར་བའི་བྱེ་བྲག་གི་མཚན་ཉིད་གཏན་ལ་དབབ་པ་དང་། སྒྲུབ་པ་ཉམས་ལེན་ཅི་རིགས་པར་འཆད་པའོ། ། གསུམ་པ་ལ། སྦྱོར་དངོས་རྗེས་གསུམ་གྱི། སྦྱོར་བ་ལ་བཀོད་པ་བཤམས་པ་དང་། དཀོན་མཆོག་ལ་གསོལ་བ་གདབ་པ་དང་། བདུད་ཚར་བཅད་ཅིང་འཁོར་ལ་བྱམས་པས་ཁྱབ་པར་བྱ་བ་གསུམ་མོ། ། དམ་པའི་ཆོས་པདྨ་དཀར་པོ་ལས། མཁས་པ་དུས་སུ་སེམས་པར་བྱེད་པ་ན། ། ཁང་བུར་ཞུགས་ཏེ་དེ་བཞིན་སྒོ་བཅད་ནས། ། ཆོས་འདི་ཐམས་ཅད་ལ་ནི་ཚུལ་བཞིན་བལྟ། ། ལངས་ནས་ཞུམ་པ་མེད་པའི་སེམས་ཀྱིས་ཤོད། ། མཁས་པ་བདེ་ལ་རྟག་ཏུ་གནས་པར་འགྱུར། ། བདེ་ལ་འདུག་ནས་དེ་བཞིན་ཆོས་ཀྱང་སྟོན། ། གཙང་ཞིང་ཡིད་དུ་འོང་བའི་ས་ཕྱོགས་སུ། ། ཡངས་པའི་སྟན་ནི་རབ་ཏུ་བཤམས་བཏིང་སྟེ། ། བཟང་པོའི་ཚོན་ཚོན་གྱིས་ལེགས་པར་ཁ་བསྒྱུར་པ། །
