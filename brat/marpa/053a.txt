གདམས་ངག་བསྐོར་ཚེ་བསྐོར་ཚེ་ཞུས། ། ལམ་རྩ་དང་རླུང་ལ་སྦར་ཟིན་བྱས། ། སངས་རྒྱས་ལག་པའི་མཐིལ་ན་གདའ། ། དུས་ལན་ཅིག་འཆི་བའི་ཚུལ་སྟོན་ཏེ། ། རྣམ་སྨིན་གྱི་ལུས་རྒྱ་གྲོལ་བའི་ཚེ། ། འདའ་ཀ་ཟབ་མོའི་གདེང་ཡོད་དེ། ། བསྲེ་འཕོ་གནད་ཀྱི་མཚམས་སྤྱར་ནས། ། དཔའ་བོ་དཔའ་མོའི་བསུ་མ་ཡོད། ། རྒྱལ་མཚན་རོལ་མོའི་སྒྲ་དང་བཅས། ། བདེ་ཆེན་མཁའ་སྤྱོད་གནས་སུ་འགྲོ། དཔལ་ནཱ་རོ་པ་དང་མཇལ་བར་ངེས། ། ད་ནི་འཆི་ཡང་ང་རེ་བྲོད། ། ཁྱེད་འདིར་བཞུགས་ཀྱི་ཇོ་བོ་སྔགས་པ་ཀུན། ། དོན་སྙན་བརྒྱུད་ཀྱི་ངོ་སྤྲོད་མི་མངའ་བས། ། བཤད་རྒྱུད་ཚིག་གི་ཐ་སྙད་ལ། ། ཚེ་གཅིག་གི་སངས་རྒྱས་མ་རེ་མཛོད། ། ལར་སྙིང་ནས་དམ་ཆོས་མཛད་དགོངས་ན། ། ནཱ་རོ་དང་མཻ་ཏྲིའི་བརྒྱུད་པ་ཟུངས། ། བུ་སློབ་མ་བས་ཀྱང་ཡང་སློབ་བཟང་། ། བདེ་ནས་བདེ་བར་འགྲོ་སྟེ་མཆི། ། དབྱངས་སྙན་དུ་བྱོན་ནམ་འདིར་བཞུགས་རྣམས། ། དོན་འཁྲུལ་པར་གདའ་ན་བཟོད་པར་བཞེས། ཞེས་གསུངས་པས། ཡང་ཁོང་རྣམས་ན་རེ། ལོ་ཙྪ་བ་ཁྱེད་ལ་སྤྱིར་བལ་པོ་ཕན་ཆད་ན་བླ་མ་དུ་བཞུགས། བླ་མ་གཙོ་བོ་རང་གང་ལ་བགྱིད། ། མངའ་བདག་གི་ཞབས་དྲུང་དུ་བསྡད་པ་ལ་སྐུ་ཆེ་བའི་ཡོན་ཏན་ཇི་འདྲ་མཐོང་། གདམས་པའི་གཙོ་བོ་ནི་ཅི་ཞུས་འདྲིར་བྱུང་བས། ཡང་རྗེ་ལོ་ཙྪ་བས་དེའི་ལན་དུ་མགུར་འདི་བཞེས་སོ། ། བྲམ་ཟེ་ཆེན་པོའི་གདུང་འཚོབ་། ། མཐའ་བྲལ་གཉུག་མའི་དོན་རྟོགས་ཕྱིར། ། དཔེ་མེད་ནམ་མཁའི་རྣལ་འབྱོར་པ། ། མཚན་མཻ་ཏྲི་ཞེས་བྱར་ཡོངས་སུ་གྲགས། །
