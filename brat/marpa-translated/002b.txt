རྒྱལ་སྲས་ཐུ་བོ་ཆོས་ཀྱི་བློ་གྲོས་ལ། ། བདག་ནི་དད་པས་དབང་མེད་གུས་ཕྱག་འཚལ། ། ཉིན་མཚན་ཁོར་ཡུག་སྒོ་གསུམ་རིམ་གཉིས་དང་། ། གཞན་དོན་བསྒྲུབ་ལ་བརྩོན་པའི་མཆོད་པ་འབུལ། ། ཐོག་མེད་དུས་ནས་། ། སྒོ་གསུམ་ཉེས་ལྟུང་ཉོན་མོངས་དབང་གྱུར་གནོང་འགྱོད་དྲག་པོས་བཤགས། ། དམ་པ་ཁྱོད་ཀྱིས་ཆོས་ཕྱིར་དཀའ་སྤྱད་སོགས། ། མཛད་མཆོག་རྣམས་ལ་སྙིང་ནས་རྗེས་ཡི་རང་། ། བདག་སོགས་འགྲོ་ཀུན་སྨིན་ཅིང་གྲོལ་བྱའི་ཕྱིར། ། གང་ལ་གང་འདུལ་ཆོས་འཁོར་བསྐོར་བཞིག་དུ། ། ཇི་སྲིད་སེམས་ཅན་སྡུག་བསྔལ་མ་བསྒྲལ་བར། ། སྐྱབས་མགོན་མྱ་ངན་མི་འདའ་བཞུགས་སུ་གསོལ། ། དགེ་བས་འགྲོ་ཀུན་སངས་རྒྱས་ཐོབ་ཕྱིར་བསྔོ། ། རྒྱལ་བ་སྲས་བཅས་ཐུགས་དགོངས་ཡོངས་རྫོགས་ཤོག། སྐྱེས་མཆོག་སྒྲ་བསྒྱུར་མར་པ་ལོ་ཙྪ་ཡི། ། ཡོན་ཏན་རྣམ་ཐར་ནམ་མཁའ་ལྟ་བུ་ལས། ། ངོ་མཚར་མདོ་ཙམ་འགྲོ་ལ་ཕན་ཕྱིར་བཀོད། ། བླ་མ་ཡི་དམ་མཁའ་འགྲོས་གནང་དུ་གསོལ། ། བརྩེ་བས་གནང་ནས་བྱིན་གྱིས་བརླབ་པར་མཛོད། ། དེ་ཡང་མ་རིག་པའི་འགྲོ་བ་མཐའ་དག་འདུལ་བའི་སླད་དུ་རྒྱལ་བ་གོང་མ་རྣམས་བཀའ་བགོས་མཛད་ནས། ཐུགས་ཀྱི་སྲས་བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་ཀུན་ཏུ་བཟང་པོ་དེ་ཉིད་མངགས་པས། རྒྱ་གར་ཏོཾ་བྷི་ཧེ་རུ་ཀ་ཞེས་བྱ་བར་སྐྱེ་བ་བཞེས་ནས་འགྲོ་བ་མང་པོའི་དོན་མཛད་རྗེས། སླར་ཡང་གངས་ཅན་ལྗོངས་ཀྱི་འགྲོ་བ་རྣམས་འདུལ་བ་ལ་མངགས་ཕྱིར་མིར་སྐྱེ་བ་བཞེས་པས་ཏེ་བོད་གངས་ཅན་གྱི་སྐྱེ་འགྲོ་ཀུན་གྱི་དཔལ་དུ་ཤར་ནས། སངས་རྒྱས་ཀྱི་བསྟན་པ་ཉིན་མོར་མཛད་པའི་སྐྱེས་བུ་དམ་པ་།
