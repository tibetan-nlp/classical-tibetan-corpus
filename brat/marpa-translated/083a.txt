འོད་གསལ་མ་བུ་མཇལ་བ་ཡིན། ། མེ་ཏོག་སྣ་ཚོགས་བཀྲ་བ་དེ། ། འབྲས་བུ་སྐྱོན་བྲལ་ཉམས་མྱོང་ཡིན། ། སྤྱིར་རྨི་ལམ་མི་ངན་རྨི་ལམ་བཟང་། ། འདིར་ཚོགས་གྲྭ་པ་བུ་སློབ་ཀུན། ། སྒོས་ལྷུན་སྟུག་གངས་རི་མཐོན་པོ་། ། ཤར་དུ་ཀ་ཆེན་བཙུགས་པ་དེ། ། དོལ་གྱི་ཚུལ་སྟོང་དབང་ངེ་ཡིན། ། ཀ་ཐོག་ཏུ་སེང་ཆེན་འགྱིང་བ་དེ། ། སྙན་བརྒྱུད་ཀྱི་གདམས་ངག་ཐེབས་པ་ཡིན། ། སྡེར་བཞི་གངས་ལ་བགྲད་པ་དེ། ། ཚད་མེད་བཞི་དང་ལྡན་པ་ཡིན། ། སྤྱན་མིག་གྱེན་དུ་གཟིགས་པ་དེ། ། འཁོར་བ་ལ་ཕྱི་ཕྱག་འཚལ་བ་ཡིན། ། སེང་གངས་དཀར་ལྟོངས་སུ་འཕྱོ་བ་དེ། ། ཐར་པའི་གླིང་དུ་གཤེགས་པ་ཡིན། ། ཤར་རྨི་ལམ་མི་ངན་རྨི་ལམ་བཟང་། ། འདིར་ཚོགས་གྲྭ་པ་བུ་སློབ་ཀུན། ། ལྷོ་རུ་ཀ་ཆེན་བཙུགས་པ་དེ། ། གཞུང་གི་རྔོག་སྟོན་ཆོས་རྡོར་ཡིན། ། ཀ་ཐོག་ཏུ་སྟག་མོ་ངར་བ་དེ། ། ཁོ་མི་གཞི་སྟག་དང་འདྲ་བ་ཡིན། ། སྟག་གྲ་འཛུམ་མཐའ་ནས་རྒྱས་པ་དེ། ། སྙན་བརྒྱུད་ཀྱི་གདམས་ངག་ཐེབས་པ་ཡིན། ། ཁོང་འཛུམ་ལན་གསུམ་བྱས་པ་དེ། ། སྐུ་གསུམ་རང་ངོ་འཕྲོད་པ་ཡིན། ། སྡེར་བཞི་ནགས་ལ་བགྲད་པ་དེ། ། ཕྲིན་ལས་རྣམ་བཞི་འགྲུབ་པ་ཡིན། ། སྤྱན་མིག་གྱེན་དུ་གཟིགས་པ་དེ། ། འཁོར་བ་ལ་ཕྱི་ཕྱག་འཚལ་བ་ཡིན། ། སྟག་ནགས་ཀྱི་གསེབ་ཏུ་འཕྱོ་བ་དེ། ། ཐར་པའི་གླིང་དུ་གཤེགས་པ་ཡིན། ། ཐང་ཤིང་བརྒྱུད་དུ་འབྲེལ་བ་དེ། ། བུ་ཚ་བོས་བརྒྱུད་པ་ཟིན་པ་ཡིན། ། ལྷོ་རྨི་ལམ་མི་ངན་རྨི་ལམ་བཟང་། ། འདིར་ཚོགས་གྲྭ་པ་བུ་སློབ་ཀུན། ། ནུབ་ཏུ་ཀ་ཆེན་བཙུགས་པ་དེ། ། གཙང་རོང་གི་མེས་སྟོན་ཚོན་པོ་ཡིན། །
