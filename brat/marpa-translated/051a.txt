ལྟ་སྟངས་སྡེགས་མཛུབ་མཛད་མ་ཐག། ཁྭ་ཏ་རེངས་ནས་ས་ལ་ལྷུང་། ། བར་ཆད་བདུད་ལས་རྒྱལ་ལོ་གསུངས། ། ངོ་མཚར་ཆེ་ན་དེ་ཆེ་མོད། ། ཁྱོད་འདི་རུ་མ་འདུག་བོད་ལ་སོང་། ། ལྗོངས་བྱང་ཕྱོགས་ཁ་བ་ཅན་རྒྱུད་ན། ། གདུལ་བྱ་སྣོད་ལྡན་ཡོད་དོ་ཞེས། ། ལུང་བསྟན་རྗེས་སུ་གནང་བ་ཐོབ། ། ངོ་མཚར་ཆེ་ན་དེ་ཆེ་མོད། ། སྤྲུལ་སྐུ་ནཱ་རོ་པཎ་ཆེན་ལ། ། ངོ་མཚར་ཅན་གྱི་མཐོང་ལུགས་བརྒྱད། ། ཁྱོད་རྡོ་རྗེ་སྤུན་རྣམས་མ་གཏོགས་པ། ། ཁོང་གཞན་ལ་བཟླས་ན་ཡིད་མི་ཆེས། ། ད་དུས་བསྟན་པའི་སྙིགས་མ་ལ། ། ལོག་རྟོག་སྐྱེ་བོ་ཕྲག་དོག་ཆེ། ། ཡོན་ཏན་བརྗོད་ན་སྒྲོ་སྐུར་འདེབས། ། དེ་ཕྱིར་རིང་མིན་གཞན་རྣམས་ལ། ། བཀའ་མཆིད་མི་སྤེལ་གསང་བར་ཞུ། ། དབྱངས་མཆོད་པར་འབུལ་ལོ་བླ་མ་རྗེ། ། ཐུགས་དགེས་པར་མཛོད་ཅིག་སྙིང་གྲོགས་རྣམས། ། ཞེས་གསུངས་པས་ཀུན་ཐུགས་མཉེས་པར་གྱུར་ཏོ། ། ཡང་དེའི་ཕྱི་ནུབ་སྤྱི་ཐེར་བའི་དུས་མཆོད་དང་། ཚེས་བཅུའི་མཆོད་པ་ལྷན་ཅིག་བྱེད་པའི་སར། ། དགེ་སློང་འཇིགས་མེད་གྲགས་པས་དབུ་མཛད་རྣལ་འབྱོར་ཕོ་མོ་ཉི་ཤུ་ཙམ་ཚོགས་པའི་གྲལ་དུ་དགེ་སློང་འཇིགས་མེད་གྲགས་པ་ན་རེ། སྤྱིར་ཡང་ཁྱེད་བོད་གླུ་མཁས་པ་ཡིན་ཞིང་། ཁྱད་པར་དུ་ལོ་ཙྪ་བ་ཁྱེད་རྒྱ་གར་དུ་ཡུན་རིང་དུ་བཞུགས་པ་ལ་སྲོག་ལ་བར་ཆད་མེད་པར་སློབ་གཉེར་མཐར་ཕྱིན་འདུག་པས། བཀྲ་ཤིས་པའི་གླུ་ཞིག་ལེན་དགོས། ཁྱེད་ཀྱི་བླ་མ་མཻ་ཏྲི་པའི་ལྟ་བ་གཙོ་བོར་སྨྲ་བ་ཡིན་ཟེར་བས། ཁོང་གི་དགོངས་པ་ཇི་ལྟར་བཞེད་འདྲིར་བྱུང་བ་ལ། དེའི་ལན་དུ་མཻ་ཏྲི་པའི་བཞེད་གཞུང་གི་རྟོགས་པའི་མགུར་འདི་བཞེངས་པར་གདའ། །
