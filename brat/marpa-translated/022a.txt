སེམས་ལ་ལྷན་པ་གདབ་ཏུ་མེད། ། སྣང་སྲོད་སེམས་ཀྱི་རང་བཞིན་ཤེས། ། གཉེན་པོས་བཅོས་ཏེ་སྒོམ་དུ་མེད། ། སེམས་ཀྱི་གནས་ལུགས་དྲན་དུ་མེད། ། དེའང་མ་བཅོས་ངང་ལ་བཞགདེ་དོན་མཐོང་ན་གྲོལ་བའི་ཕྱིར། ། བུ་ཆུང་གདོལ་པའི་སྤྱོད་པ་ལྟོས། ། ཤ་ཟ་སྨྱོན་པ་གང་དགའར་ཐེངས། ། སྙེམས་དང་བྲལ་བའི་སེང་གེ་བཞིན། ། སེམས་ཀྱི་གླང་པོ་ཡན་དུ་ཆུག། བུང་བ་མེ་ཏོག་རྒྱུ་བར་ལྟོས། ། འཁོར་བར་སྐྱོན་དུ་མ་ལྟ་བས། ། མྱ་ངན་འདས་པ་ཐོབ་བྱར་མེད། ། ཐ་མལ་ཤེས་པ་རང་ལུགས་དེ། ། མ་བཅོས་སོ་མ་ཉིད་དུ་ཞོག། བྱ་བ་རྣམས་ལ་དཔྱོད་མ་རྟོགས། ། ཕྱོགས་དང་ཆ་ལ་མི་གནས་ཤིང་། ། སྤྲོས་བྲལ་ནམ་མཁའི་དཀྱིལ་དུ་ལྟོས། ། ཆོས་ཟད་པར་འཁྱོལ་བ་སྙིང་པོའི་དོན། ། ལྟ་བའི་ཡང་རྩེ་ཕྱག་རྒྱ་ཆེ། ། སེམས་གནད་དུ་བསྣུན་པའི་བརྡ་དོན་ཞིག། བྲམ་ཟེ་ཆེན་པོའི་ཞལ་ནས་ཐོས། ། སྐད་ཅིག་དེ་ལ་གཉིད་ཀྱང་སད། ། དེ་མ་བརྗེད་དྲན་པའི་ལྕགས་ཀྱུས་ཟིན། ། མ་རིག་གཉིད་ཀྱི་མུན་ཁུང་དུ། ། རིག་པ་ཡེ་ཤེས་མཐོངས་ཕྱེ་ནས། ། སྤྲིན་མེད་མཁའ་ལ་ཉི་ཤར་ལྟར། ། འཁྲུལ་པའི་མུན་ནག་སངས་ནས་ཐལ། ། དུས་གསུམ་སངས་རྒྱས་ཞལ་མཇལ་ཀྱང་། ། ཕྱིན་ཆད་དྲི་རྒྱུ་མེད་དོ་སྙམ། ། སེམས་ཁོ་ཐོག་ཆོད་པའི་ཉམས་ཤིག་ཤར། ། ཡིད་འགྱུ་བ་ཟད་པ་ངོ་མཚར་ཆེ། ། ཨེ་མ་ཡི་དམ་མཁའ་འགྲོའི་ལུང་བསྟན་དང་། ། བླ་མས་གསུངས་པ་ཟབ་པོའི་དོན། ། ངག་སྨྲ་རུ་མི་བཏུབ་གསུངས་ན་ཡང་། །
