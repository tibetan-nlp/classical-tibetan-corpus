རྒྱལ་པོ་དེ་ཕྱིར་ཕོ་བྲང་དུ་འོངས་ནས། བཙུན་མོས་འདི་སྐད་ཅེས་གསོལ་ཏོ། ། བདག་ཉི་མ་གཅིག་ཅི་དགའ་བར་བགྱིད་དུ་གནང་བར་དབང་བསྐུར་ཅིག་བྱས་པ་དང་། རྒྱལ་པོས་ཀྱང་དེ་བཞིན་དུ་གནང་ངོ་ཞེས་བསྒོ་བ་དང་། མི་མང་པོ་བཟློག་སྟེ་ལྷའི་རྟེན་བཅོམ་ནས་ས་ཡང་བསྙམས་སོ། ། དེ་ནས་ལྷ་རྟེན་སྲུང་བའི་ལྷ་དེ་ཤིན་ཏུ་མི་དགའ་སྟེ། འཁྲུགས་ནས། རྒྱལ་པོ་ལ་གནོད་པར་བྱ་བར་བསམས་ཏེ། ཕོ་བྲང་གི་ནང་ན་སོང་བ་ལས་ཕོ་བྲང་སྲུང་བའི་ལྷ་དེས་ནང་དུ་མ་བཏང་ངོ་། ། དེའི་ཚེ་ན་དྲང་སྲོང་གཅིག་རི་དྲང་སྲོང་ལྷགས་ཞེས་བྱ་བ་ལ་གནས་ཏེ། རྒྱལ་པོ་རྐང་བཀྲ་དེས་དྲང་སྲོང་དེ་ལ་མཆོད་པ་བྱས་པས་ཉིན་གཅིག་བཞིན་དུ་ནམ་མཁའ་ལ་འཕུར་ཞིང་ཕོ་བྲང་དུ་འོངས་ཏེ། ཟས་བཟང་པོ་ནི་མི་འདོད་ཀྱིས། ཟས་ཐ་མལ་བ་ཙམ་ཟོས་ནས་ཕྱིར་འགྲོའོ། ། ཉི་མ་གཅིག་དྲང་སྲོང་དེ་མི་འོང་བར་ལྷ་དེས་ཤེས་པས། དྲང་སྲོང་གི་ལུས་སུ་སྤྲུལ་ཏེ། ཕོ་བྲང་གི་ནང་དུ་འགྲོ་བར་བསམས་པ་ལས། ཕོ་བྲང་སྲུང་བའི་ལྷས་ཤེས་ནས་ནང་དུ་མ་བཏང་བ་དང་། སྒོ་ནས་བོས་ཏེ་ནང་དུ་འོང་དུ་མི་སྟེར་རོ། ། ཞེས་བྱས་པ་དང་། རྒྱལ་པོས་ཐོས་ནས་དྲང་སྲོང་ཕྱི་རོལ་ནས་འབོད་པ་མྱུར་དུ་ནང་དུ་ཐོང་ཤིག་ཅེས་བཀའ་སྩལ་པ་དང་། ཕོ་བྲང་སྲུང་བའི་ལྷས་ཀྱང་རྒྱལ་པོའི་བཀའ་བཞིན་དུ་མ་བཀག་སྟེ་ནང་དུ་བཏང་ངོ་། ། ནང་དུ་འོངས་པ་དང་དྲང་སྲོང་གི་སྟན་ལ་འདུག་ནས། རྒྱལ་པོས་ཀྱང་རྟག་ཏུ་ཁ་ཟས་ཅི་ཟ་བ་ཙམ་དུ་སྦྱར་ཏེ་ཕུལ་ལོ། ། དེའི་ཚེ་ན་དྲང་སྲོང་སྤྲུལ་པ་དེས་ཁ་ཟས་ཟར་མ་བཏུབ་ནས། རྒྱལ་པོ་ལ་འདི་སྐད་ཅེས་སྨྲས་སོ། ། ཁ་ཟས་འདི་ནི་ངན་ཆེས་ཀྱི་ཤ་དང་ཉ་ནི་ཟའོ་ཞེས་སྨྲས་པ་དང་། རྒྱལ་པོས་ཀྱང་འདི་སྐད་ཅེས་སྨྲས་སོ། ། དྲང་སྲོང་ཆེན་པོ་སྔན་ཅད་ནི་ཁ་ཟས་ཐ་མལ་པ་ཟ་བས་ན་ཤ་དང་ཉ་མ་སྦྱར་བ་ཡིན་ནོ། ། དྲང་སྲོང་སྤྲུལ་པས་སྨྲས་པ། ད་ཕྱིན་ཅད་ནི་ཁ་ཟས་ངན་པ་འདི་དག་མི་འདོད་ཀྱིས་ཤ་དང་ཉ་སྦྱོར་ཅིག་ཅེས་སྨྲས་ནས་སོང་ངོ་། ། དེ་ནས་རྒྱལ་པོས་ཀྱང་དྲང་སྲོང་སྤྲུལ་པས་སྨྲས་པ་བཞིན་དུ་ཤ་དང་ཉ་སྦྱར་ནས། ཕྱི་དེ་ཉིན་དྲང་སྲོང་རྙིང་པ་འོང་བ་ལ་ཕུལ་ལོ། ། དྲང་སྲོང་དེ་ཞེ་སྡང་སྐྱེས་ཏེ་རྒྱལ་པོ་ལ་ཁྲོས་པ་དང་།
