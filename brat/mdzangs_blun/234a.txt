རྒྱ་མཚོའི་ལྷ་དེ་ནི་མ་འགགས་པ་ཡིན་ནོ། ། ཀུན་དགའ་བོ་འདི་ཀླུའི་རྒྱལ་པོ་བྱེད་པ་ན་ཡང་ང་ལ་བསྙེན་བཀུར་བྱས་ཏེ། དུས་ལ་མཁས་པ་ད་ལྟར་གྱི་བར་དུ་ཡང་དུས་ལ་མཁས་ཀྱིས། འདོད་པ་གསུམ་ཡང་ཡིད་བཞིན་དུ་གནང་ངོ་། ། ཞེས་བཀའ་སྩལ་པ་དང་། ཀུན་དགའ་བོ་དགའ་མགུ་རངས་ནས། པུས་མོའི་ལྷ་ང་ས་ལ་བཙུགས་ཏེ། བདག་གིས་ནམ་འཚོའི་བར་དུ་དེ་བཞིན་གཤེགས་པ་ལ་བསྙེན་བཀུར་བགྱིའོ་ཞེས་གསོལ་ལོ། ། བཅོམ་ལྡན་འདས་ཀྱིས་དེ་སྐད་ཅེས་བཀའ་སྩལ་པ་དང་། འཁོར་མང་པོ་རྣམས་ལ་ལ་ནི་རྒྱུན་དུ་ཞུགས་པའི་འབྲས་བུ་ཐོབ་པ་ནས། དགྲ་བཅོམ་པའི་བར་དུ་ཐོབ་བོ། ། ལ་ལ་ནི་རང་སངས་རྒྱས་སུ་གྱུར་ཏོ། ། ལ་ལ་ནི་བླ་ན་མེད་པའི་བྱང་ཆུབ་ཏུ་སེམས་བསྐྱེད་དོ། ། ལ་ལ་ནི་ཕྱིར་མི་ལྡོག་པའི་ས་ཐོབ་ནས། རྗེས་སུ་ཡི་རངས་ནས་མངོན་པར་དག འོ། ། སྦྱིན་པ་ཆེན་པོ་རྒྱ་མཚོར་ཞུགས་པའི་ལེའུ་སྟེ་སུམ་ཅུ་པའོ། ། ། ། འདི་སྐད་བདག་གིས་ཐོས་པའི་དུས་གཅིག་ན། བཅོམ་ལྡན་འདས་རྒྱལ་པོའི་ཁབ་ནྱ་གྲོ་དྷའི་ཚལ་ན་བཞུགས་སོ། ། དེའི་ཚེ་ན་བཅོམ་ལྡན་འདས་ཚངས་པ་དང་བརྒྱ་བྱིན་ལ་སོགས་པས་བསྐོར་ཅིང་བཞུགས་པ་ལས། བཅོམ་ལྡན་འདས་ཀྱི་མཚན་དང་དཔེ་བྱད་དང་གསེར་གྱི་མདོག་ལྟ་བུའི་སྐུ་ལ་འཁོར་མང་པོ་ཐམས་ཅད་ངོ་མཚར་དུ་འཛིན་ཏེ། བསྟོད་ཅིང་བསྔགས་སོ། ། དེ་ལྟར་བཅོམ་ལྡན་འདས་འཁོར་ཀུན་ལས་འཕགས་ཏེ། ཐམས་ཅད་ཀྱི་མཆོག་གཙོ་བོར་གྱུར་པས། དགེ་སློང་དག་ཐེ་ཙོམ་དུ་གྱུར་ནས། བཅོམ་ལྡན་འདས་ལ་ཞུས་པ་དང་། བཅོམ་ལྡན་འདས་ཀྱིས་བཀའ་སྩལ་པ། ད་ལྟར་འབའ་ཞིག་ཏུ་འཁོར་ཀུན་ལས་འཕགས་ཏེ། མཆོག་གཙོར་གྱུར་པ་མ་ཡིན་ནོ། ། སྔོན་འདས་པའི་དུས་ན་ཡང་འདི་དག་ཀུན་ལས་འཕགས་ཏེ་གཙོ་བོར་གྱུར་ཏོ། ། དེ་ནས་དགེ་སློང་དག་གིས་སྔོན་ཅི་ལྟར་གཙོ་བོར་གྱུར་པ་བསྟན་པར་གསོལ་པ་དང་། དེ་ལ་བཅོམ་ལྡན་འདས་ཀྱིས་འདི་སྐད་ཅེས་བཀའ་སྩལ་ཏོ། ། སྔོན་འདས་པའི་དུས་ན་བསྐལ་པ་གྲངས་མེད་དཔག་ཏུ་མེད་བསམ་གྱིས་མི་ཁྱབ་པའི་ཕ་རོལ་ན། འཛམ་བུའི་གླིང་འདིར་རྒྱལ་པོ་ཆེན་པོ་རྒྱལ་ཕྲན་བརྒྱད་ཁྲི་བཞི་སྟོང་ལ་དབང་བྱེད་པ་ཞིག་བྱུང་སྟེ།
