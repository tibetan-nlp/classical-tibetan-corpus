རྒྱང་མ་ནས་དེ་བཞིན་གཤེགས་པ་མེ་ཏོག་གི་སྡོང་པོ་རྒྱས་པ་ལྟ་བུའམ། སྐར་མའི་ནང་ན་ཟླ་བ་ལྟ་བུར་མཐོང་ངོ་། ། དེའི་ཚེ་བཅོམ་ལྡན་འདས་ཀྱིས་ཐོགས་པ་མེད་པའི་ཡེ་ཤེས་ཀྱིས་བདག་འདུལ་བའི་དུས་ལ་བབ་པར་ཤེས་ནས། བཞེངས་ཏེ་ཁོ་མོ་བསུའོ། ། ཁོ་མོ་སྒྲེན་མོར་འདུག་པས་ངོ་ཚ་སྟེ། ལག་པས་ནུ་མ་བཀབ་ནས་ས་ལ་འདུག་གོ། ། དེ་ནས་བཅོམ་ལྡན་འདས་ཀྱིས་ཀུན་དགའ་བོ་ལ་བཀའ་སྩལ་པ། བུད་མེད་འདི་ལ་གོས་བསྐོན་ཅིག་ཅེས་བཀའ་སྩལ་པ་དང་། ཁོ་མོས་གོས་གྱོན་ནས། བཅོམ་ལྡན་འདས་ཀྱི་ཞབས་ལ་མགོ་བོས་ཕྱག་འཚལ་ཏེ། བདག་ལ་ཐུགས་བརྩེ་བར་དགོངས་ནས། བདག་རབ་ཏུ་བྱུང་བར་གནོང་ཤིག་ཅེས་གསོལ་པ་དང་། བཅོམ་ལྡན་འདས་ཀྱིས་ཀུན་དགའ་བོ་ལ་བཀའ་སྩལ་ཏེ། བུད་མེད་འདི་སྐྱེ་དགུའི་བདག་མོ་ལ་གཏད་དེ། རབ་ཏུ་ཕྱུང་ཤིག་ཅེས་བཀའ་སྩལ་ཏོ། ། དེ་ནས་སྐྱེ་དགུའི་བདག་མོས་ཁོ་མོ་རབ་ཏུ་ཕྱུང་ནས། ཁོ་མོ་ལ་བདེན་པ་བཞིའི་ཆོས་བསྟན་ཏེ། ཁོ་མོས་ཐོས་མ་ཐག་ཏུ་བརྩོན་འགྲུས་ནན་ཏན་བྱས་ནས། དགྲ་བཅོམ་པའི་འབྲས་བུ་ཐོབ་སྟེ། འདས་པ་དང་། མ་འོངས་པ་དང་། ད་ལྟར་གྱི་བྱ་བ་ཀུན་རྟོགས་པར་གྱུར་ཏོ། ། དགེ་སློང་མ་དེ་རྣམས་ཀྱིས་སྨྲས་པ། སྔོན་སྡིག་པའི་ལས་ཅི་བགྱིས་ན། དེ་ལྟ་བུའི་རྣམ་པར་སྨིན་པ་མྱོང་བ་སྟོན་ཅིག་ཅེས་སྨྲས་པ་དང་། དགེ་སློང་མ་ཨུད་པ་ལས་ཁྱོད་རྟག་པར་ཉོན་ཅིག ། ཁྱོད་ལ་བསྟན་པར་བྱའོ། ། སྔོན་འདས་པའི་དུས་ཡུན་རིང་པོ་ཞིག་ན། ཁྱིམ་བདག་ཤིན་ཏུ་ཕྱུག་པ་ཞིག་ལ་བུ་མེད་པ་ལས། ཆུང་མ་ཞིག་བླངས་ཏེ། ཆུང་མ་དེ་ལ་ཤིན་ཏུ་བྱམས་པས། རིང་པོ་མ་ལོན་པར་ཆུང་མ་དེ་སེམས་ཅན་དང་ལྡན་པར་གྱུར་ཏེ། ཟླ་བ་ཚང་ནས་ཁྱེའུ་ཞིག་བཙས་ཏེ། ཁྱོ་ཤུག་གཉིས་དེ་བུ་ལ་བྱམས་ཤིང་སྡུག་པར་འཛིན་ཏོ། ། དེ་ནས་ཆེན་མས་འདི་སྙམ་དུ་བསམས་སོ། ། ཁོ་མོ་ནི་རིགས་ཆེ་ཞིང་བཙུན་མོད་ཀྱི་བུ་མེད་པས་རིགས་འཚོབ་པ་མེད་དོ། ། གལ་ཏེ་ཁྱེའུ་འདི་ཆེར་སྐྱེས་ནས་རིགས་འཚོབ་ཅིང་། ནོར་དང་ལོངས་སྤྱོད་ཀྱི་བདག་པོར་འགྱུར་ཏེ།
