རྒྱ་མཚོའི་ནང་ན་བགེགས་ཡ་ང་བ་མང་དུ་ཡོད་པའི་གཏམ་བྱས་སོ། ། བདག་ཀྱང་རྨོངས་པས་ལོ་བཅུ་གཉིས་ཀྱི་བར་དུ་ལྷ་ཀུན་མཆོད་དེ། ཁྱོད་ལྟ་ཞིག་ངའི་བུར་སྐྱེས་ན། ངེད་མ་བོར་བར་ཟན་ཟོ་ཤིག་ཅེས་ཞག་གཅིག་གཉིས་ནས་ཞག་དྲུག་གི་བར་དུ་ཚིག་རྣམ་པ་སྣ་ཚོགས་ཀྱིས་གྲོས་བཏབ་ཀྱང་མ་ཉན་ཏེ། སྔར་ཅི་སྨྲས་པ་ལས་མ་ཟློགས་སོ། ། ཕ་མ་ཡང་དངངས་ནས་འདི་སྐད་ཅེས་བགྲོས་སོ། ། བདག་གི་བུ་ཐོག་མ་བཙས་ནས་ད་ལྟ་ཙམ་གྱི་བར་དུ་བྱ་བ་བརྩམས་པ་མཐར་མ་གྲུབ་ཀྱི་བར་དུ་མི་གཏོང་བས། རྒྱ་མཚོར་བཏང་ན་གལ་ཏེ་ཕྱིར་ལོག་ཀྱང་སྲིད་ཀྱིས། མ་བཏང་ན་ཞག་བདུན་ལོན་ནས་འཕྲལ་དུ་བྱེ་བྲལ་བྱེད་དེ། ད་ནི་བཏང་དགོས་སོ་སྙམ་བགྲོས་ནས། བུའི་གན་དུ་སོང་སྟེ། ལག་པ་ནས་བཟུང་ནས། བུ་ལོངས་ལ་ཟན་ཟོ་ཤིག ། ཡིད་བཞིན་དུ་འགྲོ་བར་གནང་ངོ་ཞེས་སྨྲས་པ་དང་། དགེ་དོན་ལངས་ཏེ། ཟན་ཟོས་བཏུང་བ་འཐུངས་སོ། ། དེ་ནས་ཕྱི་རོལ་ཏུ་བྱུང་སྟེ། མི་མང་པོ་ལ་ཁོ་བོ་ནི་རྒྱ་མཚོར་རིན་པོ་ཆེ་ལེན་དུ་འགྲོའོ། ། ཁྱོད་སུ་འགྲོ་བ་དག་ཀྱང་འགྲོགས་ཏེ། འདོང་གིས་དེད་དཔོན་ཡང་ཁོ་བོས་བྱས་ལ་ཡོ་བྱད་ཀྱང་ཁོ་བོས་སྦྱར་རོ་ཞེས་བསྒྲགས་སོ། ། དེའི་ཚེ་ན་ཡུལ་དེ་ན་ཚོང་པ་ལྔ་བརྒྱ་སྙེད་ཀྱིས་བདག་ཅག་ཀྱང་རྒྱ་མཚོར་རིན་པོ་ཆེ་ལེན་དུ་མཆིའོ། ། ཞེས་ཐམས་ཅད་འདུས་ནས་ཆས་སོ། ། དེའི་ཚེ་ན་ཡུལ་དེ་ན་དེད་དཔོན་རྒན་པོ་ཞིག་སྔོན་ཅད་ཀྱང་རྒྱ་མཚོར་ལན་དུ་མ་ཞིག་ཕྱིན་པ་ཡོད་དེ། རྒྱལ་བུ་ཉིད་དེའི་གན་དུ་སོང་ནས་འདི་སྐད་ཅེས་སྨྲས་སོ། ། ཁོ་བོ་རྒྱ་མཚོར་འགྲོ་བའི་ས་མཁན་དུ་འགྲོ་དགོས་ཀྱིས། ཁྱོད་ཀྱང་ཆོས་ཤིག་ཅེས་བསྒོ་བ་དང་། དེད་དཔོན་གྱིས་བདག་ནི་རྒས་འཁོགས་པའི་སྟེང་དུ་མིག་གིས་ཀྱང་མི་མཐོང་བས། འགྲོ་མི་ནུས་སོ། ། ཞེས་སྨྲས་སོ། ། དེ་ནས་རྒྱལ་བུ་ཕའི་ཐད་དུ་སོང་ནས། རྒྱལ་པོ་ལ་འདི་སྐད་ཅེས་གསོལ་ཏོ། ། དེད་དཔོན་རྒན་པོ་ཞིག་མཆིས་པ་སྔོན་ཅད་རྒྱ་མཚོར་ལན་དུ་མ་ཞིག་ཕྱིན་ཏེ། རྒྱ་མཚོའི་རྒྱུས་མཆིས་ཀྱི་བདག་རྒྱ་མཚོར་འགྲོ་བའི་ས་མཁན་དུ་བཀའ་སྩོལ་ཅིག་ཅེས་གསོལ་པ་དང་། རྒྱལ་པོ་དེ་ཉིད་དེད་དཔོན་ལོང་བའི་གན་དུ་སོང་ནས།
