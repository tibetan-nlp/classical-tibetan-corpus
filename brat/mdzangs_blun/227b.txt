ཡོ་བྱད་ཀྱང་ཁོ་བོས་སྦྱར་རོ་ཞེས་བསྒྲགས་སོ། ། དེའི་ཚེ་ན་མི་ལྔ་བརྒྱ་ཙམ་ཡང་འགྲོ་བར་ཁས་བླངས་ནས་སོ་སོ་ནས་ཡོ་བྱད་སྦྱར་ཏེ། ནམ་འགྲོ་བའི་དུས་ཀྱང་བྱས་ནས། དུས་བཞིན་དུ་ལམ་དུ་ཞུགས་ཏེ་ཆས་སོ། ། ལམ་དུ་འཇུག་པའི་ཚེ་ཁྱེའུའི་ཕ་མ་དང་། རྒྱལ་པོ་དང་། རྒྱལ་བུ་དང་། བློན་པོ་ལ་སོགས་ཏེ་ཡུལ་མི་ཐམས་ཅད་ཀྱིས་ཀྱང་བསྐྱལ་ནས་རྫོངས་བཏབ་བོ། ། ཕ་མ་ཡང་ངུས་ཏེ་ཕྱི་ཕྱག་བྱས་ནས་ཞག་དུ་མའི་ལམ་འདས་པ་དང་། འབྲོག་ས་སྟོང་དུ་ཕྱིན་ཏེ་ནུབ་མོའི་དགོངས་ལམ་དུ་མི་རྒོད་ཀྱིས་བཅོམ་སྟེ། ནོར་རྫས་ཀུན་ཕྲོགས་སོ། དེ་ནས་སོང་སོང་བ་ལས་གྲོང་ཁྱེར་ང་ཤུ་པར་ཞེས་བྱ་བར་ཕྱིན་ནས་གྲོང་ཁྱེར་དེ་ན་བྲམ་ཟེ་ཀ་པྱི་ལི་ཞེས་བྱ་བ་དང་ཕྲད་དེ། བྲམ་ཟེ་དེའི་ཁྱིམ་དུ་སོང་ནས་བྲམ་ཟེ་དེ་ལས་སྦྱིན་པ་ཆེན་པོས་གསེར་སྲང་སུམ་སྟོང་བསླང་བར་བྱས་སོ། ། དེའི་ཚེ་ན་བྲམ་ཟེ་དེ་ལ་བུ་མོ་གསེར་གྱི་མདོག་ཅན་སྐྲ་མཐོན་མཐིང་གར་འདུག་པ། རྒྱལ་ཕྲན་བརྒྱད་ཁྲི་བཞི་སྟོང་སྙེད་ཀྱི་རྒྱལ་བུ་ཐམས་ཅད་ཀྱིས་བསླངས་ན་ཡང་མ་སྟེར་བ་ཅིག་ཡོད་དོ། ། དེའི་ཚེ་ན་སྦྱིན་པ་ཆེན་པོ་བྲམ་ཟེ་དེའི་སྒོར་སོང་སྟེ། བྲམ་ཟེ་ཀ་པྱི་ལི་འདུག་གམ་ཞེས་དྲིས་པའི་ཚེ། བུ་མོས་ཁྱིམ་གྱི་ནང་ནས་སྦྱིན་པ་ཆེན་པོའི་སྐད་ཐོས་པ་ཙམ་གྱིས། ཕྱི་སྒོ་ན་སྐད་གྲག་པ་འདི་ཁོ་མོའི་ཁྱིམ་ཐབ་ཡིན་ནོ། ། ཞེས་ཕ་མ་ལ་སྨྲས་སོ། ། དེ་ནས་བྲམ་ཟེ་སྒོར་བྱུང་ནས་ཕྲད་པ་དང་། ཁྱེའུ་དེ་ལ་འདི་ནི་ངེས་པར་གཞན་པས་འཕགས་སོ་སྙམ་བསམས་སོ། ། ཁྱེའུ་དེས་གསེར་བསླངས་པ་དང་། བྲམ་ཟེས་ཀྱང་སྦྱིན་པར་གནང་ནས་ལག་པ་ཡ་གཅིག་ཏུ་གསེར་ཐོགས་ལག་པ་ཡ་གཅིག་ཏུ་ནི་བུ་མོ་ཁྲིད་དེ། བུ་མོ་འདི་ཡང་ཁྱེད་ཀྱི་ཆུང་མར་སྦྱིན་གྱིས་ལོང་ཤིག་ཅེས་སྨྲས་པ་དང་། སྦྱིན་པ་ཆེན་པོས་སྨྲས་པ། བདག་ནི་རྒྱ་མཚོ་ཆེན་པོར་རིན་པོ་ཆེ་ལེན་དུ་འགྲོ་སྟེ། རྒྱ་མཚོའི་ནང་ན་བགེགས་ཤིན་ཏུ་མང་བས་ཁོ་བོ་འཁོར་ཏམ་མི་འཁོར་གཏོལ་མེད་དེ། ཁྱེད་ཀྱི་བུ་མོ་ཆུང་མར་ལེན་པའི་ཐུགས་མི་ཐུབ་བོ། ། བྲམ་ཟེས་སྨྲས་པ། ཁྱོད་བདེ་བར་འཁོར་ན་བུ་མོ་འདི་ཆུང་མར་ལོང་ཤིག་ཅེས་སྨྲས་པ་དང་། སྦྱིན་པ་ཆེན་པོས་ཀྱང་འཁོར་ན་དེ་བཞིན་བགྱིའོ་ཞེས་སྨྲས་སོ། ། དེ་ནས་བྲམ་ཟེས་ཀྱང་གསེར་སྲང་སུམ་སྟོང་དང་གཞན་ཡང་ཡོ་བྱད་ཅི་དགོས་པ་ཀུན་སྦྱར་ཏེ།
