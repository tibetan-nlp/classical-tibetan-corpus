རྒྱལ་པོས་འདི་ནི་མི་རུང་སྙམ་བསམས་ནས་བདག་ཅག་གཉིས་ཉིན་རེས་ཀྱི་སྤྱན་དྲང་ངོ་ཞེས་བསྒོ་ནས་ཉི་མ་གཅིག་ནི་རྒྱལ་པོས་སྤྱན་དྲངས། ཉི་མ་གཅིག་ནི་བློན་པོས་སྤྱན་དྲངས་ཏེ། གཉི་གའི་རེ་བ་བསྐངས་སོ། ། དེའི་ཚེ་བློན་པོ་ཆེན་པོ་དེས་དེ་བཞིན་གཤེགས་པ་ལ་ཆོས་གོས་སྨད་གསུམ་ཕུལ་ནས་དགེ་སློང་གི་དགེ་འདུན་སྟོང་ཕྲག་དགུ་བཅུ་རྣམས་ལ་ཡང་ཆོས་གོས་བདུན་པ་རེ་རེ་སྦྱར་ཏེ་ཕུལ་ལོ། ། ཀུན་དགའ་བོ་དེའི་ཚེ་དེའི་དུས་ན་བློན་པོ་ཆེན་པོས་ཆོས་གོས་སྦྱར་ཏེ། སངས་རྒྱས་དང་དགེ་འདུན་རྣམས་ལ་ཕུལ་བ་དེ་གཞན་དུ་མ་སེམས་ཤིག ། དེ་ནི་ད་ལྟར་ང་ཡིན་ནོ། ། ང་ནི་སྐྱེ་བ་ཐམས་ཅད་དུ་བསོད་ནམས་བྱེད་པ་ལ་སྐྱོ་བ་མེད་པས་དེ་དག་ཐམས་ཅད་ཀྱི་རྣམ་པར་སྨིན་པ་ཡང་མྱོང་ནས་ཆུད་མ་ཟོས་སོ། ། ཀུན་དགའ་བོ་ལ་སོགས་པ་འཁོར་མང་པོ་རྣམས་བཅོམ་ལྡན་འདས་ཀྱིས་གསུངས་པ་ཐོས་ནས། བསོད་ནམས་ཀྱི་ལས་བྱེད་པ་ལ་བརྩོན་པར་གྱུར་ཏེ། མངོན་པར་དགའ་ནས་རྗེས་སུ་ཡི་རང་ངོ་། ། བྲམ་ཟེས་ལྷན་པ་ཕུལ་བའི་ལེའུ་སྟེ་བཞི་བཅུ་གསུམ་པའོ། ། ། ། འདི་སྐད་བདག་གིས་ཐོས་པའི་དུས་གཅིག་ན། བཅོམ་ལྡན་འདས་མཉན་ཡོད་ན། རྒྱལ་བུ་རྒྱལ་བྱེད་ཀྱི་ཚལ་མགོན་མེད་ཟས་སྦྱིན་གྱི་ཀུན་དགའ་ར་བ་ན་བཞུགས་སོ། ། དེའི་ཚེ་དགེ་སློང་དག་དབྱར་ཚུལ་ལས་བྱུང་ནས། བཅོམ་ལྡན་འདས་ག་ལ་བ་དེར་སོང་སྟེ་ཕྱིན་ནས། བཅོམ་ལྡན་འདས་ལ་ཕྱག་འཚལ་ནས་སྙུན་མི་མངའ་འམ་ཞེས་སྙུན་གསོལ་ཏོ། ། བཅོམ་ལྡན་འདས་ཀྱིས་ཀྱང་བྱམས་པའི་ཐུགས་ཀྱིས་སྤྲོ་བ་བསྐྱེད་དེ། དགེ་སློང་ཁྱོད་སྡུག་བསྔལ་བར་མ་གྱུར་ཏམ། ཞེས་ཤིན་ཏུ་ཐུགས་བརྩེ་ཞིང་བྱམས་པའི་ཐུགས་ཀྱིས་བཀའ་སྩལ་པ་དང་། ཀུན་དགའ་བོས་བཅོམ་ལྡན་འདས་ལ་འདི་སྐད་ཅེས་གསོལ་ཏོ། ། འདི་ལྟར་བཅོམ་ལྡན་འདས་ཀྱིས་དགེ་སློང་དག་ལ་ཐུགས་བརྩེ་ཞིང་བྱམས་པའི་ཐུགས་བསྐྱེད་ནས་ཡུན་ཅི་སྲིད་ཅིག་ལགས། བཅོམ་ལྡན་འདས་ཀྱིས་ཀུན་དགའ་བོ་ལ་བཀའ་སྩལ་པ།
