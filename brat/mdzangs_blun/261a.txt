རྒྱལ་པོས་སྨྲས་པ། འདི་ལྟར་བྱས་པ་ང་ནི་ཉེས་མོད་ཀྱི་ད་ཕྱིན་ཅད་འདི་ལྟར་མི་བྱ་བས། འདིའི་ཚེ་བཏང་དུ་རུང་ངམ་མི་རུང་ཞེས་བསྒོ་བ་དང་། བློན་པོ་རྣམས་ཀྱིས་དེ་ལ་འདི་སྐད་ཅེས་སྨྲས་སོ། ། གལ་ཏེ་གནམ་ལ་ཁ་བ་ནག་པོ་བབ་པའམ། ཁྱོད་ཀྱི་མགོ་ལ་སྦྲུལ་ནག་པོ་སྐྱེས་ཀྱང་བཏང་བ་མེད་ཀྱིས་དམ་སྨྲ་ཤིག ། རྒྱལ་པོ་རྐང་བཀྲས་ཀྱང་བློན་པོས་སྨྲས་པའི་ཚིག་ཐོས་ནས། ད་ནི་གདོན་མི་ཟ་བར་སྲོག་དང་འབྲལ་ལོ་སྙམ་སྟེ་ཡང་བློན་པོ་རྣམས་ལ་འདི་སྐད་ཅེས་སྨྲས་སོ། ། ད་ནི་ཁྱོད་ཀྱིས་བསད་པར་ཆད་ན། སྐད་ཅིག་ཁོ་ན་ཞིག་ཁོ་བོ་བདེ་བར་སེམས་སུ་ཆུག་ལ། གདོད་སོད་ཅིག་ཅེས་བསྒོའོ། ། དེ་ནས་བློན་པོས་ཡང་ཅུང་ཟད་དེ་བསྡད་དེ་སེམས་སུ་བཅུག་པ་དང་། རྒྱལ་པོ་དེས་ཡིད་ལ་འདི་སྙམ་དུ་ཡི་དམ་བཅས་སོ། ། བདག་གི་སྔན་ཅད་དགེ་བ་ཅི་བྱས་པ་དང་། རྒྱལ་པོའི་སྲིད་ཆོས་བཞིན་བྱས་པ་དང་། དྲང་སྲོང་ལ་སྦྱིན་པ་བྱས་པའི་བསོད་ནམས་ཅི་ཡོད་པ་ཐམས་ཅད་ཀྱི་མཐུས། བདག་ནམ་མཁའ་ལ་འཕུར་བའི་སྲིན་པོར་གྱུར་ཅིག་ཅེས་བསྔོས་ཏེ། དེ་སྐད་ཅེས་སྨྲས་ནས། བསྔོས་མ་ཐག་ཏུ་སྲིན་པོར་གྱུར་ཏེ། སྟེང་གི་ནམ་མཁའ་ལ་འཕགས་ནས་བློན་པོ་རྣམས་ལ་འདི་སྐད་ཅེས་སྨྲས་སོ། ། ཁྱོད་ཀྱི་མཐུས་བསྡོས་ཏེ་ང་ལ་གསད་པར་བྱས་ཀྱང་ངའི་བསོད་ནམས་ཀྱིས་རང་སོས་པར་གྱུར་ཏེ། ད་ཕྱིན་ཅད་ནི་ཁྱོད་བཟོད་པར་བྱོས་ཤིག ། ཁྱོད་ཀྱི་ཆུང་མ་དང་བུ་སྡུག་ཅིང་ཕོངས་པ་ཐམས་ཅད་བཟའ་བར་བྱའོ་ཞེས་སྨྲས་ནས། འཕུར་ཏེ་རི་ལ་བབས་ནས་ཕྱིན་ཅད་མིའི་ནང་དུ་མི་ཟ་བར་གྱུར་ཏོ། ། མི་མང་པོ་ཡང་འཇིགས་སྐྲག་ནས་རྟག་ཏུ་བྲོས་ཤིང་གབ་པར་གྱུར་ཏོ། ། དེ་ནས་སྲིན་པོ་མི་ལ་ཟ་ཞིང་སྲིན་པོ་མང་དུ་བསྡོངས་ནས་འཁོར་དང་གཡོག་ཀྱང་མང་བར་གྱུར་ཏེ། སྲིན་པོའི་འཁོར་རྣམས་ཀྱིས་སྲིན་པོ་རྐང་བཀྲ་ལ་འདི་སྐད་ཅེས་གསོལ་ཏོ། ། བདག་ཅག་ནི་ཁྱོད་ཀྱི་འཁོར་དང་གཡོག་ཏུ་གྱུར་ན། བདག་ཅག་ལ་སྟོན་མོ་ཆེན་མོ་བྱིན་ཅིག་ཅེས་གསོལ་པ་དང་། སྲིན་པོ་རྐང་བཀྲས་ཀྱང་དེ་བཞིན་དུ་གནང་ནས།
