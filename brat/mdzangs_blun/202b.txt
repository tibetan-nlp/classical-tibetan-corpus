རྒྱལ་པོས་དེ་སྐད་སྨྲས་པ་ཐོས་ནས་རབ་ཏུ་དགའ་མགུ་རངས་ཏེ། ཕྱག་འཚལ་ནས། ངལ་བར་མ་གྱུར་ཏམ་ཞེས་དྲིས་ཏེ། ནོར་ཅི་འདོད་པ་དང་། ཡུལ་ལམ། རྒྱལ་སྲིད་དམ། གྲོང་ཁྱེར་རམ། ཆུང་མའམ། བུ་ཕོའམ། བུ་མོའམ། རིན་པོ་ཆེའམ། ཐེག་པའམ། གླང་པོ་ཆེའམ། རྟའམ། རིན་པོ་ཆེ་སྣ་བདུན་ནམ། ཕོ་བྲང་ངམ། བྲན་ཕོའམ། བྲན་མོའམ། ལས་བྱེད་པའི་མིའམ། ཅི་འདོད་པ་ཐམས་ཅད་སྦྱིན་ནོ་ཞེས་སྨྲས་སོ། ། བྲམ་ཟེས་སྨྲས་པ། ཕྱི་རོལ་གྱི་ལོངས་སྤྱོད་ཀྱི་སྦྱིན་པ་ནི་སྦྱིན་པའི་བསོད་ནམས་མང་དུ་འགྱུར་མོད་ཀྱི་སྦྱིན་པ་ཆེན་པོར་མི་འགྱུར་ཏེ། གལ་ཏེ་ལུས་ཀྱི་ཤ་ཁྲག་སྦྱིན་པ་བསོད་ནམས་ཀྱི་མཆོག་ཡིན་པས། བདག་ནི་རྒྱལ་པོའི་དབུ་འཚལ་བའི་སླད་དུ་ལམ་རིང་པོ་འདིར་མཆིས་ཀྱིས། ཡིད་དང་མི་འགལ་བར་མཛད་ན་སྦྱིན་པར་མཛོད་ཅིག ། ཅེས་གསོལ་པ་དང་། རྒྱལ་པོས་དེ་སྐད་སྨྲས་པ་ཐོས་ནས་དཔེ་མེད་པར་དགའ་བར་གྱུར་ཏོ། ། བྲམ་ཟེས་སྨྲས་པ་བདག་ལ་དབུ་སྩལ་ན། ནམ་སྩལ་བར་བཀའ་སྩལ་དུ་གསོལ། རྒྱལ་པོས་སྨྲས་པ། ད་སྟེ་ཞག་བདུན་ན་ཁྱོད་ལ་མགོ་སྦྱིན་ནོ་ཞེས་བསྒོའོ། ། དེ་ནས་བློན་པོ་ཟླ་བ་ཆེན་པོས་རིན་པོ་ཆེ་སྣ་བདུན་ལས་བྱས་པའི་མགོ་རྣམས་ཁྱེར་ཏེ་འོངས་ནས། བྲམ་ཟེའི་མདུན་དུ་ལྟོ་བ་ས་ལ་ཕབ་སྟེ་འདི་སྐད་ཅེས་སྨྲས་སོ། ། རྒྱལ་པོའི་དབུ་ནི་ཤ་དང་། ཁྲག་དང་། རུས་པ་སྟེ། མི་གཙང་བ་ཡིན་ཏེ་ཅི་ཞིག་བྱ། རིན་པོ་ཆེ་སྣ་བདུན་གྱི་མགོ་འདི་དག་གིས་མགོ་འདི་བརྗེའོ། ། རིན་པོ་ཆེ་སྣ་བདུན་གྱི་མགོ་འདི་དག་གིས་ཁྱོད་ཚེ་གཅིག་ཏུ་ཕྱུག་པར་འགྱུར་རོ། ། བྲམ་ཟེས་སྨྲས་པ། བདག་ལ་འདི་དག་མི་དགོས་ཀྱིས། རྒྱལ་པོའི་དབུ་དགོས་ཀྱིས་བདག་གི་རེ་བ་སྐོངས་ཤིག ། ཟླ་བ་ཆེན་པོས་ཐབས་རྣམ་པ་སྣ་ཚོགས་ཀྱིས་གསོལ་བ་བཏབ་ཀྱང་མ་བཏུབ་པས། མྱ་ངན་གྱིས་ནོན་ཏེ་འཁམས་ནས་སྙིང་བདུན་དུ་གས་ཏེ་ཚེ་འཕོས་སོ། ། དེའི་ཚེ་ན་རྒྱལ་པོ་དེས་བློན་པོ་ཞིག་གླང་པོ་ཆེ་དཔག་ཚད་བརྒྱ་སྟོང་དུ་འགྲོ་བ་ལ་བསྐྱོན་ནས་རྒྱལ་ཕྲན་ཐམས་ཅད་ལ་སྦྲན་ཏེ། ད་སྟེ་ཞག་བདུན་ན་རྒྱལ་པོ་ཟླ་འོད་བྲམ་ཟེ་ལ་དབུ་སྦྱིན་པ་མཛད་ཀྱིས་མཐོང་བར་འདོད་པ་ཐམས་ཅད་མྱུར་དུ་ཤོག་ཤིག །
