རྒྱལ་པོ་ལྔ་བརྒྱ་བསད་དེ་སྟོན་མོ་བྱ་བར་བགྲོས་ནས་སོང་སོང་ནས་རྒྱལ་པོ་བླངས་ཏེ་བྲག་ཕུག་ཏུ་བཅུག་གོ། ། བཞི་བརྒྱ་དགུ་བཅུ་རྩ་དགུ་ནི་རྙེད་དེ། རྒྱལ་པོ་གཅིག་མ་ཚང་བ་ལ་རྒྱལ་པོ་དེ་དག་བྲག་ཕུག་ན་གཅིག་ལ་གཅིག་འདི་སྐད་ཅེས་སྨྲས་སོ། ། བདག་ཅག་ནི་ད་ཕངས་ཏེ་སྐྱབས་སུ་གྱུར་པ་ཡང་མེད་དེ། གལ་ཏེ་སུ་ཏ་སོ་མ་སྟོབས་དང་ལྡན་པ་དེ་འདིར་འོངས་ན་བདག་ཅག་ཐར་ཀྱང་སྲིད་ཀྱིས་སུ་ཏ་སོ་མ་འདིར་འོངས་པར་སྲིན་པོ་ལ་གསོལ་བར་བྱའོ་ཞེས་བགྲོས་ནས། སྲིན་པོའི་རྒྱལ་པོ་ལ་འདི་སྐད་ཅེས་གསོལ་ཏོ། ། རྒྱལ་པོ་ཁྱོད་ཀྱི་སྟོན་མོ་བྱེད་ན། བདག་ཅག་མིའི་རྒྱལ་པོ་འབའ་ཞིག་བསད་དེ་སྟོན་མོ་བྱས་པས་སྟོན་མོ་བཟང་པོར་མི་འགྱུར་གྱིས། སུ་ཏ་སོ་མ་ཞེས་བྱ་བ་ཡོན་ཏན་ཆེན་པོ་དང་ལྡན་ཏེ། དེ་རྙེད་ན་གདོད་སྟོན་མོ་ཆེན་པོར་འགྱུར་རོ་ཞེས་གསོལ་པ་དང་། སྲིན་པོའི་རྒྱལ་པོས་འདི་ནི་ཅི་དགའ་ཞེས་སྨྲས་པ་དང་། ནམ་མཁའ་ལ་འཕུར་ཏེ་སུ་ཏ་སོ་མ་ལེན་དུ་སོང་ངོ་། ། དེའི་ཚེ་ན་སུ་ཏ་སོ་མ་བུ་མོ་གཞོན་ནུ་དང་གྲོང་ཁྱེར་གྱི་ཕྱི་རོལ་ཏུ་ཁྲུས་བྱེད་དུ་འགྲོ་བ་ལས། ལམ་དུ་བྲམ་ཟེ་སློང་བ་ཞིག་དང་ཕྲད་དེ། སྦྱིན་པར་བྱས་ནས་བྲམ་ཟེ་ལ་འདི་སྐད་ཅེས་སྨྲས་སོ། ། བདག་ཁྲུས་ཞིག་བྱེད་དུ་འགྲོ། ད་ཕྱིར་ལོག་ནས་སྦྱིན་པར་བྱའོ་ཞེས་སྨྲས་ནས། ཁྲུས་བྱེད་དུ་སོང་བ་ལས་ཁྲུས་བྱེད་པ་ན་སྲིན་པོའི་རྒྱལ་པོ་འོངས་ཏེ་བྲག་ཕུག་ཏུ་ཁྱེར་རོ། ། དེའི་ཚེ་སུ་ཏ་སོ་མ་མི་དགའ་སྟེ། ངུས་པ་དང་རྐང་བཀྲས་འདི་སྐད་ཅེས་སྨྲས་སོ། ། ཁྱོད་ནི་གཞན་པས་ལྷག་པར་ཡོན་ཏན་དང་ལྡན་པའི་མཆོག་གོ་ཞེ་ན། སྐྱེས་བུ་ཆེན་པོ་ནི་ཅིས་ཀྱང་མི་སྐྲག་པ་ཡིན་ན། ཅིའི་ཕྱིར་མི་དགའ་སྟེ། བྱིས་པ་ཆུང་ངུ་ལྟར་ངུ་ཞེས་སྨྲས་སོ། ། སུ་ཏ་སོ་མས་སྲིན་པོ་ལ་སྨྲས་པ། བདག་ནི་ལུས་ལ་ཆགས་ཤིང་། སྲིན་པོའི་འཇིགས་པས་མི་དགའ་བ་མ་ཡིན་གྱིས། བདག་ཐོག་མ་ནས་བརྫུན་དུ་སྨྲ་བ་མ་མྱོང་ན། ད་ཅི་ཁྲུས་ལ་འགྲོ་བའི་ལམ་དུ་བྲམ་ཟེ་སློང་བ་ཞིག་དང་ཕྲད་དེ། བདག་གིས་ཀྱང་ཁྲུས་བྱས་ནས་ཕྱིར་ལོག་སྟེ་སྦྱིན་པ་སྦྱིན་ནོ་ཞེས་སྨྲས་ནས། ཁྲུས་བྱེད་དུ་སོང་བ་ལས་རྒྱལ་པོ་ཁྱོད་ཀྱིས་བདག་འདིར་ཁྱེར་ཏེ་འོངས་པས།
