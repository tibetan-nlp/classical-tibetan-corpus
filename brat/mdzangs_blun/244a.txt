རྒྱལ་བུ་དོགས་པ་སྐྱེས་ཏེ། ཆུང་མ་ལ་འདི་སྐད་ཅེས་སྨྲས་སོ། ། ཁྱོད་བདག་གི་ཆུང་མ་བྱ་ཞེས་ཁྱོ་ཤུག་ཏུ་འདུས་ན། ཉིན་པར་ནི་རྟག་ཏུ་འགྲོའོ། ། མཚན་མོ་འབའ་ཞིག་ནི་ཕྱིར་འོང་ན། འདིར་མི་རྟག་པ་ལ་ཕྱིར་བལྟས་ན། ཁྱོད་གཞན་དུ་སེམས་པ་ཡོད་པར་ངེས་སོ་ཞེས་སྨྲས་པ་དང་། བུ་མོས་དམ་བཅས་པ་བདག་ནི་ཁྱོད་འབའ་ཞིག་ལ་སེམས་ཏེ། བསམ་པ་གཞན་ན་བ་སྤུ་ཙམ་ཡང་མེད་དོ། ། བདག་གིས་སྨྲས་པའི་ཚིག་བདེན་ན། བདེན་པར་སྨྲས་པ་འདིས་ཁྱོད་ཀྱི་མིག་ཡ་གཅིག་མཐོང་བར་གྱུར་ཅིག ། ཅེས་སྨྲས་མ་ཐག་ཏུ་མིག་ཡ་གཅིག་མཐོང་བར་གྱུར་ཏོ། ། དེ་ནས་གཏམ་དེ་དག་གླེངས་ནས། རྒྱལ་བུ་ལ་ཁྱོད་ཀྱི་ཕ་མ་དང་ཡུལ་གང་ན་འདུག་ཅེས་དྲིས་པ་དང་། རྒྱལ་བུས་ཆུང་མ་ལ་སྨྲས་པ། ཁྱོད་ཀྱི་རྒྱལ་པོ་རིན་པོ་ཆེ་གོ་ཆ་ཞེས་བྱ་བ་ཞིག་ཐོས་སམ། ཆུང་མས་སྨྲས་པ། བདག་གིས་ཐོས་སོ། ། འོ་ན་དེ་ནི་ངའི་ཕ་ཡིན་ནོ། ། རྒྱལ་བུ་དགེ་དོན་ཞེས་བྱ་བ་ཐོས་སམ། ཆུང་མས་སྨྲས་པ་ཐོས་སོ། ། འོ་ན་རྒྱལ་བུ་དགེ་དོན་ཅེས་བྱ་བ་ནི་བདག་ཡིན་ནོ་ཞེས་སྨྲས་པ་དང་། ཆུང་མ་དངངས་ནས། འོ་ན་ཅིའི་ཕྱིར་འདི་ལྟར་ཉོན་མོངས་པར་གྱུར་ཞེས་དྲིས་པ་ལས། རྒྱལ་བུས་ཅི་ལྟར་གྱུར་པ་ཞིབ་ཏུ་གཏམ་བྱས་པ་དང་། བུ་མོས་རྒྱལ་བུ་ལ་འདི་སྐད་ཅེས་སྨྲས་སོ། ། རྒྱལ་བུ་སྡིག་དོན་གྱིས་ཁྱེད་ཀྱི་མིག་ཕྱུང་བ་འདི་ནི་འཇིག་རྟེན་གང་ན་ཡང་མི་སྲིད་ན། ཁྱོད་དེ་དང་ཕྲད་ན་ཇི་ལྟར་བྱ་སྙམ། རྒྱལ་བུས་སྨྲས་པ། སྡིག་དོན་གྱིས་ངའི་མིག་ཕྱུང་བ་ཉེས་མོད་ཀྱི། བདག་དེ་ལ་ཅུང་ཟད་ཙམ་མི་དགའ་བའམ་འཁོན་དུ་འཛིན་པ་མེད་དོ། ། ཆུང་མས་སྨྲས་པ། བདག་ཤིན་ཏུ་ཡིད་ཆེས་པར་དཀའ་བས་གནོད་པ་འདི་ལྟ་བུ་བྱས་པས་མི་དགའ་བ་མེད་ཅེས་སྨྲས་པ་ནི། ཚིག་ཏུ་མི་རུང་ངོ་། ། རྒྱལ་བུས་སྨྲས་པ། བདག་ནི་རྒྱལ་བུ་སྡིག་དོན་ལ་མི་དགའ་བ་སྤུ་ཉག་མ་ཙམ་ཡང་མེད་དོ། ། བདག་གི་ཚིག་འདི་བདེན་ན་བདེན་པའི་ཚིག་གིས་མིག་འདི་ཡ་གཅིག་ཀྱང་མཐོང་བར་གྱུར་ཅིག ། ཅེས་སྨྲས་པ་དང་། མིག་ཡ་གཅིག་ཀྱང་གསལ་བར་མཐོང་བར་གྱུར་ཏོ། ། དེ་ལྟར་རྒྱལ་བུས་མིག་གཉི་གས་ཀྱང་མཐོང་ནས་མཚན་དང་དབྱིབས་ཀྱང་གཞན་པས་ལྷག་པར་ཆེ་བས།
