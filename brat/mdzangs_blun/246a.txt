རྒྱལ་བུ་སྡིག་དོན་ནི་ད་ལྟ་ལྷ་སྦྱིན་ཡིན་ནོ། ། འཛམ་བུའི་གླིང་གི་མི་ལ་ངའི་དྲིན་གྱིས་ཁྱབ་པ་ནི་ང་ཐོག་མར་སངས་རྒྱས་པ་ན། ལྷའི་སྲས་པོ་བརྒྱད་ཁྲི་འོངས་པ་དང་། ངའི་ཉེ་གནས་ལུང་བསྟན་པ་རྣམས་ཡིན་ནོ། ། ཀུན་དགའ་བོ་དེའི་ཚེ་དེའི་དུས་ན་དེ་ལྟར་ང་ལ་གནོད་པ་ཆེན་པོ་བྱས་ཀྱང་དེ་ལ་སྙིང་བརྩེ་བའི་སེམས་བསྐྱེད་ན། ད་མངོན་པར་སངས་རྒྱས་ནས་ཉོན་མོངས་པའི་སྒྲིབ་པ་ནི་ཟད། ཐུགས་རྗེ་ཆེན་པོས་རྒྱས་པར་ཁྱབ་པས་ན། གནོད་པ་ཅུང་ཟད་བྱས་པ་ལ་ཐུགས་བརྩེ་བ་ལྟ་ཅི་སྨོས། བཅོམ་ལྡན་འདས་ཀྱིས་དེ་སྐད་ཅེས་གསུངས་ནས། འཁོར་མང་པོ་ལ་ལ་ནི་རྒྱུན་དུ་ཞུགས་པ་དང་། ཕྱིར་མི་ལྡོག་པ་དང་དགྲ་བཅོམ་པ་ཐོབ་བོ། ། ཁ་ཅིག་ནི་རང་སངས་རྒྱས་སུ་གྱུར། ཁ་ཅིག་ནི་བླ་ན་མེད་པའི་བྱང་ཆུབ་ཏུ་སེམས་བསྐྱེད་དོ། ། དེ་སྐད་ཅེས་གསུངས་པ་ལ་ཐམས་ཅད་རྗེས་སུ་ཡི་རངས་ཏེ་མངོན་པར་དག འོ། ། རྒྱལ་བུ་དགེ་དོན་གྱི་ལེའུ་སྟེ་སུམ་ཅུ་གསུམ་པའོ། ། ། ། འདི་སྐད་བདག་གིས་ཐོས་པའི་དུས་གཅིག་ན། བཅོམ་ལྡན་འདས་མཉན་ཡོད་ན། རྒྱལ་བུ་རྒྱལ་བྱེད་ཀྱི་ཚལ་མགོན་མེད་ཟས་སྦྱིན་གྱི་ཀུན་དགའ་ར་བ་ན། དགེ་སློང་གི་དགེ་འདུན་ཆེན་པོ་བརྒྱ་ཕྲག་ཕྱེད་དང་བཅུ་གསུམ་དང་ཐབས་ཅིག་ཏུ་བཞུགས་སོ། ། དེའི་ཚེ་ན་ཡུལ་དེ་ན་སྤྲང་པོ་ལྔ་བརྒྱ་ཞིག་རྟག་ཏུ་སངས་རྒྱས་དང་དགེ་འདུན་ལ་བརྟེན་ཏེ་འཚོའོ། ། ཕྱིས་ནས་སྤྲང་པོ་དེ་དག་འདི་སྙམ་དུ་བསམས་སོ། ། ད་བདག་ཅག་སངས་རྒྱས་ཀྱི་བསྟན་པ་ལ་རབ་ཏུ་བྱུང་ངོ་སྙམ་ནས། སངས་རྒྱས་གང་ན་བ་དེར་ལྷགས་ཏེ། བཅོམ་ལྡན་འདས་ལ་འདི་སྐད་ཅེས་གསོལ་ཏོ། ། བདག་ཅག་ནི་སངས་རྒྱས་དང་དགེ་འདུན་གྱི་ཐུགས་རྗེས་འཚོ་བར་གྱུར་ཏེ། བདག་ཅག་རབ་ཏུ་བྱུང་ངོ་ཞེས་གསོལ་པ་དང་། དེ་ལ་བཅོམ་ལྡན་འདས་ཀྱིས་འདི་སྐད་ཅེས་བཀའ་སྩལ་ཏོ། ། ངས་བསྟན་པའི་ཆོས་ནི་རྣམ་པར་དག་པ་སྟེ། བཙན་ཉན་དང་ཕྱུག་དབུལ་མེད་དེ། དཔེར་ན་ཆུ་དང་འདྲ་བས་ན། བཙན་ཉན་དང་ཕྱུག་དབུལ་བཟང་ངན་ཀུན་འཁྲུ་སྟེ།
