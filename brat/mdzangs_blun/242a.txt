ལྷ་མོ་ལྔ་བརྒྱ་སོ་སོ་ནས་འོང་འོང་ནས་རིན་པོ་ཆེ་ཐོགས་ཏེ་འབུལ་བར་བྱེད་དོ། ། ནང་གི་ལྷ་མོ་སྔ་མ་གཅིག་གིས་ལག་པ་ན་ནོར་བུ་རིན་པོ་ཆེ་སྔོན་པོ་ཞིག་ཐོགས་ཏེ། སྨྲས་པ་བཞིན་བླངས་ནས་གོས་ཀྱི་ཐ་མར་ཕུར་ཏེ། དེ་མ་ཐག་ཏུ་ཕྱིར་ལོག་གོ། ། རྒྱལ་བུ་དང་བྲལ་བའི་འོག་ཏུ་རྒྱལ་བུ་སྡིག་དོན་གྱིས་ཚོང་པ་རྣམས་ལ་རིན་པོ་ཆེ་མང་དུ་ལོང་ཤིག་ཅེས་བྱས་པས་ཚོང་པ་རྣམས་ཀྱིས་མང་དུ་བླངས་པས་རིན་པོ་ཆེ་མངས་པས་གྲུ་ཡང་བར་འགའ་ནི་ནུབ། བར་འགའ་ནི་བྱུང་ནས། རྒྱལ་བུ་དགེ་དོན་འབའ་ཞིག་ནི་ཡིད་བཞིན་གྱི་ནོར་བུ་རིན་པོ་ཆེ་ཡོད་པས་ཆུར་མ་ནུབ་བོ། ། རྒྱལ་བུ་སྡིག་དོན་སྐྲག་ནས་དགེ་དོན་ལ་ཡང་སྐྱོབས་ཤིག་ཅེས་སྨྲས་པ་དང་། སྡིག་དོན་ཁྲིད་དེ་རྒྱ་མཚོ་བརྒལ་ཏོ། ། དེ་ནས་སྐམ་སར་ཕྱིན་པ་དང་། སྡིག་དོན་གྱིས་ཕུ་བོ་ལ་འདི་སྐད་ཅེས་སྨྲས་སོ། བདག་ཅག་སྤུན་གཉིས་ཕ་མ་དང་བྲལ་ཏེ། རྒྱ་མཚོར་རིན་པོ་ཆེ་ལེན་དུ་སོང་བ་ལས་མ་སྟེས་ཏེ། རིན་པོ་ཆེ་དང་བྲལ་ནས་ལག་པ་སྟོང་པར་ཁྱིམ་དུ་འགྲོ་བ་ནི་ངོ་ཚའོ། ། རྒྱལ་བུ་དགེ་དོན་རང་བཞིན་གྱིས་དྲང་བས། ནུ་བོ་ལ་བདག་གིས་རིན་པོ་ཆེ་ཐོབ་ན་ཅིའི་ཕྱིར་ངོ་ཚ་ཞེས་སྨྲས་སོ། ། རིན་པོ་ཆེ་ཅི་ལྟ་བུ་ཞིག་རྙེད་པ་བདག་ལ་བསྟན་དུ་གསོལ་ཞེས་སྨྲས་པ་དང་། ཕུ་བོས་རིན་པོ་ཆེ་བཀྲོལ་ཏེ་ནུ་བོ་ལ་བསྟན་ཏོ། ། ནུ་བོས་རིན་པོ་ཆེ་མཐོང་ནས་འདི་སྙམ་དུ་བསམས་སོ། ། བདག་གི་ཕ་མ་ཡང་བུར་འདྲ་བ་ལས་ཕུ་བོ་ལ་བྱམས་ཏེ། བདག་ལ་མི་བྱམས་པའི་སྟེང་དུ་བདག་གཉིས་འགྲོགས་ཏེ། རིན་པོ་ཆེ་ལེན་དུ་སོང་བ་ལས། ཕུ་བོས་རིན་པོ་ཆེ་རྙེད་ཀྱིས། བདག་ནི་ཅི་ཡང་མ་རྙེད་པས་ད་ཁྱིམ་དུ་འོངས་ན་བདག་ལ་ཕ་མས་དེ་བས་ཀྱང་མི་བྱམས་པར་འགྱུར་གྱིས། བདག་གིས་ཕུ་བོ་གཉིད་ལོག་པའི་ཚེ་བསད་ལ། ནོར་བུ་རིན་པོ་ཆེ་ཁྱེར་ལ་འགྲོའོ། ། གལ་ཏེ་ཕ་མས་དྲིས་ན་ཕུ་བོ་ནི་རྒྱ་མཚོའི་ནང་དུ་ནུབ་བོ། ། ཞེས་བྱ་སྙམ་བསམས་ནས། ཕུ་བོ་ལ་འདི་སྐད་ཅེས་སྨྲས་སོ། ། བདག་ཅག་སྤུན་གཉིས་རིན་པོ་ཆེ་ཐོགས་ཏེ། གཉི་ག་ཅིག་ཅར་ཉལ་བའི་མི་རིགས་ཀྱིས། རེས་ཀྱིས་ཉལ་ལ་མེལ་ཙེ་བྱའོ་ཞེས་སྨྲས་པ་དང་།
