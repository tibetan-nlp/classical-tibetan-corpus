མི་མང་པོ་ལ་འདི་སྐད་ཅེས་བཀའ་སྩལ་ཏོ། ། བུ་མོ་འདི་ལྟར་གཟུགས་བྱད་ལེགས་ཤིང་ཤིན་ཏུ་བཟང་བ་མཛེས་པ་འདི་ལ་འདོད་ཆགས་དང་མ་བྲལ་བ་སུ་ཞིག་ཆགས་པར་མི་འགྱུར་ན། དགེ་ཚུལ་འདི་འབྲས་བུ་ཡང་མ་ཐོབ་སྟེ། སྐྱེ་བ་དང་། འཇིག་པར་བཅས་པའི་ལུས་ཅན་འདིས་ཀྱང་། ཚུལ་ཁྲིམས་ཀྱི་ཕྱིར། ལུས་དང་སྲོག་བཏང་ཕོད་པ་ནི་ངོ་མཚར་ཆེའོ། ། ཞེས་བསྒྲགས་ནས། དེའི་སློབ་དཔོན་ཡང་སྤྱན་དྲངས་ཏེ། འཁོར་མང་པོ་ལ་དམ་པའི་ཆོས་སྟོན་དུ་བཅུག་ནས། འཁོར་དེའི་ནང་དུ་འདུས་པ་ཐམས་ཅད་ཀྱིས་ཆོས་བསྟན་པ་ཐོས་ནས། ཁ་ཅིག་ནི་རབ་ཏུ་བྱུང་ཞིང་ཚུལ་ཁྲིམས་བསྲུང་བར་གསོལ་ཏོ། ། ཁ་ཅིག་ནི་བླ་ན་མེད་པའི་བྱང་ཆུབ་ཏུ་སེམས་བསྐྱེད་དེ། ཐམས་ཅད་རྗེས་སུ་ཡི་རངས་ནས་མངོན་པར་དག འོ། ། དགེ་ཚུལ་གྱིས་ཚུལ་ཁྲིམས་བསྲུངས་པའི་ལེའུ་སྟེ་བཅུ་དྲུག་པའོ། ། ། ། དེའི་ཚེ་བཅོམ་ལྡན་འདས་མཉན་ཡོད་ན། རྒྱལ་བུ་རྒྱལ་བྱེད་ཀྱི་ཚལ་མགོན་མེད་ཟས་སྦྱིན་གྱི་ཀུན་དགའ་ར་བ་ན། དགེ་སློང་ཆེན་པོའི་དགེ་འདུན་དང་། ཐབས་ཅིག་ཏུ་བཞུགས་ཏེ། ཆོས་སྟོན་ཏོ། ། དེའི་ཚེ་ན་ཡུལ་དེ་ན་ཁྱིམ་བདག་ཆེན་པོ། ། བྱུག་པ། ནོར་མང་བ། གསེར་དང་། དངུལ་དང་། རིན་པོ་ཆེ་སྣ་བདུན་དང་། གླང་པོ་ཆེ་དང་། རྟ་དང་། བ་ལང་དང་། ལུག་དང་། བྲན་ཕོ་དང་། བྲེན་མོ་དང་། ལས་བྱེད་པའི་མི་རབ་ཏུ་མང་བ། བང་མཛོད་རབ་ཏུ་མང་བ་ཞིག་ལ། བུ་ཁྱེའུ་ནི་མེད་ཀྱིས། བུ་མོ་ནི་ལྔ་ཡོད་དེ། དེ་ཡང་མངའ་བྱད་མཛེས་པ་ཤེས་པའི་བློ་དང་ལྡན་པ་ཤ་སྟག་གོ། ། དེའི་ཚེ་ཁྱིམ་བདག་དེའི་ཆུང་མ་སྦྲུམ་མ་ལས། ཁྱིམ་བདག་དེ་ཚེ་འཕོས་པར་གྱུར་ཏོ། དེའི་ཚེ་རྒྱལ་པོའི་ཡུལ་ཆོས་ཀྱིས། བུ་ཕོ་མེད་ན་ལོངས་སྤྱོད་ཅི་བདོག་པ་ཐམས་ཅད་རྒྱལ་པོས་བཞེས་ཏེ། རྒྱལ་པོས་ཀྱང་བློན་པོ་ཅིག ། དེའི་ནོར་ཕྱུགས་དང་། ལོངས་སྤྱོད་ཅི་ཡོད་པ་ཐམས་ཅད་བརྩིས་ནས། བླར་བཞེས་པར་བཀའ་སྩལ་པ་དང་། བུ་མོ་དེ་དག་གིས་བསམས་པ། བདག་ཅག་གི་མའི་རུམ་མི་བདེ་བ་ཕོའམ།
