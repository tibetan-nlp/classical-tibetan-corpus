རྒྱ་མཚོ་ཡང་གྲིབ་མས་མུན་པར་གྱུར་པ་དེ་ལ་མཽ་ད་གལ་འཛེགས་ཏེ། རྩིབ་རུས་གཅིག་གི་སྟེང་ནས་ཕན་ཚུན་དུ་འགྲོ་བའི་ཕྱི་བཞིན་དུ། བདག་ཀྱང་སོང་བ་མཐོང་ནས་འདི་སྙམ་དུ་བསམས་སོ། ། བདག་ནི་སློབ་དཔོན་དང་དེའུ་རེ་བྲལ་གྱིས་། སྔར་མཐོང་བའི་དངོས་པོ་དེ་དག་ཞུའོ་སྙམ་བསམས་ན། སློབ་དཔོན་བདག་གིས་སྔར་མཐོང་བའི་དངོས་པོ་དེ་དག་བསྟན་དུ་གསོལ་ཞེས་ཞུས་པ་དང་། མཽད་གལ་གྱིས་ཀྱང་ད་དུས་ལ་བབ་ཀྱིས་ཁྱོད་ལ་བསྟན་པར་བྱའོ། ། སྔར་མཐོང་བའི་བུད་མེད་ཤེས་པར་འདོད་ན་དེ་ནི་རྒྱལ་པོའི་ཁབ་ནས་སར་ཕག་ཆེན་པོ་ཞེས་བྱ་བའི་ཆུང་མ་ཡིན་ཏེ། བྱང་བཞིན་བཟང་བས་ཁྱིམ་ཐབ་ཀྱང་ལྷག་པར་བྱམས་སོ། ། དེའི་ཚེ་སར་ཕག་དེ་རྒྱ་མཚོ་ཆེན་པོར་འཇུག་པ་ལས་ཆུང་མ་གཏོང་མ་ཕོད་པས། ཆུང་མ་ཡང་ཁྲིད་དེ་ཚོང་པ་ལྔ་བརྒྱ་ཙམ་དང་། རྒྱ་མཚོར་ཞུགས་ནས་དེའི་ཚེ་དེའི་ཆུང་མ་ཡང་རྟག་ཏུ་ཤིང་གསུམ་མགོ་བསྡུས་ཏེ་མདུང་ཁྱིམ་ལྟར་བྱས་ཏེ་དེ་ལ་མེ་ལོང་བཞག་སྟེ། མེ་ལོང་ལ་བལྟས་ན་བདག་བྱད་བཞིན་བཟང་བར་མཐོང་ནས། ཤིན་ཏུ་ཆགས་པ་སྐྱེས་སོ། ། དེའི་ཚེ་རྒྱ་མཚོ་ནང་ནས་རུས་སྦལ་ཆེན་པོ་ཞིག་གིས་གྲུ་བརྫིས་ཏེ། གྲུ་ཞིག་ནས་ཚོང་པ་ལྔ་བརྒྱ་ཙམ་ཐམས་ཅད་ཚེའི་དུས་བྱས་སོ། ། རྒྱ་མཚོའི་ཆོས་ཉིད་ཀྱིས་རོ་ཞག་ལོན་པར་མི་གནས་པར་ཆུས་ཕུལ་ནས་རྒྱ་མཚོའི་འགྲམ་དུ་ཕྱིན་ཏོ། ། སེམས་ཅན་དེ་དག་ཚེའི་དུས་བྱས་པ་ན་ཅི་ལ་ཆགས་པ་དེར་སྐྱེ་བར་འགྱུར་རོ། ། འོ་ན་སུ་ཞིག་སེམས་ཅན་དམྱལ་བར་སྐྱེ་བར་དགའ་ན་སེམས་ཅན་དམྱལ་བར་མང་དུ་སྐྱེ་བར་འགྱུར། སྨྲས་པ། ། སེམས་ཅན་གང་དཀོན་མཆོག་གསུམ་གྱི་ནོར་བརྐུས་སམ། ཕ་མའི་ལོངས་སྤྱོད་བརྐུས་སམ། མིའི་སྲོག་གཅོད་པའི་ཁ་ན་མ་ཐོ་བ་ཆེན་པོ་བྱས་ན། ཤིན་ཏུ་འབར་བའི་སེམས་ཅན་དམྱལ་བ་ཆེན་པོར་ལྷུང་བར་འགྱུར་རོ། ། མི་དེ་རླུང་ངམ་གྲང་ནད་ཀྱིས་ཟིན་པས་ཡིད་ལ་མེ་འདོད་པར་བསམས་སོ། ། མེའི་ནང་དུ་འགྲོ་བར་བསམས་མ་ཐག་ཏུ་ཚེའི་དུས་བྱས་ནས་ཤིན་ཏུ་འབར་བའི་སེམས་ཅན་དམྱལ་བར་འགྲོའོ། ། དེ་ལ་གང་སངས་རྒྱས་ལ་ཕུལ་བའི་མར་མེ་འམ་རིན་པོ་ཆེ་བརྐུས་སམ། དགེ་འདུན་གྱི་སྒྲོན་མ་འམ་ཤིང་བརྐུས་སམ། དགེ་འདུན་གྱི་གནས་ཁང་བཤིག་གམ།
