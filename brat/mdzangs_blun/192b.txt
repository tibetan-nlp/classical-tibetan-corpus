འདི་སྐད་ཅེས་བཀའ་སྩལ་ཏོ། ། ངས་བལྟས་ན་འཁོར་འདིའི་ནང་ན་སྦྱིན་པ་རྣམ་པར་དག་པ་ནི་འདི་ལས་འདའ་བ་མེད་དོ། ། ཞེས་བཀའ་སྩལ་ཏོ། ། འཁོར་མང་པོ་རྣམས་ངོ་མཚར་དུ་གྱུར་ཏོ། ། དེ་ནས་རྒྱལ་པོའི་བཙུན་མོ་རབ་ཏུ་དགའ་སྟེ། ལུས་ལ་གྱོན་པའི་གོས་དང་། རྒྱན་རྣམས་ཕུད་པས་བུད་མེད་ལ་བསྐྱལ་ལོ། ། རྒྱལ་པོས་ཀྱང་གོས་དང་རྒྱན་ཕུད་ནས་དེའི་ཁྱིམ་ཐབ་ལ་བསྐྱལ་ཏེ། ཁྱོ་ཤུག་གཉིས་འཁོར་གྱི་ནང་དུ་ཤོག་ཤིག་ཅེས་བསྒོའོ། ། སངས་རྒྱས་བསོད་སྐྱབས་ཀྱིས་འཁོར་མང་པོ་ལ་ཆོས་བཤད་པས་ཕལ་ཆེར་རྣམ་པར་གྲོལ་བར་གྱུར་ཏོ། ། བཅོམ་ལྡན་འདས་ཀྱིས་ཀུན་དགའ་བོ་ལ་བཀའ་སྩལ་པ། དེའི་ཚེ་དེའི་དུས་ན་བུད་མེད་དབུལ་མོ་དེ་ནི་དགེ་སློང་མ་དཀར་མོ་འདི་ཡིན་ནོ། ། དེའི་ཚེ་བསམ་པ་ཐག་པས་རས་བྱིན་པའི་དབང་གིས་བསྐལ་པ་དགུ་བཅུ་རྩ་གཅིག་གི་བར་གང་ནས་གང་དུ་སྐྱེས་ཀྱང་རྟག་ཏུ་རས་དང་བཅས་སུ་སྐྱེས་ཏེ། བྲེལ་ཕོངས་མེད་ཅིང་ཡིད་བཞིན་དུ་འབྱོར་པར་གྱུར་ཏོ། ། སངས་རྒྱས་དེ་ལ་དམ་པའི་ཆོས་ཐོས་ནས་རྣམ་པར་གྲོལ་བར་སྨོན་ལམ་བཏབ་པས། འདིར་ང་དང་ཕྲད་དེ་དགྲ་བཅོམ་པར་གྱུར་ཏོ། ། དེ་བས་ན་ཁྱོད་ཀྱང་ཆོས་ཉན་པ་དང་། སྦྱིན་པ་བྱེད་པ་ལ་བརྩོན་འགྲུས་གྱིས་ཤིག ། ཅེས་བཀའ་སྩལ་ནས། འཁོར་མང་པོ་བཅོམ་ལྡན་འདས་ཀྱིས་དེ་སྐད་ཅེས་གསུངས་པ་ལ་རྗེས་སུ་ཡི་རངས་ནས་མངོན་པར་དག འོ། ། དབུལ་མོས་རས་ཕུལ་བའི་ལེའུ་སྟེ་བཅོ་བརྒྱད་པའོ། ། ། ། དེའི་ཚེ་བཙུན་པ་ཀཱ་ཏྱཱ་ཡ་ན་ཆེན་པོ་ཡུལ་ཨ་པ་དེ་ན་འདུག་པ་ལས། ཡུལ་དེ་ན་ཁྱིམ་བདག་ལོངས་སྤྱོད་དང་ལྡན་པ་སེར་སྣ་ཅན་སྙིང་རྗེ་མེད་པ་ཞིག་ལ་བྲན་མོ་རྒན་མོ་ཞིག་ཡོད་དེ། བྲན་མོ་དེ་ངན་ལས་ཐམས་ཅད་བྱེད་ཀྱང་གོས་ནི་ལུས་ལ་མི་གཡོགས། ཟས་ནི་ཁར་མི་མྱོང་སྟེ། རྟག་ཏུ་བརྡུང་བཙོགས་པས་རྒན་མོས་མ་བཟོད་ནས། སྲོག་འདོར་དུ་འགྲོའོ་སྙམ་ནས་བུམ་པ་ཐོགས་ཏེ་ཆུ་འཆུར་སོང་བ་དང་། ཆུ་འགྲམ་དུ་སྐད་ཆེན་པོ་ཕྱུང་སྟེ་ངུས་སོ། ། ཀཱ་ཏྱཱ་ཡ་ན་དེར་འོངས་ཏེ། རྒན་མོ་ཅིའི་ཕྱིར་འདི་ལྟར་མི་དགའ་ཞེས་དྲིས་པ་དང་། འཕགས་པ་བདག་རྒས་པའི་སྟེང་དུ་ལས་ནི་ཆེ།
