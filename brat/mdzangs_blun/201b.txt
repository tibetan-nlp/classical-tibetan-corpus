བྲམ་ཟེ་རྣམས་ཀྱིས་དེ་སྐད་ཅེས་བསྒོ་བ་ཐོས་ནས། སོ་སོ་ནས་གཅིག་ལ་གཅིག་འདི་སྐད་ཅེས་སྨྲས་སོ། ། རྒྱལ་པོ་ཟླ་འོད་འདི་ནི་སྙིང་རྗེ་དང་ལྡན་ཏེ། འགྲོ་བ་ཐམས་ཅད་ལ་དྲིན་གྱིས་ཁྱབ་ཅིང་དབུལ་ཕོངས་པ་ལ་ཡང་ཕ་མ་ལྟ་བུར་གྱུར་པས། བདག་ཅག་ནི་ཚེ་འཕོས་ལ་ཐུག་ཀྱང་དེ་ལ་སུ་ཞིག་གནོད་པ་བྱེད་མི་ནུས་སོ། ། ཞེས་སྨྲས་སོ། ། འཚོ་བ་ལ་ཡང་མ་བལྟས་ཏེ་སོ་སོར་གྱེས་སོ། ། རྒྱལ་པོ་བྱི་མ་སེ་ན་དེ་སྐད་ཅེས་སྨྲས་པ་ཐོས་ནས། ལྷག་པར་མི་དགའ་སྟེ་འདི་སྐད་ཅེས་བཀར་བཏགས་སོ། ། གང་སུ་ལ་ཡང་རུང་ངའི་ཕྱིར་རྒྱལ་པོ་ཟླ་འོད་ཀྱི་མགོ་བཅད་དེ་འོངས་ན་རྒྱལ་སྲིད་ཕྱེད་ཕྱེད་ཀྱང་སྦྱིན་ཞིང་བུ་མོ་ཡང་ཆུང་མར་སྦྱིན་ཞེས་བཀར་བཏགས་པ་དང་། དེའི་ཚེ་རི་བྲགས་ཀྱི་གོང་ན་བྲམ་ཟེ་ལེའུ་ཏེ་ཙ་ཞེས་བྱ་བས་ཐོས་ཏེ། རྒྱལ་པོ་དེ་གང་ན་བ་དེར་འོངས་ནས་རབ་ཏུ་དགའ་སྟེ། དེ་སྐད་དུ་བདག་གིས་ནི་རྒྱལ་པོའི་དགོངས་པ་བཞིན་དུ་བསྒྲུབ་ནུས་སོ་ཞེས་གསོལ་ཏོ། ། རྒྱལ་པོས་སྨྲས་པ། དེ་ལྟར་བསྒྲུབ་ན་ཉི་མ་གང་ལ་འགྲོ་བར་བྱ། བྲམ་ཟེས་སྨྲས་པ། བདག་གི་ལམ་རྒྱགས་སྩོལ་ཅིག་དང་། ད་སླན་ཅད་ཞག་བདུན་ན་མཆིའི་ཞེས་གསོལ་པ་དང་། བྲམ་ཟེ་དེ་ཞག་བདུན་གྱི་བར་དུ་བདག་གི་ལུས་བསྲུང་བའི་སྔགས་བྱས་ནས་རྒྱལ་པོ་ལ་མཆི་ཕྱག་བྱས་ཏེ། ལམ་དུ་ཞུགས་ནས་སོང་ངོ་། ། དེའི་ཚེ་རྒྱལ་པོ་ཟླ་འོད་ཀྱི་ཡུལ་ན་ལྟས་ངན་རྣམ་པ་སྣ་ཚོགས་བྱུང་སྟེ། ས་ཀུན་གས་པ་དང་། གློག་འགྱུ་བ་དང་། སྐར་མ་ལྷུང་བ་དང་། ཉིན་མཚན་དུ། ཡུལ་རྨུགས་དང་། འབྲུག་གྲག་པ་དང་། སེར་བ་དང་། གནམ་ལྕགས་དང་། བྱ་མང་པོས་ནམ་མཁའ་ལ་སྙིང་རྗེ་རྗེའི་སྐད་ཟེར་ཞིང་ནམ་མཁའ་ལ་རང་ལྷུང་བ་དང་། སྟག་དང་། གཟིག་དང་། སྤྱང་ཀི་དང་། གཅན་གཟན་དང་། རི་དགས་ལ་སོགས་པ་རྒྱུགས་མཆོངས་ཤིང་སྐད་མི་སྙན་པ་ཟེར་བ་དང་། རྒྱལ་ཕྲན་བརྒྱད་ཁྲི་བཞི་སྟོང་གི་རྨི་ལམ་དུ་རྒྱལ་པོ་ཆེན་པོའི་གསེར་གྱི་རྒྱལ་མཚན་ཆག་པ་དང་། གསེར་གྱི་རྔ་རལ་བར་རྨིས་སོ། ། བློན་པོ་ཟླ་བ་ཆེན་པོ་ཞེས་བྱ་བའི་རྨི་ལམ་དུ། ཡི་དགས་ཞིག་གིས་རྒྱལ་པོའི་གསེར་གྱི་ཅོད་པན་ཕྲོགས་པ་རྨིས་ཏེ།
