རྒྱལ་པོ་མིག་མངས་ཀྱིས་རྩེ་བ་ཟིན་པས་བློན་པོ་ལ་ཉེས་པ་བྱེད་པའི་མི་དེ་ག་རེ་ཞལ་ཅེ་གཅོད་དོ་ཞེས་བྱས་ན། བློན་པོ་རྣམས་ཀྱིས་བདག་ཅག་གིས་ཁྲིམས་དང་སྦྱར་ཏེ་བཀུམ་ལགས་སོ་ཞེས་སྨྲས་སོ། ། རྒྱལ་པོས་དེ་སྐད་ཅེས་སྨྲས་བ་ཐོས་ནས། འཁམས་ཏེ་ས་ལ་བགྱེལ་བ་དང་། བློན་པོ་རྣམས་ཀྱིས་ཆུས་བཏབ་ནས་ཕྱིར་ལངས་ཏེ། མཆི་མ་བླགས་ནས་འདི་སྐད་ཅེས་སྨྲས་སོ། ། ཕོ་བྲང་གི་སླས་དང་། གླང་པོ་ཆེ་དང་། རྟ་དང་རིན་པོ་ཆེ་སྣ་བདུན་ལ་སོགས་པ་ཐམས་ཅད་ནི་འདིར་ལུས་ཀྱི། ང་འབའ་ཞིག་སེམས་ཅན་དམྱལ་བར་སྡུག་བསྔལ་རྣམས་མྱོང་བར་འོང་བ་ལྟ། བདག་རྒྱལ་པོ་མ་བྱས་པའི་ཚེ་ན་ཡང་ཕོ་བྲང་འདི་ན་རྒྱལ་པོའི་སྲིད་འཚོ་ཞིང་བཞུགས། ། བདག་རིང་པོར་མི་ཐོགས་པར་ཤི་ན་ཡང་དེ་བཞིན་དུ་རྒྱལ་པོའི་སྲིད་འཚོ་བར་འགྱུར་ཏེ། བདག་ནི་མི་བསད་བས་ན་རིགས་ངན་གྱི་རྒྱལ་པོ་ཡིན་ཏེ། སྐྱེ་བ་ཐམས་ཅད་དུ་བདག་གར་འགྲོ་བར་འགྱུར། ད་ནི་རྒྱལ་པོ་ཡང་མི་འདོད་དོ་ཞེས་སྨྲས་ནས། རྒྱལ་པོའི་ཐབས་བཏང་སྟེ་རི་ལ་བདག་སྲུང་ཞིང་རྒྱལ་པོ་དེ་འདུག་པ་ལས། རྒྱལ་པོ་དེའི་ཚེའི་དུས་བྱས་ན། རྒྱ་མཚོ་ཆེན་པོར་ཆུ་སྲིན་ཞིག་ཏུ་སྐྱེས་ཏེ། ལུས་ཆེ་ཁྱད་དཔག་ཚད་བདུན་བརྒྱ་ཡོད་དོ༑ ། རྒྱལ་པོ་འམ་བློན་པོ་འང་རུང་མཐུ་རྩལ་སྐྱེས་ཏེ། སེམས་ཅན་གྱི་སྲོག་གཅོད་དམ། འདོད་པ་དང་བྲལ་བའི་ལས་བྱས་སམ། སེམས་ཅན་ལ་འཚེ་བའི་ལས་བྱས་ན། ཤི་མ་ཐག་ཏུ་ཕལ་ཆེར་ཆུ་སྲིན་དུ་སྐྱེས་ཏེ་དེའི་ལུས་ལ་སྲིན་བུ་མང་པོ་གནས་སོ། ། དེའི་ལུས་གཡའ་བས་ཤེལ་གྱི་རི་ལ་སྙོགས་ཤིང་དྲུད་པ་དང་། སྲིན་བུ་མང་པོ་ཤི་བའི་ཁྲག་གིས་རྒྱ་མཚོ་ཡང་རྒྱང་གྲགས་བརྒྱ་ཙམ་དམར་པོར་བྱས་སོ། ། སྡིག་པའི་ལས་དེས་དེ་དག་ཚེ་འཕོས་ནས་སེམས་ཅན་དམྱལ་བ་ཆེན་པོར་ལྟུང་བར་འགྱུར་རོ། ། ཆུ་སྲིན་དེ་ལན་ཅིག་གཉིད་ཀྱིས་ལོག་ན་ལོ་བརྒྱ་ན་ལན་ཅིག་སངས་ཏེ། བཀྲེས་པ་དང་སྒོམ་པས་ཉེན་ནས་ཁ་གདངས་པ་དང་། རྒྱ་མཚོའི་ཆུ་ཁར་འགྲོ་བ་ནི་ཆུ་ཆེན་པོ་རྒྱ་མཚོར་འབབ་པ་དང་འདྲའོ། ། དེའི་ཚེ་ན་ཚོང་པ་ལྔ་བརྒྱ་ཙམ་ཞིག་རྒྱ་མཚོར་ཞུགས་ཏེ། རིན་པོ་ཆེ་ལེན་དུ་སོང་བ་ལས་ཆུ་སྲིན་ཁ་གདངས་པ་དང་ཕྲད་ནས། གྲུ་ཡང་ཤིན་ཏུ་མགྱོགས་པར་ཆུ་སྲིན་གྱིས་ཟ་འདོད་པའི་ཁར་སོང་ངོ་། །
