སངས་རྒྱས་ཀྱི་སྐུ་གཟུགས་རི་མོར་བྲིས་ཤིག་ཅེས་བསྒོ་བ་དང་། རི་མོ་མཁན་དག་སངས་རྒྱས་ཀྱི་སྤྱན་སྔར་སོང་སྟེ། སངས་རྒྱས་ཀྱི་མཚན་བལྟས་ནས་རི་མོ་བྲིས་པ་ལས། མཚན་གཅིག་བྲིས་ན་གཅིག་མི་འདྲ་བར་གྱུར་ཏེ། རི་མོ་མཁན་གྱིས་རི་མོ་བྲི་བར་མི་ནུས་སོ། ། སངས་རྒྱས་ཉིད་ཀྱིས་ཚོན་རྩི་སྦྱར་ཏེ། སྐུ་གཟུགས་བྲིས་ནས། དེའི་རི་མོ་མཁན་དེ་ལ་བསྟན་ཏེ། སྐུ་གཟུགས་མཚན་དང་ལྡན་པ་བརྒྱད་ཁྲི་བཞི་སྟོང་བྲིས་ནས། རྒྱལ་ཕྲན་རེ་རེ་ལ་སྐུ་གཟུགས་རེ་རེ་བསྐུར་ཏེ། སྐུ་གཟུགས་དེ་ལ་རྒྱལ་པོ་དང་། ཡུལ་མི་ཐམས་ཅད་ཀྱིས་མེ་ཏོག་དང་། སྤོས་དང་ཡོ་བྱད་ཐམས་ཅད་ཀྱིས་མཆོད་ཅིང་ཕྱག་འཚོལ་ཅིག་ཅེས་བསྒོའོ། ། རྒྱལ་ཕྲན་དེ་དག་དང་ཡུལ་མི་དེ་དག་གིས། དེ་བཞིན་གཤེགས་པའི་སྐུ་གཟུགས་མཐོང་ནས་རབ་ཏུ་དགའ་མགུ་ཡི་རངས་ནས། བསྙེན་བཀུར་བྱས་སོ། ། དེའི་ཚེ་རྒྱལ་པོ་གསལ་ཐུབ་དེ་ནི་ད་ལྟར་ང་ཡིན་ཏེ། དེ་ལྟར་དེ་བཞིན་གཤེགས་པའི་སྐུ་གཟུགས་བརྒྱད་ཁྲི་བཞི་སྟོང་བྲིས་ཏེ། རྒྱལ་ཕྲན་དང་ཡུལ་མི་ཀུན་ལ་སྤེལ་བའི་བསོད་ནམས་ཀྱིས། གང་དང་གང་དུ་སྐྱེས་ཀྱང་མཐོ་རིས་སུ་བརྒྱ་བྱིན་ལྷའི་དབང་པོར་སྐྱེས་ཏེ། བྱད་གཟུགས་བཟང་བ་དང་། མཚན་སུམ་ཅུ་རྩ་གཉིས་དང་དཔེ་བྱད་བཟང་པོ་བརྒྱད་ཅུ་ཐོབ་སྟེ་མངོན་པར་སངས་རྒྱས་སོ། ། མྱ་ངན་ལས་འདས་ནས་ཀྱང་རིང་བསྲེལ་ལ་མཆོད་རྟེན་བརྒྱད་ཁྲི་བཞི་སྟོང་སྙེད་བྱེད་པར་འགྱུར་རོ། ། བཅོམ་ལྡན་འདས་ཀྱིས་དེ་སྐད་གསུངས་པ་ལ་ཀུན་དགའ་བོ་དང་འཁོར་མང་པོ་རྣམས་མངོན་པར་དགའ་ནས་རྗེས་སུ་ཡི་རང་ངོ་། ། རྒྱལ་པོ་ཨ་ཤོ་ཀའི་ལེའུ་སྟེ་ཉི་ཤུ་བདུན་པའོ། ། ། ། འདི་སྐད་བདག་གིས་ཐོས་པའི་དུས་གཅིག་ན། བཅོམ་ལྡན་འདས་མཉན་ཡོད་ན། རྒྱལ་བུ་རྒྱལ་བྱེད་ཀྱི་ཚལ་མགོན་མེད་ཟས་སྦྱིན་གྱི་ཀུན་དགའ་ར་བ་ན་བཞུགས་སོ། ། དེའི་ཚེ་ན་དགེ་སློང་དག་དབྱར་སོ་སོའི་ཚུལ་དུ་ཞུགས་ནས། ཚུལ་ནས་བྱུང་བ་དང་བཅོམ་ལྡན་འདས་གང་ན་བ་དེར་ལྷགས་སོ། ། བཅོམ་ལྡན་འདས་དགེ་སློང་དེ་དག་དང་ཡུན་རིང་དུ་མ་ཕྲད་པས་དེ་དག་ལ་ཐུགས་བརྩེ་བར་དགོངས་ཏེ། འཁོར་ལོ་རྩིབས་སྟོང་དང་ལྡན་པའི་ཕྱག་བཏེག་སྟེ། དགེ་སློང་དེ་དག་ལ་བསྟི་སྟང་དང་བཅས་པས་ཁྱེད་ན་བར་མ་གྱུར་ཏམ།
