རྒྱལ་བུ་སྡོང་དུམ་གྱི་ཆུང་མས་ཀྱང་ཁྱོ་ལ་བསྟོད་དོ། ། ཁོ་མོའི་ཁྱོ་ནི་དཔའ་ལ་སྟོབས་ཀྱང་གྱད་དང་མཉམ་སྟེ། ལུས་ཀྱི་ཤ་ཡང་ཤིན་ཏུ་འཇམ་མོ་ཞེས་སྨྲས་པ་དང་། ཕུ་ནུ་མོས་སྨྲས་པ། ཁྱོད་མ་སྨྲ་ཤིག ། ཁྱོད་ཀྱི་ཁྱོ་ནི་ཤིན་ཏུ་ཡང་མི་སྡུག་སྟེ། སྡོང་དུམ་དང་འདྲ་བས་ཅི་སྟེ་ཁྱོད་ཉིན་པར་མཐོང་ན་དངངས་པར་འགྱུར་རོ་ཞེས་སྨྲས་སོ། ། སྡོང་དུམ་གྱི་ཆུང་མས་ཕུ་ནུ་མོ་དེ་དག་དེ་སྐད་ཅེས་སྨྲས་པ་ཐོས་ནས། ཡིད་ལ་བཟུང་སྟེ་མཚན་མོ་མར་མེ་སྦར་ནས། ཁྱོ་གཉིད་ཀྱིས་ལོག་པའི་འོག་ཏུ་མར་མེ་ཕྱུང་སྟེ། བལྟས་ན་ཤིན་ཏུ་མི་སྡུག་པར་མཐོང་ནས་དངངས་སྐྲག་སྟེ། མོད་ལ་ཕྱིར་ཕ་མའི་གནས་སུ་བྲོས་སོ། ། རྒྱལ་བུ་སྡོང་དུམ་གྱིས་ནམ་ནངས་ནས་གཟོད་ཚོར་ཏེ་མི་དགའ་ནས་གཞུ་དང་དུང་ཐོགས་ཏེ། ཆུང་མ་འི་ཡུལ་དུ་ཆུང་མ་སྙགས་སུ་སོང་ངོ་། ། ཡུལ་དེར་ཕྱིན་ནས་བློན་པོ་ཞིག་ལ་གནས་ཏེ། འདུག་འདུག་པ་ལས་དུས་གཞན་ཞིག་ན། རྒྱལ་ཕྲན་དྲུག་གི་རྒྱལ་པོས་ཀྱང་རྒྱལ་པོ་ལི་ཤི་པ་ཙ་ལ་བུ་མོ་བཟང་མོ་ཞིག་ཡོད་ཅེས་ཐོས་ནས། སོ་སོ་ནས་དམག་དག་ཆས་ཏེ་བུ་མོ་སློང་དུ་འོངས་སོ། ། དེའི་ཚེ་རྒྱལ་པོ་ལི་ཤི་པ་ཙ་མི་དགའ་ནས། བློན་པོ་རྣམས་བསྡུས་ཏེ་བགྲོས་ནས། གལ་ཏེ་བུ་མོ་འདི་རྒྱལ་པོ་གཅིག་ལ་བྱིན་ན་ནི་གཙིག་མི་དགའ་ཞིང་འཁོན་དུ་འཛིན་པས། ཅི་ལྟར་བྱས་ན་དགྲ་འདི་དག་མེད་པར་འགྱུར་ཞེས་བགྲོས་པ་དང་། བློན་པོ་གཅིག་གིས་བུ་མོ་འདི་དུམ་བུ་དུམ་བུར་བྱས་ལ། དུམ་བུ་རེ་རེ་བྱིན་ལ་དེ་དག་ཡིད་ཚིམ་པར་འགྱུར་རོ་ཞེས་སྨྲས་སོ། ། བློན་པོ་ཁ་ཅིག་གིས་འདི་སྐད་དུ་ལ་ལ་ཞིག་གིས་དམག་འདི་བཟློག་ནུས་ན། བུ་མོ་ཡང་ཆུང་མར་བྱིན་ལ་ཡུལ་ཕྱེད་ཀྱང་བྱ་དགར་བྱིན་ནོ། ། ཞེས་བཀར་བཏགས་ན་རུང་ངོ་ཞེས་སྨྲས་སོ། ། རྒྱལ་པོས་ཀྱང་གྲོས་དེ་མཉན་ཏེ་བཀར་བཏགས་པ་དང་། དེའི་ཚེ་རྒྱལ་པོ་སྡོང་དུམ་གཞུ་དང་དུང་ཐོགས་ཏེ། གྲོང་ཁྱེར་གྱི་ཕྱི་རོལ་ཏུ་དགྲ་གང་ན་འདུག་པར་སོང་ནས་དུང་བུས་ཏེ་གཞུ་རྒྱུད་སྤྲིངས་པ་ཙམ་གྱིས་ཕ་རོལ་པོའི་དམག་ཐམས་ཅད་སྐྲག་ནས་གཡུལ་འགུལ་ཡང་མ་ནུས་སོ། །
