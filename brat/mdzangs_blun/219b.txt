ཁོ་བོ་ཁྱོད་ཀྱི་སྟེང་དུ་འགྲེང་སྟེ་ལྷུང་བཟེད་དུ་བླུགས་སོ་ཞེས་སྨྲས་ནས། གྲོགས་པོས་ཀྱང་དེ་བཞིན་དུ་བྱའོ་ཞེས་སྨྲས་ཏེ། ཁྱེའུ་དེ་གྲོགས་པོའི་ཕྲག་པ་ལ་འཛེགས་ནས། ས་སྙིམ་པ་གང་སངས་རྒྱས་ལ་ཕུལ་ལོ། ། སངས་རྒྱས་ཀྱིས་ཀྱང་ལྷུང་བཟེད་སྨད་ནས་ས་ཕུལ་བ་བཞེས་སོ། ། བཞེས་ནས་ཀུན་དགའ་བོ་ལ་གཏད་དོ། ། ས་འདི་འཇིམ་པ་གྱིས་ལ་གཙུག་ལག་ཁང་ལ་སྐུས་ཤིག་ཅེས་བཀའ་སྩལ་ཏོ། ། ཀུན་དགའ་བོ་བྱིས་པ་ཆུང་ངུས་དགའ་བའི་སེམས་ཀྱིས་ད་ལྟར་ང་ལས་ཕུལ་བ་བཞེས་ནས། ལྷ་ཁང་ལ་བསྐུས་པས། དེའི་བསོད་ནམས་ཀྱིས་ང་མྱ་ངན་ལས་འདས་ཏེ་ལོ་བརྒྱ་ལོན་པའི་འོག་ཏུ་རྒྱལ་པོ་ཨ་ཤོ་ཀ་ཞེས་བྱ་བར་འགྱུར་རོ། ། ཁྱེའུ་ཅིག་ཤོས་ནི་དེའི་བློན་པོར་འགྱུར་ཏེ། འཛམ་བུའི་གླིང་ལ་དབང་བྱེད་ཅིང་དཀོན་མཆོག་གསུམ་གྱི་ཡོན་ཏན་ཡང་། ཡུལ་ཐམས་ཅད་དུ་བསྒྲགས་ནས་རིང་བསྲེལ་ལ་ཡང་རྒྱས་པར་མཆོད་པ་བྱེད་དོ། ། འཛམ་བུའི་གླིང་དུ་མཆོད་རྟེན་བརྒྱད་ཁྲི་བཞི་སྟོང་ཅིག་ཆར་བྱེད་པར་འགྱུར་རོ། ། ཀུན་དགའ་བོ་དགའ་མགུ་རངས་ནས་བཅོམ་ལྡན་འདས་ལ་འདི་སྐད་ཅེས་གསོལ་ཏོ། ། དེ་བཞིན་གཤེགས་པས་སྔོན་བསོད་ནམས་ཅི་ཞིག་མཛད་པས་རིང་བསྲེལ་ལ་ཡང་མཆོད་རྟེན་དེ་སྙེད་ཅིག་བགྱིད་པར་འགྱུར། བཅོམ་ལྡན་འདས་ཀྱིས་བཀའ་སྩལ་པ། རྟག་པར་ཉོན་ལ་ཤིན་ཏུ་ཡིད་ལ་ཟུང་ཤིག་དང་། ཁྱོད་ལ་བསྟན་པར་བྱའོ། ། སྔོན་བསྐལ་པ་གྲངས་མེད་པ་འདས་པ་ན། འཛམ་བུའི་གླིང་འདིར་རྒྱལ་པོ་གསལ་ཐུབ་ཅེས་བྱ་བ་རྒྱལ་ཕྲན་བརྒྱད་ཁྲི་བཞི་སྟོང་སྙེད་ལ་དབང་བྱེད་པ་ཞིག་བྱུང་ངོ་། ། དེའི་ཚེ་སངས་རྒྱས་པུར་ཤ་ཞེས་བྱ་བ་ཞིག་འཇིག་རྟེན་དུ་བྱུང་སྟེ། རྒྱལ་པོ་དེས་བློན་པོ་དང་བཅས་སུ་ཡོ་བྱད་རྣམ་པ་བཞི་སྦྱར་ཏེ། སངས་རྒྱས་དང་དགེ་སློང་གི་དགེ་འདུན་ལ་ཕུལ་ལོ། ། དེའི་ཚེ་ན་རྒྱལ་པོ་དང་ཡུལ་མི་རྣམས་ནི། རྟག་ཏུ་སངས་རྒྱས་དང་ཕྲད་དེ། ཕྱག་འཚལ་བ་དང་། བསྙེན་བཀུར་བྱེད་དེ། རྒྱལ་ཕྲན་གཞན་དག་ནི་བསོད་ནམས་ཀྱང་མི་བྱེད་པར། མཐའ་འཁོབ་ཀྱི་རྒྱལ་ཕྲན་དང་། ཡུལ་མི་དག་བསོད་ནམས་བྱེད་དུ་གཞུག་པའི་ཕྱིར། སངས་རྒྱས་ཀྱི་སྐུ་གཟུགས་བྲིས་ལ། རྒྱལ་ཕྲན་རྣམས་ལ་སྦྱིན་ནོ་སྙམ་བསམས་ནས། རི་མོ་མཁན་མང་པོ་བསྡུས་ཏེ།
