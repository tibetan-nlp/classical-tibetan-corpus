རྒྱང་མ་ནས་དགེ་སློང་ཁྱོད་ཅུང་ཟད་ཅིག་སྡོད་ཅིག་ཅེས་བོས་སོ། ། བཅོམ་ལྡན་འདས་ཀྱིས་ཀྱང་རྒྱང་མ་ནས། ང་ནི་རྟག་ཏུ་སྡོད་ན་ཁྱོད་མི་སྡོད་ཞེས་བཀའ་སྩལ་པ་དང་། སོར་འཕྲེང་ཅན་གྱིས་ཇི་ལྟར་ཁྱོད་ནི་སྡོད་ལ་ཁོ་བོ་ནི་མི་སྡོད་ཅེས་སྨྲས་པ་དང་། བཅོམ་ལྡན་འདས་ཀྱིས་འདི་སྐད་ཅེས་བཀའ་སྩལ་ཏོ། ། ངའི་དབང་པོ་རྣམས་ནི་རྟག་ཏུ་ཞི་ཞིང་ཏིང་ངེ་འཛིན་གྱི་དབང་ཐོབ་པས་སྡོད་དོ། ། ཁྱོད་ནི་སློབ་དཔོན་གྱིས་ལོག་པར་བསྟན་ཏེ། སྡིག་པ་ལ་ཡང་ཡིད་ཆེས་པས། ཁྱོད་ཀྱིས་སེམས་ཀྱང་བསླུས་ཏེ་མི་སྡོད་པས། ཉིན་མཚན་དུ་གསོད་པའི་ལས་བྱེད་དེ། སྡིག་པ་མཐའ་མེད་པའི་ལས་བྱེད་དོ། ། སོར་འཕྲེང་ཅན་གྱིས་ཐོས་པ་ཙམ་གྱིས་སེམས་ཀྱིས་རྟོགས་ཤིང་ཆུད་དོ། ། རལ་གྲི་ས་ལ་བོར་ཏེ་རྒྱང་མ་ནས་སྐྱབས་སུ་གསོལ་ལོ་ཞེས་ཕྱག་འཚལ་ཏོ། ། དེ་ནས་བཅོམ་ལྡན་འདས་ཀྱང་དེར་གཤེགས་ཏེ་སངས་རྒྱས་ཀྱི་སྐུར་སྤྲུལ་ནས། འོད་ཟེར་ལྷང་ངེ་ལྷན་ནེར་མཚན་སུམ་ཅུ་རྩ་གཉིས་ཀྱིས་གསལ་བར་བརྒྱན་པར་བསྟན་ཏོ། ། སོར་འཕྲེང་ཅན་གྱིས་སངས་རྒྱས་ཀྱི་སྐུ་འོད་ཟེར་དང་མཚན་དུ་ལྡན་པ་མཐོང་ནས། ལུས་ས་ལ་བརྡབས་ནས་ཉེས་པ་མཐོལ་ཞིང་འགྱོད་དེ། བཅོམ་ལྡན་འདས་ཀྱིས་ཆོས་བསྟན་པས་ཆོས་ལ་ཆོས་ཀྱི་མིག་རྣམ་པར་དག་པར་གྱུར་ཏོ། ། དད་པས་རབ་ཏུ་བྱུང་བར་གསོལ་པ་དང་། བཅོམ་ལྡན་འདས་ཀྱིས་ལེགས་པར་འོངས་སོ་ཞེས་བཀའ་སྩལ་པས། སྐྲ་དང་ཁ་སྤུ་རང་བྱི་སྟེ་དགེ་སློང་དུ་གྱུར་ཏོ། ། དེ་ལ་བཅོམ་ལྡན་འདས་ཀྱིས་ཅི་རིགས་པའི་ཆོས་བསྟན་པས་སེམས་ཀྱི་དྲི་མ་ཀུན་ཟད་ནས་དགྲ་བཅོམ་པར་གྱུར་ཏེ། བཅོམ་ལྡན་འདས་ཀྱིས་རྒྱལ་བུ་རྒྱལ་བྱེད་ཀྱི་ཚལ་དུ་ཁྲིད་དོ། ། དེའི་ཚེ་ཡུལ་དེ་ན་གནས་པའི་མི་ལ་སོགས་པའི་སྲོག་ཆགས་ཐམས་ཅད་སོར་འཕྲེང་ཅན་གྱི་མིང་ཐོས་པ་ཙམ་གྱིས་མི་སྐྲག་པ་མེད་དེ། མི་དང་སྲོག་ཆགས་སྦྲུམ་མ་རྣམས་ཀྱང་སྐྲག་ནས་བུ་བཙའ་བའི་ཐབས་མི་ཐུབ་བོ། ། དེའི་ཚེ་བལ་གླང་མོ་ཞིག་བུ་མི་ཕྱིན་པ་དང་། བཅོམ་ལྡན་འདས་ཀྱིས་སོར་འཕྲེང་ཅན་ལ་བཀའ་སྩལ་པ། ཁྱོད་བལ་གླང་མོའི་དྲུང་དུ་སོང་ལ་འདི་སྐད་དུ་བདག་ཐོག་མ་སྐྱེས་ནས་ད་ཙམ་གྱི་བར་དུ་མི་གཅིག་ཀྱང་མ་བསད་དོ་ཞེས་བདེན་པའི་ཚིག་སྨྲོས་ཤིག་ཅེས་བཀའ་སྩལ་པ་དང་།
