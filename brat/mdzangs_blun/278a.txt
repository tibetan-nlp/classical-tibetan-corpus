ཁྱིམ་བདག་དེས་སྨྲས་པ། སངས་རྒྱས་ཞེས་བྱ་བ་ཅི་ལྟ་བུ། རྒྱལ་པོས་སྨྲས་པ། ཅི་ཁྱོད་ཀྱིས་མ་ཐོས་སམ། གྲོང་ཁྱེར་སེར་སྐྱའི་རྒྱལ་པོ་ཟས་གཙང་མའི་སྲས་རྒ་བ་དང་། ན་བ་དང་། ཤི་བ་ལ་སྐྱོ་ནས་རབ་ཏུ་བྱུང་སྟེ། བྱང་ཆུབ་ལ་ནན་ཏན་བྱས་པས་མངོན་པར་སངས་རྒྱས་ཏེ། མཚན་སུམ་ཅུ་རྩ་གཉིས་དང་། དཔེ་བྱད་བཟང་པོ་བརྒྱད་ཅུས་སྐུ་བརྒྱན་པ་ནས། རྫུ་འཕྲུལ་གྱི་རྐང་པ་དང་ཡེ་ཤེས་ནི་དཔེ་ཟླ་མེད་དེ། ལྷ་དང་མིའི་གཙོ་བོར་གྱུར་བས་སངས་རྒྱས་ཞེས་བྱའོ། ། ཁྱིམ་བདག་གིས་དེ་སྐད་ཅེས་སྨྲས་པ་ཐོས་ནས། ཤིན་ཏུ་དད་པའི་སེམས་སྐྱེས་ནས། རྒྱལ་པོ་ལ་སངས་རྒྱས་ད་ལྟར་གང་ན་བཞུགས་ཞེས་གསོལ་བ་དང། རྒྱལ་བོས་སྨྲས་པ། ད་ལྟར་རྒྱལ་པོའི་ཁབ་འོད་མའི་ཚལ་ན་བཞུགས་སོ། ། རྒྱལ་པོ་གཤེགས་ཟིན་མ་ཐག་ཏུ་ཁྱིམ་བདག་དེ་ཡང་སངས་རྒྱས་གང་ན་བཞུགས་པ་དེར་སོང་ནས་ཕྱིན་པ་དང་། སངས་རྒྱས་ཀྱི་སྐུ་ལ་བརྟགས་ན་རྒྱལ་པོ་ཀུན་བས་ཀྱང་ཤིན་ཏུ་ལྷག་པར་མཐོང་ནས། དགའ་མགུ་སྟེ་རངས་ནས། མགོ་བོས་ཕྱག་འཚལ་ནས་སྙུན་གསོལ་ཏོ། ། དེ་ལ་བཅོམ་ལྡན་འདས་ཀྱིས་ཅི་རིགས་པར་ཆོས་བསྟན་པས། ཁྱིམ་བདག་དེ་ཡང་རྒྱུན་དུ་ཞུགས་པ་ཐོབ་པར་གྱུར་ཏེ། དེས་པུས་མོ་གཉིས་ས་ལ་བཙུགས་ནས། བཅོམ་ལྡན་འདས་ལ་རབ་ཏུ་བྱུང་བར་གསོལ་བ་དང་། བཅོམ་ལྡན་འདས་ཀྱིས་ཀྱང་ལེགས་པར་འོངས་སོ་ཞེས་བཀའ་སྩལ་ནས། སྐྲ་དང་ཁ་སྤུ་རང་བྱི་སྟེ། དགེ་སློང་དུ་གྱུར་ཏོ། ། དེ་ནས་ཡང་བདེན་པ་བཞིའི་ཆོས་ཅི་རིགས་པར་བསྟན་པས་སེམས་ཀྱི་དྲི་མ་ཡོངས་སུ་དག་སྟེ། དགྲ་བཅོམ་པར་གྱུར་ཏོ། ། དེ་ནས་ཀུན་དགའ་བོ་དང་འཁོར་མང་པོ་རྣམས་ཀྱིས་བཅོམ་ལྡན་འདས་ལ་འདི་སྐད་ཅེས་གསོལ་ཏོ། ། དགེ་སློང་དན་བྱི་ལས་བསོད་ནམས་ཅི་ཞིག་བགྱིས་ན་མིའི་ནང་དུ་སྐྱེས་ལ། ལྷའི་ལོངས་སྤྱོད་དང་ལྡན་ཡང་ཁྱིམ་ལ་མ་ཆགས་ཏེ། རབ་ཏུ་བྱུང་བར་དགའ་ནས་དེ་མ་ཐག་ཏུ་འབྲས་བུ་ཐོབ་པར་གྱུར། བཅོམ་ལྡན་འདས་ཀྱིས་ཀུན་དགའ་བོ་ལ་བཀའ་སྩལ་པ། ཤིན་ཏུ་ལེགས་པར་ཉོན་ཅིག་སྔོན་འདས་པའི་ཕ་རོལ་ན་བསྐལ་པ་དགུ་བཅུ་རྩ་གཅིག་འདས་པའི་ཕ་རོལ་ན།
