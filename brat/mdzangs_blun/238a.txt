རྒྱལ་པོ་རབ་ཏུ་དགའ་ནས། ཁྱོད་ཀྱང་བདག་གི་རིགས་སུ་སྐྱེ་བར་ཤོག་ཤིག ། ཅེས་སྨྲས་སོ། ། དེ་ནས་རྒྱལ་པོ་ཕྱིར་ལོག་ནས་དུས་ཕྱི་ཞིག་ན་དྲང་སྲོང་གསེར་གྱི་མདོག་ཅན་ཚེའི་དུས་བྱས་ནས། བཙུན་མོ་ཆེན་མོའི་མངལ་དུ་སྐྱེས་སོ། ། བུད་མེད་མཁས་པས་བུ་ཡོད་པར་ཤེས་ནས་བུ་ཕོར་ཤེས་ཏེ་བདག་ལ་ངེས་པར་བུ་ཕོ་ཡོད་དོ། ། ཞེས་རྒྱལ་པོ་ལ་སྨྲས་པ་དང་། རྒྱལ་པོས་ཀྱང་བཙུན་མོའི་རིམ་གྲོ་བསྐྱེད་དོ། ། ཟླ་བ་དགུ་ཚང་ནས་བུ་ཁྱེའུ་གསེར་གྱི་མདོག་ཅན། སྐྲ་མཐོན་མཐིང་ཁར་འདུག་པ། མཚན་དང་ལྡན་པ་ཞིག་བཙས་སོ། ། རྒྱལ་པོ་རབ་ཏུ་དགའ་ནས་བཙས་སྟོན་བྱས་ཏེ། མཚན་མཁན་གྱིས་མཚན་བཏགས་ནས། བུའི་མིང་ཡང་དགེ་དོན་དུ་བཏགས་སོ། ། དེ་ནས་བཙུན་མོ་གཅིག་ཀྱང་སེམས་ཅན་དང་ལྡན་པར་གྱུར་ཏེ། དྲང་སྲོང་ཅིག་ཤོས་ཀྱང་ཚེའི་དུས་བྱས་ནས་བཙུན་མོའི་མངལ་དུ་སྐྱེས་སོ། ། དེ་ནས་ཡང་བཙས་སྟོན་བྱས་ཏེ། མཚན་མཁན་གྱིས་མཚན་བལྟས་ནས། ཁྱེའུའི་མིང་སྡིག་དོན་དུ་བཏགས་སོ། ། རྒྱལ་པོ་བུ་ལ་བྱམས་ཤིང་སྡུག་པས་བུ་འདུག་པའི་གནས་དབྱར་དགུན་སྟོན་དཔྱིད་སོ་སོར་བྱས་སོ། ། ཁྱེའུ་ཆེར་སྐྱེས་ནས་རིག་པ་བཅོ་བརྒྱད་ལ་སོགས་ཏེ། གཙུག་ལག་ཐམས་ཅད་ལ་མཁས་པར་གྱུར་ཏོ། ། ཕྱིས་དགེ་དོན་གྱིས་རྒྱལ་པོ་ལ་གསོལ་ཏེ། བདག་ཕྱི་རོལ་ཏུ་འཆག་ཅིང་ལྟད་མོ་ལྟར་འགྲོའོ་ཞེས་བྱས་པ་དང་། དེའི་ཕ་མས་ཀྱང་བུ་འཆག་ཏུ་འགྲོའོ་ཞེས་ལམ་སྲང་ཕྱག་དར་བྱེད་དུ་བཅུག་སྟེ། རྒྱལ་མཚན་དང་། བ་དན་བསྒྲེངས་ནས་མེ་ཏོག་གིས་གཏོར་ཏོ། ། བདུག་པ་དང་། སྤོས་ཀྱིས་བདུག་པ་ལ་སོགས་པས་ཡོ་ལང་བཤམས་ནས། དགེ་དོན་གླང་པོ་རིན་པོ་ཆེ་སྣ་བདུན་གྱིས་བརྒྱན་པ་ལ་བསྐྱོན་ཏེ། དྲིལ་བསྒྲགས་རྫ་རྔ་བརྡུངས་རོལ་མོ་དང་སིལ་སྙན་དཀྲོལ་ཏེ། རྐྱ་པ་བརྒྱ་སྟོང་སྙེད་ཀྱིས་བསྐོར་ཏེ། གྲོང་ཁྱེར་གྱི་ཕྱི་རོལ་ཏུ་སོང་ངོ་། ། ཡུལ་མི་ཐམས་ཅད་ལམ་གྱི་གཡས་གཡོན་ན་ཁང་བུ་བརྩེགས་པའི་སྟེང་ན་མིག་མི་འཛུམས་པར་ལྟ་ཞིང་། འདི་བྱིན་དང་གཟི་བརྗིད་ནི་ཚངས་པ་དང་འདྲའོ་ཞེས་བསྒྲགས་སོ། ། དེ་ནས་སོང་སོང་བ་ལས་མུ་ལྟོ་བ་དག་ཅིག་གིས་གོས་ཧྲུལ་པོ་གྱོན་ནས་སྣོད་ཆག་པ་ཐོགས་ཏེ།
