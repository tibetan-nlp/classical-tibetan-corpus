རྒྱལ་པོ་ཆེན་པོ་དེའི་ཚེ་དེའི་དུས་ན་མར་འཚག་པ་འི་མི་དེ་རྒྱལ་བུ་སྡོང་དུམ་ཡིན་ནོ། དེའི་ཆུང་མ་ནི་སྡོང་དུམ་གྱི་ཆུང་མ་ཉིད་ཡིན་ནོ། དེའི་ཚེ་རང་སངས་རྒྱས་ལ་ལྡོང་དུམ་དང་འདྲ་ཞེས་ཚིག་ངན་པ་སྨྲས་ཏེ། ཁྲོས་བཞིན་དུ་མར་གྱི་ཚིགས་མ་བྱིན་པས་སྐྱེ་པ་ཐམས་ཅད་དུ་ཤིན་ཏུ་མི་སྡུག་པར་གྱུར་ཏོ། ། ཕྱིས་འགྱོད་ཚངས་བྱས་ཏེ་མར་བཟང་པོ་ཕུལ་བས་སྐྱེ་བ་ཐམས་ཅད་དུ་ཕྱིས་བཟང་ཞིང་སྡུག་པར་གྱུར་ཏོ། ། མར་ཕུལ་བས་ན་སྐྱེ་བ་ཐམས་ཅད་དུ་མཐུ་སྟོབས་དག་ལྡན་ཏེ་ཐམས་ཅད་དུ་ཐུབ་པར་གྱུར་ཏོ། ། དེའི་བསོད་ནམས་ཀྱིས་ནི་འཁོར་ལོས་སྒྱུར་བའི་རྒྱལ་པོར་གྱུར་ཏེ། གླིང་བཞི་ལ་དབང་ཞིང་འདོད་པ་ཡིད་བཞིན་དུ་སྤྱོད་པར་གྱུར་ཏོ། ། དགེ་བ་དང་མི་དགེ་བའི་ལས་རྣམ་པར་སྨིན་པ་ནི་ནམ་ཡང་མེད་པར་མི་འགྱུར་ཏེ། དེ་བས་ན་ལུས་དང་ངག་དང་ཡིད་ཀྱི་ལས་བསྲུང་བ་ལནན་ཏན་བྱའོ། ། བཅོམ་ལྡན་འདས་ཀྱིས་དེ་སྐད་ཅེས་བཀའ་སྩལ་ནས། རྒྱལ་པོ་གཟུགས་ཅན་སྙིང་པོ་ལ་སོགས་ཏེ་རྒྱལ་པོ་དང་། ། བློན་པོ་དང་། འཁོར་རྣམ་པ་བཞི་དང་། ལྷ་དང་ཀླུ་ལ་སོགས་པས་ཐོས་ནས། ལ་ལ་ནི་རྒྱུན་དུ་ཞུགས་པའི་འབྲས་བུ་ཐོབ་བོ། ། ལ་ལ་ནི་ལན་ཅིག་ཕྱིར་ལྡོག་པ་དང་། ཕྱིར་མི་ལྡོག་པ་དང་། དགྲ་བཅོམ་པའི་འབྲས་བུ་ཐོབ་བོ། ། ལ་ལ་ནི་རང་སངས་རྒྱས་ཀྱི་དགེ་བའི་རྩ་བ་བསྐྱེད། ལ་ལ་ནི་བླ་ན་མེད་པའི་བྱང་ཆུབ་སེམས་བསྐྱེད་དེ་ཕྱིར་མི་ལྡོག་པའི་ས་ལ་གནས་པར་གྱུར་ནས། འཁོར་ཐམས་ཅད་བཅོམ་ལྡན་འདས་ཀྱིས་གསུངས་པ་ལ་ཡི་རངས་ནས་མངོན་པར་དག འོ། ། སྟོན་པ་དྲུག་བཏུལ་བའི་ལེའུ་སྟེ་བཅུ་གསུམ་པའོ། ། ། ། བམ་པོ་བཞི་བ། འདི་སྐད་བདག་གིས་ཐོས་པའི་དུས་གཅིག་ནས། བཅོམ་ལྡན་འདས་རྒྱལ་པོའི་ཁབ་བྱ་རྒོད་སྤུངས་པའི་རི་ལ་བཞུགས་སོ། ། དེའི་ཚེ་བཅོམ་ལྡན་འདས་ཀྱི་སྐུ་ལ་བསེར་མའི་སྙུན་གྱིས་བཟུང་ནས། སྨན་པ་འཚོ་བྱེད་ཀྱིས་སྨན་རྣམ་པ་སུམ་རྩ་གཉིས་དང་མར་དུ་སྦྱར་ཏེ༑ ཉིན་རེ་བཞིན་སྨན་སྲང་སུམ་ཅུ་རྩ་གཉིས་གསོལ་ལོ། ། ལྷ་སྦྱིན་མ་རུངས་པ་དེ་ཕྲག་དོག་གི་སེམས་བསྐྱེད་ནས། སངས་རྒྱས་དང་མཉམ་པའོ་ཞེས་བསྒྲགས་ཏེ། བཅོམ་ལྡན་འདས་སྨན་གསོལ་ཞེས་ཐོས་ནས་སྨན་འདོད་ཅིང་། སངས་རྒྱས་དང་འདྲ་བར་བྱ་བའི་ཕྱིར། སྨན་པ་འཚོ་བྱེད་ལ་བཅོམ་ལྡན་འདས་ལ་གསོལ་བའི་སྨན་ང་ལ་ཡང་སྦྱོར་ཏེ་བྱིན་ཅིག་ཅེས་བསྒོའོ། །
