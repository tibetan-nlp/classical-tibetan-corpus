དེ་ལུས་ངལ་ནས་ཆང་ཚོང་གི་ཁྱིམ་དུ་སོང་སྟེ། ཆང་བསླངས་ནས་ཆང་ཚོང་མ་དེ་ལ་བུ་ཕོ་ཞིག་བཙས་ཏེ། བུ་གོས་ཀྱིས་གཡོགས་ཏེ་བསྙལ་བ་ལས། དབྱུག་པ་ཅན་དེའི་སྟེང་དུ་འདུག་པ་དང་། ཁྱེའུ་དེ་སྲོག་དང་བྲལ་བར་གྱུར་ཏོ། ། དེས་ཀྱང་དབྱུག་པ་ཅན་བཟུང་སྟེ། ཁྱོད་ཀྱིས་ངའི་བུ་བསད་ཀྱིས་བུ་བྱིན་ཅིག ། དེས་སྨྲས་པ། ངས་མ་བསད་ན་ཁྱོད་ཀྱི་བུ་ངས་ཇི་ལྟར་བྱིན། དེས་ཚུར་ཤོག་རྒྱལ་པོའི་དྲུང་དུ་འདོང་དང་། ངེའུ་ཅག་གི་ཞལ་ཅེ་གཅོད་དོ་ཞེས་སྨྲས་ནས། དེ་དག་དོང་བ་ལས་གནས་གཞན་ཞིག་ན་ཤིང་ཤ་ཀོ་ཏ་ཀ་ལ་བྱ་རོག་ཅིག་འདུག་པ་དེས་དབྱུག་པ་ཅན་མཐོང་ནས། གར་འགྲོ་ཞེས་སྨྲས་པ་དང་། ང་ནི་མི་འགྲོ་ན་འདི་དག་གིས་ང་ཁྲིད་དོ། ། གང་དུ་ཁྲིད་རྒྱལ་པོའི་དྲུང་དུའོ། ། འོ་ན་ཁོ་བོའི་འཕྲིན་ཞིག་ཀྱང་རྒྱལ་པོ་ལ་བྱོས་ཤིག ། གནས་འདི་ཞེས་བགྱི་བ་ན་ཤིང་ཤ་ཀོ་ཏ་ཀ་དེ་ལ་བྱ་རོག་ཅིག་འདུག་པ། བདག་ཤིང་གཞན་གྱི་ཁ་ན་འདུག་པའི་ཚེ་སྐད་མི་སྙན་ལ། ཤིང་འདིའི་ཁར་འོངས་ན་ངོ་མཚར་དུ་སྐད་སྙན་ན། དེ་ཅིའི་ཕྱིར་དེ་སྐད་མཆིའོ་ཞེས་སྨྲོས་ཤིག ། དེ་ནས་ཡང་གནས་གཞན་ཞིག་ཏུ་སྦྲུལ་ཞིག་གིས་མཐོང་ནས་སྔ་མ་བཞིན་དུ་བདག་གི་འཕྲིན་འདི་ཞུས་ཤིག ། བདག་ནམ་ཁུང་ནས་ཕྱིར་འབྱུང་བའི་ཚེ་ནི་བདེ། སླར་འཇུག་པའི་ཚེ་ནི་སྡུག་བསྔལ་ན་ཅིའི་ཕྱིར་དེ་ལྟར་གྱུར་ཅེས་ཞུས་ཤིག ། དེ་དག་དོང་བ་ལས། གནས་གཞན་ཞིག་ན་ན་ཆུང་མ་གཞོན་ནུ་ཞིག་གིས་མཐོང་ནས། སྔ་མ་བཞིན་དུ་ཁོ་མོའི་འཕྲིན་ཡང་ཞུས་ཤིག ། ནམ་ཁོ་མོ་ཕ་མའི་ཁྱིམ་ན་འདུག་པའི་ཚེ། གྱོས་པོའི་ཁྱིམ་འདོད་གྱོས་པོའི་ཁྱིམ་ན་འདུག་པའི་ཚེ་ཕ་མའི་ཁྱིམ་འདོད་ན། དེ་ཅིའི་ཕྱིར་ཞུས་ཤིག་ཅེས་སྨྲས་ནས། དེ་དག་དོང་བ་ལས་རྒྱལ་པོའི་དྲུང་དུ་ཕྱིན་པ་དང་། དེ་དག་རྒྱལ་པོའི་རྐང་པ་ལ་མགོ་བོས་ཕྱག་འཚལ་ཏེ། ཕྱོགས་གཅིག་ཏུ་འདུག་གོ། ། དེ་ནས་རྒྱལ་པོས་དེ་དག་ལ་ཁྱོད་ཅི་ལ་འོངས་ཞེས་དྲིས་པ་དང་། དེ་དག་གིས་དབྱུག་པ་ཅན་དང་། ཁྱིམ་བདག་རྩོད་པ་དེ་དག་ཐམས་ཅད་སྨྲས་སོ། ། རྒྱལ་པོས་དབྱུག་པ་ཅན་ལ་སྨྲས་པ། ཁྱོད་ཀྱིས་གླང་བརྙས་སམ། བརྙས་སོ། ། འོ་ན་ཕྱིར་བྱིན་ནམ།
