རྒྱལ་བུའི་མིང་སྡོང་དུམ་ཞེས་བྱ་བ་ཡང་མིང་རྗེས་ནས་རྒྱལ་བུ་སུ་ལན་ཤན་ཞེས་མིང་བརྟགས་སོ། ། དེ་ནས་དུས་གཞན་ཞིག་ན་རྒྱལ་བུ་དེས་འདི་སྙམ་དུ་བསམས་སོ། ། ད་ནི་དམག་དང་ཆས་ཏེ་ཕོ་བྲང་ཞིག་བྱའོ་སྙམ་དུ་བསམས་ནས། གང་ནས་གཞི་ཡངས་ཤིང་མཉམ་པ་ཞིག་ཏུ་སོང་སྟེ། འདིར་ཕོ་བྲང་གྱིས་ཅེས་བསྒོ་པ་དང་། ཀླུའི་རྒྱལ་པོ་བཞི་ཞིག་མིར་སྤྲུལ་ནས་འདིར་ཕོ་བྲང་མཛད་ན་ཅི་ལས་མཛད་ཅེས་དྲིས་ན། རྒྱལ་བུ་སུ་ལན་ཤན་གྱིས་སྨྲས་པ། ས་ལས་བྱའོ་ཞེས་བསྒོ་བ་དང་། ཀླུའི་རྒྱལ་པོ་བཞིས་ཅིའི་ཕྱིར་རིན་པོ་ཆེ་ལས་མི་མཛད་ཅེས་སྨྲས་སོ། ། རྒྱལ་པོས་སྨྲས་པ། ཕོ་བྲང་ཆེན་པོ་ཞིག་བྱ་དགོས་ན་རིན་པོ་ཆེ་དེ་སྙད་ཅིག་ག་ལས་རྙེད་ཅེས་སྨྲས་པ་དང་། ཀླུའི་རྒྱལ་པོས་བདག་ཅག་གིས་འབུལ་ལོ་ཞེས་སྨྲས་ནས་དེའི་ངོས་བཞིར་ཆུ་མིག་རེ་རེ་སྤྲུལ་ཏེ་འདི་སྐད་ཅེས་སྨྲས་སོ། ། ཤར་ཕྱོགས་ལོགས་ཀྱི་ཆུ་མིག་ལས་བཅུག་ཏེ་བྱས་ན་ཐམས་ཅད་བཻདཱུརྱར་འགྱུར་རོ། ། ལྷོ་ཕྱོགས་ལོགས་ཀྱི་ཆུ་མིག་ལས་བཅུས་ཏེ་བྱས་ན་ཐམས་ཅད་གསེར་དུ་འགྱུར་རོ། ། ནུབ་ཕྱོགས་ལོགས་ཀྱི་ཆུ་མིག་ལས་བཅུས་ཏེ་བྱས་ན། ཐམས་ཅད་དངུལ་དུ་འགྱུར་རོ། ། བྱང་ཕྱོགས་ངོས་ཀྱི་ཆུ་མིག་ལས་བཅུས་ཏེ་བྱས་ན་ཐམས་ཅད་ཤེལ་དུ་འགྱུར་རོ། ། ཞེས་སྨྲས་སོ། ། རྒྱལ་བུས་ཀྱང་དེ་དག་གིས་སྨྲས་པ་བཞིན་ཆུ་མིག་ལས་བཅུས་ནས་དེ་བཞིན་དུ་གྱུར་ཏོ། ། དེ་ནས་ཕོ་བྲང་གི་ར་བ་ཁོར་ཁོར་ཡུག་དཔག་ཚད་བཞི་བརྒྱ་པ་ཅིག་བྱས་ནས་དེའི་ནང་དུ་ཕོ་བྲང་དཔག་ཚད་བཞི་བཅུ་པ་ཞིག་བྱས་ཏེ། ལམ་གྱི་སྲང་ན་ཁང་བུ་བརྩེགས་པ་དང་སྐྱེད་ཤིང་དང་ཁྲུས་ཀྱི་རྫིང་བུ་ཡང་རིན་པོ་ཆེ་སྣ་བཞི་ལས་བྱས་སོ། ། བཟང་ཁྱད་ལྷའི་ཕོ་བྲང་དང་འདྲ་བར་བྱས་སོ། ། ཟིན་མ་ཐག་ཏུ་རིན་པོ་ཆེ་སྣ་བདུན་ཡང་ཡོད་པར་གྱུར་ཏེ། གླིང་བཞི་ལ་དབང་བྱེད་ཅིང་དགེ་བ་རྒྱས་པར་བྱས་སོ། ། རྒྱལ་པོ་ཆེན་པོ་དེའི་དུས་ན། རྒྱལ་པོ་མ་ཧ་ཤ་ཀུལ་ནི་ད་ལྟར་ངའི་ཡབ་རྒྱལ་པོ་ཟས་གཙང་མ་ཡིན་ནོ། ། དེའི་ཚེ་དེའི་དུས་ན་བཙུན་མོ་དེ་ནི་ད་ལྟར་ངའི་ཡུམ་ལྷ་མཛེས་མ་ཡིན་ནོ། ། རྒྱལ་བུ་མི་སྡུག་པ་སྡོང་དུམ་ནི་ད་ལྟར་ང་ཡིན་ནོ། ། དེའི་ཆུང་མ་དེ་ནི་སྐྱེ་དགུའི་བདག་མོ་ཡིན་ནོ། ། དེའི་ཆུང་མ་དེའི་ཕ་དེ་ནི་འོད་སྲུང་ཡིན་ནོ། ། རྒྱལ་པོ་དྲུག་གིས་དམག་དྲངས་ཏེ་བུ་མོ་འདོད་པ་དེ་དག་ནི་སྟོན་པ་དྲུག་པོ་འདི་དག་ཡིན་ཏེ།
