བཟའ་བ་དང་། བཏུང་བ་ཡིད་བཞིན་དུ་སྩལ་བ་དང་། གཉིས་ནི་དགེ་སློང་ནད་པའི་གཡོག་བགྱིད་པའི་བཟའ་བ་དང་། བཏུང་བ་ཡིད་བཞིན་དུ་སྦྱར་བ་དང་། གསུམ་པ་ནི་དགེ་སློང་འགྲོན་པོའི་ཡོ་བྱད་སྦྱོར་བ་དང་། བཞི་ནི་དགེ་སློང་འགྲོན་དུ་འགྲོ་བའི་རྒྱགས་དང་ཡོ་བྱད་སྦྱར་བར་འཚལ་ལོ། ། དེ་ཅིའི་སླད་དུ་ཞེ་ན། དགེ་སློང་ནད་པ་དག་ནི་ནད་ཀྱི་རྐྱེན་སྨན་དང་། བཟའ་བ་དང་། བཏུང་བ་དང་། མི་ལྡན་པ་ནད་འཚོ་དཀའ་སྟེ། གལ་ཏེ་ན་ཡང་ཚེའི་བར་ཆད་དུ་ཡང་འགྱུར་རོ། ། དགེ་སློང་ནད་གཡོག་ཀྱང་ཁ་ཟས་མེད་པའི་རྐྱེན་གྱིས་བསོད་སྙོམས་བགྱིད་དོ། ། བསོད་སྙོམས་ལ་དུས་མ་མཆིས་ཏེ། གལ་ཏེ་འཕྱིས་སམ་འདོད་པ་མ་རྙེད་ན། ནད་པ་ཞེ་སྡང་སྐྱེས་ཏེ། ནད་འཚོ་དཀའ་བས། དེ་ལ་ཡང་ཁ་ཟས་འབུལ་བར་འཚལ་ལོ། ། ཤུལ་རིང་པོ་ནས་མཆིས་པའི་དགེ་སློང་ཡུལ་མ་འདྲིས་ཏེ། བཤེས་མ་མཆིས་པས་བསོད་སྙོམས་བགྱིད་དོ། ། བསོད་སྙོམས་བགྱིད་པ་ན་ཁྱིའམ་མ་རུངས་པས་གཙེས་ན་ཞེ་སྡང་སྐྱེས་ཏེ། བཟོད་པ་ཉམས་པར་འགྱུར་བས་དེ་ལ་ཡང་ཁ་ཟས་འབུལ་བར་འཚལ་ལོ། ། དགེ་སློང་འགྲོན་དུ་མཆི་བ་གྲོགས་དང་ལྡན་པ་ཡང་རྒྱགས་མ་མཆིས་པས་གྲོགས་པོ་དང་བྲལ་ན། ཤུལ་དུ་གཅན་གཟན་མང་པོའམ། མི་རྒོད་མ་རུངས་པ་མཆིས་ཏེ། གཅིག་པུ་སོང་ན་དེ་ཡང་ལུས་ཀྱི་བར་ཆད་དུ་འགྱུར་བས། དེའི་དགོས་པ་ཡང་སྦྱར་བར་འཚལ་ལོ། ། དེའི་ཚེ་བཅོམ་ལྡན་འདས་ཀྱིས་ལྷ་མཚམས་མས་གསོལ་པ་བཞི་གསན་ནས། ཤིན་ཏུ་ལེགས་སོ་ཞེས་བསྔགས་ཏེ། ཁྱོད་ཀྱིས་གསོལ་པ། བཞི་པོ་འདིའི་བསོད་ནམས་རྒྱ་ཆེན་པོ་ནི་སངས་རྒྱས་ལ་མཆོད་པ་བྱས་པ་དང་། ཐ་མི་དད་དོ་ཞེས་བཀའ་སྩལ་ནས། དགེ་འདུན་དང་བཅས་ཏེ། སླར་རྒྱལ་བུ་རྒྱལ་བྱེད་ཀྱི་ཚལ་དུ་གཤེགས་སོ། ། བཅོམ་ལྡན་འདས་གཤེགས་པའི་འོག་ཏུ་སྒྲོམ་བུ་ཕྱེས་ཏེ་བལྟས་ན། བུའི་མགོ་སུམ་ཅུ་རྩ་གཉིས་ཡོད་པར་མཐོང་ནས། འདོད་པ་དང་བྲལ་བའི་ཕྱིར་མི་དགའ་བར་ནི་མ་གྱུར་ཏོ། ། འདི་སྐད་དུ་ཀྱེ་མ་མི་ཅིག་སྐྱེས་པ་ལ་ཚེ་ཟད་པ་ཡང་ཡོད་དེ་དེ་རིང་པོར་མ་ཐོགས་སོ། །
