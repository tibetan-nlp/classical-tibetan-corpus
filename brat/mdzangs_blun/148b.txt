རྒྱལ་པོས་ཀྱང་སྲས་མོ་དེ་དབུལ་པོའི་ཆུང་མར་བྱིན་ཏེ། དེའི་གནས་ཁང་ལ་སོགས་པ་བྱས་ནས། སྒོ་རིམ་པ་བདུན་དུ་བྱས་ཏེ། དམག་པ་ལ་འདི་སྐད་ཅེས་བསྒོའོ། ། ཁྱོད་གང་དུ་འགྲོ་བའི་ཚེ། ཁྱོད་ཉིད་ཀྱིས་སྒོ་ཆོད་ལྡེ་མིག་ཀྱང་ཁྱེར་ཅིག ། ངའི་བུ་མོ་རབ་ཏུ་མི་སྡུག་པ་འདི་ནི་ཅི་ནས་ཀྱང་ཕྱི་མི་གཞན་གྱིས་མི་མཐོང་བར་སྒོ་རབ་ཏུ་སྡོམས་ཤིག་ཅེས་བསྒོ་ནས། ནོར་དང་ལོངས་སྤྱོད་ལ་སོགས་པ་ཡོ་བྱད་ཐམས་ཅད་ཅི་དགོས་པ་ཆོག་པར་བྱིན་ཏེ། བློན་པོར་བཅུག་གོ། ། དེ་ནོར་བུ་རིན་པོ་ཆེ་དང་ལོངས་སྤྱོད་མང་པོ་དང་ལྡན་པར་གྱུར་པས། དུས་ཕྱི་མ་ཞིག་ན་རིགས་ཆེ་ཞིང་མཐོ་བ་གཞན་དང་སྟོན་མོ་བྱ་བར་ཐམས་ཏེ། སྟོན་མོ་ལ་ཁྱོ་ཤུག་དང་། བུ་ཚ་ཡང་འདུས་ཏེ་རྩེ་བར་བྱས་པ་ལས། གཞན་ཀུན་ནི་ཆུང་མ་དང་འགྲོགས་ཏེ་འོངས་ན་བློན་པོ་དེ་འབའ་ཞིག་གི་ཆུང་མ་མ་འོངས་པས་གྲོགས་པོ་དག་ཐེ་ཚོམ་དུ་གྱུར་ཏེ། ཡང་ན་ནི་མི་འདིའི་ཆུང་མ་མངའ་བྱད་ཤིན་ཏུ་མཛེས་ཤིང་བཟང་མོ་ཞིག་གོ། ། ཡང་ན་ནི་ཤིན་ཏུ་མི་སྡུག་ཅིང་གཞན་གྱིས་མཐོང་དུ་མི་རུང་བ་ཞིག་ཡིན་པས། ཁྲིད་དེ་མ་འོངས་སོ་སྙམ་ནས་ཐེ་ཙོམ་ཟ་བས། ད་དག་དེའི་ཆུང་མ་བལྟ་བར་བགྲོས་ནས། ཡིད་འཐུན་པར་གསང་སྟེ། མི་འདི་ཆང་གིས་ར་རོ་བར་བྱས་ཏེ་མི་དྲན་པའི་འོག་ཏུ་ལྡེ་མིག་བརྐུས་ཏེ། མི་ལྔ་ཙམ་ཞིག་དེའི་ཆུང་མར་བལྟ་བར་བཏང་ངོ་ཞེས་བསྒོས་སོ། ། དེའི་ཚེ་ན་མི་དེའི་ཆུང་མས་སྔོན་བདག་གིས་སྡིག་པ་ཅི་བགྱིས་ན་ཁྱིམ་ཐབ་སྡང་སྟེ། འདི་འདྲའི་མུན་ཁང་དུ་བཅུག་ནས་ཉི་མ་དང་ཟླ་བ་ཙམ་ཡང་བལྟར་མེད་དེ། མི་གཞན་ཡང་མི་མཐོང་སྙམ་སྟེ། ཤིན་ཏུ་མི་དགའ་ནས་ཡང་འདི་སྙམ་དུ་བསམས་སོ། ། སངས་རྒྱས་བཅོམ་ལྡན་འདས་འཇིག་རྟེན་ན་བཞུགས་ཏེ། སེམས་ཅན་གང་སྡུག་བསྔལ་གྱིས་གདུངས་པ་དག་གི་མངོན་དང་སྐྱབས་མཛད་དེ་ཐར་བར་མཛད་དོ་ཞེ་ན། བཅོམ་ལྡན་འདས་ལ་བདག་རྒྱང་མ་ནས་ཕྱག་འཚལ་ལོ། ། ཐུགས་བརྩེ་བར་དགོངས་ཏེ། བདག་གི་མདུན་དུ་མངོན་སུམ་དུ་གཤེགས་ཤིང་། བདག་ལ་ཅུང་ཟད་ཅིག་ཞལ་སྟོན་པར་གྱུར་ཀྱང་ཅི་མ་རུང་སྙམ་བསམས་སོ། ། བུ་མོ་དེའི་བསམ་པ་ཤིན་ཏུ་དག་ཅིང་མོས་པར་མཁྱེན་ནས་བཅོམ་ལྡན་འདས་དེའི་ཁྱིམ་དུ་གཤེགས་ཏེ།
