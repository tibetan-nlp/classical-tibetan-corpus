རྒྱལ་པོས་དེ་སྐད་སྨྲས་པ་ཐོས་ནས་བསུས་ཏེ། ཁྱོད་ལམ་ན་ངལ་བར་མ་གྱུར་ཏམ། ཉོན་མ་མོངས་སམ། བདེ་བར་འོངས་སམ། ཞེས་སྤྲོ་བའི་ཚིག་གིས་སྨྲས་ནས། རྒྱལ་སྲིད་དམ། རིན་པོ་ཆེའམ། རྟའམ། ཐེག་པའམ། གོས་སམ། བཟའ་བའམ། བཏུང་བའམ། ནད་ཀྱི་རྐྱེན་རྩིའམ། ཅི་དང་ཅི་དགོས་པ་སློང་ཤིག་དང་། དེ་དང་དེ་དག་ཆོག་པར་སྦྱིན་ནོ་ཞེས་སྨྲས་པ་དང་། བྲམ་ཟེས་སྨྲས་པ། ཕྱི་རོལ་གྱི་ཟང་ཟིང་བྱིན་པས་ནི་བསོད་ནམས་ཀྱི་དམ་པར་མི་འགྱུར་གྱི། ནང་གི་སྦྱིན་པ་བྱས་ན་འབྲས་བུ་ཆེ་སྟེ། བདག་མིག་སྟོར་ནས་ཡུན་རིང་དུ་ལོན་པས་རྟག་ཏུ་མདོངས་ཤིང་འཐོམས་པར་འགྱུར་གྱི། བདག་ནི་རྒྱལ་པོའི་སྤྱན་འཚལ་ལོ་ཞེས་སྨྲས་སོ། ། དེ་སྐད་ཅེས་སྨྲས་པ་དང་། རྒྱལ་པོ་རབ་ཏུ་དགའ་ནས། བྲམ་ཟེ་ལ་ཁྱོད་ལ་མིག་སྦྱིན་ནོ་ཞེས་བསྒོའོ། ། བྲམ་ཟེས་སྨྲས་པ། རྒྱལ་པོས་བདག་ལ་སྩལ་ན་ནམ་སྩོལ་ཅེས་སྨྲས་པ་དང་། ད་སྟེ་ཞག་བདུན་ན་སྦྱིན་ནོ་ཞེས་བསྒོ་ནས། དྲིལ་བསྒྲགས་ནས། རྒྱལ་ཕྲན་བརྒྱད་ཁྲི་བཞི་སྟོང་ལ་སྦྲན་ནོ། ། རྒྱལ་པོ་མིག་འབྱེད་ད་སྟེ་ཞག་བདུན་ན་བྲམ་ཟེ་ལ་མིག་སྦྱིན་པ་བྱེད་ཀྱིས། ལྟ་འདོད་ན་ཐམས་ཅད་ཤོག་ཤིག་ཅེས་བསྒོའོ། ། དེའི་ཚེ་ན་འཛམ་བུའི་གླིང་གི་མི་དེ་དག་འདི་སྐད་ཅེས་བསྒོ་བ་ཐོས་ནས། ཐམས་ཅད་མྱུར་བར་དེར་ལྷགས་སོ། ། རྒྱལ་ཕྲན་བརྒྱད་ཁྲི་བཞི་སྟོང་སྙེད་ཀྱང་སོ་སོའི་བློན་པོ་དང་ཆས་ཏེ། རྒྱལ་པོའི་ཐད་དུ་ལྷགས་ནས། ལྟོ་བ་ས་ལ་ཕབ་སྟེ། མཆི་མ་བླགས་ནས་འདི་སྐད་ཅེས་གསོལ་ཏོ། ། བདག་ཅག་འཛམ་བུའི་གླིང་གི་མི་ཐམས་ཅད་རྒྱལ་པོ་ལ་བརྟེན་ནས་འཚོ་ན་ཅིའི་སླད་དུ་མི་གཅིག་གི་ཕྱིར་བདག་ཅག་ཀུན་སྤངས་ཞེས་གསོལ་ཏོ། ། བློན་པོ་རྣམས་དང་། བཙུན་མོ་དང་། ཕོ་བྲང་གི་སླས་རྣམས་ཀྱང་མགོ་བོ་ས་ལ་རྡེབས་ཤིང་། ས་ལ་ཁ་སྦུབ་ཏུ་ཉལ་ནས། བྲམ་ཟེ་མི་གཅིག་གི་སླད་དུ་བདག་ཅག་ཀུན་མ་སྤངས་ཤིག་ཅེས་གསོལ་ནས། ཆོ་ངེས་འདེབས་སོ། ། རྒྱལ་བུ་ལྔ་བརྒྱ་རྣམས་ཀྱང་ངུས་ནས། བདག་ཅག་ལ་མགོན་མ་མཆིས་སོ། ། སྐྱབས་མ་མཆིས་སོ། །
