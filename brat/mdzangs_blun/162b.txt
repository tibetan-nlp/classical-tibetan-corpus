ཁོང་དུ་ཆུད་ཅིང་རྟོགས་ནས་སངས་རྒྱས་སུ་སྨོན་ལམ་འདེབས་པ་དང་། འབྲས་བུ་ཐོབ་པ་དང་། མཐོ་རིས་ཀྱི་ལྷར་སྐྱེ་བར་སྨོན་པར་ཡང་གྲངས་མེད་དོ། ། དེའི་ཕྱི་ཉིན་ཤཱཀྱའི་རིགས་རྣམས་ཀྱིས་སངས་རྒྱས་སྤྱན་དྲངས་ནས། བཅོམ་ལྡན་འདས་ཀྱིས་དེའི་ཉིན་པར་འཁོར་དེ་དག་ཐམས་ཅད་སོ་སོ་ནས། འཁོར་ལོས་སྒྱུར་བའི་རྒྱལ་པོར་བསྒྱུར་ཏེ། རིན་པོ་ཆེ་སྣ་བདུན་དང་བུ་སྟོང་དང་ལྡན་པར་སྤྲུལ་ནས། རྒྱལ་ཕྲན་དང་། བློན་པོ་རྣམས་ཀྱིས་བསྟི་སྟང་དུ་བྱས་པར་མཐོ་སྟེ། སོ་སོ་ནས་ངོ་མཚར་རྨད་དུ་བྱུང་བའི་སྤྲོ་བ་སྐྱེས་ནས། སྤྲོ་བ་དེ་ལ་བཅོམ་ལྡན་འདས་ཀྱིས་ཆོས་རྣམ་པར་ཕྱེ་ཞིང་བསྟན་པས། སོ་སོའི་ཡིད་ཚིམ་པར་གྱུར་ཏེ། བླ་ན་མེད་པའི་བྱང་ཆུབ་ཏུ་སེམས་བསྐྱེད་པ་དང་། ལ་ལས་འབྲས་བུ་ཐོབ་པ་དང་། ལྷ་དང་མིར་སྐྱེས་པ་གྲངས་མེད་པར་གྱུར་ཏོ། ། དེའི་ཕྱི་ཉིན་བརྒྱ་བྱིན་གྱིས་སངས་རྒྱས་སྤྱན་དྲངས་ཏེ། སེང་གེའི་ཁྲི་བཤམས་ནས། བཅོམ་ལྡན་འདས་དེ་ལ་བཞུགས་པ་དང་། བརྒྱ་བྱིན་གྱིས་གཡོན་ནས་མཆོད་པ་བྱེད། ཚངས་པའི་རྒྱལ་པོས་ནི་གཡས་ནས་མཆོད་པ་བྱེད་དོ། ། འཁོར་མང་པོ་རྣམས་ཀྱང་ཅང་མི་སྨྲ་བར་འཁོད་པ་ལ། བཅོམ་ལྡན་འདས་ཀྱིས་ཕྱག་གིས་སེང་གེ་འི་ཁྲི་ལ་མནན་པ་དང་། སེང་གེའི་ཁྲི་ལས་གླང་པོ་ཆེའི་སྐད་ལྟ་བུའི་སྒྲ་ཆེན་པོ་ཞིག་བྱུང་བ་དང་། དེ་མ་ཐག་ཏུ་སྲིན་པོ་ཆེན་པོ་ལྔ་ཞིག་བྱུང་ནས། སྟོན་པ་དྲུག་གི་ཁྲི་དྲངས་ནས་བཤིག་གོ། ། ཕྱག་ན་རྡོ་རྗེ་ནི་རྡོ་རྗེའི་རྩེ་མོ་ལ་མེ་འབར་བ་ཐོགས་ཏེ༑ སྟོན་པ་དྲུག་གི་མགོར་གཟས་སོ། ། སྟོན་པ་དྲུག་པོ་འཇིགས་སྐྲག་ནས་བྱེར་ཞིང་བྲོས་ཏེ་སྐྱེངས་ནས། ཁ་ཅིག་ནི་ཆུར་མཆོངས་ནས་ཚེ་འཕོས་སོ། ། སྟོན་པ་དྲུག་གི་འཁོར་ལས་སྟོང་ཕྲག་དགུ་བཅུ་ནི། སངས་རྒྱས་ལ་སྐྱབས་སུ་གསོལ་ནས། རབ་ཏུ་འབྱུང་བར་གསོལ་བ་དང་༑ བཅོམ་ལྡན་འདས་ཀྱིས་ལེགས་པར་འོངས་སོ། ། ཞེས་བཀའ་སྩལ་པ་དང་། སྐྲ་དང་ཁ་སྤུ་རང་བྱི་སྟེ་དགེ་སློང་དུ་གྱུར་ཏོ། ། བཅོམ་ལྡན་འདས་ཀྱིས་དེ་དག་ལ་ཅི་རིགས་པར་ཆོས་བསྟན་པས་ཟག་པ་ཀུན་ཟད་དེ། ཉོན་མོངས་པ་ལས་རྣམ་པར་གྲོལ་ནས་དགྲ་བཅོམ་པར་གྱུར་ཏོ། །
