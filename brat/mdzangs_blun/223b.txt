ཁང་བུ་བརྩེགས་པའི་ཤར་ཕྱོགས་ཀྱི་སྐར་ཁུང་ནས་ཉི་མའི་འོད་ཟེར་བྱུང་བ་ན། ནུབ་ཕྱོགས་ཀྱི་རྩིག་ངོས་ལ་ཤར་ཅིང་ལྷན་པ་བཞིན་དུ་བཅོམ་ལྡན་འདས་ཀྱི་ཐུགས་ཀུན་དགའ་བོ་ལ་དགོངས་ཤིང་དགྱེས་པར་ཉེ་གནས་ཀུན་གྱིས་རྟོགས་ནས། ཚེ་དང་ལྡན་པ་ཤཱ་རིའི་བུ་དང་། མཽད་གལ་གྱི་བུ་ཆེན་པོ་གཉིས་ཀུན་དགའ་བོ་གང་ན་བ་དེར་སོང་སྟེ་དྲུང་དུ་ཕྱིན་པ་དང་། ཀུན་དགའ་བོ་དང་ལྷན་ཅིག་ཏུ་ཡིད་མགུ་ཞིང་དགའ་བའི་གཏམ་རྣམ་པ་དུ་མ་བྱས་ནས། ཕྱོགས་གཅིག་ཏུ་འཁོད་དོ། ། ཕྱོགས་གཅིག་ཏུ་འཁོད་ནས་ཤཱ་རིའི་བུ་དང་། མཽད་གལ་ཡ་ནས་ཚེ་དང་ལྡན་པ་ཀུན་དགའ་བོ་ལ་འདི་སྐད་ཅེས་སྨྲས་སོ། ། ཚེ་དང་ལྡན་པ་ཀུན་དགའ་བོ་བཅོམ་ལྡན་འདས་ལ་བསྙེན་བཀུར་བྱ་བར་དད་དམ། བཅོམ་ལྡན་འདས་ཀྱང་ཁྱོད་ཀྱིས་བསྙེན་བཀུར་བྱ་བར་དགྱེས་ཏེ། དེ་བཞིན་གཤེགས་པ་ལ་བསྙེན་བཀུར་བྱས་ན་ཁྱོད་ལ་ཕན་པའི་དོན་ཆེར་འགྱུར་རོ། ། བཙུན་པ་ཤཱ་རིའི་བུ་བདག་ནི་བཅོམ་ལྡན་འདས་ལ་བསྙེན་བཀུར་བྱ་བར་མི་དད་དོ། ། དེ་ཅིའི་ཕྱིར་ཞེ་ན། སངས་རྒྱས་བཅོམ་ལྡན་འདས་ནི་བསྒྲུབ་པར་དཀའ་བཟོད་པར་དཀའོ། ། འདི་ལྟ་སྟེ་མེ་འབར་བ་ནི་བསྒྲུབ་པར་ཡང་དཀའ། བཟོད་པར་ཡང་དཀའ་བ་ཡིན་ཏེ་དེ་ལྟར་དེ་བཞིན་དུ་སངས་རྒྱས་བཅོམ་ལྡན་འདས་བསྒྲུབ་པར་དཀའ་བས་ན། བདག་བཅོམ་ལྡན་འདས་ལ་བསྙེན་བཀུར་བགྱིས་པར་མི་དད་དོ། ། དེ་ནས་ཚེ་དང་ལྡན་པ་མཽད་གལ་ཡ་ནའི་བུ་ཆེན་པོས་ཀུན་དགའ་བོ་ལ་འདི་སྐད་ཅེས་སྨྲས་སོ། ། ཀུན་དགའ་བོ་ཁྱོད་ལ་བཅོམ་ལྡན་འདས་དགོངས་ཤིང་དགྱེས་པ་ནི་འདི་ལྟ་སྟེ། ཁང་བུ་བརྩེགས་པའི་ཤར་ཕྱོགས་ཀྱི་སྐར་ཁུང་ནས་ཉི་མའི་འོད་ཟེར་བྱུང་ནས་ནུབ་ཕྱོགས་ཀྱི་རྩིག་ངོས་ན་ཤར་ཅིང་ལྷན་པ་བཞིན་དུ་བཅོམ་ལྡན་འདས་ཀྱི་ཐུགས་ཀྱང་ཀུན་དགའ་བོ་ཁྱོད་ཀྱིས་བསྙེན་བཀུར་བྱ་བར་དགྱེས་ཀྱི་དད་པར་གྱིས་ཤིག ། དེ་བཞིན་གཤེགས་པ་ལ་བསྙེན་བཀུར་བྱས་ན། ཁྱོད་ལ་ཕན་པའི་དོན་ཆེར་འགྱུར་རོ་ཞེས་སྨྲས་སོ། ། བཙུན་པ་མཽད་གལ་ཡ་ན་ཆེན་པོ་ཅི་སྟེ་བདག་ལ་བཅོམ་ལྡན་འདས་ཀྱིས་འདོད་པ་འདི་གསུམ་གནང་ན་བཅོམ་ལྡན་འདས་ལ་བསྙེན་བཀུར་བྱ་བར་འདོད་དོ། །
