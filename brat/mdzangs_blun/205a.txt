བསྒོམ་པ་ལ་ནན་ཏན་བྱས་པས་ཁ་ཅིག་གིས་ནི་དགེ་སྦྱོང་གི་འབྲས་བུ་ཐོབ་པར་གྱུར་ཏོ། ། ཁ་ཅིག་ནི་བླ་ན་མེད་པ་ཡང་དག་པར་རྫོགས་པའི་བྱང་ཆུབ་ཐོབ་པར་གྱུར་ཏེ། བཅོམ་ལྡན་འདས་ཀྱིས་གསུངས་པ་ལ་རྗེས་སུ་ཡི་རངས་ནས་མངོན་པར་དག འོ། ། རྒྱལ་པོ་ཟླ་འོད་ཀྱིས་མགོ་སྦྱིན་པའི་ལེའུ་སྟེ་ཉི་ཤུ་གཉིས་པའོ། ། ། ། འདི་སྐད་བདག་གིས་ཐོས་པའི་དུས་གཅིག་ན། བཅོམ་ལྡན་འདས་མཉན་ཡོད་ན། རྒྱལ་བུ་རྒྱལ་བྱེད་ཀྱི་ཚལ་མགོན་མེད་ཟས་སྦྱིན་གྱི་ཀུན་དགའ་ར་བ་ན་བཞུགས་སོ། ། དེའི་ཚེ་རྒྱལ་པོ་གསལ་ཐུབ་ལ་བློན་པོ་རི་དགས་ཞེས་བྱ་བ་ཡོད་དེ། ། རབ་ཏུ་ཕྱུག་པ་ལོངས་སྤྱོད་ཆེ་བ། དེ་ལ་བུ་ཕོ་བདུན་ཡོད་དོ། ། བུ་ཕོ་དྲུག་གི་ནི་ཆུང་མ་ལོན་ཏེ། ནང་གི་བུ་ཐ་ཆུངས་གཅིག་ཆུང་མ་མ་ལོན་ནས་ཕས་འདི་སྙམ་དུ་བསམས་སོ། ། བདག་ནི་རྒས་འཁོགས་ཏེ། བུ་གཅིག་གི་ཆུང་མ་མ་ལོན་པ་འདི་ལ་བཟང་མོ་མཆོག་ཞིག་བཙལ་ལོ་སྙམ་བསམས་ནས། དེའི་ཚེ་ན་ཁྱིམ་བདག་དེ་ལ་བྲམ་ཟེ་མཛའ་ཞིང་བཤེས་པ་ཞིག་ཡོད་དེ། དེ་དང་ཕྲད་ནས་གཏམ་དུ་གླེངས་པ། བདག་གི་བུ་ནང་གི་ཐ་ཆུངས་འདི་ད་དུང་ཆུང་མ་མ་ལོན་ཏེ། སུ་ལ་བསླང་ཡང་མི་ཤེས་ན། ཁྱོད་ནི་ཡུལ་ཀུན་ཏུ་ཕྱིན་ན་བུ་མོ་བཟང་ཞིང་མཛེས་པ། བློ་དང་ལྡན་པ། ཚེ་དང་དབང་ཐང་ཡོད་པ་ལྟ་ཅི་མཐོང་། བདག་གི་བུ་འདིའི་ཡིད་དུ་འོང་བ་ཞིག་ཁྱོད་ཀྱིས་བཙལ་དགོས་སོ་ཞེས་སྨྲས་པ་དང་། བྲམ་ཟེ་དེས་ཀྱང་དེ་བཞིན་དུ་བྱའོ་སྨྲས་ནས། ཀུན་ཏུ་རྒྱུ་ཞིང་ཚོལ་བ་ལས། ཡུལ་ཤི་རི་ཏི་ཏ་ཞེས་བྱ་བ་ན་བུ་མོ་ལྔ་བརྒྱ་ཞིག་འགྲོགས་ཏེ། རྩེ་བ་ལས་མེ་ཏོག་བཟང་པོ་བཏུས་ཏེ། སངས་རྒྱས་ལ་མཆོད་པའི་ཕྲེང་བ་བྱེད་པ་མཐོང་ངོ་། ། བྲམ་ཟེ་དེས་ཀྱང་བུ་མོ་དེ་དག་གར་སོང་བའི་ཕྱི་བཞིན་དུ་བལྟས་ན་བུ་མོ་དེ་དག་ཆུ་ཆུང་ངུ་ཞིག་འབོག་པ་ལས། གཞན་ཐམས་ཅད་ནི་ལྷམ་ཕུད་ཀྱི་བུ་མོ་གཅིག་ལྷམ་མ་ཕུད་པར་ཆུ་བརྒལ་ཏོ། ། དེ་ནས་ཡང་སོང་སོང་བ་ལས་ཆུ་ཆེན་པོ་ཞིག་བརྒལ་ནས། བུ་མོ་གཞན་ནི་གོས་ཕུད་ནས་ཆུར་ཞུགས། བུ་མོ་དེ་འབའ་ཞིག་གོས་ཀྱང་མ་ཕུད་པར་ཆུ་བརྒལ་ཏོ། །
