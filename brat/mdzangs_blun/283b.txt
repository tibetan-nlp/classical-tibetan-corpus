ལྷག་པར་བྱས་པ་མང་དུ་འཆང་ཞིང་། ཆོག་མི་ཤེས་པར་བསྩོགས་པར་གྱུར་པ། བཅོམ་ལྡན་འདས་ཀྱིས་གཟིགས་ནས། དགེ་སློང་རྣམས་རྙེད་པ་ལ་ཆགས་པའི་སྐྱོན་འདི་དག་གསུངས་སོ། ། གང་རྙེད་པ་ལ་ཆགས་པ་ནི་ལུས་ཞིག་ནས་ཚེ་འཕོས་མ་ཐག་ཏུ་ངན་སོང་གསུམ་དུ་ལྟུང་སྟེ། སྡུག་བསྔལ་ཚད་མེད་པ་མྱོང་བར་འགྱུར་རོ། ། དེ་བས་ན་ངས་མངོན་དུ་དྲན་པ་སྔོན་འདས་པའི་དུས་ན། འདོད་པའི་དབང་གིས་ལྟུང་བར་གྱུར་ཏེ། སྡུག་བསྔལ་མང་པོ་མྱོང་ངོ་། ། དེ་ནས་ཀུན་དགའ་བོས་པུས་མོ་ས་ལ་བཙུགས་ཏེ། ཐལ་མོ་སྦྱར་ནས། བཅོམ་ལྡན་འདས་ལ་འདི་སྐད་ཅེས་གསོལ་ཏོ། ། སྔོན་བཅོམ་ལྡན་འདས་ལ་འདོད་པ་ལ་ཆགས་པའི་དབང་གིས་ལྟུང་བར་གྱུར་པ་ཅི་ལྟ་བུ་ལགས། བཅོམ་ལྡན་འདས་ཀྱིས་བཀའ་སྩལ་པ། སྔོན་འདས་པའི་དུས་ཀྱི་བསྐལ་པ་གྲངས་མེད་ཚད་མེད་མཐའ་ཡས་བསམ་གྱིས་མི་ཁྱབ་དཔག་ཏུ་མེད་པའི་ཕ་རོལ་ན། འཛམ་བུའི་གླིང་འདིར་རྒྱལ་པོ་གུས་པ་དེ་ཞེས་བྱ་བ་འཛམ་བུའི་གླིང་གི་རྒྱལ་ཕྲན་བརྒྱད་ཁྲི་བཞི་སྟོང་ལ་དབང་བྱེད་དེ། བཙུན་མོ་དང་ཕོ་བྲང་སླས་ནི་སྟོང་ཕྲག་ཉི་ཤུ་ཡོད་དོ། ། བློན་པོ་ཆེན་པོ་ནི་སྟོང་ཕྲག་བཅུ་ཡོད་དོ། ། དུས་གཞན་ཞིག་ན་རྒྱལ་པོའི་སྤྱི་བོ་ལ་སྒོ་ང་འདྲ་བའི་ཆུ་བུར་ཅིག་བྱུང་སྟེ། གསལ་བ་ཡོངས་སུ་དག་པ་ན་ཚ་མེད་དོ། ། བགས་ཀྱིས་ཇེ་ཆེ་ཇེ་ཆེར་སྐྱེས་ཏེ། ཀུ་བ་ཙམ་དུ་གྱུར་པ་དང་བཀས་པ་ན། ནང་ནས་ཁྱེའུ་བལྟ་ན་སྡུག་པ་མཚན་དང་ལྡན་པ་སྐྲ་མཐོན་མཐིང་དུ་འདུག་པ། ལུས་ཀྱི་མདོག་གསེར་ལྟ་བུར་འདུག་པ་ཞིག་བྱུང་བར་གྱུར་ཏེ། མཚན་མཁན་གྱིས་བལྟས་ནས་བུ་འདི་ནི་ངོ་མཚར་ཆེ་བའི་ཡོན་ཏན་གྱི་མཚན་དང་ལྡན་ཏེ། གདོན་མི་ཟ་བར་འཁོར་ལོས་སྒྱུར་བའི་རྒྱལ་པོར་འགྱུར་ཞིང་གླིང་བཞི་ལ་དབང་བྱེད་པར་འགྱུར་རོ་ཞེས་སྨྲས་ནས། ཁྱེའུའི་མིང་ཡང་སྤྱི་བོ་སྐྱེས་ཞེས་བཏགས་སོ། ། ཆེར་སྐྱེས་ནས་ཡོན་ཏན་དང་ལྡན་པས་རྒྱལ་པོས་ཀྱང་ཡུལ་གཅིག་གི་རྒྱལ་ཕྲན་དུ་བཅུག་གོ། ། དུས་གཞན་ཞིག་ན་རྒྱལ་པོ་ཆེན་པོ་ནད་ཀྱིས་བཏབ་པ་དང་།
