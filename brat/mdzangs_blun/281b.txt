རྒྱལ་བུ་རྒྱལ་བྱེད་ཀྱི་ཚལ་མགོན་མེད་ཟས་སྦྱིན་གྱི་ཀུན་དགའ་ར་བ་ན་བཞུགས་སོ། ། དེའི་ཚེ་བཅོམ་ལྡན་འདས་ཀྱིས་ཀུན་དགའ་བོ་ཞབས་འབྲིང་དུ་ཁྲིད་དེ། བསོད་སྙོམས་ལ་གཤེགས་པ་ན། བཅོམ་ལྡན་འདས་གསོལ་བའི་ན་བཟའ་རྡོལ་བ་ཅུང་ཟད་ཅིག་ཡོད་དེ། དེ་ཡང་སེམས་ཅན་འདུལ་བར་རིགས་པའི་དོན་དུ་དགོངས་སོ། ། བསོད་སྙོམས་བསླངས་ནས་སླར་གཤེགས་པའི་ཚེ། བྲམ་ཟེ་ཞིག་འོངས་ནས། བཅོམ་ལྡན་འདས་ལ་སྤྱི་བོས་ཕྱག་འཚལ་ནས། བཅོམ་ལྡན་འདས་ཀྱི་མཚན་ལྷང་ངེ་ལྷན་ནེར་གསལ་བ་ལ་བརྟགས་ནས། བཅོམ་ལྡན་འདས་ཀྱི་ན་བཟའ་ཅུང་ཟད་ཅིག་རྡོལ་བ་མཐོང་ནས་སྦྱིན་པ་བྱ་བར་བསམས་ཏེ། ཁྱིམ་ན་རས་དཀར་པོ་ཞིག་ཡོད་པ་ལས་དུམ་བུ་གཅིག་བཅད་ནས། བཅོམ་ལྡན་འདས་ལ་ཕུལ་ཏེ། རས་ཀྱི་དུམ་བུ་འདིས་ན་བཟའ་ཞིག་ལྷན་པ་མཛོད་ཅིག་ཅེས་གསོལ་ཏོ། ། བཅོམ་ལྡན་འདས་ཀྱིས་ཀྱང་དེ་ལ་ཐུགས་བརྩེ་བའི་སླད་དུ་བཞེས་པ་དང་། བྲམ་ཟེ་རབ་ཏུ་དགའ་ནས་མགུ་རངས་པར་གྱུར་ཏོ། ། བཅོམ་ལྡན་འདས་ཀྱིས་དེ་ལ་ཐུགས་བརྩེ་བའི་སླད་དུ་ལུང་བསྟན་ཏེ། བྲམ་ཟེ་ཁྱོད་མ་འོངས་པའི་དུས་ན་བསྐལ་པ་གྲངས་མེད་པ་བརྒྱ་འདས་ནས་མངོན་པར་འཚང་རྒྱ་སྟེ། མཚན་ཡང་མངོན་པར་ཤེས་པ་བཟང་པོ་ཞེས་བྱ་སྟེ། སྟོབས་བཅུ་རྫོགས་པར་འགྱུར་རོ་ཞེས་ལུང་བསྟན་པ་དང་། རབ་ཏུ་དགའ་ནས་སོང་ངོ་། ། དེའི་ཚེ་ཡུལ་དེ་ན་ཆོ་རིགས་བཙུན་པའི་ཚོང་དཔོན་དང་། ཁྱིམ་བདག་ལ་སོགས་པས་འདི་སྙམ་དུ་བསམ་མོ། ། ཅིའི་སླད་དུ་བཅོམ་ལྡན་འདས་སྦྱིན་པ་ཅུང་ཟད་ཙམ་གྱི་ལན་རབ་ཏུ་ཆེ་བ་བྱིན་སྙམ་བསམས་ནས། སོ་སོ་ནས་རས་དཀར་པོ་བཤམས་ནས། དེ་བཞིན་གཤེགས་པའི་ན་བཟའ་རྣམ་པ་སྣ་ཚོགས་བྱས་ནས། སངས་རྒྱས་ལ་ཕུལ་ལོ། ། ཀུན་དགའ་བོས་བཅོམ་ལྡན་འདས་ལ་གསོལ་པ། བཅོམ་ལྡན་འདས་ཀྱིས་སྔོན་དགེ་བའི་རྩ་བ་ཅི་ཞིག་མཛད་ན། ཐམས་ཅད་ཀྱིས་ན་བཟའ་མང་པོ་ཕུལ་ཞེས་གསོལ་པ་དང་། བཅོམ་ལྡན་འདས་ཀྱིས་འདི་སྐད་ཅེས་བཀའ་སྩལ་ཏོ། །
