འདི་ལྟ་བུ་སུ་ཡིན་སྙམ་སྟེ་འདུག་པ་ལས་ལངས་ནས་བསུ་སྟེ། བསྟི་སྟང་དང་བཅས་པས་ཕྱག་བྱས་ནས་རིན་པོ་ཆེའི་ཁྲི་ལ་བཞག་སྟེ། ཁ་ཟས་རྣམ་པ་སྣ་ཚོགས་བྱིན་ནོ། ། ཁ་ཟས་ཟོས་ནས་ཁྱོད་འདིར་ཅི་ལ་འོངས་ཅི་ལྟར་ཕྱིན་དྲིས་སོ། ། བྱང་ཆུབ་སེམས་དཔས་སྨྲས་པ། འཛམ་བུའི་གླིང་གི་མི་རྣམས་ནི་དབུལ་ཕོངས་པས་ནོར་དང་། ཟས་དང་། གོས་ཀྱི་ཕྱིར་གཅིག་གིས་གཅིག་རྣམ་པ་སྣ་ཚོགས་ཀྱིས་འཚེ་ཞིང་གནོད་པ་བྱེད་དེ། ཚེ་འཕོས་མ་ཐག་ཏུ་ངན་སོང་གསུམ་དུ་ལྷུང་ངོ་། ། བདག་ནི་འདི་ལ་སྙིང་བརྩེ་སྟེ། དེ་དག་གི་མགོན་བྱ་བའི་ཕྱིར་ལམ་རིང་པོ་ནས་ཉོན་མོངས་པ་དང་བསྡོས་ཏེ། རྒྱལ་པོ་ཆེན་པོ་ཁྱོད་ལ་ཡིད་བཞིན་གྱི་ནོར་བུ་རིན་པོ་ཆེ་སློང་དུ་འོངས་སོ། ། ཡིད་བཞིན་གྱི་ནོར་བུ་རིན་པོ་ཆེས་སེམས་ཅན་ཐམས་ཅད་ལ་ཕན་གདགས་པའི་བསོད་ནམས་ཀྱིས་མངོན་པར་རྫོགས་པར་འཚང་རྒྱ་བར་འཚལ་གྱིས། བདག་གི་ཡིད་དང་མི་འགལ་བར་རིན་པོ་ཆེ་སྩལ་དུ་གསོལ། ། ཀླུའི་རྒྱལ་པོས་སྨྲས་པ། ཡིད་བཞིན་གྱི་ནོར་བུ་རིན་པོ་ཆེ་ནི་ཤིན་ཏུ་རྙེད་པར་དཀའ་བ་ཡིན་ཏེ། ཁྱོད་དོན་འདིའི་ཕྱིར་འོང་ན། འདིར་ཟླ་བ་གཅིག་ཅིག་འདུག་སྟེ་བདག་གི་མཆོད་པ་ཡང་བཞེས་ལ་བདག་ལ་ཆོས་བསྟན་ན། ཡིད་བཞིན་གྱི་ནོར་བུ་རིན་པོ་ཆེ་ཡང་སྦྱིན་ནོ་ཞེས་བྱས་པ་དང་། བྱང་ཆུབ་སེམས་དཔས་ཀྱང་དེ་བཞིན་དུ་བྱའོ་ཞེས་སྨྲས་སོ། ། དེ་ནས་ཀླུའི་རྒྱལ་པོས་ཉིན་གཅིག་བཞིན་དུ་ཁ་ཟས་རོ་བརྒྱ་དང་ལྡན་པ་རྣམ་པ་སྣ་ཚོགས་སྦྱར་ཏེ་ཕུལ་ནས། རོལ་མོས་བྱང་ཆུབ་སེམས་དཔའ་མཆོད་དོ། ། བྱང་ཆུབ་སེམས་དཔས་ཀྱང་དེ་ལ་དྲན་པ་ཉེ་བར་གཞག་པའི་ཆོས་ཀྱི་ཡན་ལག་ཟླ་བ་གཅིག་གི་བར་དུ་བསྟན་ནས། ཕྱིར་འགྲོའོ་ཞེས་བྱས་པ་དང་། ཀླུའི་རྒྱལ་པོ་དགའ་ནས་སྤྱི་བོའི་གཙུག་ཕུད་ནས་ཡིད་བཞིན་གྱི་ནོར་བུ་རིན་པོ་ཆེ་བཀྲོལ་ཏེ་ཕུལ་ལོ། ། སྐྱེས་བུ་ཆེན་པོ་འགྲོ་བ་ལ་ཕན་པའི་དོན་དུ་ཡི་དམ་ལ་ཤིན་ཏུ་བརྟན་པ་ཁྱོད་གདོན་མི་ཟ་བར་འཚང་རྒྱའོ། ། ནམ་མངོན་པར་སངས་རྒྱས་པ་ན་བདག་ཀྱང་ཁྱོད་ཀྱི་ཉེ་གནས་མཆོག་ཏུ་གྱུར་ཅིག་ཅེས་སྨྲས་སོ། །
