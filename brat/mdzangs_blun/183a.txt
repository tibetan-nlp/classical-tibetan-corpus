ཡུལ་ཨན་ཏ་ཞེས་བྱ་བ་ན། དགེ་སློང་བསོད་སྙོམས་པ་དགོན་པ་ལ་གནས་པར་དགའ་བ། སྤྱོད་ལམ་དང་ལྡན་པ་ཞིག་ཡོད་དོ། ། དགེ་སློང་བསོད་སྙོམས་པ་ནི་སངས་རྒྱས་ཀུན་གྱིས་བསྔགས་ཀྱིས། ཕལ་གྱི་ནང་ན་གནས་པ་མ་ཡིན་ནོ། ། དེ་ཅིའི་ཕྱིར་ཞེ་ན། དགེ་སློང་བསོད་སྙོམས་པ་ནི་འདོད་པ་ཉུང་ཞིང་ཆོག་ཤེས་པ་སྟེ། ཅི་ཡང་མི་སོགས་ཅིང་། ཟ་ཅིག་པ། བླ་གབ་མེད་པ། ཆོས་གོས་གསུམ་པ་ལ་སོགས་སྤྱོད་དོ། ། བཙུན་པའོ། ། གཙོ་བོའོ། ། དགེ་འདུན་གྱི་ནང་ན་གནས་པའི་དགེ་སློང་ནི། འདོད་པ་མང་བ། ཆོག་མི་ཤེས་པ། སེར་སྣ་བྱེད་པ། ཅི་ལ་ཡང་ཆགས་པས་གྲགས་པ་ཆེན་པོ་ཐོབ་པར་མི་འགྱུར་རོ། ། དགེ་སློང་བསོད་སྙོམས་པདེ་ཡོན་ཏན་སྤྱོད་པར་ལྡན་ཏེ། དགེ་སློང་གི་འབྲས་བུ་ཐོབ་ནས། མངོན་པར་ཤེས་བ་དྲུག་དང་། རིག་པ་གསུམ་དང་། རྣམ་པར་ཐར་བ་བརྒྱད་ལ་གནས་ཏེ། སྤྱོད་ལམ་ཀུན་མཛེས་པ་ཀུན་ཏུ་གྲགས་པར་གྱུར་ཏོ། ། དེའི་ཚེ་ཡུལ་ཨན་ཏ་ཞེས་བྱ་བ་ན། དགེ་བསྙེན་དཀོན་མཆོག་གསུམ་ལ་ཤིན་ཏུ་དད་པ། སྲོག་མི་གཅོད་པ། མ་བྱིན་པར་མི་ལེན་པ། འདོད་པས་ལོག་པར་མི་གཡེམ་པ། བརྫུན་དུ་མི་སྨྲས་པ། སྨྱོ་བའི་ཁུ་བ་ཆང་མི་འཐུབ་འི་ཁྲིམས་ལྔ་བསྲུང་བ། ཡོན་ཏན་དང་ལྡན་པ་ཞིག་ཡུལ་དེ་ན་འདུག་པ་ལས། དགེ་བསྙེན་དེས་དགེ་སློང་བསོད་སྙོམས་པ་དེ། ཚེ་གཅིག་གི་བར་དུ་འཚོ་བ་སྦྱར་བའི་བསོད་ནམས་ནི་མྱོང་བར་འགྱུར་རོ་སྙམ་ནས། གལ་ཏེ་བདག་གིས་ཁྱིམ་དུ་སྤྱན་དྲངས་ཏེ། ལམ་དུ་གྲང་བ་དང་ཚབས་ཉོན་མོངས་ཏེ། བསྒོམ་པའི་བར་ཆད་དུ་ཡང་འགྱུར་ལ། ཕྱིས་རྣམ་པར་སྨིན་པ་མྱོང་བ་ན་ཡང་། བདག་གིས་བསམས་ཤིང་དོན་དུ་གཉེར་དགོས་པ་ཐོབ་པར་འགྱུར་རོ། གལ་ཏེ་འཚོ་བ་གནས་སུ་བསྐྱལ་ན། ཕྱིས་རྣམ་པར་སྨིན་པ་ཡང་དོན་དུ་གཉེར་མི་དགོས་པར་ཐོབ་པར་འགྱུར་རོ་སྙམ་བསམས་ནས། དགེ་བསྙེན་དེ་དད་པའི་སེམས་ཀྱིས་ཞལ་ཟས་རོ་བརྒྱ་དང་ལྡན་པའི་དྲི་བསུང་དང་། ཁ་དོག་ཏུ་ལྡན་པ་སྣ་ཚོགས་བཤམས་ཏེ།
