ཁ་ཅིག་ནི་རྐང་འགྲོས་སྤེལ་ཏེ། སྤོགས་བྱས་ན་ནོར་མང་དུ་རྙེད་ཟེར་རོ། ། ཁ་ཅིག་ནི་ལམ་རིང་པོར་ཚོང་སྤོགས་བྱས་ན་ནོར་མང་དུ་རྙེད་ཅེས་ཟེར་རོ། ། ཁ་ཅིག་ནི་རྒྱ་མཚོར་ཞུགས་ཏེ། ནོར་བུ་རིན་པོ་ཆེ་བླངས་ན་རིན་པོ་ཆེ་མང་དུ་འགྱུར་རོ་ཞེས་ཟེར་རོ། ། དགེ་དོན་གྱིས་དེ་དག་སྨྲས་པ་ཐོས་ནས་ཞིང་རྨོ་བ་དང་། རྐང་འགྲོས་སྤེལ་བ་དང་། ལམ་རིང་པོར་ཚོང་པ་ལ་འགྲོ་བ་ནི་ངའི་ཆ་མ་ཡིན་ནོ། ། རྒྱ་མཚོར་འཇུག་པ་འབའ་ཞིག་ངའི་ཆ་ཡིན་ནོ་ཞེས་སྨྲས་ནས། རྒྱ་མཚོར་འཇུག་གོ། ། ཞེས་བསམས་ཏེ། ཕ་མ་ལ་འདི་སྐད་ཅེས་སྨྲས་སོ། ། བདག་ནི་རྒྱ་མཚོ་ནས་རིན་པོ་ཆེ་མང་པོ་བླངས་ཏེ། དབུལ་ཕོངས་པ་ལ་སྦྱིན་པ་ཡིད་བཞིན་དུ་བྱའོ། ། བདག་རྒྱ་མཚོར་རིན་པོ་ཆེ་ལེན་དུ་འགྲོ་བར་གནོང་ཤིག་ཅེས་སྨྲས་སོ། ། དེའི་ཕ་མས་དེ་སྐད་ཅེས་སྨྲས་པ་ཐོས་ནས་འཁམས་ཏེ་བུ་ལ་འདི་སྐད་ཅེས་སྨྲས་སོ། ། འཇིག་རྟེན་ན་དབུལ་ཕོངས་པ་དག་བློ་མེད་དེ། ལུས་དང་། སྲོག་དང་བསྡོས་ནས། འགྲོ་བར་བྱེད་དོ། ། ཁྱོད་ཅིས་ཕོངས་ན། དེ་ལྟ་བུའི་སེམས་ཁྱོད་སྦྱིན་པར་འདོད་ན། ཁྱིམ་ན་ཡོད་པའི་ནོར་ཐམས་ཅད་སྦྱིན་པར་བྱོས་ཤིག ། རྒྱ་མཚོར་ནི་འགྲོ་མི་རུང་སྟེ། རྒྱ་མཚོའི་ནང་ན་ཆུ་རླབས་དང་། དབའི་གློང་འཁྱིལ་བ་དང་། ཆུ་སྲིན་དང་། ཀླུ་གདུག་པ་དང་། སྲིན་མོ་དང་། རི་ཆུའི་མདོག་ཅན་ལ་སོགས་ཏེ་ཤིན་ཏུ་ཡ་ང་བའི་བགེགས་མང་དུ་ཡོད་པས་ཐར་དཀའ་ན། ཁྱོད་ཅིས་ཕོངས་ཤིང་སྲོག་དང་སྡོ་བར་བྱེད། འདི་ལ་ནི་ཉན་པ་མེད་ཀྱིས་མ་སེམས་པར་འདུག་ཅིག་ཅེས་བསྒོའོ། ། དགེ་དོན་གྱིས་དེ་སྐད་ཅེས་བསྒོ་བ་ཐོས་ནས། བསམ་པ་བཞིན་དུ་མ་གྲུབ་པས་ཡི་དམ་བཅས་ཏེ། བསམ་པ་འདི་མ་གྲུབ་ཀྱི་བར་དུ། ཕ་མའི་མདུན་དུ་ས་ལ་ཁ་སྦུབ་ཏུ་ཉལ་ཏེ། མི་ལྡང་ངོ་སྙམ་བསམས་ནས། ཡབ་དང་ཡུམ་གྱིས་མཁྱེན་པར་མཛོད་ཅིག ། བདག་ནི་བསམ་པ་བཞིན་དུ་མ་གྲུབ་ཀྱི་བར་དུ། ཁ་སྦུབ་ཏུ་ཉལ་བ་འདི་ལས་མི་ལྡང་ངོ་ཞེས་སྨྲས་སོ། ། དེ་ནས་ཕ་མ་མི་དགའ་སྟེ། གཉེན་བཤེས་དང་བགྲོས་ཏེ།
