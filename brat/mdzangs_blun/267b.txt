གལ་ཏེ་ཁྱིམ་ན་གནས་ན་འཁོར་ལོས་སྒྱུར་བའི་རྒྱལ་པོར་འགྱུར་རོ། ། གལ་ཏེ་རབ་ཏུ་བྱུང་ན་མངོན་པར་འཚང་རྒྱའོ་ཞེས་སྨྲས་ཏེ། མཚན་མཁན་གྱིས་རྒྱལ་པོ་ལ་འདི་སྐད་ཅེས་གསོལ་ཏོ། ། རྒྱལ་བུ་འདི་ཐོག་མ་བཙས་པ་ན་ངོ་མཚར་ཡ་མཚན་གྱི་ལྟས་ཅི་བྱུང་ཞེས་དྲིས་པ་ན། རྒྱལ་བུ་འདི་བཙས་པའི་སྤྱི་བོ་ནས་རིན་པོ་ཆེ་ཤིན་ཏུ་གསལ་བ་ཞིག་ཡོད་དོ། ། ཞེས་སྨྲས་ནས། རྒྱལ་བུའི་མིང་རིན་ཆེན་གཙུག་ཕུད་ཅེས་བཏགས་སོ། ། ཕྱིས་ཆེར་སྐྱེས་ནས་རབ་ཏུ་བྱུང་སྟེ། བྱང་ཆུབ་ལ་ནན་ཏན་བྱས་ནས་མངོན་པར་སངས་རྒྱས་ཏེ། སེམས་ཅན་གྲངས་མེད་པའི་དོན་བྱས་སོ། ། དེའི་ཚེ་རྒྱལ་པོ་ཡབ་ཀྱིས་སངས་རྒྱས་དང་། དགེ་སློང་གི་དགེ་འདུན་དང་བཅས་པ་ཟླ་བ་གསུམ་དུ་སྤྱན་དྲངས་ནས་མཆོད་སྟོན་གསོལ་ཏོ། ། དེའི་ནང་ན་དགེ་སློང་འཕགས་པའི་བཤེས་གཉེན་ཞེས་བྱ་བ་ཞིག་ཟླ་བ་གསུམ་དུ་མེ་མར་སྦྱོར་བའི་ཡོན་བདག་ཏུ་ཁས་བླངས་ནས། ཉིན་གཅིག་བཞིན་དུ་གྲོང་རྒྱུ་ཞིང་གྲོང་ཁྱེར་དུ་སོང་ནས་ཚོང་དཔོན་དང་། ཁྱིམ་བདག་དང་མི་མང་པོ་ལས་མར་དང་འབྲུ་མར་དང་། མེ་མར་དང་སྙིང་པོ་བསླངས་སོ། ། དེའི་ཚེ་ན་རྒྱལ་པོ་དེ་ལ་བུ་མོ་ཐུབ་པ་ཞེས་བྱ་བ་ཞིག་ཡོད་དེ། ཡང་ཐོག་གི་སྟེང་ན་དགེ་སློང་དེ་ཉིན་གཅིག་བཞིན་དུ་གྲོང་རྒྱུ་ཞིང་ཉོན་མོངས་པ་མཐོང་ནས་དེ་ལ་སྙིང་བརྩེ་སྟེ། མི་ཞིག་མངགས་ནས། བཙུན་པ་ཁྱོད་ཅི་ཞིག་བསྒྲུབས་ན། རྟག་ཏུ་འདི་ལྟ་བུ་ཉོན་མོངས་ཞེས་དྲིས་ན་དགེ་སློང་དེས་འདི་སྐད་ཅེས་སྨྲས་སོ། ། བདག་ནི་ཟླ་བ་གསུམ་གྱི་བར་དུ་སངས་རྒྱས་དང་། དགེ་འདུན་གྱི་མར་མེ་སྦྱོར་བའི་ཡོན་བདག་ཏུ་ཁས་བླངས་ཏེ། དེ་བས་ན་གྲོང་རྒྱུ་ཞིང་དགེ་བསྙེན་རྣམས་ལས་མར་དང་འབྲུ་མར་དང་། མེ་མར་གྱི་སྙིང་པོ་དང་མར་མེ་འཛུགས་པའི་ཡོ་བྱད་རྣམས་བླང་ངོ་ཞེས་བུ་མོ་ལ་སྤྲིང་བ་དང་། བུ་མོ་དེ་རབ་ཏུ་དགའ་ནས། དགེ་སློང་འཕགས་པའི་བཤེས་གཉེན་ལ་འདི་སྐད་ཅེས་བསྒོའོ། ། ད་ཕྱིན་ཅད་དེ་ལྟ་བུར་རྟག་ཏུ་འགྲོ་མི་དགོས་ཀྱིས། མེ་མར་གྱི་ཡོ་བྱད་ཁོ་མོས་སྦྱར་གྱི། ལོང་ཤིག་ཅེས་བསྒོ་བ་དང་། དགེ་སློང་དེ་ཡང་དེ་ལྟར་བྱའོ་ཞེས་སྨྲས་ནས། དེ་ཕྱིན་ཅད་རྟག་ཏུ་མེ་མར་དང་འབྲུ་མར་དང་།
