བདག་གིས་བརྫུན་དུ་སྨྲས་སོ་སྙམ་སྟེ་མི་དག འོ། ། ལུས་དང་སྲོག་ལ་ཆགས་པ་ནི་མ་ཡིན་ཏེ། རྒྱལ་པོ་ཁྱོད་བདག་ལ་ཐུགས་བརྩེ་བའི་ཕྱིར་ཞག་བདུན་ཞིག་བཤོལ་ཏེ། བྲམ་ཟེ་དེ་ལ་སྦྱིན་པ་བྱས་ནས་ཕྱིར་ཡང་འགུམ་དུ་མཆིའོ་ཞེས་གསོལ་ཏོ། ། སྲིན་པོས་སྨྲས་པ། ཁྱོད་ད་བཏང་ན་ཕྱིར་འོང་དུ་བཏུབ་བམ། སྨྲས་པ་གལ་ཏེ་མ་འོང་ན་ཡང་ཁྱོད་ལ་བླངས་པའི་མཐུ་ཡོད་དོ་ཞེས་སྨྲས་ནས། ཕྱིར་བཏང་ངོ་། ། སུ་ཏ་སོ་མ་ཕྱིར་ཁྱིམ་དུ་ཕྱིན་ཙམ་ན། བྲམ་ཟེ་དེ་ཡང་ད་དུང་འདུག་སྟེ་རབ་ཏུ་དགའ་ནས། བྲམ་ཟེ་ལ་སྦྱིན་པ་སྦྱིན་ནོ། ། བྲམ་ཟེ་དེས་ཀྱང་སུ་ཏ་སོ་མ་རིང་པོར་མི་ཐོགས་པར་སྲོག་དང་བྲལ་བར་འགྲོ་བས་ཕྱིར་ཡུལ་ལ་ཆགས་ན་མི་དགའ་བར་ཤེས་ནས། དེ་ལ་ཚིགས་སུ་བཅད་དེ་སྨྲས་པ། བསྐལ་པའི་མཐའ་ལ་ཐུག་པ་ན། ། གནམ་དང་ཀླུང་ཡང་རབ་འབར་ཏེ། ། རི་རབ་ལྷུན་པོ་རྒྱ་མཚོ་ཡང་། ། ཐམས་ཅད་ཐལ་བ་རྡུལ་དུ་འགྱུར། ། ལྷ་དང་ཀླུ་དང་ལྷ་མ་ཡིན། ། ཐམས་ཅད་འཇིག་ཅིང་ཉམ་ཉེས་ཏེ། ། གནམ་དང་ས་ཡང་མེད་འགྱུར་ན། ། རྒྱལ་སྲིད་ལྟ་ཞིག་ག་ལ་རྟག ། སྐྱེ་རྒ་ན་དང་འཆི་བ་ཡི། ། འཁོར་བའི་མཐའ་ནི་ཟད་མི་ཤེས། ། རེ་ཞིག་འདོད་པ་མ་གྲུབ་ན། ། མྱ་ངན་གྱིས་ནི་གནོད་པར་འགྱུར། ། འདོད་པ་སྐྱོན་དུ་རབ་ཆེ་སྟེ། ། ཉེས་པ་གཞན་ནས་འོང་བ་མིན། ། ཁམས་གསུམ་སྡུག་བསྔལ་བཅས་པ་ན། ། ཡུལ་སྲིད་ཡིད་བརྟན་རུང་རེ་སྐན། ། ཡོད་པ་ཡེ་ནས་མེད་པ་སྟེ། ། རྒྱུ་དང་རྐྱེན་གྱིས་འགྲུབ་པར་འགྱུར། ། དར་བ་ཡོད་ན་རྒུད་པ་ཡོད། ། བདེན་པ་ངེས་པར་བརྫུན་འགྱུར་རོ། ། སོ་སོའི་སྐྱེ་བོ་མཐའ་ཐོག་དག ། སྒྱུ་མ་ལ་ནི་ཀུན་གནས་ཏེ། ། ཁམས་གསུམ་ཅི་ལྟར་སྟོང་བཞིན་དུ། ། ཡུལ་གྱི་སྲིད་ཀྱང་དེ་བཞིན་ནོ། ། རྣམ་པར་ཤེས་པ་གཟུགས་མེད་དོ། ། གདུག་པའི་སྦྲུལ་བཞི་ལ་གནས་ཏེ། ། མ་རིག་པ་ལས་སྐྱེས་པ་ལ། ། བདེ་བའི་གནས་སུ་ཀུན་འཛིན་ཏོ། ། གཟུགས་ལ་རྟག་པའི་བདག་མེད་དེ། ། སེམས་ལ་རྟག་པའི་ཁྱིམ་ཡོད་མིན། ། ལུས་དང་སེམས་ཀྱང་བྲལ་གྱུར་ན། །
