རྒྱལ་བུ་རྒྱལ་བྱེད་ཀྱི་ཚལ་མགོན་མེད་ཟས་སྦྱིན་གྱི་ཀུན་དགའ་ར་བ་ན། དགེ་སློང་གི་དགེ་འདུན་ལ་ཆོས་སྟོན་ཅིང་བཞུགས་སོ། དེའི་ཚེར་ཡུལ་དེ་ན་ཁྱིམ་བདག་གི་ཆུང་མ་ཞིག་ལ་བུ་མོ་ཞིག་བཙས་ནས། བུ་མོ་གཞན་པས་འཕགས་པ་བཞིན་བཟང་བ་བལྟ་ན་སྡུག་པ་མཛེས་པ་ཞིག་སྟེ། ཐོག་མ་བཙས་ནས་རས་དཀར་པོ་འཇམ་པོས་དཀྲིས་ཏེ་བྱུང་ངོ་། ། དེ་ལ་ཕ་མ་ཡ་མཚན་དུ་གྱུར་ཏེ། མཚན་མཁན་ལ་བསྟན་ནས། མཚན་མཁན་གྱིས་བལྟས་ནས་བསོད་ནམས་ཆེན་པོ་དང་ལྡན་པར་རྟོགས་ནས་མིང་ཡང་དཀར་མོར་བཏགས་སོ། ། བུ་མོ་དཀར་མོ་ཅི་ཙམ་དུ་སྐྱེས་པ་བཞིན་དུ་རས་ཀྱང་ལུས་ཀྱི་ཚད་དུ་སྐྱེས་སོ། ། བུ་མོ་དེ་ཕྱིས་ཆེར་སྐྱེས་ནས། ཡུལ་མི་ལས་མང་དུ་ཆུང་མར་སློང་བ་དང་། ཕ་མས་ཀྱང་བུ་མོ་བག་མར་བཏང་བའི་རྒྱན་རྣམས་བརྡུང་སྙམ་ནས་མགར་བ་གཉེར་ཏེ་གསེར་དངུལ་བཅལ་ལོ། ། བུ་མོས་གསེར་དངུལ་བཅལ་བ་མཐོང་ནས་འདི་ཅི་ཞིག་བྱ་ཞེས་ཕ་མ་ལ་དྲིས་པ་དང་། འདི་ནི་ཁྱོད་བག་མར་བཏང་བའི་རྒྱན་རྣམས་བརྡུང་ངོ་། ། ཞེས་སྨྲས་སོ། ། བུ་མོས་སྨྲས་པ། བདག་ནི་རབ་ཏུ་བྱུང་བར་དད་ཀྱི། བག་མར་མཆིས་པར་མི་དག འོ། ། ཕ་མས་ཀྱང་བུ་མོའི་ཡིད་དང་སྦྱར་ནས་དེ་མ་ཐག་ཏུ་རས་ཕྱུང་སྟེ། ཆོས་གོས་དྲུབ་པར་བཤམས་པ་ལས། བུ་མོས་རས་འདིས་ཅི་ཞིག་བྱ་ཞེས་དྲིས་པ་དང་། ཁྱོད་ཀྱི་ཆོས་གོས་དྲུབ་པོ་ཞེས་སྨྲས་སོ། ། བུ་མོས་སྨྲས་པ་བདག་ལ་ཆོས་གོས་གྱོན་པ་ཡོད་ཀྱི་དྲུབ་མི་དགོས་སོ༑ ། བཅོམ་ལྡན་འདས་གང་ན་བཞུགས་པར་ཁྲིད་དེ་འདོང་ངོ་ཞེས་སྨྲས་པ་དང་། ཕ་མས་ཀྱང་བཅོམ་ལྡན་འདས་གང་ན་བཞུགས་པར་ཁྲིད་དེ་དདོང་ངོ་ཞེས་སྨྲས་ནས། བཅོམ་ལྡན་འདས་ཀྱི་སྤྱན་སྔར་ཁྲིད་དེ་སངས་རྒྱས་གང་ན་བ་དེར་ཕྱིན་པ་དང་། བཅོམ་ལྡན་འདས་ལ་མགོ་བོས་ཕྱག་འཚལ་ཏེ། རབ་ཏུ་བྱུང་བར་གསོལ་ལོ། ། བཅོམ་ལྡན་འདས་ཀྱིས་ལེགས་པར་འོངས་སོ། ། ཞེས་བཀའ་སྩལ་པས་སྐྲ་དང་། ཁ་སྤུ་རང་བྱི་སྟེ་ལུས་ལ་གྱོན་པའི་རས་དཀར་པོ་ཡང་ཆོས་གོས་སྨད་ལྔར་གྱུར་ནས། དགེ་སློང་མ་སྐྱེ་དགུའི་བདག་མོ་ལ་གཏང་ངོ་། ། དགེ་སློང་མ་དེས་བརྩོན་འགྲུས་བརྩམས་པས་རིང་པོར་མ་ལོན་པར་དགྲ་བཅོམ་པར་གྱུར་ཏེ།
