མི་དེས་སྨྲས་པ། ཁྱོད་ཀྱིས་ཁོ་བོ་བོས་ཏེ་ཅི་ཞིག་བྱ། ཁྱོད་ནི་ཤིན་ཏུ་གདུག་པ་ཅན་ཡིན་པས་ཁོ་བོ་བསད་དུ་འོང་ངོ་། ། སྦྲུལ་གྱིས་སྨྲས་པ། ཁོ་བོས་གནོད་པ་བྱ་ན་ཁྱོད་དྲུང་དུ་མ་འོངས་ཀྱང་གནོད་པ་བྱ་ནུས་སོ། ། ཞེས་སྨྲས་པ་དང་། མི་དེ་སྐྲག་ནས་སྦྲུལ་གྱི་དྲུང་དུ་བསྙེན་ཏོ། ། སྦྲུལ་གྱིས་སྨྲས་པ། ཁོ་བོ་ལ་འདི་ན་གསེར་གྱི་བུམ་པ་ཞིག་ཡོད་ཀྱིས། ཁྱོད་ལ་བསོད་ནམས་ཤིག་བྱ་བར་བཅོལ་བར་ནུས་སམ། ཅི་སྟེ་མ་བྱས་ན་ཁྱོད་ལ་གནོད་པར་བྱའོ། ། མི་དེས་ཁོ་བོས་ནུས་སོ་ཞེས་སྨྲས་པ་དང་། སྦྲུལ་གྱིས་མི་དེ་གསེར་གྱི་བུམ་པའི་དྲུང་དུ་ཁྲིད་དེ། གསེར་གྱི་བུམ་པ་གཅིག་བསྐུར་ཏེ། འདི་སྐད་ཅེས་སྨྲས་སོ། ། ཁྱོད་གསེར་འདི་ཁྱེར་ལ་དགེ་འདུན་རྣམས་ལ་མཆོད་སྟོན་གསོལ་ཅིག ། མཆོད་སྟོན་གསོལ་བའི་ཉིན་པར་ཁོ་བོ་ཡང་འོང་གིས་བསུས་ཤིག་ཅེས་བསྒོའོ། ། དེ་ནས་མི་དེས་གསེར་ཁུར་ཏེ་དགེ་འདུན་གྱི་ནང་དུ་སོང་ནས། དགེ་སློང་གི་གཙང་སྦྱོར་ལ་གསེར་ཕུལ་ཏེ། འདི་ནི་སྦྲུལ་གདུག་པ་ཞིག་གི་གསེར་ལགས་ཀྱིས་གསེར་འདིས་དགེ་འདུན་ལ་མཆོད་སྟོན་གསོལ་ཅིག་ཅེས་མཆིའོ། ། གསེར་འདིས་མཆོད་སྟོན་སྦྱར་ཏེ། དེ་ནས་མཆོད་སྟོན་ནམ་གསོལ་བ་ཞག་དུས་བརྩིས་ནས། དེའི་ཉིན་པར་མི་དེ་གཟེབ་ཞིག་ཐོགས་ཏེ། ། སྦྲུལ་གདུག་པ་དེ་གང་ན་འདུག་པ་དེར་སོང་ངོ་། ། སྦྲུལ་གདུག་པ་དེས་ཀྱང་མི་དེ་འོང་བ་མཐོང་ནས་རབ་ཏུ་དགའ་བར་གྱུར་ཏོ། ། མི་དེས་སྦྲུལ་དེ་གཟེབ་ཀྱི་ནང་དུ་བཅུག་ནས་ཁྱེར་ཏེ་སོང་བ་ལས་ལམ་དུ་མི་ཅིག་དང་ཕྲད་དེ། མི་དེས་ཁྱོད་གང་ནས་འོངས་ཞེས་ལན་གསུམ་གྱི་བར་དུ་དྲིས་ཀྱང་སྦྲུལ་ཐོགས་པ་དེས་ཅང་མ་སྨྲས་པས། སྦྲུལ་ཁྲོས་ནས་གདུག་པ་ཕྱུང་བ་ལས་ཡང་འདི་སྙམ་བསམས་སོ། ། མི་འདི་ཅི་ཞིག་ཉེས་ན། ཕ་རོལ་པོས་ལན་གསུམ་གྱི་བར་དུ་དྲིས་ན་གཏམ་གྱི་ལན་ཡང་མི་སྟེར་སྙམ་སྟེ། ཕྱིར་ཡང་ཁྲོས་ནས་གདུག་པ་བྱུང་བ་ལས་ཕྱིར་དགག་སྟེ། ཡང་འདི་སྙམ་དུ་བསམས་སོ། ། མི་འདི་བདག་གི་ཕྱིར་བསོད་ནམས་བྱེད་ན། བྱས་པ་གཟོ་བའི་དྲིན་ཡང་དྲན་པ་བྱས་པར་གདུག་པ་མི་འབྱུང་ངོ་སྙམ་བསམས་ཏེ། མི་འདིས་བདག་ལ་ཕན་པའི་དོན་བྱེད་ཀྱི་ཉེས་པ་ཞིག་བྱས་ན་ཡང་བཟོད་པར་བྱའོ། ། ཞེས་ས་དབེན་པར་ཕྱིན་པ་དང་།
