ད་ཁྱོད་ཀྱིས་འདིར་བོར་ཏེ་མི་འགྲོ་ན་བདག་ཅག་ཀྱང་ཕྱིར་ཁྱིམ་དུ་མི་ཕྱིན་ཏེ། ཐམས་ཅད་འཁྱམ་པར་འགྱུར་རོ། ། སྦྱིན་པ་ཆེན་པོས་སྨྲས་པ། ཚོང་པ་ཁྱོད་མ་ཉེས་པར་ཁོ་བོས་ཡུལ་དུ་ཕྱིན་པའི་སྨོན་ལམ་བཏབ་བོ་ཞེས་པོག་ཕོར་ལག་ཏུ་བླངས་ཏེ། ཕྱོགས་བཞིར་མཆོད་ནས་འདི་སྐད་ཅེས་སྨྲས་སོ། ། བདག་ནི་རིན་པོ་ཆེ་བླངས་ཏེ། སེམས་ཅན་ཕོངས་པ་ལ་ཕན་གདགས་པར་མ་ཟད་ཀྱིས། དེ་ལྟར་སྦྱིན་པ་བྱེད་པའི་བསོད་ནམས་དེས་སངས་རྒྱས་སུ་འགྲུབ་པའི་ཕྱིར་རོ། ། བདག་གིས་སྨྲས་པ་འདི་བདེན་ཏེ་སྨོན་ལམ་བཞིན་དུ་འགྲུབ་ན་ཚོང་པ་འདི་དག་རིན་པོ་ཆེ་དང་བཅས་སུ་ཉེ་ཞོ་མེད་པར་འཛམ་བུའི་གླིང་དུ་བདེ་བར་ཕྱིན་པར་ཤོག་ཤིག ། ཅེས་སྨོན་ལམ་བཏབ་ནས་ཐག་པ་བཅད་དེ་གཡོར་ཤིང་བསྒྲེང་ནས་ཕྱིར་འཛམ་བུའི་གླིང་དུ་བདེ་བར་སོང་ངོ་། ། དེ་དག་སོང་བའི་འོག་ཏུ་སྦྱིན་པ་ཆེན་པོ་ཆུར་ཞུགས་ཏེ་སོང་བ་ལས་ཞག་བདུན་གྱི་བར་དུ་པུས་མོ་ཙམ་ལས་མ་ནུབ་བོ། ། དེ་ནས་ཡང་ཞག་བདུན་གྱི་བར་དུ་བརླ་དྲུང་ཙམ་ལས་མ་ནུབ་བོ། ། དེ་ནས་ཡང་ཞག་བདུན་སོང་བ་ལས་རྐེད་པ་ཙམ་ལས་མ་ནུབ་བོ་ཞག་བདུན་གྱི་བར་དུ་ཕྲག་པ་ཙམ་དུ་ཕྱིན་ཏོ་དེ་ནས་ཞག་བདུན་གྱི་བར་དུ་ནི་རྐྱལ་ལོ། ། དེ་ནས་རི་ཞིག་གི་དྲུང་དུ་ཕྱིན་པ་དང་། ཤིང་ལ་འཇུས་ཏེ་རི་ལ་འཛེགས་ནས། ཞག་བདུན་གྱིས་རིའི་རྩེ་མོར་ཕྱིན་ཏོ། ། རིའི་རྩེ་མོར་ཞག་བདུན་སོང་སོང་ནས་ཡང་ཞག་བདུན་གྱིས་ཤོད་དུ་ཆུ་འགྲམ་དུ་ཕྱིན་ཏོ། ། ཆུ་འགྲམ་དུ་ཕྱིན་པ་དང་། ཆུའི་ནང་ནས་པད་མ་གསེར་གྱི་མདོག་ཅན་ཡོད་པ་ལ་སྦྲུལ་གདུག་པས་པད་མའི་རྩ་བ་ལ་དཀྲིས་ཤིང་འདུག་པར་མཐོང་ངོ་། ། དེ་ནས་བྱང་ཆུབ་སེམས་དཔའ་སྐྱིལ་མོ་ཀྲུང་བཅས་ཏེ་འདུག་ནས་སེམས་རྩེ་གཅིག་ཏུ་བྱམས་པའི་ཏིང་ངེ་འཛིན་ལ་ཞུགས་ཏེ། སྦྲུལ་གདུག་པ་འདི་དག་ཀྱང་སྔོན་ཞེ་སྡང་དང་ཕྲག་དོག་བྱས་པའི་རྐྱེན་གྱིས་འདི་ལྟ་བུའི་ལུས་མི་སྡུག་པ་མྱོང་ངོ་སྙམ་ལྷག་པར་དེ་ལ་སྙིང་བརྩེ་བའི་སེམས་བསྐྱེད་པས་སྦྲུལ་གདུག་པ་དེའི་གདུག་པ་ཀུན་ཞི་ནས། བྱང་ཆུབ་སེམས་དཔའ་པད་མའི་སྟེང་ནས་གོམ་པ་བོར་ཏེ་སོང་སོང་བ་ལས་ཞག་བདུན་གྱིས་སྦྲུལ་གདུག་པ་དེ་དག་བརྟོལ་ཏོ། ། དེ་ནས་ཡང་སོང་སོང་བ་ལས་སྲིན་པོ་དང་ཕྲད་ནས་སྲིན་པོ་དེ་དག་མིའི་དྲི་ཚོར་བས་ཀུན་ཏུ་ཚོལ་ལོ། །
