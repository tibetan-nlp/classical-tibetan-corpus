བསླབ་པ་མང་པོ་དང་། རྩོད་པ་ཞི་བྱེད་དང་བཅས་པ་དང་། རྗེས་བཅས་དང་། སྤང་གནང་དང་། རབ་བྱུང་དང་། གསོ་སྦྱོང་དང་། དགག་དབྱེ་དང་། ཕྲ་མོ་དང་། གླེང་གཞི་དང་། མོས་པ་ནི་འདིའོ་ཞེས་འདུལ་བ་བསྡུས་ནས། ཉེ་བར་འཁོར་ཁྲི་ལས་བབས་སོ། ། དེ་ནས་འོད་སྲུངས་ཀྱིས་མ་འོངས་པའི་གང་ཟག་གི་དོན་དུ། མ་མོ་བསྡུ་བའི་ཕྱིར་གསོལ་བ་གཉིས་ཀྱི་ལས་བྱས་ཏེ། མ་མོ་ཞེས་བྱ་ནི་བདག་ཁོ་ནས་སྨྲ་བར་བྱ་སྟེ། མ་མོ་ལྟ་བུ་ནི་ཤེས་བྱའི་མཚན་ཉིད་རབ་ཏུ་གསལ་བར་བྱེད་པ་སྟེ། འདི་ལྟ་སྟེ་དྲན་པ་ཉེ་བར་གཞག་པ་བཞི་དང་། ཞེས་སོགས་སྔར་བཞིན་ནོ། ། དེ་ནས་ལྷ་རྣམས་ཀྱིས་དགྲ་བཅོམ་པ་ལྔ་བརྒྱས་སྡེ་སྣོད་གསུམ་བསྡུས་པས་ལྷ་འཕེལ་ལྷ་མིན་འགྲིབས་སོ། ། ཞེས་སྒྲོགས་ཤིང་ལྔ་བརྒྱས་ཡང་དག་པར་བསྡུས་པ་ཞེས་པའི་མིང་དུ་གྱུར་ཏོ། ། དེ་ནས་འོད་སྲུངས་ཆེན་པོས། ཀྱེ་མ་སྟོབས་བཅུ་མངའ་བའི་གསུང་ནི་ཚད་མེད་པ། འགྲོ་ལ་ཕན་ཕྱིར་ཆོས་ཀྱི་བསྟན་བཅོས་འདི་བསྡུས་པས། གང་ན་འཇིག་རྟེན་བློ་ངན་མུན་བྱེད་ཆུད་ཟོས་པ། བློ་གྲོས་རབ་རིབ་བས་ལ་བས་སྒྲོན་མར་ཡང་དག་ལྡན། ། དེ་ནས་འོད་སྲུངས་ཀྱིས་སྟོན་པའི་གསུང་ཅི་ནུས་སུ་བསྒྲུབས་ཤིང་ཕན་གདགས་པ་ཅུང་ཟད་བྱས་ཏེ། མ་ལུས་པར་སུས་ནུས་ང་མྱ་ངན་ལས་འདའ་བར་བྱའོ་སྙམ་སྟེ། བདག་གིས་ཐུབ་པའི་མདོ་སྡེ་ནི། ། ཡང་དག་བསྡུས་ནས་ཆོས་ལམ་བཞག། གཅིག་ནས་གཅིག་ཏུ་ཡུན་རིང་བར། ། སྟོན་པའི་གསུང་རབ་འདི་གནས་འགྱུར། ། ངོ་ཚ་མེད་རྣམས་ཆད་པས་བཅད། ། ངོ་ཚ་བྱེད་ནི་ཡོངས་སུ་བཟུང་། ། བདག་གིས་ཕན་དང་དགེ་བྱས་པས། ། བདག་ནི་མྱ་ངན་འདས་དུས་བབ། ། ཀུན་དགའ་བོ་ཤེས་པར་གྱིས་ཤིག། སྟོན་པས་ང་ལ་བསྟན་པ་གཏད་ནས་འདས་སོ། །
