ལྷ་རྣམས་ཀྱིས་སྤྱད་དེ། ནུབ་ནས་འཛམ་བུའི་གླིང་དུ་མ་ཕྱིན་ནོ། ། དེ་ནས་ལྷའི་དབང་པོ་བརྒྱ་བྱིན་གྱིས། སྒྲའི་བསྟན་བཅོས་ཤིག་མཛད་དེ། ཨིནྡྲ་བྱཱ་ཀ་ར་ཎ་ཞེས་གྲགས་པ་དེ་དྲང་སྲོང་ཕུར་བུས་ལེགས་པར་ཤེས་ནས་ལྷའི་བུ་རྣམས་ལ་བསྟན་པས། ལྷའི་བླ་མ་ཞེས་སམ་སྒྲ་མཁན་ཞེས་མིང་དུ་གྲགས་སོ། ། དེ་ང་ལས་སྒྲ་མཁས་པ་མེད་སྙམ་དུ་ང་རྒྱལ་བྱེད་པ་ན། ལྷའི་དབང་པོས་རྒྱ་མཚོ་ལས་ཆུ་བུམ་པ་གང་བླངས་ཏེ། དེ་ལས་ཀུ་ཤའི་རྩེས་ཆུ་ཐིགས་པ་གཅིག་བླངས་ཏེ། ཕུར་བུ་ལ་སྒྲ་སྤྱིར་རྒྱ་མཚོ་ཙམ། ངས་ཤེས་པ་བུམ་ཆུ་ཙམ། ཁྱོད་ཀྱིས་ཤེས་པ་འདི་ཙམ་མོ་ཞེས་བརྗོད་པས། ཕུར་བུ་ཡི་མུག་སྟེ། སྒྲ་ཡེ་མི་སློབ་པར་སེམས་པ་ལ། དབང་པོས་འོན་ཀྱང་ཇི་ལྟར་ཤེས་པ་སློབས་ཤིག་ཅེས་བསྒོ་པས་བཀའ་མ་བཅག་པར་དུས་བཟང་བཞི་ལ་མི་སློབ་གཞན་ལ་སློབ་པར་ཁས་བླངས་པས། བྲམ་ཟེ་རྣམས་དིང་སང་ཡང་། ཚེས་བརྒྱད་སློབ་དཔོན་ངེས་པར་འཇོམས། ། བཅུ་བཞི་ལ་ནི་སློབ་མ་འཇོམས། ། གནམ་སྟོང་ལ་ནི་རིག་པ་འཇོམས། ། ཚེས་གཅིག་ལ་ནི་ཐམས་ཅད་འཇོམས། ། ཟེར་ནས་དུས་བཟངས་བཞི་སྲུང་བ་ཡང་དེ་ཡིན་ནོ། ། བསྟན་བཅོས་དེ་འཛམ་བུའི་གླིང་དུའང་ཕྱིན་ཏེ་དར་བར་གྱུར་ཏོ། ། དེ་ནས་དེ་ནུབ་པ་དང་བྲམ་ཟེ་པཱ་ཎི་ནི་བྱུང་སྟེ། སྒྲ་བསླབ་པར་འདོད་ནས། ལག་རིས་མཁན་ལ་བསྟན་པས་སྒྲ་མི་ཤེས་པར་འདུག་ཟེར་བ་དང་། མཚོན་རྣོན་པོས་ལག་རིས་བྲིས་ཏེ། སློབ་དཔོན་འཚོལ་དུ་སོང་བས་མ་རྙེད་དོ། ། དེ་ནས་མ་ཧཱ་དེ་བ་བསྒྲུབས་པས་ཞལ་བསྟན་ནས། ཅི་འདོད་ཟེར་བ་ལ་སྒྲ་ཤེས་པར་འཚལ་ལོ། ། ཞེས་སྨྲས་པས། ལྷས་བྱིན་གྱིས་བརླབས་ཏེ་དུས་གཅིག་ཏུ།
