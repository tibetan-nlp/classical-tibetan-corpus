རྫིང་དང་ཁྲོན་པའང་ཐམས་ཅད་དུ། ། སྟོན་པའི་གཟུགས་བརྙན་ཐམས་ཅད་བརྟགས། ། གཟིངས་དང་ཟམ་པ་ཉིད་རྣམས་ནི། ། བྱེད་པར་འགྱུར་བར་ཐེ་ཚོམ་མེད། ། མཚོན་གྱིས་ལུས་ཞིག་མཐོ་རིས་འགྲོ། དེ་ནས་ཡི་གེ་ས་དང་ཀ། དང་པོ་དག་ཏུ་བརྗོད་པ་དང་། ། དེ་བཞིན་ཡི་གེ་ར་དང་གྷ། ། དང་པོ་དག་ཏུ་རབ་བརྗོད་དང་། ། དང་པོར་ཡི་གེ་བ་དང་ནི། ། ལེགས་བཞིན་ལེགས་ཏེ་བདེ་ཞེས་བསྟན། ། སྦྱིན་བྱེད་སྦྱིན་པར་བྱེད་པ་ཉིད། ། གཞན་གྱི་གྲུབ་མཐའ་སུན་བྱེད་པ། ། ཚོང་པ་སྔོན་དང་སྨན་པ་སྔོན། ། གཉིས་ཀ་སྦྱིན་བྱེད་སྣ་ཚོགས་བྱེད། ། ཡི་གེ་བ་ཞེས་སྡོམ་བརྩོན་བསྒྲགས། ། དེ་ལས་གཞན་པ་དང་ཡིག་ར། ། དང་པོ་ཡི་གེ་བ་གྲགས་དང་། ། སྟོན་པའི་གཟུགས་བརྙན་དོན་ཡང་བྱེད། ། བློ་ལྡན་དང་པོར་ཡི་གེ་མ། ། དེ་བཞིན་སྡོམ་བརྩོན་དད་པ་འབྱུང་། ། རྣམ་པ་དུ་མར་སྡོམ་བརྩོན་གསུངས། ། དེ་ཚེ་མཐའ་ཡས་འབྱུང་བར་འགྱུར། ། སྡོམ་བརྩོན་ཐམས་ཅད་བསྒྲགས་པ་ཡང་། ། སྟོན་པའི་བསྟན་པ་གསལ་བྱེད་པའོ། ། བསྟན་པ་དེ་ཚེ་ས་སྟེང་དུ། ། རྣམ་པར་ཉམས་ཤིང་སྣང་མེད་ལ། ། སྟོན་པའི་གཟུགས་བརྙན་ཡིད་དུ་འོང་། ། བྱེད་པར་འགྱུར་བ་ཐེ་ཚོམ་མེད། ། ཐམས་ཅད་བྱང་ཆུབ་ལུང་བསྟན་ཏེ། ། ང་ཡི་བྱང་ཆུབ་མཆོག་ཐོབ་འགྱུར། ། དེ་བཞིན་འཇིག་རྟེན་སྦྱིན་གནས་འགྱུར། ། སྲིད་པ་གསུམ་གྱི་མཐར་ཡང་བྱེད། ། སྔགས་དང་རྒྱུད་ལ་མངོན་བརྩོན་པས། ། གྲགས་པ་འཛིན་པར་ཤེས་པར་བྱ། ། བྲམ་ཟེ་ཆོས་ཀྱི་ངང་ཚུལ་ཅན། ། ད་ནི་རབ་ཏུ་བརྗོད་པར་བྱ། ། སྔགས་དང་རྒྱུད་ལ་མངོན་བརྩོན་པས། ། རྒྱལ་པོའི་ལུགས་ལ་ཉེ་བར་གནས། ། དེ་ཚེ་མི་བཟད་དུས་དག་ཏུ། ། འཇིག་རྟེན་ཀུན་ལ་འབྱུང་བར་འགྱུར། ། ཡི་གེ་བ་ཞེས་བྲམ་ཟེ་མཆོག། ཕྱུག་ཅིང་རིག་བྱེད་ཕ་རོལ་ཕྱིན། ། དེ་འདིར་ས་རྣམས་མ་ལུས་པར། །
