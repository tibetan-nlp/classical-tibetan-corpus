ཐོ་རངས་ཕྱི་རོལ་གྱི་ལྷ་ཁང་དུ་བལྟར་ཕྱིན་པ་དང་། དགེ་འདུན་ལ་ལ་ནི་མགོ་བཅད། ལ་ལ་ནི་རྐང་ལག་བཅད། ཁ་ཅིག་ནི་མིག་ཕྱུང་སྟེ། སྣ་ཚོགས་སུ་ཤི་བ་མཐོང་ནས། མྱ་ངན་ལངས་ཏེ། དགྲ་བཅོམ་པ་དང་སྡེ་སྣོད་གསུམ་པ་སློབ་མ་ཅན་གྱི་པུར་བཙལ་ཏེ། མཆན་ཁུང་གཡས་གཡོན་དུ་བཅུག་ནས། དགྲ་བཅོམ་པ་ནི་ངའི་མ་ཡིན། སྡེ་སྣོད་གསུམ་པ་ནི་ཆོས་ཀྱི་མཛོད་ཡིན་ཏེ། འདི་གཉིས་འདས་པས། ད་ཕྱིན་ཆད་ངའི་སྲོག་ཀྱང་གཅེས་པ་མེད་དོ། ། རྒྱལ་སྲིད་ཀྱང་སུ་འདོད་པ་ལ་སྦྱིན་ནོ། ། ཞེས་སྨྲས་ནས་མིག་ཀྱང་བཙུམས་ཏེ་མི་བལྟ་བར་གྱུར་ཏོ། ། དེ་ནས་བློན་པོ་རྣམས་ཀྱིས་རྒྱལ་པོའི་མྱ་ངན་བསང་པའི་ཕྱིར། མི་ལྔ་བརྒྱར་རབ་ཏུ་བྱུང་བར་བཅོས་ཏེ། སྐྲ་དང་ཁ་སྤུ་བྲེག་ཏུ། སྤུ་གྲིས་མ་ཆོད་ནས། མེས་བསྲེགས་ཏེ། ཕྱུགས་ནག་པོ་དང་། དམར་པོའི་ཀོ་བ་བསྐོན་ཏེ། རྒྱལ་པོའི་དྲུང་དུ་འོངས་ནས་རབ་ཏུ་བྱུང་བ་ལྔ་བརྒྱ་འོངས་སོ། ། ཞེས་གསོལ་བ་དང་། རྒྱལ་པོ་དགའ་ནས་མིག་ཕྱེ་སྟེ་བལྟས་པས། ཐམས་ཅད་ཕྱུགས་ཀྱི་ཀོ་བ་གྱོན་ཞིང་། སྐྲ་དང་ཁ་སྤུ་མེས་བསྲེགས་པ་མཐོང་ནས། དཀོན་མཆོག་གསུམ་ལ་མཆོད་པའི་ཡོ་བྱད་ཁྱེར་ལ་ཤུག་ཅེས་བསྒོ་ནས། དེ་རྣམས་ལ་མཆོད་དོ། ། དེ་ནས་རྒྱལ་པོས་དེ་དག་ལ་ཆོས་དྲིས་པས། ཆོས་ཚིག་གཅིག་ཀྱང་མི་ཤེས་ནས། རྒྱལ་པོ་ཡང་མྱ་ངན་ལངས་ཏེ་ངུའོ། ། དེ་ནས་དགེ་སློང་རྣམས་ཀྱི་རོ་ཐམས་ཅད་བསྡུས་ཏེ། བསྲེགས་ནས་མཆོད་པ་བྱེད་དོ། ། དེའི་ཚེ་འཛམ་བུ་གླིང་གི་དམ་ཆོས་རིལ་ནུབ་བོ། ། དེ་ནས་གསེར་ནི་དངུལ་ངན་པ་དང་རྡོར་འགྱུར། དངུལ་ནི་ར་གན་ངན་པ་དང་རྡོར་འགྱུར་རོ། ། ར་གན་ནི་ཟངས་སུ་འགྱུར། མུ་ཏིག་ནི་རྭ་རུ་འགྱུར། རོ་དྲུག་ལས་ཁ་བ་དང་སྐྱུར་བ་གཉིས་ལུས་སོ། །
