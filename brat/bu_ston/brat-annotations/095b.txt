བྲལ་བ་འབྲས་བུ་བློ་ལྡན་ཡིན། ། དེས་དེའི་མཁན་པོ་ལ་སྨྲས་པ། ། ཀུན་དགའ་བོ་ནི་རྒས་གྱུར་ཏེ། ། དེ་ཡི་དྲན་པའང་ཉམས་པར་གྱུར། ། སྨྲས་པ། རྒས་བ་དག་གིས་གཟིར་གྱུར་པ། ། ལུས་ཅན་དེ་ནི་དྲན་པ་ཉམས། ། ཡང་དག་རྗེས་སུ་མི་དྲན་པ། ། དེ་ཡི་བློ་ནི་རྒས་པས་ནོན། ། དེ་ཀུན་དགའ་བོས་ཐོས་ནས། དེ་ལ་སྨྲས་ན་འཐབ་མོར་འགྱུར། དེའི་མཁན་པོའི་དྲུང་དུ་བདག་འགྲོ་བར་མི་རིགས་པས་མྱ་ངན་ལས་འདའ་བར་བྱའོ་སྙམ་སྟེ། རྙིང་པ་དག་ཀྱང་སྔར་འདས་ཏེ། ། གསར་བུ་དེ་དང་བདག་མི་མཐུན། ། བདག་ནི་གཅིག་པུ་བསམ་གཏན་ལ། ། སྒོ་ང་ལས་སྐྱེས་བཞིན་དུ་གནས། ། བདག་གི་གྲོགས་དང་མཛའ་བོ་ཀུན། ། སྔོན་དུ་འདས་ཤིང་སོང་བར་གྱུར། ། ཇི་ལྟར་བསམ་གཏན་གནས་དྲན་པ། ། དེ་འདྲའི་བཤེས་གཉེན་འདི་ན་མེད། ། དེ་ན་ཤ་ནའི་གོས་ཅན་ལ། སྟོན་པས་འོད་སྲུངས་ལ། དེས་ཀྱང་ང་ལ་བསྟན་པ་གཏད་ཏོ། ། ངའང་འདའ་བས། བསྟན་པ་ཁྱོད་ཀྱིས་སྐྱོངས་ཤིག། ཡུལ་བཅོམ་བརླག་གི་རི་མུ་རུན་དར། ཚོང་དཔོན་གྱི་བུ་ནཱ་ཏ་དང་ཕ་ཏས་གཙུག་ལག་ཁང་བརྩིགས་ནས། སྦྱིན་བདག་བྱེད་པར་ལུང་བསྟན་ནོ། ། དེར་སྤོས་ཚོང་སྦས་པའི་བུ་ཉེར་སྦས་རབ་ཏུ་ཕྱུང་ཞིག། ང་འདས་ནས་ལོ་བརྒྱ་ན་དེ་མཚན་མེད་པའི་སངས་རྒྱས་སུ་འགྱུར་ཏེ། སངས་རྒྱས་ཀྱི་བྱ་བ་བྱེད་དོ། ། ཞེས་སྟོན་པས་ལུང་བསྟན་ནོ། ། དེས་དེ་བཞིན་བྱའོ་ཞེས་སོ། ། དེ་ནས་མ་སྐྱེས་དགྲ་ལ་སྦྲན་ཏེ་གངྒཱའི་དབུས་སུ་སོང་ངོ་། ། རྒྱལ་པོས་གདུགས་ཀྱི་ཡུ་བ་ཆག་པ་རྨིས་ནས། སད་པ་དང་སྦྲན་པ་ཐོས་ཏེ་བརྒྱལ་ལོ། ། སངས་པ་དང་དཔུང་དང་བཅས་ཏེ་གང་གཱར་སོང་ངོ་། ། དེའི་ཚེ་ལྷས་ཡངས་པ་ཅན་པ་ལ། འཕགས་པ་ཀུན་དགའ་བོ་འགྲོ་བའི་སྒྲོན་མ་ནི། ། སྐྱེ་བོ་ཕལ་པོ་པོ་ཆེ་ལ་ཐུགས་རྗེ་མཛད། །
