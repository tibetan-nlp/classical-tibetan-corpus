རྒྱལ་བས་ཕ་རོལ་ཕྱིན་པ་དྲུག་། ཡང་དག་བཤད་དེ་དང་པོ་གསུམ། ། མཐའ་མ་གཉིས་ཀྱིས་རྣམ་པ་གཉིས། ། གཅིག་ནི་གསུམ་ཅར་ལ་ཡང་རྟོགས། ། ཞེས་སོ། ། བཅུ་གཅིག་པ་ནི། བསོད་ནམས་བྱ་བའི་དངོས་པོ་གསུམ་སྟེ། སྦྱིན་པ་ལས་བྱུང་བའི་བསོད་ནམས་བྱ་བའི་དངོས་པོ་དང་། ཚུལ་ཁྲིམས་ལས་བྱུང་བའི་བསོད་ནམས་བྱ་བའི་དངོས་པོ་དང་། བསྒོམ་པ་ལས་བྱུང་བའི་བསོད་ནམས་བྱ་བའི་དངོས་པོ་སྟེ་གསུམ་མོ། ། དང་པོ་ནི་སྦྱིན་པའི་ཕ་རོལ་ཏུ་ཕྱིན་པའོ། ། གཉིས་པ་ནི་ཚུལ་ཁྲིམས་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པའོ། ། གསུམ་པ་ནི་བཟོད་པ་བརྩོན་འགྲུས་བསམ་གཏན་ཤེས་རབ་ཀྱི་ཕར་ཕྱིན་རྣམས་སོ། ། བཅུ་གཉིས་པ་ལམ་བདུན་ནམ་བྱང་ཕྱོགས་སུམ་ཅུ་སོ་བདུན་དུ་དབྱེ་བ་ནི་གྲགས་པ་ཉིད་དོ། ། དེ་ལྟ་བུའི་ཚུལ་གྱིས་སྟོན་པའི་གསུང་རབ་ཐམས་ཅད་ཀྱང་བཤད་པར་ནུས་གསུང་ངོ་། ། གཉིས་པ་དུས་ཇི་ཙམ་དུ་བསགས་པའི་དུས་ནི། གསུང་རབ་ནས་མི་འདྲ་བ་མང་དུ་འབྱུང་སྟེ། བསྐལ་པ་གྲངས་མེད་གསུམ་དང་། བདུན་དང་། བཅུ་དང་། སུམ་ཅུ་རྩ་གསུམ་ལ་སོགས་པ་བཤད་ལ། བཅུ་གསུམ་དུ་བཤད་པའང་ཡོད་གསུང་ངོ་། ། དེ་ལ་གྲངས་མེད་གསུམ་ནི་གསུང་རབ་ཕལ་ཆེར་ལས་འབྱུང་ཞིང་། གྲངས་མེད་ཀྱི་དོན་ཡང་། མངོན་པ་མཛོད་ཀྱི་འགྲེལ་པ་དང་ཐུབ་པ་དགོངས་རྒྱན་ལ་སོགས་པ་མང་པོ་ནས། བགྲང་བ་ལས་འདས་པའི་གྲངས་མེད་མ་ཡིན་གྱི། གྲངས་ཀྱི་གནས་གཞན་དྲུག་ཅུར་ཕྱིན་པའི་གྲངས་མེད་ལ་བཞེད་ལ། གྲངས་ནི། གཅིག་བཅུ། བརྒྱ་སྟོང་། ཁྲི་འབུམ། ས་ཡ། བྱེ་བ། དུང་ཕྱུར། ཐེར་འབུམ། ཐེར་འབུམ་ཆེན་པོ། ། ཁྲག་ཁྲིག། ཁྲག་ཁྲིག་ཆེན་པོ། ། རབ་བཀྲམ་རབ་བཀྲམ་ཆེན་པོ། ། གཏམས་གཏམས་ཆེན་པོ། ། དཀྲིགས་དཀྲིགས་ཆེན་པོ། ། མི་འཁྲུགས་པ་མི་འཁྲུགས་པ་ཆེན་པོ། །
