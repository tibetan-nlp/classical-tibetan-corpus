རྒྱུད་ཀྱི་དོན་མ་རྟོགས་པར། སྦྱོར་སྒྲོལ་ལ་སོགས་ལོག་པར་སྤྱོད་དོ། ། གཉིས་པ་བསྟན་པ་ཕྱི་དར་གྱི་ཚུལ་ནི། དར་མས་བསྟན་པ་བསྣུབས་པའི་དུས་སུ་འཇད་ཀྱི་གྱེལ་མི་གཙང་རབ་གསལ། བོ་དོང་པ་གཡོ་དགེ་འབྱུང་སྟོད་ལུངས་པ་སྨར་ཤཱཀྱ་མུ་ནི་དང་གསུམ། དཔལ་ཆུ་བོ་རི་ན་སྒོམ་ཞིང་ཡོད་པ་ན། བཙུན་པ་ཁྱི་ར་བྱེད་པ་མཐོང་སྟེ། གཏམ་དྲིས་པས། རྒྱལ་པོས་བསྟན་པ་བསྣུབས་པར་ཤེས་ཏེ། འདུལ་བ་དྲེའུ་ཁལ་ཞིག་བཀལ་ཏེ་། མངའ་རིས་སྟོད་དུ་བྲོས་ནས། གར་ལོག་ནས་ལམ་ཐོན་ནས། ཧོར་ཡུལ་དུ་ཕྱིན་ཏེ། སྐད་རིགས་མི་གཅིག་པའི་སར་བསྟན་པ་དར་བར་བྱ་སྙམ་པ་ལ། དར་བར་མ་ནུས་ཏེ། མདོ་སྨད་ལྷོ་ངོས་ཀྱི་བེ་རོ་ཚ་ཚོ་ན་ཚུར་བྱོན་ཏེ་རྨ་ལུང་རྡོ་རྗེ་བྲག་ར་ཨ་ཆུང་གནམ་རྫོང་དག་དིག་ཤེལ་གྱི་ཡང་དགོན་ན་བསྒོམ་ཞིང་ཡོད་པ་ལས། རྨ་ཆུ་ཁའི་ཕྱུགས་རྫིས་མཐོང་སྟེ། དགོངས་ཀ་མི་འདུས་པ་རྣམས་ལ་སྨྲས་པས། ལས་དང་སྐལ་བར་ལྡན་པ་མོས་པས་སྤྱོད་པའི་ས་པ། མུ་གཟུ་གསལ་འབར་བྱ་བས་བལྟར་ཕྱིན་པས། དད་ནས་རབ་ཏུ་འབྱུང་བར་ཞུས་པས། འདུལ་བའི་པོ་ཏི་གཏད་ནས་འདི་ལྟོས། དད་ན་རབ་ཏུ་དབྱུང་གསུང་། བལྟས་པས་དད་དེ་མཆི་མ་བྱུང་ངོ་། ། དེར་གཙང་གིས་མཁན་པོ་བྱས། གཡོས་སློབ་དཔོན་བྱས་ཏེ་དགེ་ཚུལ་དུ་བསྒྲུབས་སོ། ། མཁན་སློབ་ལས་མིང་སྦྱར་ཏེ། དགེ་བ་རབ་གསལ་དུ་བཏགས། ཕྱིས་ཐུགས་རབ་ཆེ་བས་དགོངས་པ་རབ་གསལ་དུ་གྲགས་སོ། ། དེ་ནས་བསྙེན་རྫོགས་ཞུས་པས། དགེ་སློང་ལྔ་མ་ཚང་བས་མི་འོང་གསུང་བ་དང་། སྔར་ལྷ་ལུང་དཔལ་གྱི་རྡོ་རྗེ་བ་དགེ་སློང་གསུམ།
