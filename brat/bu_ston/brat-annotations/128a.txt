རྒྱལ་པོ་ཁྲོས་ཏེ། ཁྱོད་ཀྱི་རལ་གྲི་ཕྱུང་ཞིག་ཅེས་སྨྲས་པས། ལྷ་ཉིད་ལ་གནོད་པར་འགྱུར་རོ་ཞེས་གསོལ་བས། གནོད་ཀྱང་སླའི་ངེས་པར་ཕྱུང་ཞིག་ཟེར་རོ། ། འོ་ན་ལྷའི་སྤྱན་གཅིག་བཀབ་སྟེ་གཅིག་གིས་གཟིགས་ཤིག་ཅེས་གསོལ་པས། དེ་ལྟར་བྱས་ཏེ། རལ་གྲི་ཕྱུང་བས་དེའི་འོད་ཟེར་གྱིས་མ་བཟོད་དེ། མིག་གཅིག་ཞར་བ་དང་། བཟོད་པར་གསོལ་ཏེ་སྐྱབས་སུ་སོང་སྟེ། ཆོས་ལ་བཙུད་དོ། ། དེ་ནས་དཔལ་གྱི་རིའི་ལྷོ་ཕྱོགས་སུ་བྱོན་ཏེ། ཨུ་ཚ་སྨའི་སྤྱོད་པ་ཞེས་པ་སྤྲང་པོའི་ཆ་བྱད་གཅེར་བུ་ཁྲུད་མ་ཕོ་བའི་ཚིགས་མས་འཚོ་བར་བྱེད་པ་ན། ཁ་ཏི་བི་ཧ་རའི་རྒྱལ་པོའི་བྲན་མོ་ཀ་ཙ་ལ་ཧས་ཁྲུད་མ་ཕོ་བའི་ཆུ་ཐོར་སློབ་དཔོན་ལ་ཕོག་པས་ཁོལ་བ་མཐོང་སྟེ། དེའི་ཚེ་མུ་སྟེགས་ཀྱི་སྟོན་པ་ཤད་ཀ་ར་དེ་པ་ཞེས་བྱ་བས། གནངས་ནམ་མཁའ་ལ་དབང་ཕྱུག་ཆེན་པོའི་དཀྱིལ་འཁོར་འབྲི་བས། དེ་བཤིག་མ་ནུས་ན་ནང་པའི་ཆོས་ལྷ་རྟེན་བསྲེགས་ནས། ཕྱི་རོལ་པའི་སྒོར་འཇུག་དགོས་སོ་ཞེས་རྒྱལ་པོ་ལ་གསོལ་པ་དང་། རྒྱལ་པོས་དགེ་འདུན་བསྡུས་ཏེ། དོན་དེ་བསྙད་པས་སུས་ཀྱང་བཤིག་པར་ཁས་ལེན་མ་ནུས་པས། རྒྱལ་པོ་མྱ་ངན་དུ་གྱུར་པ་ན། བྲན་མོས་སྔར་མཐོང་པ་དེ་བཤད་པས། ཚོལ་ཅིག་གསུངས་ནས། ཕྱོགས་ཀུན་ཏུ་བཙལ་བས་ཤིང་སྡོང་གཅིག་གི་དྲུང་ནས་རྙེད་དེ། དོན་དེ་ཞུས་པས། ངས་ནུས་ཀྱིས། ། ཆུ་བུམ་པ་གང་རས་ཡུག་གཉིས། མེ་རྣམས་སྟ་གོན་གྱིས་ཤིག་གསུངས་ནས། དེ་ལྟར་བྱས་སོ། ། སང་དགོང་ཁ་ཐིག་བཏབ་སྟེ་སོང་བས་ཐམས་ཅད་ཐེ་ཚོམ་དུ་གྱུར་ཏོ། ། གནངས་སྔ་དྲོ་དཀྱིལ་འཁོར་བྲིས་ཏེ་ཤར་སྒོ་ཙམ་ཟིན་པ་ན། སློབ་དཔོན་གྱིས་ཏིང་ངེ་འཛིན་མཛད་པས་ཚུབ་མ་ཆེན་པོ་ལངས་ཏེ།
