ཇི་སྐད་དུ། བྱེ་བ་བརྒྱད་ཅུར་རྒྱལ་བ་ཉིད། ། སངས་རྒྱས་ཉིད་དུ་བསྟན་གྱུར་ཀྱང་། ། ད་དུང་ཆོག་པའི་འདུ་ཤེས་བསལ། ། བྱང་ཆུབ་མཆོག་ཏུ་ཐུགས་ཀྱང་འཇུག། སྟོང་གསུམ་དྲུག་ཅུ་རྩ་གཅིག་གི། སངས་རྒྱས་ཞིང་རྣམས་དག་གྱུར་པས། ། ཇི་ལྟར་ཐུབ་པ་ཐབས་མཁས་པས། ། རྒྱལ་བ་ཁྱོད་ནི་ཀུན་གྱིས་འཚལ། ། ད་དུང་དང་པོ་ཐུགས་བསྐྱེད་པར། ། དེ་དང་དེར་ནི་ཡོངས་བསྟན་ཏེ། ། ད་དུང་དུ་ཡང་འདྲེན་པ་ཁྱོད། ། སངས་རྒྱས་མང་པོ་སྟོན་པར་མཛད། ཅེས་སོ། ། དམ་པའི་ཆོས་པད་མ་དཀར་པོ་ལས་ཀྱང་། བསྐལ་པ་བྱེ་སྟོང་བསམ་གྱིས་མི་ཁྱབ་པ། ། དེ་ཡི་ཚད་ནི་ནམ་ཡང་མེད་པ་ནས། ། བྱང་ཆུབ་མཆོག་རབ་འདི་ནི་ངས་ཐོབ་སྟེ། ། ང་ནི་རྟག་ཏུ་ཆོས་ཀྱང་རབ་འཆད་དོ། ། ཞེས་སྔར་དུ་སངས་རྒྱས་པར་གསུངས་སོ། ། དེ་མཉམ་པ་ཉིད་ལ་དགོངས་པ་ཡིན་དུ་ཆུག་ནའང་། གྲུབ་མཐའི་འདོད་ཚུལ་གསུམ་ལས། ཉན་ཐོས་ཀྱི་ལུགས་གོང་དུ་བསྟན་ཅིང་། ཐེག་ཆེན་གྱི་ལུགས་ཀྱིས་སངས་རྒྱས་རྣམས་འོག་མིན་དུ་སངས་རྒྱས་ནས་འདོད་ཁམས་སུ་མཛད་པ་བཅུ་གཉིས་སྟོན་པར་གསུངས་ཏེ། ལང་ཀར་གཤེགས་པར་། འདོད་པའི་ཁམས་དང་གཟུགས་མེད་དུ། ། སངས་རྒྱས་རྣམ་པར་འཚང་མི་རྒྱ། ། གཟུགས་ཀྱི་ཁམས་ཀྱི་འོག་མིན་དུ། ། འདོད་ཆགས་བྲལ་ཁྱོད་འཚང་རྒྱའོ། ། ཞེས་པ་དང་། སྟུག་པོ་བཀོད་པའི་མདོ་ལས། སངས་རྒྱས་ཐམས་ཅད་འོག་མིན་དུ། ། སངས་རྒྱས་མ་གྱུར་འདོད་ཁམས་སུ། ། སངས་རྒྱས་མཛད་པ་མི་མཛད་དོ། ། ཞེས་དང་། རྒྱུད་བླ་མར། ཐུགས་རྗེ་ཆེན་པོས་འཇིག་རྟེན་མཁྱེན། ། འཇིག་རྟེན་ཀུན་ལ་གཟིགས་ནས་ནི། ། ཆོས་ཀྱི་སྐུ་ལས་མ་གཡོས་པར། ། སྤྲུལ་པའི་རང་བཞིན་སྣ་ཚོགས་ཀྱིས། །
