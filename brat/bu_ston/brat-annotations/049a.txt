ཡང་ན་རྫ་མཁན་གྱི་ཁྱེའུ་སྣང་བྱེད་དུ་གྱུར་པའི་ཚེ། དེ་བཞིན་གཤེགས་པ་ཤཱཀྱ་ཐུབ་པ་ཆེན་པོ་ལ་ཁམ་པའི་བུམ་པ་གཅིག་འགྲོན་བུ་ལྔ་མཆིལ་ལྷམ་ཟུང་གཅིག་གདུགས་གཅིག་རྣམས་ཕུལ་ཏེ། དེ་བཞིན་གཤེགས་པ་ཁྱེད་སྐུ་ཅི་འདྲ་དང་། ། འཁོར་དང་སྐུ་ཚེའི་ཚད་དང་ཞིང་ཁམས་དང་། ། དེ་བཞིན་མཚན་མཆོག་བཟང་པོ་ཅི་འདྲ་བ། ། དེ་འདྲ་ཁོ་ནར་བདག་ཀྱང་འགྱུར་བར་ཤོག། ཅེས་སྨོན་ལམ་བཏབ་སྟེ་སེམས་བསྐྱེད་དོ། ། གཉིས་པ་ཚོགས་བསགས་ཚུལ་ལ་གསུམ་སྟེ། གང་བསག་བྱའི་ཚོགས་ངོས་གཟུང་། ཇི་ཙམ་བསགས་པའི་དུས། ཇི་ལྟར་བསགས་པའི་ཚུལ་དང་གསུམ་མོ། ། དང་པོ་ལ་ཚོགས་ཀྱི་རང་བཞིན། དེ་ཕར་ཕྱིན་ལ་སྦྱར་བ། ངེས་ཚིག་ལས། སྒྲུབ་ཚུལ། འབྲས་བུ། དམིགས་ཡུལ། རྣམ་གྲངས་ཀྱི་སྒོ་ནས་དབྱེ་བ་དང་བརྒྱད་དོ། ། དང་པོ་ནི། མདོ་སྡེ་རྒྱན་ནས། བྱང་ཆུབ་སེམས་དཔའ་རྣམས་ཀྱི་ཚོགས། ། བསོད་ནམས་ཡེ་ཤེས་མཚུངས་པ་མེད། ། གཅིག་གིས་འཁོར་བར་མངོན་མཐོ་བྱེད། ། གཞན་གྱིས་ཉོན་མོངས་མེད་པར་འཁོར། ། ཞེས་པས་མངོན་མཐོ་ངེས་ལེགས་ཀྱི་རྒྱུར་གྱུར་པའི་དགེ་བ་ཟག་བཅས་ཟག་མེད་རྣམས་ནི་ཚོགས་ཀྱི་རང་བཞིན་ནོ། ། གཉིས་པ་ནི། སྦྱིན་དང་ཚུལ་ཁྲིམས་བསོད་ནམས་ཀྱི། ། ཚོགས་ཡིན་ཤེས་རབ་ཡེ་ཤེ་ཀྱི། ། གསུམ་པོ་གཞན་ནི་གཉིས་ཀ་སྟེ། ། ལྔ་ཆར་ཡང་ནི་ཡེ་ཤེས་ཚོགས། ། ཞེས་སོ། ། གསུམ་པ་ནི། རྒྱུན་དུ་བསྒོམ་པ་ཐོབ་ནས་ནི། ། དགེ་བ་ཡང་དང་ཡང་དུ་བསྒྲུབ། ། ཅེས་པས། སཾ་སྐྲི་ཏ་ལ་སཾ་བྷ་ར་ཞེས་པ་ཚོགས་ཡིན་པས། སཾ་ཏ་ན་རྒྱུན་དུ་བྷཱ་ཝ་ན་ཞེས་པ་བསྒོམ་པ་། ར་ཏི་ནི་ཡང་ནས་ཡང་དུ་སྟེ། རྒྱུན་མི་ཆད་པར་ཡང་ཡང་བསྒོམ་པ་ཐོབ་པས།
