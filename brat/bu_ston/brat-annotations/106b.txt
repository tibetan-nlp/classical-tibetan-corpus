དུས་མཐའ་འཇིག་རྟེན་ཉམས་པ་ན། ། སྡོམ་བརྩོན་རྒྱལ་པོའི་ཚུལ་གྱིས་ནི། ། འབྱུང་བར་འགྱུར་བར་ཐེ་ཚོམ་མེད། ། དཔེར་ན་མ་ཁོལ་ཞེས་བྱ་དང་། ། མེ་ཏོག་ཅེས་བྱར་རྣམ་པར་གྲགས། ། ཡི་གེ་མ་ཞེས་གཞོན་ནུ་ཞེས། ། དང་པོ་ཐ་མར་ཆོས་ལ་དགའ། ། ཀླུ་བོས་ཞེས་ཀྱང་རྣམ་པར་གྲགས། རིན་ཆེན་འབྱུང་ཞེས་བྱ་མིང་དང་། ། ཡི་གེ་ག་ཞེས་གཞོན་ནུ་ཞེས། ། ཡི་གེ་བ་ཞེས་ཆོས་སེམས་བྱེད། ། ཡི་གེ་ཨཱ་མིང་བདག་ཉིད་ཆེ། ། སྟོན་པའི་བསྟན་པ་གཅེས་འཛིན་བྱེད། ། བློ་ལྡན་ཡོན་ཏན་ཀུན་གྱིས་བཀུར། ། དང་པོའི་ཡི་གེ་ལ་ཞེས་བསྟན། ། རྒྱལ་པོ་སངས་རྒྱས་པ་ཡི་ཕྱོགས། ། སྟོན་པའི་བསྟན་པ་གསལ་བར་བྱེད། ། སྡོམ་བརྩོན་ཡི་གེ་ཨ་ཞེས་བསྒྲགས། ། བྲམ་ཟེ་ལས་ནི་རབ་ཏུ་བྱུང་། ། གྲོང་ཁྱེར་གནས་བཅས་གནས་པ་སྟེ། ། དེ་བཞིན་ཚེ་ནི་བརྒྱད་ཅུར་འཚོ། ། དང་ཡིག་ཡི་གེ་ཨ་དགེ་སློང་། ། ལྷོ་ཡི་ཕྱོགས་སུ་འབྱུང་བར་འགྱུར། ། བློ་ལྡན་ཚེ་ནི་དྲུག་ཅུར་འཚོ། ། ཀ་ཤི་ཞེས་བྱའི་གྲོང་དུ་གནས། ། ཡི་གེ་དྷ་ཞེས་སྟོམ་བརྩོན་ཉིད། ། ལྷོ་ཡི་ཕྱོགས་སུ་རྣམ་བར་བསྒྲགས། ། ཕ་རོལ་རྐོལ་བ་བཟློག་ནས་ཀྱང་། ། དེ་བཞིན་སྡོམ་བརྩོན་སྔགས་ཀྱང་འགྲུབ། ། གཞན་ཡང་རབ་བྱུང་དམ་པ་ནི། ། སིང་ག་ལ་སིང་ག་ལ་སིང་ག་ལ་སིང་ག་ལ་སིང་ག་ལ་སིང་ག་ལ་སིང་ག་ལ་སིང་ག་ལ་སིང་ག་ལ་སིང་ག་ལ་སིང་ག་ལ་སིང་ག་ལ་སིང་ག་ལ་སིང་ག་ལ་སིང་ག་ལ་སིང་ག་ལ་སིང་ག་ལ་སིང་ག་ལ་སིང་ག་ལ་སིང་ག་ལ་སིང་ག་ལ་སིང་ག་ལ་ཡི་གྲོང་ན་གནས། ། འཕགས་པ་མིན་ལ་འཕགས་པའི་མིང་། ། སིང་ག་ལ་ཡི་གླིང་ན་གནས། ། ཕ་རོལ་རྒོལ་བ་བཟློག་ནས་ནི། ། མུ་སྟེགས་རྣམས་ཀྱི་སྔགས་སུན་ཕྱུང་། ། དེ་དུས་འཇིགས་སུ་རུང་བ་ཡི། ། དུས་ཀྱི་མཐའ་ལ་འབྱུང་བར་འགྱུར། ། ཡི་གེ་ད་དང་ལ་དང་པོར། ། སྡོམ་བརྩོན་གང་ཞིག་རབ་གསུངས་དང་། ། ཡི་གེ་ར་དང་བ་དང་པོར། ། དགེ་སློང་གང་ཞིག་རབ་འབྱུང་གསུངས། ། སྟོན་པའི་བསྟན་པ་གསལ་བྱེད་པ། ། འབྱུང་བར་འགྱུར་བར་ཐེ་ཚོམ་མེད། ། རྒྱལ་པོ་ནག་པོ་ཞེས་བྱ་བར། ། ཡི་གེ་མ་ཞེས་སྡོམ་བརྩོན་འབྱུང་། ། མཆོད་རྟེན་ཀུན་དགའ་ར་བ་གནས། །
