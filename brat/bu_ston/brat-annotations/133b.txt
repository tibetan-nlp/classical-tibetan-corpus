རྒྱལ་བ་ལེགས་སོ། ། ད་ཕྱིན་ཆད་ཁྱོད་ཀྱིས་རྒྱལ་སྲིད་ཟུང་ཤིག། ང་ནི་རབ་ཏུ་འབྱུང་ངོ་ཞེས་བསྒོས་ནས། རྒྱལ་སྲིད་བུས་བཟུང་ངོ་། ། དེ་ནས་ལོ་བཅུ་གཉིས་ཀྱི་བར་དུ་རྒྱལ་པོ་གསུམ་གྱི་དམག་དང་འཐབས་པ་ལས་དམག་མང་པོའང་རིམ་གྱིས་གཏུགས། རྒྱལ་པོ་གསུམ་ཡང་བཟུང་ནས་མེད་པར་བྱས་སོ། ། དེ་ཕན་ཆད་འཛམ་བུ་གླིང་གི་རྒྱལ་པོ་ཆེན་པོ་བྱས་པར་གྱུར་ཏོ། ། དེ་ནས་རྒྱལ་པོས། བློན་པོ་རྣམས་ལ་ང་འཛམ་བུའི་གླིང་གི་རྒྱལ་པོ་བྱེད་པ་ནི་དགའ་ན། སེམས་ཅན་མང་པོ་འདི་སྙེད་ཅིག་བསད་པའི་སྡིག་པ་ཆེ་བས། མི་དགའ་བས། སྡིག་པ་འདི་ཇི་ལྟ་བུ་བྱས་ན་འབྱང་བར་འགྱུར། ཞེས་དྲིས་པ་དང་། བློན་པོ་རྣམས་ན་རེ། ཡུལ་པཱ་ཌ་ལི་པུ་ཏྲ་ན་ཆོས་ཀྱི་མཁན་པོ་སྡེ་སྣོད་གསུམ་ཤེས་པ་བྲམ་ཟེ་ཨ་ཀ་ན་དྲ་དྷའི་བུ་སློབ་མ་ཅན་ཞེས་བགྱི་བ་ཞིག་དགོན་པ་ན་གནས་པ་དེ་སྤྱན་དྲངས་ན། སྡིག་པ་འབྱང་བར་བགྱིད་ནུས་སོ་ཞེས་ཟེར་བ་དང་། རྒྱལ་པོ་དགའ་ནས། དགེ་སློང་སློབ་མ་ཅན་སྤྱན་དྲངས་ནས་རྒྱལ་པོས་ཕྱག་བྱས་ཏེ། སྡིག་པ་རྣམས་ཐབས་ཅི་ཞིག་གིས་འབྱང་ཞེས་ཞུས་པ་དང་། དེས་དཀོན་མཆོག་གསུམ་ལ་ལོ་བཅུ་གཉིས་སུ་མཆོད་ཅིང་སྐྱབས་སུ་གསོལ་ན་སྡིག་པ་འབྱང་ངོ་ཞེས་བྱས་སོ། ། དེ་ནས། འཛམ་བུའི་གླིང་ན་དགེ་སློང་འདུག་གོ་ཅོག་ལ་འགུགས་པ་བཏང་སྟེ། ཐམས་ཅད་ཀོའུ་ཤཱམ་བཱིར་བསྡུས་ནས། ཡུལ་གཞན་དུ་ཆོས་སྤྱོད་དུ་མི་སྟེར་བར་གྱུར་ཏོ། ། དགེ་སློང་རྣམས་ཀྱང་ལམ་དུ་གཅན་གཟན་དང་། མི་རྒོད་དང་ཆུ་ལ་སོགས་པས་ཕལ་ཆེར་ཆུད་ཟོས་ནས། རྒྱལ་པོའི་དྲུང་དུ་འབུམ་ཙམ་ཞིག་གིས་ཕྱིན་པར་འགྱུར་རོ། ། དེ་དག་ཕྱིན་པ་དང་རྒྱལ་པོས་མཆོད་སྟོན་གསོལ་ཡོན་ཕུལ་ཏེ་མཆོད་དོ། ། དེ་ནས་དགེ་འདུན་རྣམས་ཕན་ཚུན་ཁྱོད་ཀྱི་མཁན་པོ་ག་རེ་སློབ་མ་གར་སོང་།
