འོད་གཅིག་གིས་ནི་གསལ་བར་གྱུར། ཞེས་དྲིས་པས། ལན་དུ། ཉི་མའི་འོད་ནི་རྒྱུན་དུ་རྣོུ། ། བུ་གཅིག་སྣང་བའི་འོད་འདི་དག། ཡན་ལག་རེག་ན་འོད་བསིལ་བས། ། ངེས་པར་ཐུབ་པ་དེ་ཡི་འོད་ཅེས་སོ། ། དེ་ནས་ཉོན་མོངས་མེད་སེར་སྐྱར་ཞུགས་ཏེ། ། རྒྱལ་ཆེན་བདག་ཅག་འདིར་མཆིས་པ། ། ཁྱེད་ཀྱི་སྲས་ནི་མཐོང་བར་འཚལ། ། འཇིག་རྟེན་གཙོ་བོ་རྣམ་འདྲེན་པ། ། ཐུབ་པ་དེ་ནི་བལྟ་བར་འཚལ། ཞེས་པ་དང་། མནལ་དུ་སོང་བ་ནའང་། སྤྱན་གཟིགས་པས་སྨྲས་པ། རིགས་དང་ལྡན་པའི་རྟ་རྣམས་ནི། ། མཚན་མོའི་བཞི་ཆ་ཉལ་བ་ལྟར། ། དགོས་པ་དོན་གཉེར་རྣམས་ཀྱི་གཉིད། ། མིག་ལ་ཡུན་རིང་མི་གནས་སོ། ། ཞེས་སོ། ། མཚན་མཁན་དག་འཁོར་ལོས་བསྒྱུར་བར་ལུང་སྟོན་ན་ཅིར་འགྱུར། ། ས་བདག་རྟོག་པ་པོ་རྣམས་བློ་འཁྲུལ་ཏེ། ། རྩོད་པའི་དུས་སུ་འཁོར་ལོས་བསྒྱུར་མི་འབྱུང་། ། བསོད་ནམས་མཆོག་གི་ཆོས་ཀྱི་མཛོད་འདི་ནི། ། ཉེས་ལས་རྣམ་བརྒལ་སངས་རྒྱས་ཉིད་དུ་འགྱུར། ཞེས་སོ། ། གཞོན་ནུ་སྐྱེ་དགུའི་བདག་མོ་ལ་གཏད་དེ། པང་ན་འཚོ་བའི་མ་མ་བརྒྱད། ནུ་མ་བསྣུན་པ་བརྒྱད། རྩེན་བའི་མ་མ་བརྒྱད། དྲི་མ་འཕྱི་བ་བརྒྱད་དེ། སོ་གཉིས་ལ་གཏད་དེ་བསྐྱེད་བསྲིངས་སོ། ། དེ་ལྟ་བུའི་ཆོ་འཕྲུལ་དེ་རི་བོ་གངས་ཅན་ན་གནས་པ་དྲང་སྲོང་ནག་པོ་ཞེས་པ་མངོན་ཤེས་ལྔ་དང་ལྡན་པ་ཞིག་གིས་ཚོར་ནས། སྣག་ཚ་མེས་བྱིན་དང་བཅས་པས་རྫུ་འཕྲུལ་གྱིས་སེར་སྐྱའི་གྲོང་ཁྱེར་དུ་ཕྱིན་ཏེ། དེའི་མཚན་བལྟས་པས། དེ་ཅིར་འགྱུར་བ་ཤེས་ཏེ། ངོ་མཚར་དང་མྱ་ངན་སྐྱེས་ནས་དེའི་ཞབས་ལ་ཕྱག་འཚལ་ཞིང་། སྐོར་བ་བྱས་ནས་གདོང་མཆི་མ་དང་བཅས་ཏེ་ལུང་བསྟན་པ། བཙས་གྱུར་གཞོན་ནུ་འདི་ཡི་མཚན་ལྟས་ནི། །
