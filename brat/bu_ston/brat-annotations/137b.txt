གང་ནས་བྱོན་དྲིས་པས་ནམ་མཁའ་ལ་མཛུབ་མོ་སྦྲེང་ཞིང་། སྐད་མ་གོ་ནས། ཤིང་གི་ཁྲི་ལ་བཞག་སྟེ་མི་བཞིའི་གཉའ་བ་ལ་ཁུར་ནས། འདི་ལ་ཁོ་བོ་ཅག་གི་ཇོ་བོ་བྱའོ་ཟེར་ཏེ། མིང་ཡང་གཉའ་ཁྲི་བཙན་པོར་བཏགས། རྒྱལ་པོ་སྔ་ཤོས་ཡིན་ནོ། ། དེའི་སྲས་དམུ་ཁྲི་བཙན་པོ། དེའི་སྲས་དིང་ཁྲི། དེའི་སྲས་སོ་ཁྲི། དེའི་སྲས་ཡེ་ཁྱི། དེའི་སྲས་གདགས་ཁྲི། དེའི་སྲས་སྲི་ཁྲི་བཙན་པོ་དང་བདུན་ལ་ཁྲི་བདུན་ཟེར་ཞིང་། ཐ་མ་དེའི་རིང་ལ་རྡོལ་བོན་དར་རོ། ། དེའི་སྲས་གསུམ་གྱི་བྱ་ཁྲི་ལ་སྤུ་ལྡེ་གུང་རྒྱལ་ཞེས་བཏགས། དེའི་སྲས་ཨ་ཤོ་ལེགས། དེའི་སྲས་ཨི་ལེགས་ཀྱིས་འཆིང་བར་སྟག་རྩེ་བརྩིགས་ཏེ། མཁར་སྔ་ཤོས་ཡིན་ནོ། ། དེའི་སྲས་དེ་ཤོ་ལེགས། དེའི་སྲས་གུ་རུབ་ལེགས། དེའི་སྲས་འབྲོང་རྗེ་ལེགས། དེའི་སྲས་ཐོ་ཤོ་ལེགས་ཏེ། དྲུག་ལ་སའི་ལེགས་དྲུག་ཟེར་རོ། ། དེའི་སྲས་རྒྱལ་ཟ་རྣམ་ཟིན་ལྡེ། དེའི་སྲས་ལྡེ་སྣལགནམ་། ནས་བཟུང་སྟེ། ཁྲི་ཐོག་རྗེ་ཐོག་བཙན་གྱི་བར་ཡན་ཆད་རྒྱལ་རབས་ཉི་ཤུ་རྩ་དྲུག་ན། ལྷ་ཐོ་ཐོ་རི་གཉན་བཙན་བྱོན་པའི་ཚེ། དགུང་ལོ་དྲུག་ཅུ་ཐུབ་པ་ན། ཕོ་བྲང་ཡམ་བུ་བླ་སྒང་གི་རྩེ་ན། བཞུགས་པ་ན་ནམ་མཁའ་ནས་ཟ་མ་ཏོག་ཅིག་བབས་པ་ཁ་ཕྱེ་བས། ཟ་མ་ཏོག་བཀོད་པ། པང་གོང་ཕྱག་རྒྱ་པ། གསེར་གྱི་མཆོད་རྟེན་ཞིག་བྱུང་སྟེ། མིང་གཉན་པོ་གསང་བར་བཏགས་ཏེ་མཆོད་པས། ལོ་བརྒྱ་ཉི་ཤུར་ཐུབ་སྟེ། དམ་པའི་ཆོས་ཀྱི་དབུ་བརྙེས་ཤིང་། དེ་ཡན་ཆད་ནི་ཆབ་སྲིད་བོན་གྱིས་བསྐྱངས་སོ། ། དེའི་རྨི་ལམ་ན། མི་རབས་ལྔན། འདི་ཤེས་པ་འབྱུང་ཞེས་ལུང་བསྟན་ནོ། ། དེའི་སྲས་ཁྲི་གཉན་ཟུང་བཙན།
