ཙྭ་ཅན་གྱི་གྱད་རྣམས་ལ་སྦྲན་ཏེ། གྱད་རྣམས་ཀྱིས་ཉི་མ་བདུན་གྱི་བར་ལ་སྟོ་གོན་བྱས་ནས། བདུན་པ་ལ། གྱད་ཀྱི་ཆུང་མ་དང་། བུ་མོ་རྣམས་ཀྱིས་བླ་རེ་བྲེས། གྱད་གཞོན་ནུ་དག་གིས་ཁྱོགས་བཟུང་། ལྷ་རྣམས་ཀྱིས་དྲི་དང་ཕྲེང་བ་དང་བདུག་པ་ལ་སོགས་པས་བཀུར་སྟིར་བྱས་ཏེ། རྩྭ་ཅན་གྱི་ནུབ་སྒོ་ནས་དབུས་ལ་ཞུགས། ཤར་སྒོ་ནས་འཐོན་དབྱིག་ལྡན་རྒལ་གྱད་ཀྱི་དབུ་རྒྱན་བཏགས་པའི་མཆོད་རྟེན་གྱི་དྲུང་དུ་བཞུགས་སུ་གསོལ་ཏེ། ལྷ་རྣམས་ཀྱིས་མེ་ཏོག་པུས་ནུབ་ཙམ་གཏོར་རོ། ། དེའི་ཚེ། འཚོ་བ་ཞིག་གིས་མེ་ཏོག་མང་པོ་བླངས་ཏེ། སྡིག་པ་ཅན་གྱི་གྲོང་དུ་སོང་བ་དང་། འོད་སྲུངས་ཆེན་པོ་གདུང་མ་ཉམས་པ་ལ་ཕྱག་འཚལ་འདོད་ནས་འོངས་པ་དང་ཕྲད་དེ། སྟོན་པ་འདས་པའི་གཏམ་བསྙད་པས། རྒན་ཞུགས་ཤིག་འདི་སྐད་དུ། ཀྱེ་དགེ་སློང་དག་འདི་ནི་བྱ་བའོ། ། འདི་ནི་བྱ་བ་མ་ཡིན་པའོ། ། ཞེས་སྨྲ་བ་ལས་ཐར་རོ། ། ད་ནི་ནུས་པ་བྱ། མི་ནུས་པ་མི་བྱའོ། ། ཞེས་སྨྲས་པ་དེ། ལྷས་བསྒྲིབས་ཏེ། འོད་སྲུངས་མ་གཏོགས་པས་མི་ཐོས་པ་བྱས་སོ། ། འོད་སྲུངས་ཀྱིས་དགེ་འདུན་རྣམས་མྱུར་དུ་སྐུ་གདུང་མ་ཉམས་པ་ལ་བལྟ་བའི་ཕྱིར་རིངས་པར་འགྲོའོ་ཞེས་བསྐུལ་ལོ། ། དེའི་ཚེ་གྱད་རྣམས་ཀྱིས། ། གདུང་འཁོར་ལོས་བསྒྱུར་རྒྱལ་ལ་བྱ་བ་ལྟར་བྱས་ཏེ། ཞུགས་ལ་གཞེན་པར་འདོད་ཀྱང་མེ་མ་འབར་བ། མ་འགག་པས། འོད་སྲུངས་མ་སླེབ་པ་ཡིན་པར་ཤེས་ཏེ། སྨྲས་པ་དང་། འོད་སྲུངས་སླེབ་སྟེ། སྒྲོམ་ཕྱེ་རས་དང་ཤིང་བལ་བསལ་ཏེ། སྐུ་མ་ཉམས་པ་ལ་ཕྱག་བྱས་སོ། ། དེའི་ཚེ་ཉན་ཐོས་ཆེན་པོ་ཀུན་ཤེས། སྐུལ་བྱེད། སྟོབས་བཅུ་འོད་སྲུངས། འོད་སྲུངས་ཆེན་པོ་སྟེ་བཞི་ལས།
