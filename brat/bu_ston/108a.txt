ངེས་པར་བྱང་ཆུབ་གཞོལ་བར་འགྱུར། ། ཞེས་པ་འདི་གཉིས། དབྱིག་གཉེན་དང་ཕྱོགས་གླང་ལ་སྦྱོར་བ་མང་སྟེ་མིན་པ་འདྲ། སྡོམ་བརྩོན་ལས་ལོགས་སུ་ཕྱེ་བས་བཤད་པའི་ཕྱིར་དང་། བྱེད་ལས་དང་མཚན་ཉིད་མེད་པའི་ཕྱིར་རོ། ། དབུས་ཀྱི་ཡུལ་དུ་བསྟན་པ་ནི། བྲམ་ཟེ་གང་བ་བཟང་ཞེས་པ། ། འདུལ་བ་དང་ནི་ཚུལ་བཟང་ཉིད། ། གང་བ་བཅོམ་བརླག་ཉིད་དུ་གནས། ། དང་ཡིག་བྷ་ནི་ནོར་གྱི་སྤྱན། ། རྒྱལ་པོ་རྣམས་ཀྱིས་བསྔགས་མཆོད་བྱེད། ། བྲམ་ཟེ་འདི་དག་བསྟན་པ་ནི། ། སྟོན་པའི་བསྟན་པ་མཆོད་བྱེད་པའོ། ། ཞེས་གསུངས་སོ། ། ཤར་ཕྱོགས་རྒྱ་མཚོར་བརྟེན་ནས་ནི། ། དེ་འདིར་འབྲོག་གི་མཐར་ཐུག་པར། ། བློ་ལྡན་ལོ་ཧི་ཏ་ལས་གཞན། ། བྱང་དུ་དེ་བཞིན་ཁ་བ་ཅན། ། ཞེས་པ་འདིས་ཡ་འབྲོག་ཏུ་འཁྲུངས་པའི་ལོ་ཙྪ་བ་ཆེན་པོ་བསྟན་པར་འདོད་དེ། རྒྱལ་པོ་ལུང་བསྟན་པའི་སྐབས་ཡིན་པས་བརྟག་དགོས་ཏེ། ། དེའི་འཕྲོ་ལས་ནུབ་ཏུ་ཀ་ཤི་ཉམས་དགའི་གྲོང་། རྩེ་མོ་ཞེས་བྱའི་གྲོང་དེ་བཞིན། ། འདི་ཡི་ནང་དུ་རྒྱལ་པོ་ནི། ། སྟོན་པའི་བསྟན་པ་གསལ་བྱེད་པ། ། དེ་ནི་སེང་གེ་ལྔ་ཞེས་བྱ། ། ཞེས་པས་སོ། ། གཞན་ཡང་སྟོན་པ་འདས་ནས་ལོ་བརྒྱ་ན་གྲོང་ཁྱེར་མེ་ཏོག་ཅན་དུ་རྒྱལ་པོ་མྱ་ངན་མེད་འབྱུང་སྟེ། ལོ་བརྒྱ་ལྔ་བཅུར་འཚོ་ཞིང་། ལོ་བརྒྱད་ཅུ་རྩ་བདུན་དུ་མཆོད་རྟེན་མཆོད་པར་བྱེད་དོ། ། དེ་ནས་རྒྱལ་པོ་མྱ་ངན་མེད་བྲལ་ཞེས་པ་ལོ་བདུན་ཅུ་རྩ་དྲུག་ཏུ་མཆོད་རྟེན་མཆོད་དོ། ། དེ་ནས་རྒྱལ་པོ་དཔའ་བོའི་སྡེ་ལོ་བདུན་ཅུར་རྒྱལ་སྲིད་བྱེད་དོ། ། དེ་ནས་རྒྱལ་པོ་དགའ་བོ་ལོ་ལྔ་བཅུ་རྩ་དྲུག་རྒྱལ་སྲིད་བྱེད་ཅིང་། དེའི་གྲོགས་པོ་བྲམ་ཟེ་པཱ་ཎི་ནི་ཡིན་ནོ། ། དེ་ནས་རྒྱལ་པོ་ཟླ་བ་སྦས་པ་འབྱུང་། དེ་ནས་དེའི་བུ་སྙིང་པོ་ཐིག་ལེ་ཞེས་བྱ་བས་ལོ་བདུན་ཅུར་རྒྱལ་སྲིད་བྱེད།
