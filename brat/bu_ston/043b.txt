བརྒྱད་ཁྲི་བས་སྔ་བར་མི་འབྱུང་སྟེ། སྐྱོ་བ་ཤས་ཆུང་བས་ཆོས་དོན་དུ་མི་གཉེར་བས་སོ། ། བརྒྱ་པ་བས་འཕྱི་བར་མི་འབྱུང་སྟེ། ཚེ་དང་སེམས་ཅན་དང་ཉོན་མོངས་པ་དང་། ལྟ་བ་དང་། དུས་ཀྱི་སྙིགས་མ་ལྔ་བདེ་བས་སོ། ། ཡར་འཕེལ་ལ་མི་འབྱུང་སྟེ། མི་དགེ་བ་ངང་གིས་སྤོང་ཞིང་། དངོས་པོ་རྣམས་འཕེལ་བས་སྐྱོ་བ་མི་སྐྱེ་བས་སོ། ། ཇི་སྐད་དུ། མར་ནི་འགྲིབ་པ་ཡི། ། བརྒྱ་ཡི་བར་ལ་དེ་དག་འབྱུང་། ། ཞེས་པས་སོ། ། དེ་དག་ཀྱང་བསྐལ་པ་བཟང་པོ་འདིའི་རྣམ་གཞག་ལ་དགོངས་ཀྱི། ཐམས་ཅད་དུ་མི་བསམ་སྟེ། སྙིང་རྗེ་པད་མ་དཀར་པོར། འཇིག་རྟེན་གྱི་ཁམས་མཐེ་བོང་ཅན་ན་ཚེ་ལོ་བཅུ་པའི་དུས་སུ་མི་མཐེ་བོང་ཙམ་གྱི་ནང་དུ་སངས་རྒྱས་སྐར་མ་ལ་དགའ་བ་ཞེས་བྱ་བ་སྐུ་ཁྲུ་གང་བ་སོར་བདུན་གྱི་ཚད་ཙམ་པ་ཡོད་པར་བཤད་ཅིང་། བརྒྱད་ཁྲི་བས་སྔ་བའི་སངས་རྒྱས་མང་པོ་བཤད་པས་སོ། ། འདིའི་ཚེ་ལོ་བཞི་ཁྲིའི་དུས་སུ་འཁོར་བ་འཇིག། སུམ་ཁྲིའི་དུས་སུ་གསེར་ཐུབ། ཉི་ཁྲིའི་དུས་སུ་འོད་སྲུངས། བརྒྱ་པའི་དུས་སུ་ཤཱཀྱ་ཐུབ་པ། བརྒྱད་ཁྲི་པའི་དུས་སུ་བྱམས་པ་འབྱུང་ངོ་། ། མདོ་བསྐལ་བཟང་དུ་སངས་རྒྱས་རེ་རེའི་ཡུལ་དང་རིགས་དང་འོད་དང་ཡབ་ཡུམ་དང་སྲས་དང་རིམ་གྲོ་པ་དང་ཤེས་རབ་ཅན་དང་། རྫུ་འཕྲུལ་ཅན་གྱི་མཆོག་གཉིས་དང་། འཁོར་འདུས་པ་དང་། ཚེ་ཚད་དང་། བསྟན་པ་གནས་པའི་ཚད་དང་། སྐུ་གདུང་སྟེ། མཆོག་ཟུང་སོ་སོར་བགྲངས་པའི་དོན་བཅུ་གསུམ་བཅུ་གསུམ་གྱིས་བསྟན་ཅིང་། ཁ་ཅིག་གིས་གདུང་སྨོས་ཏེ་བཅུ་བཞིའོ། ། འཁོར་འདུས་པའང་ཁ་ཅིག་ལ་རེ་རེ་དང་། ཁ་ཅིག་ལ་མང་པོ་དང་སྐུ་གདུང་ལ་རྒྱས་པར་འགྱུར་བ་དང་། རིལ་པོ་གཅིག་ཏུ་འདུག་པ་གཉིས་སོ། ། སྟོན་པ་འདི་ལ་དཔེར་བརྗོད་ན། མཆོག་ཏུ་དགའ་བའི་རྒྱལ་པོ་ང་སྐྱེ་བའི་ཡུལ་ནི་སེར་སྐྱ་ཞེས་བྱའོ། །
