དགའ་ལ་ཆགས་པ་སྐྱེས་བུ་མཁས་བ་ཅིར་མི་རུང་། ། གལ་ཏེ་རྒ་མེད་ན་མེད་འཆི་བ་མེད་གྱུར་ཀྱང་། ། འོན་ཀྱང་ཕུང་པོ་ལྔ་འཛིན་སྡུག་བསྔལ་ཆེན་ཡིན་ན། ། རྒ་ནང་འཆི་བ་རྟག་ཏུ་འབྲང་བ་སྨོས་ཅི་དགོས། ། ལེགས་པར་སླར་ལོག་རབ་ཏུ་ཐར་བར་བྱ་བར་བསམ། ། ཡང་། བྱང་སྒོར་དགེ་སློང་གཟིགས་ཏེ་སྨྲས་པ། ཁ་ལོ་བསྒྱུར་བ་སྐྱེས་བུ་སེམས་ཞི་རབ་ཞི་ཞིང་། ། མིག་ཕབ་འགྲོ་ལ་གཉའ་ཤིང་གང་ཙམ་ལྟ་བྱེད་ཅིང་། ། ངུར་སྨྲིག་གོས་གྱོན་ཤིན་ཏུ་རབ་ཏུ་ཞི་བར་སྤྱོད། ། ལྷུང་བཟེད་ཐོགས་ནས་མི་རྒད་མ་ཁེངས་སུ་ཞིག་ཡིན། དེས་སྨྲས་པ། ལྷ་གཅིག་མི་འདི་དགེ་སློང་ཞེས་ནི་བགྱི་བ་ལགས། ། འདོད་པའི་དགའ་བ་སྤངས་ཤིང་ཤིན་ཏུ་དུལ་བར་སྤྱོད། ། བདག་ཉིད་ཞི་བ་ཚོལ་ཞིང་རབ་ཏུ་བྱུང་བར་གྱུར། ། འདོད་ཆགས་ཞེ་སྡང་བྲལ་བར་བསོད་སྙོམས་སྤྱོད་ཅིང་རྒྱུ། ། བྱང་ཆུབ་སེམས་དཔས་སྨྲས་པ། འདི་ནི་དགེ་བར་ལེགས་སྨྲས་ང་ཡང་འདོད་པ་སྟེ། ། རབ་ཏུ་བྱུང་བ་མཁས་པ་རྣམས་ཀྱིས་རྟག་ཏུ་བསྔགས། ། དེ་ནི་བདག་ལ་ཕན་ཞིང་སེམས་ཅན་གཞན་ལའང་ཕན། ། འཚོ་བ་བདེ་ཞིང་ཉམས་མཐུན་མཐུན་འབྲས་བུར་འགྱུར། ། དེ་ནས་བྱང་ཆུབ་སེམས་དཔའ་ནང་དུ་ཞུགས་སོ། ། རྒྱལ་པོས་ཀྱང་དེ་དག་མཐོང་ཐོས་པས། བསྲུང་བའི་ཕྱིར། ར་བ་དང་། འོབས་དང་། སྒོ་བརྟན་པོ་བཙུག། གྲོང་ཁྱེར་གྱི་བཞི་མདོར་བསྲུང་བའི་དབུང་བཞག། བཙུན་མོ་རྣམས་ཀྱང་གླུ་དབྱངས་དང་རྩེ་དགའ་སྦྱོར་དུ་བཅུག་གོ། དེའི་ཚེ་བྱང་ཆུབ་སེམས་དཔའ་རབ་ཏུ་འབྱུང་བའི་ལྟས་སུ། བྱ་སྐད་མི་འབྱིན། པདྨ་རྙིངས། ཤིང་ལ་མེ་ཏོག་མི་འཆགས། ། གླིང་བུ་དང་པི་ཝང་རྒྱུད་ཚད། རྫ་རྔ་ལ་སོགས་བརྡུངས་ཀྱང་། སྒྲ་མེད་པར་གྱུར་པས། རྒྱལ་པོ་ཡང་སེམས་ཁོང་དུ་ཆུད་ཅིང་འདུག་གོ།
