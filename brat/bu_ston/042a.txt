རྒྱལ་པོ་སྲས་དང་བཙུན་མོར་བཅས་པ་ཞུགས་ཏེ། ནམ་མཁའ་ལ་ཕྱིན་ཏེ་སངས་རྒྱས་ཀྱི་སྤྱན་སྔར་འཁོད་པ་དང་། བཅོམ་ལྡན་འདས་ཀྱིས་རྒྱལ་པོ་ཆེན་པོ་དད་པ་ནི་ཡོངས་སུ་མི་ཉམས་ཤིང་ཁྱད་པར་དུ་འགྲོ་བར་འགྱུར་བའོ་ཞེས་པ་ལ་སོགས་པའི་ཆོས་བསྟན་པ་མཉན་ནས་རང་གི་ཁྱིམ་དུ་ཕྱིན་པ་དང་། ཟླ་བ་ཉ་ལ་བབ་པ་དང་། བཙུན་མོ་མ་སྨད་པཔང་བར་ཁྱེའུ་ཆོས་སེམས་དང་། དཔེ་མེ་པའི་པང་བར་ཆོས་ཀྱི་བློ་གྲོས་གཉིས་རྫུས་ཏེ་བྱུང་ངོ་། ། དེ་གཉིས་དང་བཅས་ཏེ་སངས་རྒྱས་ལ་ཆོས་མཉན་ནོ། ། དེའི་ཚེ་རྒྱལ་པོ་འདི་སྙམ་དུ། རྒྱལ་བུ་འདི་ཐམས་ཅད་བྱང་ཆུབ་ཏུ་ཞུགས་པ་ཤ་སྟག་ཡིན་ན། གང་ཐོག་མར་སངས་རྒྱ་བརྟག་པར་བྱའོ་སྙམ་སྟེ། གཞོན་ནུ་དེ་ཐམས་ཅད་ཀྱི་མིང་བྲིས་ནས་རིན་པོ་ཆེ་སྣ་བདུན་གྱི་བུམ་པར་བཅུག་སྟེ། ཞག་བདུན་དུ་མཆོད་པ་བྱས་ཏེ། བཙུན་མོའི་འཁོར་དང་སྲས་སྟོང་དང་གཞོན་ནུ་གཉིས་ཀྱི་མདུན་དུ་ནང་མིས་བུམ་པའི་ནང་ནས་མིང་རྣམས་ཕྱུང་ངོ་། ། ཐོག་མར་གཞོན་ནུ་རྣམ་པར་དག་པའི་བློ་གྲོས་ཀྱི་མིང་བྱུང་བས། ས་གཡོས་ཤིང་རོལ་མོའི་སྒྲ་བྱུང་སྟེ། དེ་ནི་འཁོར་བ་འཇིག་གོ། དེ་ནས་གཞོན་ནུ་རྣམ་པར་རྒྱལ་བའི་སྡེ་གསེར་ཐུབ་བོ། ། དབང་པོ་ཞི་བ་འོད་སྲུངས་སོ། ། དོན་ཐམས་ཅད་དུ་གྲུབ་པ་ནི་ཤཱཀྱ་ཐུབ་པའོ། ། སྐ་རགས་ཅན་ནི་བྱམས་པའོ། ། མཆོག་གི་བློ་གྲོས་ནི་སེང་གེའོ། ། གློག་གི་ལྷ་ནི་ཏོག་ཅེས་བྱ་བའོ། ། བཟང་པོ་ནི་མེ་ཏོག་དམ་པའོ། ། འོད་ཀྱི་དཔལ་ནི་མེ་ཏོག་གོ། རྣམ་པར་སྣང་བའི་པད་མ་ནི་སྐར་རྒྱལ་ལོ། ། དྲི་མ་མེད་པའི་འོད་ནི་སྤྱན་ལེགས་སོ། ། རྡུལ་མེད་ནི་ལག་བཟང་ངོ་། ། བློ་གྲོས་རྒྱལ་པོ་ནི་འོད་དོ། ། དགེ་བ་བཀོད་པའི་རྒྱལ་པོ་ནི་སྐར་མ་ལ་དགའ་བའོ། །
