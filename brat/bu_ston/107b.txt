རྩོད་པའི་ཕྱིར་ནི་རབ་ཏུ་རྒྱུ། ། མཚོ་གསུམ་ས་ཡི་མཐར་ཐུག་པར། ། ཕ་རོལ་མུ་སྟེགས་རྩོད་ལ་དགའ། ། གང་ཕྱིར་དང་པོ་གཙོ་བོའི་ཚིག། སེམས་ཅན་རྣམས་ལ་ཕན་འདོད་པས། ། གཞོན་ནུས་ཡོངས་སུ་གསུངས་གྱུར་པ། ། ཡི་གེ་དྲུག་པའི་སྔགས་ཀྱང་ཟློས། ། ཆོ་ག་རབ་འབྱམས་འདི་དག་ལས། ། དེས་ནི་དེ་ཕན་དེ་ལ་བསྟན། ། རྒྱལ་དང་ལེགས་པར་རྒྱལ་བ་དང་། ། གྲགས་ལྡན་དགེ་བ་དེ་ལས་གཞན། ། རིགས་ལྡན་ཆོས་དང་ལྡན་པ་དང་། ། རྩོད་ལྡན་ལེགས་པར་ཁྱབ་འཇུག་དང་། ། སྦྲང་རྩི་སྦྲང་རྩི་བཟང་པོ་དང་། ། དེ་བཞིན་གྲུབ་དང་ལེགས་བྲལ་དང་། ། ར་གྷ་པ་དང་དམངས་རིགས་དང་། ། དེ་བཞིན་ཀ་ཤིར་སྐྱེས་པ་གཞན། ། གཞོན་ནུ་འདི་ཡི་བཟླས་པ་ལ། ། ཐམས་ཅད་ཀུན་ཡང་འབྱུང་བར་འགྱུར། ། དེ་དག་ཀུན་ཀྱང་བཟང་པོ་སྟེ། ། བློ་ལྡན་མང་དུ་ཐོས་པ་ཡིན། ། ཞེས་པ་དང་། ། དེ་འོག་རྣམ་པར་གྲགས་པ་ནི། ། བྲམ་ཟེ་དང་པོའི་ཡི་གེ་པི། ། གྲོང་ཁྱེར་མེ་ཏོག་ཅེས་བྱ་བར། ། དེ་ནི་ཁྲོ་བོ་གྲུབ་པར་འགྱུར། ། ཞེས་དང་། ། དེ་འོག་རྣམ་པར་གྲགས་པ་ནི། ། བྲམ་ཟེ་ཡི་གེ་དང་པོ་ཨ། ། ཆོས་དོན་དགེ་བ་ལྡན་བདག་ཉིད། ། བདེན་པར་སྨྲ་ཞིང་དབང་པོ་ཐུལ། ། ཞེས་དང་། ། དེ་འོག་རྣམ་པར་གྲགས་པ་ནི། ། བྲམ་ཟེ་ཆོས་དོན་སེམས་བྱེད་པ། ། དེ་ནས་མཐར་ནི་དང་ཡིག་ཤ། ། མ་ལའི་ཡུལ་དུ་དེ་ནི་འབྱུང་། ། ཞེས་དང་། ། དེ་འོག་རྣམ་པར་གྲགས་པ་ནི། ། ལྷོ་ཡི་ཕྱོགས་སུ་བྲམ་ཟེ་ནི། ། ཡི་གེ་དང་པོ་བ་བསྟན་པ། ། སྟོན་པའི་བསྟན་ལ་དེ་གཙོ་བྱེད། ། མཆོད་རྟེན་ཀུན་དགའ་ར་བ་གནས། ། སྟོན་པའི་གཟུགས་བརྙན་ཡིད་འོང་གིས། ། ས་ནི་རྒྱ་མཚོ་གཉིས་མཐར་ཐུག། ཐམས་ཅད་ཀྱི་ནི་རྒྱན་དུ་བྱས། ། དེ་འོག་རྣམ་པར་གྲགས་པ་ནི། ། བྲམ་ཟེའི་གཙོ་པོ་ནོར་ཆེན་ལྡན། ། དང་པོ་ཡི་གེ་ན་ཞེས་པ། ། ལྷོ་ཡི་ཕྱོགས་སུ་ཡང་དག་བསྟེན། ། བདག་ཉིད་ཆེན་པོ་སྔགས་ཟློས་ལྡན། །
