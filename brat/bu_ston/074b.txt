བདུད་བཅོམ་ནས་ནི་བྱང་ཆུབ་མྱུར་དུ་རེག་པར་འགྱུར། ། མྱ་ངན་ལས་འདས་ཞི་བ་བསིལ་བའི་དངོས་པོ་ཐོབ། ། ཅེས་ཆེད་དུ་བརྗོད་དོ། ། ལྷ་རྣམས་ཀྱིས་རྒྱ་ཆེར་བསྟོད་ནས་སྐྱིལ་ཀྲུང་མ་བཤིག་པར་ཞག་བདུན་དུ་བྱང་ཆུབ་ཤིང་ལ་གཟིགས་སོ། ། ཞག་བདུན་པ་གཉིས་པ་ལ། ། སྟོང་གསུམ་རྒྱང་རིང་དུ་འཆག་གོ། གསུམ་པ་ལ། ངས་འདིར་སྡུག་བསྔལ་མཐར་ཕྱུང་སྟེ། བྱང་ཆུབ་ཀྱི་སྙིང་པོ་ལ་སྤྱན་མི་འཛུམས་པར་གཟིགས་སོ། ། བཞི་པ་ལ་ཤར་ནུབ་ཀྱི་བར་དུ་རྒྱང་མི་རིང་བར་འཆག་གོ། དེ་ནས་བདུད་སྡིག་ཅན་གྱིས་མྱ་ངན་ལས་འདའ་བར་གསོལ་བ་བཏབ་ཀྱང་། ། གདུལ་བྱའི་དོན་བྱས་ཟིན་གྱི་བར་དུ་མི་འདའ་བར་ཞལ་ཀྱིས་བཞེས་ཏེ། མྱ་ངན་དུ་གྱུར་ཅིང་བུ་མོ་དགའ་མ་ལ་སོགས་པའང་དྲུང་དུ་སོང་བ་ན་རྒན་མོར་མཛད་དོ། ། ཞག་བདུན་པ་ལྔ་པ་ལ་ཡུལ་ངན་ཆེན་པོ་བྱུང་སྟེ། ཀླུ་བཏང་བཟུང་གི་གནས་ན་བཞུགས་སོ། ། དྲུག་པ་ལ་ཤིང་ནྱ་གྲོ་ངྷ་ར་སྐྱོང་དུ་བཞུད་དེ། ཀུན་ཏུ་རྒྱུ་རྣམས་ལ། ཆོས་ཐོས་པ་དང་མཐོང་བ་དང་། ། དབེན་པར་དགའ་པ་བདེ་བ་ཡིན། ། སྲོག་ཆགས་འབྱུང་པོ་རྣམས་ལ་བསྡམས། ། གནོད་པ་མི་བྱེད་འཇིག་རྟེན་བདེ། ། ཞེས་སོགས་གསུངས་སོ། ། བདུན་པ་ལ། ཤིང་སྤྲོལ་རྒྱའི་དྲུད་དུ་བཞུགས་ཏེ། ཆོང་པ་ག་གོན་དང་བཟང་པོས། སྤྲང་རྩི་དང་སྐྱོ་མ་དང་། བུ་རམ་ཤིང་བཤུས་པ་དག་ཕུལ་བ་དང་། ལག་པས་བླངས་ན་མ་ལེགས། སྔོན་གྱི་སངས་རྒྱས་ཀྱིས་སྣོད་ཇི་ལྟ་བུར་བཞེས་སྙམ་པ་དང་། རྒྱལ་ཆེན་བཞིས་གསེར་ལ་སོགས་པའི་ལྷུང་བཟེད་ཕུལ་བ་མ་བཞེས་པས། རྣམ་སྲས་ལ་སྔོན་པོའི་རིས་ཀྱི་ལྷས་བྱིན་པའི་རྡོའི་ལྷུང་བཟེད་བཞི་ཡོད་པ་རེ་རེས་བཟུང་སྟེ། མེ་ཏོག་གིས་བཀང་སྟེ་ཕུལ་བས། ཕྱག་བརྐྱངས་ཏེ། བདེ་བར་གཤེགས་ལ་ལྷུང་བཟེད་ཕུལ་ཅིག་དང་།
