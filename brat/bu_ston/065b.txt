སྙམ་སྟེ། སྡེ་བུད་མེད་ཀྱི་ཡོན་ཏན་ཡི་གེ་ལ་བྲིས་ཏེ། འདི་འདྲ་བ་བདག་ལ་གསོལ། ཞེས་པ་དང་། རྒྱལ་པོས་མདུན་ན་འདོན་ལ་ཁྱོད་སོང་ལ། གང་ན་ཡོན་ཏན་འདི་དག་ཡོད་པའི་བུ་མོ་དེ་འོན་ཅིག། རྒྱལ་རིགས་བྲམ་ཟེའི་བུ་མོ་དང་། ། རྗེའུ་རིགས་དམངས་རིགས་དེ་བཞིན་ཏེ། ། སུ་ལ་ཡོན་ཏན་འདི་ཡོད་པའི། ། བུ་མོ་དེ་ནི་འདིར་འོན་ཅིག། ང་ཡི་བུ་འདི་རིགས་དག་དང་། ། རུས་དང་ངོ་མཚར་འཛིན་མ་ཡིན། ། ཡོན་ཏན་བདེན་པ་ཆོས་ཀྱི་རྣམས། ། དེ་ལ་འདི་ཡི་ཡིད་དགའ་འོ། ། ཞེས་སྨྲས་སོ། ། དེ་ནས་བྲམ་ཟེ་དེས་ཡུལ་ཀུན་ཏུ་སོང་སོང་བས། ཤཱཀྱ་ལག་ན་བེ་ཅོན་ཅན་གྱི་བུ་མོ་ས་འཚོ་མ་མཐོང་ནས། ཡི་གེ་དེ་བྱིན་པ་དང་། བཞིན་འཛུམ་སྟེ། བྲམ་ཟེ་ཡོན་ཏན་འདི་དག་བདག་ལ་ཀུན་ཚང་གིས། དེས་པའི་གཟུགས་ཅན་དེ་ནི་བདག་གི་ཁྱིམ་ཐབ་བོ། ། གལ་ཏེ་གཞོན་ནུ་བཞེས་ན་ཕྱི་བཤོལ་མ་མཛད་གསུངས། ། བརྒྱ་ལ་སྐྱེ་བོ་ཕལ་དང་བཞུགས་པར་གྱུར་ཀྱང་གདའ། ཞེས་སྨྲས་སོ། ། དེམ་མདུན་ན་འདོན་གྱིས། རྒྱལ་པོ་ལ་གསོལ་པས། རྒྱལ་པོས། བུད་མེད་འདི་རྫུན་ཆེ་བས་ཡིད་མ་ཆེས་ཏེ། ཞག་བདུན་ན་བུ་མོ་ཐམས་ཅད་འདུན་ཁང་དུ་ཤོག་ཅིག། གཞོན་ནུས་ཡིད་དུ་འོང་བའི་ཡོ་བྱད་དག་སྦྱིན་ནོ། ། ཞེས་སྨྲས་ཏེ། དགུང་བདུན་ནས་གཞོན་ནུས་འདུན་ཁང་དུ་སོང་བ་ལ། བུ་མོ་རྣམས་ཇི་ལྟར་འོངས་བཞིན་ཡོ་བྱད་དག་བྱིན་ཞིང་དེ་དག་གིས་ཀྱང་གཟི་བརྗིད་མ་བཟོད་པར་མྱུར་དུ་སོང་ངོ། ། དེ་ནས་ས་འཚོ་མ་འོངས་པ་དང་ཡོ་བྱད་ཟད་དེ་དེས་འཛུམ་དམུལ་ཏེ། བདག་གིས་ནོངས་པ་ཅི་བགྱིས་ནས། བདག་ལ་རྔན་སྡང་མཛད། གཞོན་ནུས། རྔན་སྡང་མིན་གྱི་ཁྱེད་འཕྱིས་པས་ལན་ནོ། ། གསུངས་ནས་སོར་གདུབ་སྲང་བརྒྱ་སྟོང་རི་བ་དེ་བྱིན་ནོ། །
