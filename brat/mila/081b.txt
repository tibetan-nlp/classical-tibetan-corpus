སྔ་ནས་དད་བཞིན་པ་སྐྱོ་སྟོན་ཤཱཀྱ་གུ་ན་ལ་དབང་དང་གདམས་པ་གནང་ནས་སྨིན་གྲོལ་ལ་བཀོད། ། དེ་ནས་བྱང་རྟ་སྒོ་ལ་བྱོན་ཚེ་ཕར་རེས་གཅུང་གི་གད་པ་སླེ་གསུམ་དུ་ཉ་མ་དཔལ་དར་འབུམ་དང་མཇལ། ཚུར་རེས་གཡས་རུ་བྱང་གི་མགྲོན་ཁང་ཞིག་ཏུ་སེ་བན་རས་པ་དང་མཇལ། དེ་ནས་ལ་སྟོད་རྒྱལ་གྱི་ཤྲཱི་ལ་སྒོམ་དུ་བྱོན་པས་འབྲི་སྒོམ་རས་པ་དང་མཇལ། དེ་ནས་སྟོན་རོ་སྙོམས་ཀྱི་ལྡོམ་བུ་ལ་བྱོན་པས་ཆུ་མིག་དངུལ་འབུམ་དུ་རས་པ་ཞི་བ་འོད་དང་མཇལ། དེ་ནས་ལྕིམ་ལུང་དུ་སྦ་ལྕག་གི་སྐོར་ལ་ངན་རྫོང་རས་པ་དང་མཇལ། དེ་ནས་ལ་ཕྱི་ལ་བཞུགས་པའི་ཚེ། བླ་མས་ལུང་བསྟན་ལྟར་མཁའ་འགྲོས་བསྐུལ་ཏེ། ཏཱི་སེ་ལ་བྱོན་པའི་ལམ་དུ་དམ་པ་རྒྱགས་ཕུ་བ་དང་མཇལ། དེ་ནས་ཡར་བྱོན་ཏེ། གློ་བོ་ཀོ་རེ་ལ་ཐོག་གི་སྐོར་ལ་མཁར་ཆུང་རས་པ་དང་མཇལ། དེ་ནས་དགུན་པུ་རངས་ཀྱི་རོང་། འབྲི་རྩེའི་གངས་འདབས་སུ་བཞུགས་དུས་རས་པ་དར་མ་དབང་ཕྱུག་དང་མཇལ་ཏེ། སོ་སོ་ཏཱི་སེ་ལ་བྱོན་ནས་ནཱ་རོ་བོན་ཆུང་རྫུ་འཕྲུལ་གྱིས་ཟིལ་གྱིས་མནན་པ་ཏི་སེའི་སྐོར། དེ་ནས་མར་བྱོན་སླར་ཡང་བྲག་སྐྱ་རྡོ་རྗེ་རྫོང་དུ་བཞུགས་པས་རོང་ཆུང་རས་པ་དང་མཇལ། དེ་ནས་མཁའ་འགྲོས་བསྐུལ་ཏེ་བྱོན་པའི་ལམ་དུ་སྦས་ཕུག་མ་མོ་རྫོང་གཟིགས། དེར་ཞག་ཤས་བཞུགས་པས་རྫི་བོ་ཞབས་ཕྱིར་འབྲངས་པ་ལ་ལུག་རྫི་རས་པ་བྱ་བའི་རྟོགས་ལྡན་བཟང་པོ་ཞིག་བྱུང་། དེ་ནས་ལ་ཕུག་པདྨ་རྫོང་དུ་གཤེན་སྒོམ་རས་པ་དང་མཇལ་ཏེ། ཕྱིས་གླང་སྒོ་ཀླུ་བདུད་རྫོང་དང་། སྦལ་ཕུག་མ་མོ་རྫོང་དུ་བཞུགས་ཚེ་ཞབས་ཏོག་ཀྱང་ཕུལ་དུ་ཕྱིན་པ་ཕུལ་བ་ཡིན། ། དེ་ནས་ཅོ་རོ་འབྲིག་མཚམས་ལ་བྱོན་པའི་སྐོར་ལ་རས་ཆུང་མ་དང་མཇལ། ། དེ་ནས་མོན་གཉི་ཤངས་འགུར་རྟ་ལ་བཞུགས་ཚེ་ཁྱི་ར་རས་པ་དང་མཇལ།
