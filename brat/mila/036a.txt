རོ་ཐམས་ཅད་སོས་ནས་ཧར་ལངས་ཏེ། སྐད་ཅིག་དེ་ཉིད་ལ་གནམ་ལ་འཕུར་དགུ་དང་ས་ལ་འཛུལ་དགུ་རྣམས་རང་རང་སོ་སོའི་གནས་སུ་སོང་བ་དང་། བླ་མ་སངས་རྒྱས་དངོས་སུ་མཐོང་སྟེ། རེ་རང་དགའ་ན་རེ་རང་དགའ་ན། ད་ལན་སེམས་ཅན་མང་པོ་རང་ཅིག་ཤི་ན་འཚེང་སྟེ་སྙམ་པ་བྱུང་། དེ་ནས་དགྱེས་པ་རྡོ་རྗེའི་དཀྱིལ་འཁོར་དུ་དབང་བསྐུར། གདམས་ངག་གནང་ནས། བླ་མའི་གདན་ས་མཐོང་བའི་གྲོག་ལྡན་ཁ་ལྷོར་ལྟ་བ་ཅིག་ན་ཕུག་རོན་ཞིག འདུག་པ་དེ་གསོས་ཏེ། འདག་སྦྱར་ལ་བསྡད་པའི་བུག་སྒོར། བླ་མས་ཆོས་གནང་ཞིང་ཡེངས་མེད་དུ་བསྒོམས་རུང་། མར་པའི་བཀས་མ་གྲོལ་བས་ཉམས་མྱོང་ཅི་ཡང་མ་སྐྱེས་པ་ལ། ལན་ཅིག་བླ་མའི་ཞལ་ནས། ཨ་བོ་མཐུ་ཆེན། རྟགས་འདི་འདྲ་དང་འདི་འདྲ་བྱུང་ངེ་གསུང་། ཅི་ཡང་མ་བྱུང་ལགས་ཞུས་པས། ཅི་ཟེར་ངའི་བརྒྱུད་པ་འདི་ལ་དམ་ཚིག་གི་སེལ་མེད་ན་ཉམས་རྟོགས་ཀྱི་ཡོན་ཏན་མྱུར་དུ་མི་སྐྱེ་བའི་མཐུ་མེད་པ་ཅིག་ཡིན་ཏེ། ཁྱེད་རང་ནི་ང་ལ་དད་ནས་འོངས་པ་ཡིན། བླ་མའི་བཀས་མ་གྲོལ་ན་ནི་ཕྱག་རྟགས་གནང་དོན་ཡང་མེད་ཅི་ཡིན་ནམ་གང་ལྟར་ནའང་སྙིང་རུས་བསྐྱེད་ལ་བསྒོམས་ཤིག་གསུང་པས། འཇིགས་ཆུམ་པ་ཅིག་བྱུང་སྟེ། དངོས་སྣང་ཞུའམ་སྙམ་པ་ལ། ཞུ་རང་མ་སྤོབས་པར། བླ་མ་མར་པའི་ཐུགས་ཟིན་པ་ཅིག་ཅིས་ཀྱང་དགོས་པ་འདུག་སྙམ་པ་བྱུང་སྙིང་རུས་བསྐྱེད་ནས་སྒོམ་ཞིང་བཞུགས་པའི་ཚེ། བླ་མ་མར་པས་སྲས་མཁར་གྱི་འཕྲོ་ལོངས་ནས། རྔོག་པའི་སར་གསུང་ཡིག་བྱུང་བ་ལ། འདིར་ངེད་ཀྱི་སྲས་མཁར་ལ་སྤེན་བད་དགོས་པ་བྱུང་བས། ཁྱེད་ཀྱིས་དེ་ནས་སྤེན་ཁུར་གྱི་འབུལ་བ་ཅི་ལྕོགས་ཅིག་སྔོན་ལ་ཁྱེར་ཤོག སྤེན་བད་དང་འགའ་རྫི་ར་ཁེལ་བ་དང་། མཁར་གྱི་རབ་གནས། མདོ་སྡེ་འབུམ་གྱི་ཆེ་འདོན་རྣམས་དུས་མཉམ་དུ་བྱེད་པ་ཡོད་པས། དེ་ལ་ཁྱེད་རང་ཡང་ཤོག དེ་ན་ངེད་ཀྱི་མི་ངན་དེ་ཡོད་འདུག་པ། དེ་ཡང་དེ་དུས་ཁྲིད་ཤོག་གསུང་བ་ཅིག་འདུག་པས།
