འོ་ན་ཁྱེད་ཀྱི་ཆ་ལུགས་སྤྲང་པོ་བས་ངན་པ་འདི་འདྲའི་ཆོས་པ་སྔར་མ་མཐོང་། འདི་ཐེག་པ་གང་གི་ཆ་ལུགས་ཡིན་ཟེར་བས། འདི་ཐེག་པ་ཀུན་གྱི་མཆོག་རབ་ཚེ་འདིར་སངས་རྒྱས་སྒྲུབ་ཕྱིར་ཆོས་བརྒྱད་རླུང་ལ་བསྐུར་བ་ཞེས་བྱ་བའི་ཐེག་པའི་ཆ་ལུགས་ཡིན་བྱས་པས། མོ་ན་རེ། ཁྱེད་དང་ཁོང་ཚོའི་ཆ་ལུགས་དང་སྤྱོད་པ་ཀུན་ཁྱེད་རང་ཟེར་བ་བཞིན་རྒྱབ་ལོག་རང་དུ་འདུག་པས་གང་རུང་ཅིག་ཆོས་མིན་པ་འདྲ། ཡིན་པ་མཉམ་པོ་ལ་ཁོང་ཚོའི་དེ་དགའ་བར་འདུག་ཟེར་བ་ལ། ངས་ཁྱོད་འཇིག་རྟེན་པ་དགའ་བའི་ཆོས་པ་དེ་ལ་ང་མི་དགའ། ང་དང་དོན་འཐུན་པའི་ཆོས་པ་ན་བཟའ་ངུར་སྨྲིག་གསོལ་བ་རྣམས་ལའང་ཆོས་བརྒྱད་ཀྱི་ཆ་བག་ཙམ་རེ་ཡོད་པ་འདྲ། མེད་རུང་སངས་རྒྱས་ཐོབ་པ་མྱུར་མི་མྱུར་ལ་ཁྱད་པར་བསམ་གྱིས་མི་ཁྱབ་པ་ཅིག་ཡོད་དེ་དེ་ཁྱོད་ཀྱིས་མི་གོདེ་བས་ཁྱོད་རང་ཡང་ནུས་ན་ཆོས་གྱིས། མི་ནུས་ན་ཁང་ཞིང་འཁྱེར་ལུགས་སྔར་ལྟར་གྱིས་ལ་སོང་བྱས་པས། མོ་ན་རེ། ང་ལ་ཁྱེད་ཀྱི་ཁང་ཞིང་མི་དགོས། སྲིང་མོ་ལ་བྱིན་མཛོད། ཆོས་ཤིག་བྱེད་ཏེ་ཁྱེད་ཀྱི་འདི་འདྲ་མི་ནུས་པར་གདའ་ཟེར་སོང་ངོ་། ། དེ་ནས་ཨ་ནེས་ང་ཁང་ཞིང་ལ་རྩིགས་མེད་པ་ཐོས་ནས། ཞག་འགའ་སོང་བ་དང་། ཨ་ནེའི་བསམ་པ་ལ་ཁོ་བླ་མའི་བཀའ་བཞིན་བྱེད་ཟེར་བས་ཞིང་འདི་ཨེ་ཁུགས་ལྟ་སྙམ་། ངའི་སར་བཙམ་ཆང་ཆུ་སྡོར་མཚོན་པ་ཅིག་ཁྱེར་ནས་བྱུང་སྟེ། ཁ་སང་ངས་མི་ཤེས་པ་ཞིག་བྱས་འདུག་སྟེ། ཚ་བོ་ཆོས་པ་ཡིན་པས་བཟོད་པར་མཛོད། ད་ཨ་ནེས་ཁྱེད་རང་གི་ཞིང་བཏབ་ནས་བརྒྱགས་སྐྱེལ་ཟེར་བ་ལ། འོ་ན་ཨ་ནེས་ཟླ་བ་རེ་ལ་ནས་ཁལ་རེའི་ཕྱེ་སྐྱོལ། དེ་ལྷག་རྣམས་ཨ་ནེ་ཁྱེར་ལ་ཞིང་ཐོབ་ཅིག་བྱས་པས། དེ་ལྟར་བྱེད་ཟེར་ཟླ་བ་གཉིས་ཙམ་ཆད་དོན་བཞིན་བསྐྱལ།
