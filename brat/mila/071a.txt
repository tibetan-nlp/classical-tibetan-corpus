ང་རང་ལུས་རི་བོའི་དགོན་པ་ན། ། བྱང་ཁོག་བསྟོད་ཀྱི་ལྷ་ཁང་ན། ། ཟུར་གསུམ་སྙིང་གི་དབུ་རྩེན། ། སེམས་ཀྱི་རྟ་ཕོ་རླུང་ལྟར་འཕྱོ། ། རྟ་འཛིན་ན་ཞགས་པ་གང་གིས་འཛིན། ། གཏོད་ན་ཕུར་བ་གང་གིས་གཏོང་། ། ལྟོགས་ན་ཟས་སུ་ཅི་ཞིག་སྟེར། སྐོམ་ན་ཆུ་རུ་ཅི་ཞིག་ལྡུད། ། འཁྱགས་ན་ར་བ་གང་དུ་འཛུད། ། རྟ་འཛིན་ན་གཉིས་མེད་ཀྱི་ཞགས་པས་འཛིན། ། གཏོང་ན་ཏིང་འཛིན་གྱི་ཕུར་བུས་གཏོང་། ། ལྟོགས་ན་བླ་མའི་གདམས་ངག་སྟེར། སྐོམ་ན་དྲན་པའི་ཆུ་རྒྱུན་ལྡུད། ། འཁྱགས་ན་སྟོང་ཉིད་ཀྱི་ར་བར་འཇུག ། ཐབས་ཤེས་གཉིས་ཀྱི་སྒ་སྲབ་སྤྲད། ། འགྱུར་མེད་བརྟན་པའི་གོང་སྨེད་བཅུག ། སྲོག་རྩོལ་རླུང་གི་འཐུར་འདའ་རྒྱུས། ། དེ་ལ་རིག་པའི་ཁྱེ་ཆུང་ཞོན། ། ཐེག་ཆེན་སེམས་བསྐྱེད་རྒྱ་རྨོག་གྱོན། ། ཐོས་བསམ་སྒོམ་པའི་འཇོལ་ཁྲབ་གྱོན། ། བཟོད་པའི་ཕུབ་ཆུང་རྒྱབ་ཏུ་ཁུར། ། ལྟ་བའི་མདུང་རིངས་ལག་ཏུ་ཐོགས། ། ཤེས་རབ་ཀྱི་རལ་གྲི་ལྔོ་ལ་བཏགས། ། ཀུན་གཞི་འཇམ་པོའི་སྨྱུག་མ་ལ། ། ཁོང་ཁྲོ་མེད་པའི་འཁྱོག་ཅིག་བསྲང་། ། ཚད་མེད་བཞི་ཡི་སྒྲོ་ཞིག་བསྐྱོན། ཤེས་རབ་རྣོན་པོའི་མདེའུ་བཙུགས། ། སྟོང་པ་ཉིད་ཀྱི་གཞུ་མོ་ལ། ། ཐབས་ལམ་ཟབ་མོས་ལྟོང་སྦྱར་ནས། ། ཟུང་འཇུག་ཡངས་པའི་མདོམ་ལ་འཇལ། ། མདའ་འཕེན་པ་རྒྱལ་ཁམས་ཡོངས་ལ་འཕེན། ། ཕོག་པ་དད་པ་ཅན་ལ་ཕོག ། ཤི་བ་ངར་འཛིན་གྱི་འགོང་བོ་ཤི། ། དེ་ཕྱིར་དགྲ་ནི་ཉོན་མོངས་འདུལ། ། གཉེན་ནི་འགྲོ་དྲུག་སེམས་ཅན་སྐྱོང་། ། རྟ་རྒྱུག་ནི་བདེ་ཆེན་ཐང་ལ་རྒྱུག ། སྙོག་ནི་རྒྱལ་བའི་གོ་འཕང་སྙོག ། ཀྱུས་སྨད་འཁོར་བའི་གཞི་རྩ་བཅད། ། ཀྱུས་སྟོད་བྱང་ཆུབ་སྐམ་སར་སླེབ། ། རྟ་དེ་ལྟར་བརྒྱུགས་པས་སངས་རྒྱས་ཐོབ། ། ཁྱེད་ཀྱི་སྐྱིད་དང་འདྲ་འམ་ལྟོས། ། འཇིག་རྟེན་གྱི་སྐྱིད་ལ་སྨོན་རྒྱུ་མེད། ། བྱས་པས་ཁོང་རྣམས་དད་པར་གྱུར་ནས་སོང་ངོ་། །
