ལྷོ་བྲག་མར་པའི་ཞབས་ལ་འདུད། ། སྤྲང་རི་ཁྲོད་ཟིན་པར་བྱིན་གྱིས་རློབས། ། ཡོན་བདག་བཟང་པོའི་ཚོགས་སོག་ལ། ། དོན་གཉིས་འགྲུབ་པའི་རྟེན་འབྲེལ་འགྲིག ། རྙེད་དཀའ་འཇིག་སླའི་ལུས་པོ་འདི། ། ཟས་དང་ཕྲད་པས་སྐུ་ཁམས་བཟང་། ། དོག་མོ་ས་ཡི་རྩི་བཅུད་དང་། ། དགུང་སྔོན་དབྱིངས་ཀྱི་སྦྲང་ཆར་གཉིས། ། འགྲོ་ལ་ཕན་པའི་རྟེན་འབྲེལ་ཡིན། ། རྟེན་འབྲེལ་གྱི་སྙིང་པོ་ལྷ་ཆོས་ཡིན། ། ཕ་མས་གསོས་པའི་སྒྱུ་ལུས་དང་། ། བླ་མ་དམ་པའི་གདམས་ངག་གཉིས། ། ལྷ་ཆོས་བྱེད་པའི་རྟེན་འབྲེལ་ཡིན། ། རྟེན་འབྲེལ་གྱི་སྙིང་པོ་སྙིང་རུས་ཡིན། ། མི་མེད་ལུང་བའི་བྲག་ཕུག་དང་། ། ཁ་ཞེ་མེད་པའི་དགེ་སྦྱོར་གཉིས། ། ཅི་བས་མས་གྲུབ་པའི་རྟེན་འབྲེལ་ཡིན། ། རྟེན་འབྲེལ་གྱི་སྙིང་པོ་སྟོང་ཉིད་ཡིན། ། མི་ལ་རས་པའི་སྒོམ་སྲན་ན་དང་། ། ཁམས་གསུམ་སེམས་ཅན་དད་པ་གཉིས། ། འགྲོ་དོན་འབྱུང་བའི་རྟེན་འབྲེལ་ཡིན། ། རྟེན་འབྲེལ་གྱི་སྙིང་པོ་སྙིང་རྗེ་ཡིན། ། བྲག་ལ་སྒོམ་པའི་སྒོམ་ཆེན་དང་། ། འཚོ་བ་སྦྱོང་བའི་ཡོན་བདག་གཉིས། ། སངས་མཉམ་པོར་རྒྱ་བ་རྟེན་འབྲེལ་ཡིན། ། རྟེན་འབྲེལ་གྱི་སྙིང་པོ་བསྔོ་བ་ཡིན། ། བླ་མ་བཟང་པོའི་ཐུགས་རྗེ་དང་། ། སློབ་མ་བཟང་པོའི་སྒོམ་སྲན་གཉིས། ། བསྟན་པ་འཛིན་པའི་རྟེན་འབྲེལ་ཡིན། ། རྟེན་འབྲེལ་གྱི་སྙིང་པོ་དམ་ཚིག་ཡིན། ། བྱིན་རླབས་མྱུར་པོའི་དབང་བསྐུར་དང་། ། དད་གུས་དྲག་པོའི་གསོལ་འདེབས་གཉིས། ། མྱུར་དུ་མཇལ་བའི་རྟེན་འབྲེལ་ཡིན། ། རྟེན་འབྲེལ་གྱི་སྙིང་པོ་བཀྲ་ཤིས་ཡིན། ། རྗེ་མི་བསྐྱོད་ཀྱི་ངོ་བོ་རྡོ་རྗེ་འཆང་། ། སྤྲང་པོའི་སྐྱིད་སྡུག་ཁྱེད་རང་མཁྱེན། ། ཅེས་པའི་གླུ་བླངས་སྟེ། དེ་ནས་ཡང་སྙིང་རུས་བྱས་བསྒོམས་པས། ཉིན་སྣང་ལ་ལུས་འདོད་དགུར་བསྒྱུར་བ་དང་། མཁའ་ལ་ལྡིང་བ་ལ་སོགས་ཏེ་རྫུ་འཕྲུལ་ཡ་མ་ཟུང་དཔག་མེད་སྟོན་ནུས་པའི་ཉམས་ཤར།
