བླ་མ་རང་ནི་སྐུ་ན་བགྲེས་གདའ་བས། ད་བཀའ་བརྒྱུད་ཀྱི་བསྟན་པ་རིན་པོ་ཆེ་ཇི་ལྟར་དར་བར་འགྱུར་བ་དང་། བདག་ཅག་སློབ་མ་རྣམས་ལ་གདུལ་བྱ་དང་ཕྲིན་ལས་ཇི་ལྟར་འབྱུང་བའི་ལུང་བསྟན་ཅིག་གནང་བར་ཞུ་ཞུས་པས། བླ་མའི་ཞལ་ནས། ང་ནཱ་རོ་པཎ་ཆེན་ཀྱི་བུ་བརྒྱུད་འདི་རྟེན་འབྲེལ་དང་རྨི་ལམ་ལ་མངའ་བརྙེས་པ་ཡིན་ཞིང་། བཀའ་བརྒྱུད་ཀྱི་བསྟན་པ་ལ་པཎ་ཆེན་ནཱ་རོའི་ལུང་བསྟན་བཟང་པོ་ཡང་ཡོད། ། ཁྱེད་བུ་ཆེན་རྣམས་ཀྱིས་རྨི་ལམ་བསྒུགས་ལ་ཤོག་ཅིག་གསུང་། དེར་བུ་ཆེན་རྣམས་ཀྱིས་རྨི་ལམ་བཟུང་ནས་ཞུས་པས། ཀུན་ལ་རྨི་ལམ་བཟང་པོ་འགའ་བྱུང་འདུག་ནའང་། ལུང་བསྟན་ཕོག་པ་མ་བྱུང་སྟེ། ང་ལ་ཀ་ཆེན་བཞིའི་རྨི་ལམ་བྱུང་ཚུལ་རྣམས་བླ་མའི་དྲུང་དུ་ཕུལ་བ་འདི་ལྟར་རོ། ། ། ། རྗེ་རྡོ་རྗེ་འཆང་གི་བཀའ་བཞིན་དུ། ། འདང་གསུམ་ཉལ་བའི་རྨི་ལམ་། ། བྱུང་ཚུལ་བླ་མའི་དྲུང་དུ་འབུལ། ། སྙན་དར་ཅིག་བདག་ལ་གསན་པར་ཞུ། ། འཛམ་གླིང་ཡངས་པའི་བྱང་ཕྱོགས་འདིར། ། ལྷུན་སྟུག་གངས་རི་ཆགས་པ་རྨི། ། གངས་རྩེ་མོ་དགུང་ལ་རེག་པ་རྨི། ། རྩེ་ལ་ཉི་ཟླ་འཁོར་བ་རྨི། ། འོད་ཀྱིས་ནམ་མཁའ་གང་བ་རྨི། ། རྩ་བས་ས་གཞི་ཁྱབ་པ་རྨི། ། ཕྱོགས་བཞིར་ཆུ་རྒྱུན་འབབ་པ་རྨི། ། ཆུ་དེས་འགྲོ་ཀུན་ཚིམ་པར་རྨི། ། ཆུ་སྣ་རྒྱ་མཚོར་འབབ་པར་རྨི། ། མེ་ཏོག་སྣ་ཚོགས་བཀྲ་བར་རྨི། ། སྤྱིར་གྱིས་རྨི་ལམ་དེ་ལྟར་རྨི། ། བླ་མ་དུས་གསུམ་སངས་རྒྱས་ལ་ཞུས་པ་ལགས། ། སྒོས་ལྷུན་སྟུག་གངས་རི་མཐོན་པོ་། ། ཤར་དུ་ཀ་ཆེན་བཙུགས་པར་རྨི། ། ཀ་ཐོག་ཏུ་སེང་ཆེན་འགྱིང་བ་རྨི། ། སེང་གཡུ་རལ་མཐའ་ནས་རྒྱས་པ་རྨི། ། སྡེར་བཞི་གངས་ལ་བགྲད་པ་རྨི། ། སྤྱན་མིག་གྱེན་དུ་གཟིགས་པ་རྨི། ། སེང་གངས་དཀར་གྱི་ལྟོངས་སུ་འཕྱོ་བ་རྨི། ། བླ་མ་དུས་གསུམ་སངས་རྒྱས་ལ་ཞུས་པ་ལགས། ། ལྷོ་རུ་ཀ་ཆེན་བཙུགས་པ་རྨི། ། ཀ་ཐོག་ཏུ་སྟག་མོ་ངར་བ་རྨི། ། སྟག་གྲ་འཛུམ་མཐའ་ནས་རྒྱས་པ་རྨི། །
