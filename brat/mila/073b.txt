ཁྱད་པར་ང་ལ་ལྷོ་བྲག་མར་པའི་བཀའ་ལུང་ལ། འཇིག་རྟེན་གྱི་ཆོས་བརྒྱད་འདུ་འཛིའི་གཡེང་བ་སྤོངས། གྱོང་ལྟོ་གོས་གཏམ་གསུམ་ལ་ཐོངས། གནས་ངེས་མེད་ཀྱི་དབེན་པ་འགྲིམས་ལ། ཚེ་འདི་བློས་བཏང་གི་བསམ་སྦྱོར་དྲག་པོས་སྒྲུབ་པ་གཙོ་བོར་ཐོན་ཅིག་གསུང་པ་ཡོད་པས་བླ་མའི་བཀའ་སྒྲུབ་པ་ཡིན། བཀའ་བཞིན་བསྒྲུབས་པས། རང་གི་འཁོར་དུ་བྱུང་བའི་སེམས་ཅན་འགའི་ཚེ་འདིའི་བདེ་སྐྱིད་ཙམ་དུ་མ་ཟད། རང་གཞན་སེམས་ཅན་ཀུན་གྱི་ནམ་གཞག་གཏན་གྱི་བདེ་སྐྱིད་ཡོད་པ་དང་། ནམ་འཆི་ངེས་མེད་དུ་འདུག་པས་ཚེ་འདིའི་བྱ་བ་འཇིག་རྟེན་ཆོས་བརྒྱད་ཀྱི་ཆེ་ཐབས་སྤངས་པ་ཡིན། བསྒྲུབས་ན་བླ་མ་བ་རི་བའི་ཕྱགས་ཕྱི་བ་ཞན་ཤོས་སུ་མ་ཟད། ཁོང་རང་ཙམ་ཞིག་ངས་ཀྱང་ཡོང་སྟེ། ཚེ་འདིར་སངས་རྒྱས་ཐོབ་པར་འདོད་ནས་སྒྲུབ་པ་ལ་ནན་ཏན་དང་བརྩོན་འགྲུས་བྱས་པ་ཡིན། ད་པེ་ཏ་རང་ཡང་ཆོས་བརྒྱད་སྤོངས་ལ་ཨ་ཇོའི་ཕྱི་བཞིན་ལ་ཕྱི་གངས་ལ་སྒོམ་དུ་འགྲོ། ཆོས་བརྒྱད་བློས་བཏང་ནས་སྒོམ་ནུས་ན་འདི་ཕྱི་གཉིས་ཀར་དུ་སྐྱིད་པའི་ཉི་མ་ཤར་བ་ཡིན་པས། ཨ་ཇོའི་གླུ་འདི་ཉོན་ཅིག་བྱས་གླུ་འདི་བླངས་སོ། ། ། ། རྗེ་དུས་གསུམ་སངས་རྒྱས་འགྲོ་བའི་མགོན། ། ཆོས་བརྒྱད་སྐྱོན་གྱིས་མ་གོས་ལྟར། ། བུ་བརྒྱུད་བྱིན་གྱིས་རློབ་མཛད་པ། ། སྒྲ་བསྒྱུར་མར་པའི་ཞབས་ལ་འདུད། ། ཚེ་འདིའི་འདོད་པས་གདུངས་པ་ཡི། ། སྲིང་མོ་ཉོན་ཅིག་པེ་ཏ་ལྕམ། ། གདུགས་ཡ་ཏོག་གསེར་གྱི་འགའ་རྗི་ར་བཀྲག་དང་གཅིག ། མ་ཐོག་རྒྱ་མོ་དར་གྱི་ཤམ་བུ་འགྲིག་དང་གཉིས། ། བར་གདུགས་རྩིབས་རྨ་བྱའི་ལྡིང་ཁང་མཛེས་དང་གསུམ། ། སེང་ལྟེང་དམར་པོའི་ཡུ་བ་འདྲོངས་དང་བཞི། ། འདི་བཞི་བཤམས་ན་ཨ་ཇོས་ཤོམས་ཙམ་གདའ། །
