གྲྭ་བུ་སློབ་རྣམས་ཀྱིས་དུང་ཁ་སྦྱོར་དུ་བུས་པས་མི་མང་པོ་འདུས་ནས། འབུལ་བ་བསམ་གྱིས་མི་ཁྱབ་པ་ཡོང་བས། འཁོར་དང་ཉེ་ཚན་ལ་ཕན་པ། མི་ལ་ངོམ་པས་ཆོག་པའི་ཆོས་པ་བཟང་པོ་རང་ཞིག་འདུག དེའི་ཕྱགས་ཕྱི་ལ་ཨེ་འཁྲིད་ལྟོས་དང་། ཁྲིད་ན་གྲྭ་པ་ཞན་ཤོས་ཤིག་ཐོབ་རུང་སྐྱིད་ཕོ་ཡོང་བ་འདུག་གིས། དེ་མིན་ཨ་ཇོའི་ཆོས་འདི་དང་། ངའི་འཁོས་ཀ་འདིས་འོ་ཅག་མིང་སྲིང་གི་ཚེ་མི་ཕྱིད་པར་འདུག་ཟེར་ངུ་ཞིང་འདུག་པས། ངས་པེ་ཏ་དེ་སྐད་མ་ཟེར་ཞིག ཁྱོད་དག་ང་གོས་མེད་པ་ཆ་ལུགས་ལྡུག་པར་གནས་པ་འདི་ལ་ངོ་ཚར་བྱེད་པར་འདུག་སྟེ། ངའི་འདི་སྤྱིར་ཡོད་པ་རང་གིས་འཚེང་སྟེ་ཆོས་དང་མཇལ་བ་ཡིན་པས་ངོ་མི་ཚ། ཁྱད་པར་ཨ་མ་ལས་སྐྱེས་ཙམ་ཉིད་ན་ཡོད་པས་ངོ་ཚ་རྒྱུ་མེད། དེ་བས་སྤྱིར་སྡིག་པར་ངོ་ཤེས་ནས་མི་འཛེམ་པར་ཕ་མའི་ཐུགས་དཀྲུག བླ་མ་དང་དཀོན་མཆོག་གི་དཀོར་ལ་འབག རང་འདོད་སྒྲུབ་ཕྱིར་ངན་གཡོ་སྣ་ཚོགས་ཀྱི་སྒོ་ནས་སེམས་ཅན་ལ་གནོད་འཚེ་བྱེད་པ་རྣམས་རང་གཞན་ཀུན་ལ་གནོད་ཅིང་། ལྷ་མི་དམ་པ་ཀུན་གྱིས་ཐུགས་ཀྱིས་ཁྲེལ་བས་འདི་ཕྱི་གཉིས་ཀར་དུ་ངོ་ཚ། ཁྱད་པར་ལུས་ལ་ངོ་ཚ་ན། ཁྱོད་རང་ཨ་མ་ལས་སྐྱེས་དུས་མེད་པའི་བྲང་ལ་པྲངས་འབུར་ཞིག་འདུག་པ་དེ་ངོ་ཚ་མོད། ། ཡང་ཁྱོད་ཀྱི་ཡིད་ལ་ང་ལྟོ་གོས་ལ་གྱོང་བསྐུར་ནས་སྒོམ་པ་འདི། ཟང་ཟིང་མ་འབྱོར་བ་ཡིན་པ་འདུག་སྙམ་སྟེ་དེ་ལྟ་མིན། སྤྱིར་གྱིས་ང་རང་འཁོར་བ་ངན་སོང་གི་སྡུག་བསྔལ་འདི་ལ་འཇིགས་དྲགས། གསོན་པོར་མེ་ནང་དུ་བཅུག་པ་ལྟ་བུའི་འཇིགས་སྐྲག་ཡོད་པས། འདུ་འཛི་དང་གཡེང་བ་ཆོས་བརྒྱད་ཀྱི་འདོར་ལེན་མཐོང་ཚེ། སྐྱུག་ནད་པ་ལ་ཟན་བྱིན་པ་ལྟར་ཞེ་ལོག ཕ་བསད་ཀྱི་ལག་དམར་པོ་མཐོང་བ་ལྟར་སེམས་མི་དགའ་བ་ཞིག་ཡོད་པས་སྤངས་པ་ཡིན།
