ཨ་ནེ་ཟམ་སྣར་སླེབ་བྱུང་སྟེ། ཚ་མོ་ཟམ་པ་མ་སྡོག་ཅིག་ཨ་ནེ་ཡོང་གི་ཡོད་དོ་ཟེར་བ་ལ། དེ་ཀས་སྡོག་པ་ཡིན་བྱས་པས། ཚ་མོ་བདེན་ཏེ། ད་ང་འགྱོད་པ་དྲག་པོ་རང་ཞིག་སྐྱེས་ནས་ཁྱེད་མིང་སྲིང་དང་ཕྲད་དུ་ཡོངས་པས་ཟམ་པ་ཚུགས། མི་འཛུགས་ན་ཨ་ཇོ་ལ་ང་སླེབ་འདུག་གྱིས་དང་ཟེར་བས། ངས་ཀྱང་ཟམ་པའི་ཚུར་སྣ་བྲག་ཐོག་ཅིག་གི་ཁར་ཕྱིན་ནས་བསྡད་པས། ཨ་ནེ་ཕྱག་འཚལ་ཞིང་འཕྲད་དགོས་པའི་ལོ་རྒྱུས་མང་པོ་བྱེད་ཅིང་འདུག་པས། ངའི་བསམ་པ་ལ་གཏིང་ཡེ་མ་ཕྲད་ན་ཆོས་པར་ཡང་མི་རུང་། ཐོག་མར་ཤགས་ཤིག་འདེབས་དགོས་སྙམ། ངས་ང་སྤྱིར་གཉེན་གྱི་ཁོ་ཐག་ཆོད། སྒོས་ཨ་ཁུ་དང་ཨ་ནེའི་ཁོ་ཐག་ཆོད་དེ། དང་པོ་སྡུག་དེ་བཏང་བའི་ཁར། ཆོས་བྱས་རྟིང་སློངས་མོ་ལ་ཡོངས་དུས་ཡང་སྡུག་དེ་བཏང་བས། ཨ་ཁུ་དང་ཨ་ནེའི་ཁོ་ཐག་ཆོད་དོ། ། དེའི་རྒྱུ་མཚན་གླུ་འདི་ལྟར་ཡིན་པས་ཉོན་ཅིག་བྱས་ཨ་ནེ་ཁྲེལ་འདེབས་ཀྱི་གླུ་འདི་ལྟར་བླངས་སོ། ། ། ། ཕ་ཀུན་ལ་བརྩེ་བའི་ཐུགས་རྗེ་ཅན། ། སྒྲ་བསྒྱུར་མར་པའི་ཞབས་ལ་འདུད། ། སྤྲང་གཉེན་མེད་བདག་གི་གཉེན་མཛོད་ཅིག ། ལས་བྱས་པ་དྲན་ནམ་ཨ་ནེ་སོམས། ། མི་དྲན་གླུ་ཡིས་གསལ་ཅིག་འདེབས། ། མ་ཡེངས་འགྱོད་སེམས་བསྐྱེད་ལ་ཉོན། ། སྐྱ་རྔ་རྩ་ཡི་ཀོ་རོན་ན། ། ཕ་བཟང་གིས་བོར་བའི་མ་བུ་གསུམ། ། ནོར་ཡོད་དགུ་ཕྲོགས་ནས་སྡུག་དགུ་བཅོལ། ། མཐར་སྲན་ཕུང་བེར་ཀས་གཏོར་བ་བཞིན། ། ཁྱེད་ཨ་ཁུ་ཨ་ནེས་བཏང་བས་ན། ། སྤྱིར་གཉེན་གྱི་ཁོ་ཐག་དེ་ནས་ཆོད། ། སྒོས་ང་ས་མཐར་འཁྱར་བའི་ཚེ། ། མ་སྲིང་དྲན་ནས་ཡུལ་དུ་ཡོངས། ། མ་ཤི་སྲིང་མོ་འཁྱར་བ་དེས། ། སྐྱོ་ཤས་སྙིང་རླུང་ལངས་པའི་མཐུས། ། རྩེ་གཅིག་སྒྲུབ་པ་བྱེད་པའི་ཚེ། ། འཚོ་བ་ཟས་ཀྱིས་བཀྲེས་ཕོངས་བྱུང་། ། གཉེན་ཨ་ནེའི་སྒོ་དྲུང་སློང་དུ་ཡོངས། །
