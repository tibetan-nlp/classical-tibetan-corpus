ཁོ་བོ་གང་ལ་སྐྱབས་སུ་འགྲོ་བ་མི་ཤེས་པས་མྱ་ངན་བྱེད་དོ། ། བི་ཤྭ་ཀརྨས་སྨྲས་པ། ལྷའི་དབང་པོ་འཛམ་བུའི་གླིང་ན་རྒྱལ་པོ་ཆེན་པོ་བྱང་ཆུབ་སེམས་དཔའི་སྤྱད་པ་སྤྱོད་པ་ཤི་བི་ཞེས་བྱ་བ་ཞིག་བཞུགས་ཏེ། ཤིན་ཏུ་ཡི་དམ་ལ་བརྟན་ཞིང་བརྩོན་འགྲུས་བརྩམས་པས་གདོན་མི་ཟ་བར་མངོན་པར་འཚང་རྒྱ་སྟེ། དེ་ལ་སྐྱབས་སུ་སོང་ན་གདོན་མི་ཟ་བར་དཔུང་གཉེན་དུ་འགྱུར་ཏེ། བགེགས་ལས་རྣམ་པར་ཐར་བར་འགྱུར་རོ། ། བརྒྱ་བྱིན་གྱིས་སྨྲས་པ། གལ་ཏེ་བྱང་ཆུབ་སེམས་དཔའ་ཡིན་ན། སྔར་བགམ་དགོས་ཀྱིས་སད་དུ་འདོང་ངོ་། ། ཁྱོད་ཀྱིས་ནི་ཕུག་རོན་ཞིག་ཏུ་སྤྲུལ་ཅིག ། ང་ནི་ཁྲ་བཞིན་དུ་སྤྲུལ་ཏེ། མགྱོགས་པར་ཁྱོད་ཀྱི་ཕྱི་བཞིན་དུ་འོང་གི་རྒྱལ་པོ་དེ་གང་ན་བཞུགས་པར་སོང་ལ། སྐྱབས་བཙལ་ནས་བགམ་སྟེ། དེ་སྒྱུ་ཡོད་དམ། བདེན་པ་རྟོགས་པར་འགྱུར་རོ། ། བི་ཤྭ་ཀརྨས་སྨྲས་པ། ལྷའི་དབང་པོ་སྐྱེས་བུ་ཆེན་པོ་བྱང་ཆུབ་སེམས་དཔའ་ལ་འཚེ་བ་མི་ཤའི་མཆོད་པའི་རིགས་ཀྱིས། ཤིན་ཏུ་དཀའ་བའི་དངོས་པོ་འདི་ལྟ་བུས་གཙེ་བའི་མི་རིགས་སོ་ཞེས་སྨྲས་པ་དང་། བརྒྱ་བྱིན་ལྷའི་དབང་པོས་ཚིགས་སུ་བཅད་པ་འདི་དག་སྨྲས་སོ། ། བདག་ནི་དེ་ལ་ངན་སེམས་མིན། གསེར་ལྟར་སད་ནས་མངོན་པར་འགྱུར། བདེན་པ་རྟོགས་པར་འདོད་པ་ན། ། བྱང་ཆུབ་སེམས་དཔའ་སད་པར་བྱ། ། ཞེས་ཚིགས་སུ་བཅད་པ་དེ་སྐད་སྨྲས་ནས། བི་ཤྭ་ཀརྨ་ནི་ཕུག་རོན་དུ་སྤྲུལ། བརྒྱ་བྱིན་ནི་ཁྲ་རུ་སྤྲུལ། ཕུག་རོན་གྱི་ཕྱི་བཞིན་དུ་འབྲངས་ཏེ། བཟུང་ལ་ཐུག་པ་ལྟར་བྱས་པ་དང་། ཕུག་རོན་སྐྲག་དངངས་ནས། རྒྱལ་པོའི་མཆན་ཁུང་དུ་ཞུགས་ཏེ། རྒྱལ་པོ་ལ་སྲོག་གི་སྐྱབས་གསོལ་ཏོ། ། ཁྲ་ཡང་མོད་ལ་ཕྱིན་ཏེ། ཕོ་བྲང་གི་དྲུག་གང་ན་བ་དེར་འདུག་ནས། རྒྱལ་པོ་ལ་འདི་སྐད་ཅེས་གསོལ་ཏོ། ། ཕུག་རོན་འདི་ནི་བདག་གི་ཟས་ལགས་ཏེ། འདིར་རྒྱལ་པོ་ལ་བབས་ཀྱིས་མྱུར་དུ་བདག་ལ་སྩོལ་ཅིག ། བདག་ནི་བཀྲེས་པས་རབ་ཏུ་ཉེན་ཏོ་ཞེས་སྨྲས་པ་དང་། རྒྱལ་པོས་སྨྲས་པ། ཁོ་བོས་ནི་གང་བདག་ལ་སྐྱབས་སུ་འོངས་སོ་ཅོག་ལ་ཕྱིར་མི་གཏོང་བར་དམ་བཅས་པས། འདི་ཡང་མི་སྦྱིན་ནོ་ཞེས་སྨྲས་པ་དང་། ཡང་ཁྲས་སྨྲས་པ།
