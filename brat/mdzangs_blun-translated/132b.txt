ཅི་དང་ཅི་འདོད་པ་ཡིད་བཞིན་དུ་སྦྱིན་ནོ་ཞེས་བསྒོའོ། ། དེ་ནས་བྲམ་ཟེའི་ལེའུ་དུ་ཆུ་ཞེས་བྱ་བ་ཕོ་བྲང་གི་སྒོར་འོངས་ནས་འདི་སྐད་ཅེས་སྨྲས་སོ། ། བདག་ལ་དམ་པའི་ཆོས་ཡོད་དོ། གང་ཐོས་པར་འདོད་པ་ལ་བསྟན་པར་བྱའོ། ། ཞེས་སྨྲས་པ་རྒྱལ་པོས་ཐོས་ནས་རབ་ཏུ་དགའ་མགུ་རངས་ཏེ། ཕྱི་རོལ་ཏུ་བསུས་ནས་རྐང་པ་ལ་མགོ་བོས་ཕྱག་འཚལ་ཏེ། ཚིག་སྙན་པར་སྨྲས་ནས། ཕོ་བྲང་ནང་དུ་ཁྲིད་དེ། དེ་སྟན་བཏིང་བ་ལ་བཞགས་ནས་ཐལ་མོས་སྦྱར་ཏེ། འདི་སྐད་ཅེས་སྨྲས་སོ། སྟོན་པ་ཆེན་པོ་བདག་ལ་དམ་པའི་ཆོས་བསྟན་དུ་གསོལ། བྲམ་ཟེས་སྨྲས་པ། བདག་གིས་ལུས་ཀུན་ཏུ་སྡུག་བསྔལ་སྣ་ཚོགས་མྱོང་ནས་ཡུན་ཡང་རིང་པོ་ཞིག་ལོན་ཏེ། གདོད་ལོབས་པས། རྒྱལ་པོ་ཆེན་པོ་ཐོས་པར་འདོད་ན། འདི་ཙམ་གྱིས་ཆོག་པ་མ་ཡིན་ནོ། ། རྒྱལ་པོ་ཐལ་མོ་སྦྱར་ཏེ། ཅི་དང་། ཅི་འདོད་པ་སྨྲོས་ཤིག་དང་། སྟོན་པ་ཁྱོད་ལ་མི་ཕངས་པ་ཐམས་ཅད་འབུལ་ལོ། ། བྲམ་ཟེས་སྨྲས་པ། གལ་ཏེ་ཁྱོད་ཀྱི་ལུས་ལ་ལྕགས་གཟེར་སྟོང་གིས་བཏབ་ན། ཁྱོད་ལ་ཆོས་སྦྱིན་ནོ་ཞེས་སྨྲས་པ་དང་། རྒྱལ་པོས་སྨྲས་པ། དེ་བཞིན་དུ་བགྱིའོ། ། དེ་སྟེ་ཞག་བདུན་ན་དངོས་པོ་འདི་དག་བསྒྲུབ་པོ་ཞེས་སྨྲས་སོ། ། དེའི་ཚེ་ན་རྒྱལ་པོ་དེས་ཕོ་ཉ་ཞིག་གླང་པོ་ཆེ་རྒྱང་གྲགས་བརྒྱད་ཁྲི་བཞི་སྟོང་སྙེད་དུ་འགྲོ་བ་ཞིག་ལ་བསྐྱོན་ཏེ། འཛམ་བུའི་གླིང་ཀུན་དུ་འདི་སྐད་ཅེས་བསྒོའོ། ། རྒྱལ་པོ་ཆེན་པོ་བྱིང་ལིང་གར་ལིས། དེ་སྟེ་ཞག་བདུན་ན་ལུས་ལ་ལྕགས་གཟེར་སྟོང་གིས་འདེབས་སོ་ཞེས་བསྒོ་བ་དང་། བློན་པོ་དང་འགྲོ་བ་མང་པོས་ཐོས་ནས་ཐམས་ཅད་རྒྱལ་པོའི་དྲུང་དུ་ལྷགས་ཏེ། འདི་སྐད་ཅེས་གསོལ་ཏོ། ། བདག་ཅག་ཕྱོགས་བཞིན་འཁོད་པའི་འགྲོ་བ་ཐམས་ཅད་རྒྱལ་པོའི་ཡོན་ཏན་དང་། བཀའ་དྲིན་གྱིས་སོ་སོ་ནས་བདེ་བ་དང་སྐྱིད་པ་ལ་ལོངས་སྤྱོད་དེ། རྒྱལ་པོ་ཆེན་པོ་ད་བདག་གི་སླད་དུ་གསོལ་པ་བཏབ་སྟེ། སྐུ་ལ་ལྕགས་གཟེར་སྟོང་གིས་མི་གདབ་ཏུ་གསོལ་ཞེས་གསོལ་པ། དེའི་ཚེ་ན་བཙུན་མོ་དང་། ཕོ་བྲང་གི་སླས་དང་། རྒྱལ་ཚབ་དང་། རྒྱལ་བུ་དང་། བློན་པོ་ཐམས་ཅད་ཀྱིས་ཀྱང་གསོལ་བ་བཏབ་སྟེ།
